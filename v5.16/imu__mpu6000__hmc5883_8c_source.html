146 void imu_mpu_hmc_event(void)
147 {
148  uint32_t now_ts = get_sys_time_usec();
149 
150  mpu60x0_spi_event(&imu_mpu_hmc.mpu);
151  if (imu_mpu_hmc.mpu.data_available) {
152  // set channel order
153  struct Int32Vect3 accel = {
154  IMU_MPU_X_SIGN * (int32_t)(imu_mpu_hmc.mpu.data_accel.value[IMU_MPU_CHAN_X]),
155  IMU_MPU_Y_SIGN * (int32_t)(imu_mpu_hmc.mpu.data_accel.value[IMU_MPU_CHAN_Y]),
156  IMU_MPU_Z_SIGN * (int32_t)(imu_mpu_hmc.mpu.data_accel.value[IMU_MPU_CHAN_Z])
157  };
158  struct Int32Rates rates = {
159  IMU_MPU_X_SIGN * (int32_t)(imu_mpu_hmc.mpu.data_rates.value[IMU_MPU_CHAN_X]),
160  IMU_MPU_Y_SIGN * (int32_t)(imu_mpu_hmc.mpu.data_rates.value[IMU_MPU_CHAN_Y]),
161  IMU_MPU_Z_SIGN * (int32_t)(imu_mpu_hmc.mpu.data_rates.value[IMU_MPU_CHAN_Z])
162  };
163  // unscaled vector
164  VECT3_COPY(imu.accel_unscaled, accel);
165  RATES_COPY(imu.gyro_unscaled, rates);
166 
167  imu_mpu_hmc.mpu.data_available = false;
168  imu_scale_gyro(&imu);
169  imu_scale_accel(&imu);
170  AbiSendMsgIMU_GYRO_INT32(IMU_MPU6000_HMC_ID, now_ts, &imu.gyro);
171  AbiSendMsgIMU_ACCEL_INT32(IMU_MPU6000_HMC_ID, now_ts, &imu.accel);
172  }
173 
174  /* HMC58XX event task */
175  hmc58xx_event(&imu_mpu_hmc.hmc);
176  if (imu_mpu_hmc.hmc.data_available) {
177  /* mag by default rotated by 90deg around z axis relative to MPU */
178  imu.mag_unscaled.x = IMU_HMC_X_SIGN * imu_mpu_hmc.hmc.data.value[IMU_HMC_CHAN_X];
179  imu.mag_unscaled.y = IMU_HMC_Y_SIGN * imu_mpu_hmc.hmc.data.value[IMU_HMC_CHAN_Y];
180  imu.mag_unscaled.z = IMU_HMC_Z_SIGN * imu_mpu_hmc.hmc.data.value[IMU_HMC_CHAN_Z];
181  imu_mpu_hmc.hmc.data_available = false;
182  imu_scale_mag(&imu);
183  AbiSendMsgIMU_MAG_INT32(IMU_MPU6000_HMC_ID, now_ts, &imu.mag);
184  }
185 }
