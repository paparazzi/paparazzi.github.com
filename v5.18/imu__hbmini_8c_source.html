93 void imu_hbmini_event(void)
94 {
95  uint32_t now_ts = get_sys_time_usec();
96 
97  max1168_event();
98 
99  if (max1168_status == MAX1168_DATA_AVAILABLE) {
100  imu.gyro_unscaled.p = max1168_values[IMU_GYRO_P_CHAN];
101  imu.gyro_unscaled.q = max1168_values[IMU_GYRO_Q_CHAN];
102  imu.gyro_unscaled.r = max1168_values[IMU_GYRO_R_CHAN];
103  imu.accel_unscaled.x = max1168_values[IMU_ACCEL_X_CHAN];
104  imu.accel_unscaled.y = max1168_values[IMU_ACCEL_Y_CHAN];
105  imu.accel_unscaled.z = max1168_values[IMU_ACCEL_Z_CHAN];
106  max1168_status = MAX1168_IDLE;
107  imu_scale_gyro(&imu);
108  imu_scale_accel(&imu);
109  AbiSendMsgIMU_GYRO_INT32(IMU_BOARD_ID, now_ts, &imu.gyro);
110  AbiSendMsgIMU_ACCEL_INT32(IMU_BOARD_ID, now_ts, &imu.accel);
111  }
112 
113  // HMC58XX event task
114  hmc58xx_event(&imu_hbmini.hmc);
115  if (imu_hbmini.hmc.data_available) {
116  imu.mag_unscaled.z = imu_hbmini.hmc.data.value[IMU_MAG_X_CHAN];
117  imu.mag_unscaled.y = imu_hbmini.hmc.data.value[IMU_MAG_Y_CHAN];
118  imu.mag_unscaled.x = imu_hbmini.hmc.data.value[IMU_MAG_Z_CHAN];
119 
120  imu_hbmini.hmc.data_available = false;
121  imu_scale_mag(&imu);
122  AbiSendMsgIMU_MAG_INT32(IMU_BOARD_ID, now_ts, &imu.mag);
123  }
124 
125 }
