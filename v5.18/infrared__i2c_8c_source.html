101 void infrared_i2c_update(void)
102 {
103 #if ! (defined SITL || defined HITL)
104  // IR horizontal
105  if (irh_trans.status == I2CTransDone && ir_i2c_hor_status == IR_I2C_IDLE) {
106  if (ValidConfWord(ir_i2c_conf_word) && !ir_i2c_conf_hor_done) {
107  irh_trans.buf[0] = ir_i2c_conf_word | IR_HOR_OC_BIT | IR_START_CONV ;
108  i2c_transmit(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 1);
109  ir_i2c_hor_status = IR_I2C_CONFIGURE_HOR;
110  } else {
111  // Read next values
112  i2c_receive(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 3);
113  ir_i2c_data_hor_available = false;
114  ir_i2c_hor_status = IR_I2C_READ_IR1;
115  }
116  }
117  // IR vertical
118  if (irv_trans.status == I2CTransDone) {
119  if (ValidConfWord(ir_i2c_conf_word) && !ir_i2c_conf_ver_done) {
120  irv_trans.buf[0] = ir_i2c_conf_word | IR_VER_OC_BIT;
121  i2c_transmit(&i2c0, &irv_trans, IR_VER_I2C_ADDR, 1);
122  } else {
123  // Read next values
124  i2c_receive(&i2c0, &irv_trans, IR_VER_I2C_ADDR, 2);
125  ir_i2c_data_ver_available = false;
126  }
127  }
128 #endif /* SITL || HITL */
129 }

133 void infrared_i2c_hor_event(void)
134 {
135 #if ! (defined SITL || defined HITL)
136  irh_trans.status = I2CTransDone;
137  switch (ir_i2c_hor_status) {
138  case IR_I2C_IDLE :
139  break;
140  case IR_I2C_READ_IR1 :
141  if (bit_is_set(irh_trans.buf[2], 7)) {
142  i2c_receive(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 3);
143  break;
144  }
145  // Read IR1 value
146  int16_t ir1 = (irh_trans.buf[0] << 8) | irh_trans.buf[1];
147  ir1 = ir1 - (IR_I2C_IR1_NEUTRAL << ir_i2c_conf_word);
148  ir_i2c.ir1 = FilterIR(ir_i2c.ir1, ir1);
149  // Select IR2 channel
150  irh_trans.buf[0] = IR_HOR_I2C_SELECT_IR2 | IR_HOR_OC_BIT | ir_i2c_conf_word | IR_START_CONV;
151  i2c_transmit(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 1);
152  ir_i2c_hor_status = IR_I2C_IR2_SELECTED;
153  break;
154  case IR_I2C_IR2_SELECTED :
155  // IR2 selected, asking for IR2 value
156  i2c_receive(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 3);
157  ir_i2c_hor_status = IR_I2C_READ_IR2;
158  break;
159  case IR_I2C_READ_IR2 :
160  // Read IR2 value
161  if (bit_is_set(irh_trans.buf[2], 7)) {
162  i2c_receive(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 3);
163  break;
164  }
165  int16_t ir2 = (irh_trans.buf[0] << 8) | irh_trans.buf[1];
166  ir2 = ir2 - (IR_I2C_IR2_NEUTRAL << ir_i2c_conf_word);
167  ir_i2c.ir2 = FilterIR(ir_i2c.ir2, ir2);
168  // Update estimator
169  ir_i2c_data_hor_available = true;
170 #ifndef IR_I2C_READ_ONLY
171  if (ir_i2c_data_ver_available) {
172  ir_i2c_data_hor_available = false;
173  ir_i2c_data_ver_available = false;
174  UpdateIRValue(ir_i2c);
175  }
176 #endif
177  // Select IR1 channel
178  irh_trans.buf[0] = IR_HOR_I2C_SELECT_IR1 | IR_HOR_OC_BIT | ir_i2c_conf_word | IR_START_CONV;
179  i2c_transmit(&i2c0, &irh_trans, IR_HOR_I2C_ADDR, 1);
180  ir_i2c_hor_status = IR_I2C_IR1_SELECTED;
181  break;
182  case IR_I2C_IR1_SELECTED :
183  // End reading cycle
184  ir_i2c_hor_status = IR_I2C_IDLE;
185  break;
186  case IR_I2C_CONFIGURE_HOR :
187  // End conf cycle
188  ir_i2c_conf_hor_done = true;
189  ir_i2c_hor_status = IR_I2C_IDLE;
190  break;
191  }
192 #endif /* !SITL && !HITL */
193 }

195 void infrared_i2c_ver_event(void)
196 {
197 #if ! (defined SITL || defined HITL)
198  irv_trans.status = I2CTransDone;
199  // Read TOP value
200  if (irv_trans.type == I2CTransRx) {
201  int16_t ir3 = (irv_trans.buf[0] << 8) | irv_trans.buf[1];
202  ir3 = ir3 - (IR_I2C_TOP_NEUTRAL << ir_i2c_conf_word);
203  ir_i2c.ir3 = FilterIR(ir_i2c.ir3, ir3);
204  ir_i2c_data_ver_available = true;
205 #ifndef IR_I2C_READ_ONLY
206  if (ir_i2c_data_hor_available) {
207  ir_i2c_data_hor_available = false;
208  ir_i2c_data_ver_available = false;
209  UpdateIRValue(ir_i2c);
210  }
211 #endif
212  }
213  if (irv_trans.type == I2CTransTx) {
214  ir_i2c_conf_ver_done = true;
215  }
216 #endif /* !SITL && !HITL */
217 }
