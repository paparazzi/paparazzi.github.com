56 void actuators_asctec_set(bool motors_on)
57 {
58 #if defined ACTUATORS_START_DELAY && ! defined SITL
59  if (!actuators_delay_done) {
60  if (SysTimeTimer(actuators_delay_time) < USEC_OF_SEC(ACTUATORS_START_DELAY)) { return; }
61  else { actuators_delay_done = true; }
62  }
63 #endif
64 
65  switch (actuators_asctec.i2c_trans.status) {
66  case I2CTransFailed:
67  actuators_asctec.nb_err++;
68  actuators_asctec.i2c_trans.status = I2CTransDone;
69  break;
70  case I2CTransSuccess:
71  case I2CTransDone:
72  actuators_asctec.i2c_trans.status = I2CTransDone;
73  break;
74  default:
75  actuators_asctec.nb_err++;
76  return;
77  }

79 #ifdef KILL_MOTORS
80  actuators_asctec.cmds[PITCH] = 0;
81  actuators_asctec.cmds[ROLL] = 0;
82  actuators_asctec.cmds[YAW] = 0;
83  actuators_asctec.cmds[THRUST] = 0;
84 #else /* ! KILL_MOTORS */
85  Bound(actuators_asctec.cmds[PITCH], ASCTEC_MIN_CMD, ASCTEC_MAX_CMD);
86  Bound(actuators_asctec.cmds[ROLL], ASCTEC_MIN_CMD, ASCTEC_MAX_CMD);
87  Bound(actuators_asctec.cmds[YAW], ASCTEC_MIN_CMD, ASCTEC_MAX_CMD);
88  if (motors_on) {
89  Bound(actuators_asctec.cmds[THRUST], ASCTEC_MIN_THROTTLE + 1, ASCTEC_MAX_THROTTLE);

91  actuators_asctec.cmds[THRUST] = 0;

93 #endif /* KILL_MOTORS */

113  actuators_asctec.new_addr;

115  break;
116  case NONE:
