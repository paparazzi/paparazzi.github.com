183 static uint16_t crc_update(uint16_t crc, uint8_t data)
184 {
185  data ^= (crc & 0xff);
186  data ^= data << 4;
187 
188  return ((((uint16_t)data << 8) | ((crc>>8)&0xff)) ^ (uint8_t)(data >> 4) ^ ((uint16_t)data << 3));
189 }
