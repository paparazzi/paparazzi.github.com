103void baro_bmp280_event(void)
104{
105 bmp280_event(&baro_bmp280);
106
107 if (baro_bmp280.data_available) {
108 uint32_t now_ts = get_sys_time_usec();
109 // send ABI message
110 AbiSendMsgBARO_ABS(BARO_BMP_SENDER_ID, now_ts, baro_bmp280.pressure);
111 AbiSendMsgTEMPERATURE(BARO_BMP_SENDER_ID, baro_bmp280.temperature);
112 baro_bmp280.data_available = false;
113 baro_alt = pprz_isa_altitude_of_pressure(baro_bmp280.pressure);
114 baro_alt_valid = true;
115 baro_press = (float)baro_bmp280.pressure;
116 baro_temp = ((float)baro_bmp280.temperature) / 100;
117
118#if defined(BMP280_SYNC_SEND)
119 int32_t up = (int32_t)(baro_bmp280.raw_pressure);
120 int32_t ut = (int32_t)(baro_bmp280.raw_temperature);
121 int32_t p = (int32_t) baro_bmp280.pressure;
122 int32_t t = (int32_t)(baro_bmp280.temperature);
123 DOWNLINK_SEND_BMP_STATUS(DefaultChannel, DefaultDevice, &up, &ut, &p, &t);
124#endif
125 }
