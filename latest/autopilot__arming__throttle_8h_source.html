73 static inline bool autopilot_arming_check_valid(void)
74 {
75  if (!PITCH_STICK_CENTERED()) {
76  autopilot.arming_status = AP_ARMING_STATUS_PITCH_NOT_CENTERED;
77  } else if (!ROLL_STICK_CENTERED()) {
78  autopilot.arming_status = AP_ARMING_STATUS_ROLL_NOT_CENTERED;
79  } else if (!YAW_STICK_CENTERED()) {
80  autopilot.arming_status = AP_ARMING_STATUS_YAW_NOT_CENTERED;
81  } else if (autopilot_get_mode() != MODE_MANUAL && !autopilot_unarmed_in_auto) {
82  autopilot.arming_status = AP_ARMING_STATUS_NOT_MODE_MANUAL;
83  } else if (THROTTLE_STICK_DOWN()) {
84  autopilot.arming_status = AP_ARMING_STATUS_THROTTLE_DOWN;
85  } else {
86  return true; // all checks valid
87  }
88  return false; // one of the checks failed
89 }

98 static inline void autopilot_arming_check_motors_on(void)
99 {
100 
101  /* only allow switching motor if not in KILL mode */
102  if (autopilot_get_mode() != AP_MODE_KILL) {
103 
104  switch (autopilot_arming_state) {
105  case STATE_UNINIT:
106  autopilot.motors_on = false;
107  autopilot_arming_delay_counter = 0;
108  autopilot_arming_state = STATE_WAITING;
109  break;
110  case STATE_WAITING: // after startup wait until throttle is down before attempting to arm
111  autopilot.motors_on = false;
112  autopilot_arming_delay_counter = 0;
113  if (THROTTLE_STICK_DOWN()) {
114  autopilot_arming_state = STATE_MOTORS_OFF_READY;
115  } else {
116  autopilot.arming_status = AP_ARMING_STATUS_THROTTLE_NOT_DOWN;
117  }
118  break;
119  case STATE_MOTORS_OFF_READY:
120  autopilot.motors_on = false;
121  autopilot_arming_delay_counter = 0;
122  if (autopilot_arming_check_valid()) {
123  autopilot_arming_state = STATE_ARMING;
124  }
125  break;
126  case STATE_ARMING:
127  autopilot.motors_on = false;
128  autopilot_arming_delay_counter++;
129  if (!autopilot_arming_check_valid()) {
130  autopilot_arming_state = STATE_MOTORS_OFF_READY;
131  } else if (autopilot_arming_delay_counter >= AUTOPILOT_ARMING_DELAY) {
132  autopilot_arming_state = STATE_MOTORS_ON;
133  } else {
134  autopilot.arming_status = AP_ARMING_STATUS_ARMING;
135  }
136  break;
137  case STATE_MOTORS_ON:
138  autopilot.motors_on = true;
139  autopilot.arming_status = AP_ARMING_STATUS_ARMED;
140  autopilot_arming_delay_counter = AUTOPILOT_ARMING_DELAY;
141  if (THROTTLE_STICK_DOWN()) {
142  autopilot_arming_state = STATE_UNARMING;
143  }
144  break;
145  case STATE_UNARMING:
146  autopilot.motors_on = true;
147  autopilot.arming_status = AP_ARMING_STATUS_DISARMING;
148  autopilot_arming_delay_counter--;
149  if (!THROTTLE_STICK_DOWN()) {
150  autopilot_arming_state = STATE_MOTORS_ON;
151  } else if (autopilot_arming_delay_counter == 0) {
152  autopilot_arming_state = STATE_MOTORS_OFF_READY;
153  if (autopilot_get_mode() != MODE_MANUAL) {
154  autopilot_unarmed_in_auto = true;
155  } else {
156  autopilot_unarmed_in_auto = false;
157  }
158  }
159  break;
160  default:
161  break;
162  }
163  } else {
164  autopilot.arming_status = AP_ARMING_STATUS_KILLED;
165  }
166 
167 }
