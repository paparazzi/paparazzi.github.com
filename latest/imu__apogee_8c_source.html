94 void imu_apogee_init(void)
95 {
97  // MPU-60X0
98  mpu60x0_i2c_init(&imu_apogee.mpu, &(IMU_APOGEE_I2C_DEV), MPU60X0_ADDR_ALT);
99  // change the default configuration
100  imu_apogee.mpu.config.smplrt_div = APOGEE_SMPLRT_DIV;
101  imu_apogee.mpu.config.dlpf_cfg = APOGEE_LOWPASS_FILTER;
102  imu_apogee.mpu.config.gyro_range = APOGEE_GYRO_RANGE;
103  imu_apogee.mpu.config.accel_range = APOGEE_ACCEL_RANGE;
104  // set MPU in bypass mode for the baro
105  imu_apogee.mpu.config.nb_slaves = 1;
106  imu_apogee.mpu.config.slaves[0].configure = &configure_baro_slave;
107  imu_apogee.mpu.config.i2c_bypass = true;
108  // set the default values
109  imu_set_defaults_gyro(IMU_BOARD_ID, NULL, NULL, MPU60X0_GYRO_SENS_FRAC[APOGEE_GYRO_RANGE]);
110  imu_set_defaults_accel(IMU_BOARD_ID, NULL, NULL, MPU60X0_ACCEL_SENS_FRAC[APOGEE_ACCEL_RANGE]);
111 
112 #if APOGEE_USE_MPU9150
113  // if using MPU9150, internal mag needs to be configured
114  ak8975_init(&imu_apogee.ak, &(IMU_APOGEE_I2C_DEV), AK8975_I2C_SLV_ADDR);
115  imu_apogee.mpu.config.nb_slaves = 2;
116  imu_apogee.mpu.config.slaves[1].configure = &configure_mag_slave;
117 #endif
118 }

131 void imu_apogee_event(void)
132 {
133  uint32_t now_ts = get_sys_time_usec();
134 
135  // If the itg3200 I2C transaction has succeeded: convert the data
136  mpu60x0_i2c_event(&imu_apogee.mpu);
137  if (imu_apogee.mpu.data_available) {
138  struct Int32Rates rates = {
139  (int32_t)( imu_apogee.mpu.data_rates.value[IMU_APOGEE_CHAN_X]),
140  (int32_t)(-imu_apogee.mpu.data_rates.value[IMU_APOGEE_CHAN_Y]),
141  (int32_t)(-imu_apogee.mpu.data_rates.value[IMU_APOGEE_CHAN_Z])
142  };
143  AbiSendMsgIMU_GYRO_RAW(IMU_BOARD_ID, now_ts, &rates, 1, imu_apogee.mpu.temp);
144  struct Int32Vect3 accel = {
145  (int32_t)( imu_apogee.mpu.data_accel.value[IMU_APOGEE_CHAN_X]),
146  (int32_t)(-imu_apogee.mpu.data_accel.value[IMU_APOGEE_CHAN_Y]),
147  (int32_t)(-imu_apogee.mpu.data_accel.value[IMU_APOGEE_CHAN_Z])
148  };
149  AbiSendMsgIMU_ACCEL_RAW(IMU_BOARD_ID, now_ts, &accel, 1, imu_apogee.mpu.temp);
150  imu_apogee.mpu.data_available = false;
151  }
152 
153 #if APOGEE_USE_MPU9150
154  ak8975_event(&imu_apogee.ak);
155  if (imu_apogee.ak.data_available) {
156  struct Int32Vect3 mag = {
157  (int32_t)( imu_apogee.ak.data.value[IMU_APOGEE_CHAN_Y]),
158  (int32_t)(-imu_apogee.ak.data.value[IMU_APOGEE_CHAN_X]),
159  (int32_t)( imu_apogee.ak.data.value[IMU_APOGEE_CHAN_Z])
160  };
161  AbiSendMsgIMU_MAG_RAW(IMU_BOARD_ID, now_ts, &mag);
162  imu_apogee.ak.data_available = false;
163  }
164 #endif
165 }
