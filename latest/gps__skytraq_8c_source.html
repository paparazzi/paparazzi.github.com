98void gps_skytraq_msg(void)
99{
100 // current timestamp
101 uint32_t now_ts = get_sys_time_usec();
102 gps_skytraq.state.last_msg_ticks = sys_time.nb_sec_rem;
103 gps_skytraq.state.last_msg_time = sys_time.nb_sec;
104 gps_skytraq_read_message();
105 if (gps_skytraq.msg_id == SKYTRAQ_ID_NAVIGATION_DATA) {
106 if (gps_skytraq.state.fix == GPS_FIX_3D) {
107 gps_skytraq.state.last_3dfix_ticks = sys_time.nb_sec_rem;
108 gps_skytraq.state.last_3dfix_time = sys_time.nb_sec;
109 }
110 AbiSendMsgGPS(GPS_SKYTRAQ_ID, now_ts, &gps_skytraq.state);
111 }
112 gps_skytraq.msg_available = false;
113}

127void gps_skytraq_read_message(void)
128{
129
130 if (gps_skytraq.msg_id == SKYTRAQ_ID_NAVIGATION_DATA) {
131 gps_skytraq.state.ecef_pos.x = SKYTRAQ_NAVIGATION_DATA_ECEFX(gps_skytraq.msg_buf);
132 gps_skytraq.state.ecef_pos.y = SKYTRAQ_NAVIGATION_DATA_ECEFY(gps_skytraq.msg_buf);
133 gps_skytraq.state.ecef_pos.z = SKYTRAQ_NAVIGATION_DATA_ECEFZ(gps_skytraq.msg_buf);
134 SetBit(gps_skytraq.state.valid_fields, GPS_VALID_POS_ECEF_BIT);
135
136 gps_skytraq.state.ecef_vel.x = SKYTRAQ_NAVIGATION_DATA_ECEFVX(gps_skytraq.msg_buf);
137 gps_skytraq.state.ecef_vel.y = SKYTRAQ_NAVIGATION_DATA_ECEFVY(gps_skytraq.msg_buf);
138 gps_skytraq.state.ecef_vel.z = SKYTRAQ_NAVIGATION_DATA_ECEFVZ(gps_skytraq.msg_buf);
139 SetBit(gps_skytraq.state.valid_fields, GPS_VALID_VEL_ECEF_BIT);
140
141 gps_skytraq.state.lla_pos.lat = SKYTRAQ_NAVIGATION_DATA_LAT(gps_skytraq.msg_buf);
142 gps_skytraq.state.lla_pos.lon = SKYTRAQ_NAVIGATION_DATA_LON(gps_skytraq.msg_buf);
143 gps_skytraq.state.lla_pos.alt = SKYTRAQ_NAVIGATION_DATA_AEL(gps_skytraq.msg_buf) * 10;
144 SetBit(gps_skytraq.state.valid_fields, GPS_VALID_POS_LLA_BIT);
145
146 gps_skytraq.state.hmsl = SKYTRAQ_NAVIGATION_DATA_ASL(gps_skytraq.msg_buf) * 10;
147 SetBit(gps_skytraq.state.valid_fields, GPS_VALID_HMSL_BIT);
148
149 // pacc;
150 // sacc;
151 gps_skytraq.state.pdop = SKYTRAQ_NAVIGATION_DATA_PDOP(gps_skytraq.msg_buf);
152 gps_skytraq.state.num_sv = SKYTRAQ_NAVIGATION_DATA_NumSV(gps_skytraq.msg_buf);
153 gps_skytraq.state.tow = SKYTRAQ_NAVIGATION_DATA_TOW(gps_skytraq.msg_buf) * 10;
154
155 switch (SKYTRAQ_NAVIGATION_DATA_FixMode(gps_skytraq.msg_buf)) {
156 case SKYTRAQ_FIX_3D_DGPS:
157 case SKYTRAQ_FIX_3D:
158 gps_skytraq.state.fix = GPS_FIX_3D;
159 break;
160 case SKYTRAQ_FIX_2D:
161 gps_skytraq.state.fix = GPS_FIX_2D;
162 break;
163 default:
164 gps_skytraq.state.fix = GPS_FIX_NONE;
165 }
166
167 if (gps_skytraq.state.fix == GPS_FIX_3D) {
168 if (distance_too_great(&gps_skytraq.ref_ltp.ecef, &gps_skytraq.state.ecef_pos)) {
169 // just grab current ecef_pos as reference.
170 ltp_def_from_ecef_i(&gps_skytraq.ref_ltp, &gps_skytraq.state.ecef_pos);
171 }
172 // convert ecef velocity vector to NED vector.
173 ned_of_ecef_vect_i(&gps_skytraq.state.ned_vel, &gps_skytraq.ref_ltp, &gps_skytraq.state.ecef_vel);
174 SetBit(gps_skytraq.state.valid_fields, GPS_VALID_VEL_NED_BIT);
175
176 // ground course in radians
177 gps_skytraq.state.course = (atan2f((float)gps_skytraq.state.ned_vel.y, (float)gps_skytraq.state.ned_vel.x)) * 1e7;
178 SetBit(gps_skytraq.state.valid_fields, GPS_VALID_COURSE_BIT);
179 // GT: gps_skytraq.state.cacc = ... ? what should course accuracy be?
180
181 // ground speed
182 gps_skytraq.state.gspeed = sqrt(gps_skytraq.state.ned_vel.x * gps_skytraq.state.ned_vel.x + gps_skytraq.state.ned_vel.y * gps_skytraq.state.ned_vel.y);
183 gps_skytraq.state.speed_3d = sqrt(gps_skytraq.state.ned_vel.x * gps_skytraq.state.ned_vel.x + gps_skytraq.state.ned_vel.y * gps_skytraq.state.ned_vel.y + gps_skytraq.state.ned_vel.z * gps_skytraq.state.ned_vel.z);
184
185 // vertical speed (climb)
186 // solved by gps_skytraq.state.ned.z?
187 }
188
189
190#ifdef GPS_LED
191 if (gps_skytraq.state.fix == GPS_FIX_3D) {
192 LED_ON(GPS_LED);
193 } else {
194 LED_TOGGLE(GPS_LED);
195 }
196#endif
197 }

201void gps_skytraq_parse(uint8_t c)
202{
203 if (gps_skytraq.status < GOT_PAYLOAD) {
204 gps_skytraq.checksum ^= c;
205 }
206 switch (gps_skytraq.status) {
207 case UNINIT:
208 if (c == SKYTRAQ_SYNC1) {
209 gps_skytraq.status = GOT_SYNC1;
210 }
211 break;
212 case GOT_SYNC1:
213 if (c != SKYTRAQ_SYNC2) {
214 gps_skytraq.error_last = GPS_SKYTRAQ_ERR_OUT_OF_SYNC;
215 goto error;
216 }
217 gps_skytraq.status = GOT_SYNC2;
218 break;
219 case GOT_SYNC2:
220 gps_skytraq.len = c << 8;
221 gps_skytraq.status = GOT_LEN1;
222 break;
223 case GOT_LEN1:
224 gps_skytraq.len += c;
225 gps_skytraq.status = GOT_LEN2;
226 if (gps_skytraq.len > GPS_SKYTRAQ_MAX_PAYLOAD) {
227 gps_skytraq.error_last = GPS_SKYTRAQ_ERR_MSG_TOO_LONG;
228 goto error;
229 }
230 break;
231 case GOT_LEN2:
232 gps_skytraq.msg_id = c;
233 gps_skytraq.msg_idx = 0;
234 gps_skytraq.checksum = c;
235 gps_skytraq.status = GOT_ID;
236 break;
237 case GOT_ID:
238 gps_skytraq.msg_buf[gps_skytraq.msg_idx] = c;
239 gps_skytraq.msg_idx++;
240 if (gps_skytraq.msg_idx >= gps_skytraq.len - 1) {
241 gps_skytraq.status = GOT_PAYLOAD;
242 }
243 break;
244 case GOT_PAYLOAD:
245 if (c != gps_skytraq.checksum) {
246 gps_skytraq.error_last = GPS_SKYTRAQ_ERR_CHECKSUM;
247 goto error;
248 }
249 gps_skytraq.status = GOT_CHECKSUM;
250 break;
251 case GOT_CHECKSUM:
252 if (c != SKYTRAQ_SYNC3) {
253 gps_skytraq.error_last = GPS_SKYTRAQ_ERR_OUT_OF_SYNC;
254 goto error;
255 }
256 gps_skytraq.status = GOT_SYNC3;
257 break;
258 case GOT_SYNC3:
259 gps_skytraq.msg_available = true;
260 goto restart;
261 default:
262 gps_skytraq.error_last = GPS_SKYTRAQ_ERR_UNEXPECTED;
263 goto error;
264 }
265 return;
266error:
267 gps_skytraq.error_cnt++;
268restart:
269 gps_skytraq.status = UNINIT;
270 return;
271}

273static int distance_too_great(struct EcefCoor_i *ecef_ref, struct EcefCoor_i *ecef_pos)
274{
275 int32_t xdiff = abs(ecef_ref->x - ecef_pos->x);
276 if (xdiff > MAX_DISTANCE) {
277 return true;
278 }
279 int32_t ydiff = abs(ecef_ref->y - ecef_pos->y);
280 if (ydiff > MAX_DISTANCE) {
281 return true;
282 }
283 int32_t zdiff = abs(ecef_ref->z - ecef_pos->z);
284 if (zdiff > MAX_DISTANCE) {
285 return true;
286 }
287
288 return false;
289}
