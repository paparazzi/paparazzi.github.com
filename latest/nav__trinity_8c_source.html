87static struct EnuCoor_f process_new_point_trinity(struct EnuCoor_f *position, float alt_sp, float uav_direction)
88{
89 struct EnuCoor_f new_point;
90 float rot_angle;
91
92 if (nav_trinity.rotation == TRINITY_RIGHT) {
93 rot_angle = -M_PI_2;
94 } else{
95 rot_angle = M_PI_2;
96 }
97
98 new_point.x = position->x + (cosf(rot_angle + uav_direction) * nav_trinity.radius);
99 new_point.y = position->y + (sinf(rot_angle + uav_direction) * nav_trinity.radius);
100 new_point.z = alt_sp;
101
102 return new_point;
103}

116static void update_target_point(struct EnuCoor_f *target, struct EnuCoor_f *border, float dt, float tau)
117{
118 // with a proper discrete transform, coeff should be exp(-dt/tau) and (1-exp(-dt/tau))
119 // but to make it simpler with the same behavior at the limits, we use tau/(dt+tau) and dt/(dt+tau)
120 // with a positive value for tau
121 if (tau > 1e-4) {
122 float alpha = dt / (dt + tau);
123 target->x = (border->x * alpha) + (target->x * (1.f - alpha));
124 target->y = (border->y * alpha) + (target->y * (1.f - alpha));
125 }
126}

132static bool nav_trinity_mission(uint8_t nb, float *params, enum MissionRunFlag flag)
133{
134 if (flag == MissionInit && nb == 8) {
135 float start_x = params[0];
136 float start_y = params[1];
137 float start_z = params[2];
138 int first_turn = params[3];
139 float circle_radius = params[4];
140 float vx = params[5];
141 float vy = params[6];
142 float vz = params[7];
143 nav_trinity_setup(start_x, start_y, start_z, first_turn, circle_radius, vx, vy, vz);
144 return true;
145 }
146 else if (flag == MissionUpdate && nb == 3) {
147 // update target 3D position (ENU frame, above ground alt)
148 float x = params[0];
149 float y = params[1];
150 float z = params[2];
151 VECT3_ASSIGN(nav_trinity.target, x, y, z);
152 return true;
153 }
154 else if (flag == MissionUpdate && nb == 2) {
155 // update horizontal speed
156 float vx = params[0];
157 float vy = params[1];
158 nav_trinity.pos_incr.x = vx * nav_dt;
159 nav_trinity.pos_incr.y = vy * nav_dt;
160 return true;
161 }
162 else if (flag == MissionUpdate && nb == 1) {
163 // update vertical speed
164 float vz = params[0];
165 nav_trinity.pos_incr.z = vz * nav_dt;
166 return true;
167 }
168 else if (flag == MissionRun) {
169 return nav_trinity_run();
170 }
171 return false; // not a valid case
172}
173#endif

205void nav_trinity_setup(float init_x, float init_y,
206 float init_z, int turn,
207 float desired_radius, float vx,
208 float vy, float vz)
209{
210 struct EnuCoor_f start = {init_x, init_y, init_z};
211 // increment based on speed
212 VECT3_ASSIGN(nav_trinity.pos_incr, vx*nav_dt, vy*nav_dt, vz*nav_dt);
213
214 nav_trinity.target = start;
215 nav_trinity.status = TRINITY_ENTER;
216 nav_trinity.inside_cloud = false;
217 nav_trinity.radius = desired_radius;
218
219 if (turn == 1) {
220 nav_trinity.rotation = TRINITY_RIGHT;
221 nav_trinity.radius_sign = 1.0f;
222 } else {
223 nav_trinity.rotation = TRINITY_LEFT;
224 nav_trinity.radius_sign = -1.0f;
225 }
226
227 nav_trinity.actual = *stateGetPositionEnu_f();
228}

230bool nav_trinity_run(void)
231{
232 float pre_climb = 0.f;
233 float time = get_sys_time_float();
234
235 NavVerticalAutoThrottleMode(0.f); /* No pitch */
236
237 switch (nav_trinity.status) {
238 case TRINITY_ENTER:
239 // init stage
240 nav_init_stage();
241 // reach target point
242 nav_route_xy(nav_trinity.actual.x, nav_trinity.actual.y, nav_trinity.target.x, nav_trinity.target.y);
243 NavVerticalAltitudeMode(nav_trinity.target.z + ground_alt, pre_climb);
244
245 if (nav_trinity.inside_cloud) {
246 nav_trinity.status = TRINITY_INSIDE_FIRST;
247 }
248 break;
249 case TRINITY_INSIDE_FIRST:
250 // prepare inside circle
251 nav_trinity.actual = *stateGetPositionEnu_f();
252 nav_trinity.direction = change_rep(stateGetHorizontalSpeedDir_f());
253 nav_trinity.circle = process_first_point_trinity(&nav_trinity.actual, nav_trinity.target.z, nav_trinity.direction);
254 // init stage
255 nav_init_stage();
256 // update border and target for recover
257 nav_trinity.estim_border = nav_trinity.actual;
258 update_target_point(&nav_trinity.target, &nav_trinity.estim_border, time - nav_trinity.last_border_time, NAV_TRINITY_BORDER_FILTER);
259 nav_trinity.last_border_time = time;
260 // fly inside
261 nav_trinity.status = TRINITY_INSIDE;
262 break;
263 case TRINITY_INSIDE:
264 // increment center position
265 VECT3_ADD(nav_trinity.circle, nav_trinity.pos_incr);
266 nav_circle_XY(nav_trinity.circle.x, nav_trinity.circle.y , nav_trinity.radius_sign * nav_trinity.radius);
267 pre_climb = nav_trinity.pos_incr.z / nav_dt;
268 NavVerticalAltitudeMode(nav_trinity.circle.z + ground_alt, pre_climb);
269
270 if (!nav_trinity.inside_cloud) {
271 // found border, start outside
272 nav_trinity.status = TRINITY_OUTSIDE_START;
273 }
274 else if (NavCircleCountNoRewind() > NAV_TRINITY_RECOVER_MAX_TURN) {
275 // most likely lost inside
276 nav_trinity.status = TRINITY_RECOVER_START;
277 }
278 break;
279 case TRINITY_OUTSIDE_START:
280 // prepare outside circle
281 nav_trinity.actual = *stateGetPositionEnu_f();
282 nav_trinity.direction = change_rep(stateGetHorizontalSpeedDir_f());
283 nav_trinity.circle = process_new_point_trinity(&nav_trinity.actual, nav_trinity.target.z, nav_trinity.direction);
284 // init stage
285 nav_init_stage();
286 // upadte border and target for recover
287 nav_trinity.estim_border = nav_trinity.actual;
288 update_target_point(&nav_trinity.target, &nav_trinity.estim_border, time - nav_trinity.last_border_time, NAV_TRINITY_BORDER_FILTER);
289 nav_trinity.last_border_time = time;
290 // fly outside
291 nav_trinity.status = TRINITY_OUTSIDE;
292 break;
293 case TRINITY_OUTSIDE:
294 // increment center position
295 VECT3_ADD(nav_trinity.circle, nav_trinity.pos_incr);
296 pre_climb = nav_trinity.pos_incr.z / nav_dt;
297 nav_circle_XY(nav_trinity.circle.x, nav_trinity.circle.y , nav_trinity.radius_sign * nav_trinity.radius);
298 NavVerticalAltitudeMode(nav_trinity.circle.z + ground_alt, pre_climb);
299
300 if(nav_trinity.inside_cloud){
301 // found border, continue inside
302 nav_trinity.status = TRINITY_INSIDE;
303 }
304 else if (NavCircleCountNoRewind() > NAV_TRINITY_RECOVER_MAX_TURN) {
305 // most likely lost outside
306 nav_trinity.status = TRINITY_RECOVER_START;
307 }
308 break;
309 case TRINITY_RECOVER_START:
310 // prepare recovery circle or line
311 nav_trinity.recover_circle = nav_trinity.target;
312 nav_trinity.actual = *stateGetPositionEnu_f();
313 // initial recovery radius
314 nav_trinity.recover_radius = nav_trinity.radius;
315 nav_trinity.max_recover_radius = 2.0f * nav_trinity.recover_radius; // FIXME ?
316 // init stage
317 nav_init_stage();
318 if (nav_trinity.inside_cloud) {
319 nav_trinity.status = TRINITY_RECOVER_INSIDE;
320 }
321 else {
322 nav_trinity.status = TRINITY_RECOVER_OUTSIDE;
323 }
324 break;
325 case TRINITY_RECOVER_INSIDE:
326 // increment border position
327 // fly in opposite direction to cover more space
328 nav_route_xy(nav_trinity.estim_border.x, nav_trinity.estim_border.y, nav_trinity.actual.x, nav_trinity.actual.y);
329 if (!nav_trinity.inside_cloud) {
330 nav_trinity.status = TRINITY_OUTSIDE_START;
331 }
332 break;
333 case TRINITY_RECOVER_OUTSIDE:
334 // increment center position
335 VECT3_ADD(nav_trinity.recover_circle, nav_trinity.pos_incr);
336 nav_circle_XY(nav_trinity.recover_circle.x, nav_trinity.recover_circle.y , nav_trinity.radius_sign * nav_trinity.recover_radius);
337 // increment recover circle radius
338 if (nav_trinity.recover_radius < nav_trinity.max_recover_radius) {
339 nav_trinity.recover_radius += 0.5;
340 }
341 // found a new border
342 if (nav_trinity.inside_cloud) {
343 nav_trinity.status = TRINITY_INSIDE;
344 }
345 break;
346 default:
347 // error, leaving
348 return false;
349 }
350 // increment border and target positions
351 VECT3_ADD(nav_trinity.estim_border, nav_trinity.pos_incr);
352 VECT3_ADD(nav_trinity.target, nav_trinity.pos_incr);
353#ifdef WP_TARGET
354 nav_move_waypoint_enu(WP_TARGET, nav_trinity.target.x, nav_trinity.target.y, nav_trinity.target.z + ground_alt);
355 RunOnceEvery(10, nav_send_waypoint(WP_TARGET));
356#endif
357
358 return true;
359}
