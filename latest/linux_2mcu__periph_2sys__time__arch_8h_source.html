57static inline unsigned int sys_time_elapsed_us(struct timespec *prev, struct timespec *now)
58{
59 time_t d_sec = now->tv_sec - prev->tv_sec;
60 long d_nsec = now->tv_nsec - prev->tv_nsec;
61 /* wrap if negative nanoseconds */
62 if (d_nsec < 0) {
63 d_sec -= 1;
64 d_nsec += 1000000000L;
65 }
66 return d_sec * 1000000 + d_nsec / 1000;
67}
