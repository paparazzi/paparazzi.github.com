105void imu_mpu9250_init(void)
106{
107 /* MPU9250 */
108 mpu9250_i2c_init(&imu_mpu9250.mpu, &(IMU_MPU9250_I2C_DEV), IMU_MPU9250_I2C_ADDR);
109 // change the default configuration
110 imu_mpu9250.mpu.config.smplrt_div = IMU_MPU9250_SMPLRT_DIV;
111 imu_mpu9250.mpu.config.dlpf_gyro_cfg = IMU_MPU9250_GYRO_LOWPASS_FILTER;
112 imu_mpu9250.mpu.config.dlpf_accel_cfg = IMU_MPU9250_ACCEL_LOWPASS_FILTER;
113 imu_mpu9250.mpu.config.gyro_range = IMU_MPU9250_GYRO_RANGE;
114 imu_mpu9250.mpu.config.accel_range = IMU_MPU9250_ACCEL_RANGE;
115
116 // Set the default scaling
117 imu_set_defaults_gyro(IMU_MPU9250_ID, NULL, NULL, &MPU9250_GYRO_SENS_F[IMU_MPU9250_GYRO_RANGE]);
118 imu_set_defaults_accel(IMU_MPU9250_ID, NULL, NULL, &MPU9250_ACCEL_SENS_F[IMU_MPU9250_ACCEL_RANGE]);
119}

126void imu_mpu9250_event(void)
127{
128 uint32_t now_ts = get_sys_time_usec();
129
130 // If the MPU9250 I2C transaction has succeeded: convert the data
131 mpu9250_i2c_event(&imu_mpu9250.mpu);
132
133 if (imu_mpu9250.mpu.data_available) {
134 // set channel order
135 struct Int32Vect3 accel = {
136 IMU_MPU9250_X_SIGN *(int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_X]),
137 IMU_MPU9250_Y_SIGN *(int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Y]),
138 IMU_MPU9250_Z_SIGN *(int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Z])
139 };
140 struct Int32Rates rates = {
141 IMU_MPU9250_X_SIGN *(int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_X]),
142 IMU_MPU9250_Y_SIGN *(int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Y]),
143 IMU_MPU9250_Z_SIGN *(int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Z])
144 };
145
146 imu_mpu9250.mpu.data_available = false;
147 AbiSendMsgIMU_GYRO_RAW(IMU_MPU9250_ID, now_ts, &rates, 1, IMU_MPU9250_PERIODIC_FREQ, NAN);
148 AbiSendMsgIMU_ACCEL_RAW(IMU_MPU9250_ID, now_ts, &accel, 1, IMU_MPU9250_PERIODIC_FREQ, NAN);
149 }
150#if IMU_MPU9250_READ_MAG
151 // Test if mag data are updated
152 if (imu_mpu9250.mpu.akm.data_available) {
153 struct Int32Vect3 mag = {
154 IMU_MPU9250_X_SIGN *(int32_t)(imu_mpu9250.mpu.akm.data.value[IMU_MPU9250_CHAN_Y]),
155 IMU_MPU9250_Y_SIGN *(int32_t)(imu_mpu9250.mpu.akm.data.value[IMU_MPU9250_CHAN_X]),
156 -IMU_MPU9250_Z_SIGN *(int32_t)(imu_mpu9250.mpu.akm.data.value[IMU_MPU9250_CHAN_Z])
157 };
158 imu_mpu9250.mpu.akm.data_available = false;
159 AbiSendMsgIMU_MAG_RAW(IMU_MPU9250_ID, now_ts, &mag);
160
161 }
162#endif
163}
