103 void imu_mpu9250_init(void)
104 {
105  /* MPU9250 */
106  mpu9250_i2c_init(&imu_mpu9250.mpu, &(IMU_MPU9250_I2C_DEV), IMU_MPU9250_I2C_ADDR);
107  // change the default configuration
108  imu_mpu9250.mpu.config.smplrt_div = IMU_MPU9250_SMPLRT_DIV;
109  imu_mpu9250.mpu.config.dlpf_gyro_cfg = IMU_MPU9250_GYRO_LOWPASS_FILTER;
110  imu_mpu9250.mpu.config.dlpf_accel_cfg = IMU_MPU9250_ACCEL_LOWPASS_FILTER;
111  imu_mpu9250.mpu.config.gyro_range = IMU_MPU9250_GYRO_RANGE;
112  imu_mpu9250.mpu.config.accel_range = IMU_MPU9250_ACCEL_RANGE;
113 
114  // Set the default scaling
115  imu_set_defaults_gyro(IMU_MPU9250_ID, NULL, NULL, MPU9250_GYRO_SENS_FRAC[IMU_MPU9250_GYRO_RANGE]);
116  imu_set_defaults_accel(IMU_MPU9250_ID, NULL, NULL, MPU9250_ACCEL_SENS_FRAC[IMU_MPU9250_ACCEL_RANGE]);
117 }

124 void imu_mpu9250_event(void)
125 {
126  uint32_t now_ts = get_sys_time_usec();
127 
128  // If the MPU9250 I2C transaction has succeeded: convert the data
129  mpu9250_i2c_event(&imu_mpu9250.mpu);
130 
131  if (imu_mpu9250.mpu.data_available) {
132  // set channel order
133  struct Int32Vect3 accel = {
134  IMU_MPU9250_X_SIGN *(int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_X]),
135  IMU_MPU9250_Y_SIGN *(int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Y]),
136  IMU_MPU9250_Z_SIGN *(int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Z])
137  };
138  struct Int32Rates rates = {
139  IMU_MPU9250_X_SIGN *(int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_X]),
140  IMU_MPU9250_Y_SIGN *(int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Y]),
141  IMU_MPU9250_Z_SIGN *(int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Z])
142  };
143 
144  imu_mpu9250.mpu.data_available = false;
145  AbiSendMsgIMU_GYRO_RAW(IMU_MPU9250_ID, now_ts, &rates, 1);
146  AbiSendMsgIMU_ACCEL_RAW(IMU_MPU9250_ID, now_ts, &accel, 1);
147  }
148 #if IMU_MPU9250_READ_MAG
149  // Test if mag data are updated
150  if (imu_mpu9250.mpu.akm.data_available) {
151  struct Int32Vect3 mag = {
152  IMU_MPU9250_X_SIGN *(int32_t)(imu_mpu9250.mpu.akm.data.value[IMU_MPU9250_CHAN_Y]),
153  IMU_MPU9250_Y_SIGN *(int32_t)(imu_mpu9250.mpu.akm.data.value[IMU_MPU9250_CHAN_X]),
154  -IMU_MPU9250_Z_SIGN *(int32_t)(imu_mpu9250.mpu.akm.data.value[IMU_MPU9250_CHAN_Z])
155  };
156  imu_mpu9250.mpu.akm.data_available = false;
157  AbiSendMsgIMU_MAG_RAW(IMU_MPU9250_ID, now_ts, &mag);
158 
159  }
160 #endif
161 }
