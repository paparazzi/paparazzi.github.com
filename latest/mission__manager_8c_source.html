78void mavlink_mission_message_handler(const mavlink_message_t *msg)
79{
80 mavlink_block_message_handler(msg);
81
82 mavlink_wp_message_handler(msg);
83
84 if (msg->msgid == MAVLINK_MSG_ID_MISSION_ACK) {
85 MAVLINK_DEBUG("Received MISSION_ACK message\n");
86 mavlink_mission_cancel_timer();
87 mission_mgr.state = STATE_IDLE;
88 MAVLINK_DEBUG("State: %d\n", mission_mgr.state);
89 }
90}

93void mavlink_mission_periodic(void)
94{
95 // FIXME: really use the SCRIPT_ITEM message to indicate current block?
96 if (mission_mgr.current_block != get_nav_block()) {
97 mission_mgr.current_block = get_nav_block();
98 mavlink_msg_script_current_send(MAVLINK_COMM_0, get_nav_block());
99 MAVLinkSendMessage();
100 }
101 // check if we had a timeout on a transaction
102 if (sys_time_check_and_ack_timer(mission_mgr.timer_id)) {
103 mavlink_mission_cancel_timer();
104 mission_mgr.state = STATE_IDLE;
105 mission_mgr.seq = 0;
106 MAVLINK_DEBUG("Warning: Mavlink mission request timed out!\n");
107 }
108}

114 MAVLinkSendMessage();
115 MAVLINK_DEBUG("Sent MISSION_ACK message\n");
116}
