132void logger_control_effectiveness_periodic(void)
133{
134 if (LogFileIsOpen()) {
135 return;
136 }
137
138 struct Int32Rates *rates = stateGetBodyRates_i();
139 struct Int32Vect3 *accel_body = stateGetAccelBody_i();
140
141 // log time, rate and accel
142 LogWrite(pprzLogFile, LogFormatHeader,
143 get_sys_time_float(),
144 rates->p,
145 rates->q,
146 rates->r,
147 accel_body->x,
148 accel_body->y,
149 accel_body->z);
150
151 // log commands
152#if LOGGER_CONTROL_EFFECTIVENESS_COMMANDS
153 for (unsigned int i = 0; i < COMMANDS_NB; i++) {
154 LogWrite(pprzLogFile, ",%d", commands[i]);
155 }
156#endif
157
158 // log actuators
159#if LOGGER_CONTROL_EFFECTIVENESS_ACTUATORS
160 for (unsigned int i = 0; i < ACTUATORS_NB; i++) {
161 LogWrite(pprzLogFile, ",%d", actuators[i].pprz_val);
162 }
163#endif
164
165 // log position
166#if LOGGER_CONTROL_EFFECTIVENESS_POS
167 struct EnuCoor_i *pos = stateGetPositionEnu_i();
168 LogWrite(pprzLogFile, LogFormatVect3, pos->x, pos->y, pos->z);
169#endif
170
171 // log speed
172#if LOGGER_CONTROL_EFFECTIVENESS_SPEED
173 struct EnuCoor_i *speed = stateGetSpeedEnu_i();
174 LogWrite(pprzLogFile, LogFormatVect3, speed->x, speed->y, speed->z);
175#endif
176
177 // log airspeed
178#if LOGGER_CONTROL_EFFECTIVENESS_AIRSPEED
179 LogWrite(pprzLogFile, ",%.2f", stateGetAirspeed_f());
180#endif
181
182 // end line
183 LogWrite(pprzLogFile,"\n");
184}

187static void open_log(FILE *file)
188{
189 // Create output folder if necessary
190 if (access(STRINGIFY(LOGGER_CONTROL_EFFECTIVENESS_FILE_PATH), F_OK)) {
191 char save_dir_cmd[256];
192 sprintf(save_dir_cmd, "mkdir -p %s", STRINGIFY(LOGGER_CONTROL_EFFECTIVENESS_FILE_PATH));
193 if (system(save_dir_cmd) != 0) {
194 printf("[logger] Could not create log file directory %s.\n", STRINGIFY(LOGGER_CONTROL_EFFECTIVENESS_FILE_PATH));
195 return;
196 }
197 }
198
199 // Get current date/time for filename
200 char date_time[80];
201 time_t now = time(0);
202 struct tm tstruct;
203 tstruct = *localtime(&now);
204 strftime(date_time, sizeof(date_time), "%Y%m%d-%H%M%S", &tstruct);
205
206 uint32_t counter = 0;
207 char filename[512];
208
209 // Check for available files
210 sprintf(filename, "%s/%s.csv", STRINGIFY(LOGGER_CONTROL_EFFECTIVENESS_FILE_PATH), date_time);
211 while ((file = fopen(filename, "r"))) {
212 fclose(file);
213
214 sprintf(filename, "%s/%s_%05d.csv", STRINGIFY(LOGGER_FILE_PATH), date_time, counter);
215 counter++;
216 }
217
218 file = fopen(filename, "w");
219 if(!file) {
220 printf("[logger] ERROR opening log file %s!\n", filename);
221 return;
222 }
223
224 printf("[logger] Start logging to %s...\n", filename);
225}
