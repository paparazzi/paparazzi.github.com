124 void imu_mpu_hmc_init(void)
125 {
126  mpu60x0_spi_init(&imu_mpu_hmc.mpu, &IMU_MPU_SPI_DEV, IMU_MPU_SPI_SLAVE_IDX);
127  // change the default configuration
128  imu_mpu_hmc.mpu.config.smplrt_div = IMU_MPU_SMPLRT_DIV;
129  imu_mpu_hmc.mpu.config.dlpf_cfg = IMU_MPU_LOWPASS_FILTER;
130  imu_mpu_hmc.mpu.config.gyro_range = IMU_MPU_GYRO_RANGE;
131  imu_mpu_hmc.mpu.config.accel_range = IMU_MPU_ACCEL_RANGE;
132 
133  // Set the default scaling
134  imu_set_defaults_gyro(IMU_MPU6000_HMC_ID, NULL, NULL, MPU60X0_GYRO_SENS_FRAC[IMU_MPU_GYRO_RANGE]);
135  imu_set_defaults_accel(IMU_MPU6000_HMC_ID, NULL, NULL, MPU60X0_ACCEL_SENS_FRAC[IMU_MPU_ACCEL_RANGE]);
136 
137  /* initialize mag and set default options */
138  hmc58xx_init(&imu_mpu_hmc.hmc, &IMU_HMC_I2C_DEV, HMC58XX_ADDR);
139 }

152 void imu_mpu_hmc_event(void)
153 {
154  uint32_t now_ts = get_sys_time_usec();
155 
156  mpu60x0_spi_event(&imu_mpu_hmc.mpu);
157  if (imu_mpu_hmc.mpu.data_available) {
158  // set channel order
159  struct Int32Vect3 accel = {
160  IMU_MPU_X_SIGN * (int32_t)(imu_mpu_hmc.mpu.data_accel.value[IMU_MPU_CHAN_X]),
161  IMU_MPU_Y_SIGN * (int32_t)(imu_mpu_hmc.mpu.data_accel.value[IMU_MPU_CHAN_Y]),
162  IMU_MPU_Z_SIGN * (int32_t)(imu_mpu_hmc.mpu.data_accel.value[IMU_MPU_CHAN_Z])
163  };
164  struct Int32Rates rates = {
165  IMU_MPU_X_SIGN * (int32_t)(imu_mpu_hmc.mpu.data_rates.value[IMU_MPU_CHAN_X]),
166  IMU_MPU_Y_SIGN * (int32_t)(imu_mpu_hmc.mpu.data_rates.value[IMU_MPU_CHAN_Y]),
167  IMU_MPU_Z_SIGN * (int32_t)(imu_mpu_hmc.mpu.data_rates.value[IMU_MPU_CHAN_Z])
168  };
169 
170  imu_mpu_hmc.mpu.data_available = false;
171  AbiSendMsgIMU_GYRO_RAW(IMU_MPU6000_HMC_ID, now_ts, &rates, 1, IMU_MPU_HMC_PERIODIC_FREQ, imu_mpu_hmc.mpu.temp);
172  AbiSendMsgIMU_ACCEL_RAW(IMU_MPU6000_HMC_ID, now_ts, &accel, 1, IMU_MPU_HMC_PERIODIC_FREQ, imu_mpu_hmc.mpu.temp);
173  }
174 
175  /* HMC58XX event task */
176  hmc58xx_event(&imu_mpu_hmc.hmc);
177  if (imu_mpu_hmc.hmc.data_available) {
178  /* mag by default rotated by 90deg around z axis relative to MPU */
179  struct Int32Vect3 mag = {
180  IMU_HMC_X_SIGN * imu_mpu_hmc.hmc.data.value[IMU_HMC_CHAN_X],
181  IMU_HMC_Y_SIGN * imu_mpu_hmc.hmc.data.value[IMU_HMC_CHAN_Y],
182  IMU_HMC_Z_SIGN * imu_mpu_hmc.hmc.data.value[IMU_HMC_CHAN_Z]
183  };
184  imu_mpu_hmc.hmc.data_available = false;
185  AbiSendMsgIMU_MAG_RAW(IMU_MPU6000_HMC_ID, now_ts, &mag);
186  }
