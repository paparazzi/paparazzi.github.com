98{
99 int16_t x = cam_gimbal.target_pos.x;

127static void default_compute_angles(struct FloatVect3 dir, float *pan, float *tilt)
128{
129 *pan = atan2f(dir.y, dir.x);
130 *tilt = asinf(-dir.z);
131}

134static void cam_gimbal_angles(struct CamGimbal *cam)
135{
136 Bound(cam->pan_angle, cam->pan_min, cam->pan_max);
137 Bound(cam->tilt_angle, cam->tilt_min, cam->tilt_max);
138
139 if (!cam->lock) {
140 float delta_pan = cam->pan_max - cam->pan_min;
141 float delta_tilt = cam->tilt_max - cam->tilt_min;
142 cam->pan_cmd = (int16_t) MAX_PPRZ * ((2.f / delta_pan) * (cam->pan_angle - cam->pan_min) - 1.f);
143 cam->tilt_cmd = (int16_t) MAX_PPRZ * ((2.f / delta_tilt) * (cam->tilt_angle - cam->tilt_min) - 1.f);
144 }
145}

148static void cam_gimbal_target(struct CamGimbal *cam)
149{
150 struct FloatRMat *ltp_to_body = stateGetNedToBodyRMat_f();
151 struct NedCoor_f pos = *stateGetPositionNed_f();
152 struct NedCoor_f target;
153 ENU_OF_TO_NED(target, cam->target_pos);
154
155 // compute looking direction in gimbal frame
156 // o: Earth frame (ltp)
157 // b: body frame
158 // g: gimbal frame
159 // D/o = normalized(Pt/o - Pg/o = Pt/o - (Pb/o + Pg/b))
160 struct FloatVect3 dir_ltp;
161 VECT3_DIFF(dir_ltp, target, pos);
162 VECT3_SUB(dir_ltp, cam->gimbal_pos);
163 float_vect3_normalize(&dir_ltp);
164 // rotate D/o to get D/g = Rg/o D/o = inv(Rb/g) Rb/o D/o
165 struct FloatVect3 dir_body;
166 float_rmat_vmult(&dir_body, ltp_to_body, &dir_ltp);
167 struct FloatVect3 dir_gimbal;
168 float_rmat_transp_vmult(&dir_gimbal, &cam->gimbal_to_body, &dir_body);
169 // compute angles from direction
170 cam->compute_angles(dir_gimbal, &cam->pan_angle, &cam->tilt_angle);
171 // apply angles
172 cam_gimbal_angles(cam);
173}

176static void cam_gimbal_nadir(struct CamGimbal *cam)
177{
178 struct EnuCoor_f target = *stateGetPositionEnu_f();

185static void cam_gimbal_waypoint_target(struct CamGimbal *cam)
186{
187 if (cam->target_wp_id < nb_waypoint) {
188 struct EnuCoor_f target;
189 target.x = WaypointX(cam->target_wp_id);
190 target.y = WaypointY(cam->target_wp_id);

192 cam_gimbal_set_target_pos(cam, target);
193 cam_gimbal_target(cam);
194 }

294void cam_gimbal_run(struct CamGimbal *cam)
295{
296 switch (cam->mode) {
297 case CAM_GIMBAL_MODE_OFF:
298 cam_gimbal_set_pan_command(cam, CAM_GIMBAL_PAN0);
299 cam_gimbal_set_pan_command(cam, CAM_GIMBAL_TILT0);
300 break;
301 case CAM_GIMBAL_MODE_JOYSTICK:
302 cam_gimbal_joystick(cam);
303 break;
304 case CAM_GIMBAL_MODE_ANGLES:
305 cam_gimbal_angles(cam);
306 break;
307 case CAM_GIMBAL_MODE_NADIR:
308 cam_gimbal_nadir(cam);
309 break;
310 case CAM_GIMBAL_MODE_TARGET:
311 cam_gimbal_target(cam);
312 break;
313 case CAM_GIMBAL_MODE_WAYPOINT:
314 cam_gimbal_waypoint_target(cam);
315 break;
316 case CAM_GIMBAL_MODE_AC_TARGET:
317 cam_gimbal_ac_target(cam);
318 break;
319 default:
320 break;
321 }
322}

326void cam_gimbal_init(void)
327{
328 // apply default settings
329 struct FloatEulers gimbal_to_body = {
330 CAM_GIMBAL_TO_BODY_PHI,
331 CAM_GIMBAL_TO_BODY_THETA,
332 CAM_GIMBAL_TO_BODY_PSI
333 };
334 struct FloatVect3 gimbal_pos = {
335 CAM_GIMBAL_POS_X,
336 CAM_GIMBAL_POS_Y,
337 CAM_GIMBAL_POS_Z
338 };
339 cam_gimbal_setup_angles(&cam_gimbal,
340 CAM_GIMBAL_PAN_MAX, CAM_GIMBAL_PAN_MIN,
341 CAM_GIMBAL_TILT_MAX, CAM_GIMBAL_TILT_MIN);
342 cam_gimbal_setup_mounting(&cam_gimbal, gimbal_to_body, gimbal_pos);
343 cam_gimbal_set_angles_callback(&cam_gimbal, default_compute_angles);
344
345 AbiBindMsgJOYSTICK(ABI_BROADCAST, &joystick_ev, joystick_cb);
346
347 register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_CAM, send_cam);
348#ifdef CAM_SHOW_COORDINATES
349 register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_CAM_POINT, send_cam_point);
350#endif
351}
