561 void float_rates_of_euler_dot_vec(float r[3], float e[3], float edot[3])
562 {
563  float sphi = sinf(e[0]);
564  float cphi = cosf(e[0]);
565  float stheta = sinf(e[1]);
566  float ctheta = cosf(e[1]);
567  r[0] = edot[0] - stheta * edot[2];
568  r[1] = cphi * edot[1] + sphi * ctheta * edot[2];
569  r[2] = -sphi * edot[1] + cphi * ctheta * edot[2];
570 }

573 void float_euler_dot_of_rates_vec(float r[3], float e[3], float edot[3])
574 {
575  float sphi = sinf(e[0]);
576  float cphi = cosf(e[0]);
577  float stheta = sinf(e[1]);
578  float ctheta = cosf(e[1]);
579  if (fabs(ctheta) < FLT_EPSILON){
580  ctheta = FLT_EPSILON;
581  }
582  edot[0] = r[0] + sphi*stheta/ctheta*r[1] + cphi*stheta/ctheta*r[2];
583  edot[1] = cphi*r[1] - sphi*r[2];
584  edot[2] = sphi/ctheta*r[1] + cphi/ctheta*r[2];
585 }
