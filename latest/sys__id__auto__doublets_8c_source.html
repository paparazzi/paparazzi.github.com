87 void periodic_sys_id_auto_doublets(void)
88 {
89  // your periodic code here.
90  // freq = 20.0 Hz
91 
92  // Run if doublet activated
93  if (sys_id_auto_doublets_activated)
94  {
95  float current_time_s = get_sys_time_float();
96  sys_id_auto_doublets_timer = current_time_s - sys_id_auto_doublets_start_time_s;
97 
98  // Check when to increase the counters and perform doublet
99  uint8_t current_index = sys_id_auto_doublets_timer / sys_id_auto_doublets_interval_time;
100 
101  // stop doublets when all doublets done
102  if (current_index >= sys_id_auto_doublets_n_doublets)
103  {
104  sys_id_auto_doublets_on_deactivation();
105  }
106  else
107  { // continue doublets when not finished yet
108  // Check if doublet should be performed and index inxreased
109  if (current_index > sys_id_auto_doublets_counter)
110  {
111  uint8_t actuator_index = current_index / sys_id_auto_doublets_n_repeat;
112  perform_sys_id_auto_doublets(actuator_index);
113  sys_id_auto_doublets_counter++;
114  }
115  }
116  }
117 }

120 {

135 void sys_id_auto_doublets_on_activation(uint8_t active)
136 {
137  sys_id_auto_doublets_activated = active;
138  // If the activation boolean is set to true
139  if (sys_id_auto_doublets_activated)
140  {
141  if (sys_id_auto_doublets_timer > 0)
142  {
143  // Do nothing if already activated
144  }
145  else
146  {
147  // Activate auto doublet if not activated yet
148  sys_id_auto_doublets_start_time_s = get_sys_time_float();
149  sys_id_auto_doublets_timer = 0;
150  perform_sys_id_auto_doublets(0);
151  }
152  }
153  else
154  { // If the activation boolean is set to false
155  sys_id_auto_doublets_on_deactivation();
156  }
157 }
