142 STABILIZATION_ATTITUDE_PHI_DGAIN,
143 STABILIZATION_ATTITUDE_THETA_DGAIN,

147 STABILIZATION_ATTITUDE_PHI_IGAIN,
148 STABILIZATION_ATTITUDE_THETA_IGAIN,
149 STABILIZATION_ATTITUDE_PSI_IGAIN);
150
151 VECT3_ASSIGN(stabilization_gains.dd,
152 STABILIZATION_ATTITUDE_PHI_DDGAIN,
153 STABILIZATION_ATTITUDE_THETA_DDGAIN,
154 STABILIZATION_ATTITUDE_PSI_DDGAIN);

157 STABILIZATION_ATTITUDE_PHI_FFDGAIN,
158 STABILIZATION_ATTITUDE_THETA_FFDGAIN,

180void stabilization_attitude_run(bool in_flight, struct StabilizationSetpoint *sp, struct ThrustSetpoint *thrust, int32_t *cmd)
181{
182 stab_att_sp_euler = stab_sp_to_eulers_i(sp);
183
184 /* update reference */
185#if USE_ATTITUDE_REF
186 attitude_ref_euler_int_update(&att_ref_euler_i, &stab_att_sp_euler);
187#else
188 INT32_EULERS_LSHIFT(att_ref_euler_i.euler, stab_att_sp_euler, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC));
189 INT_RATES_ZERO(att_ref_euler_i.rate);
190 INT_RATES_ZERO(att_ref_euler_i.accel);
191#endif
192
193 const struct Int32Eulers att_ref_scaled = {
194 OFFSET_AND_ROUND(att_ref_euler_i.euler.phi, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)),
195 OFFSET_AND_ROUND(att_ref_euler_i.euler.theta, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC)),
196 OFFSET_AND_ROUND(att_ref_euler_i.euler.psi, (REF_ANGLE_FRAC - INT32_ANGLE_FRAC))
197 };
198 const struct Int32Rates rate_ref_scaled = {
199 OFFSET_AND_ROUND(att_ref_euler_i.rate.p, (REF_RATE_FRAC - INT32_RATE_FRAC)),
200 OFFSET_AND_ROUND(att_ref_euler_i.rate.q, (REF_RATE_FRAC - INT32_RATE_FRAC)),
201 OFFSET_AND_ROUND(att_ref_euler_i.rate.r, (REF_RATE_FRAC - INT32_RATE_FRAC))
202 };
203
204 /* compute feedforward command */
205 stabilization_att_ff_cmd[COMMAND_ROLL] =
206 OFFSET_AND_ROUND(stabilization_gains.dd.x * att_ref_euler_i.accel.p, 5) +
207 stabilization_gains.ffd.x * rate_ref_scaled.p;
208 stabilization_att_ff_cmd[COMMAND_PITCH] =
209 OFFSET_AND_ROUND(stabilization_gains.dd.y * att_ref_euler_i.accel.q, 5) +
210 stabilization_gains.ffd.y * rate_ref_scaled.q;
211 stabilization_att_ff_cmd[COMMAND_YAW] =
212 OFFSET_AND_ROUND(stabilization_gains.dd.z * att_ref_euler_i.accel.r, 5) +
213 stabilization_gains.ffd.z * rate_ref_scaled.r;
214
215 /* with FFD gain of 100, reference rate of 180deg/s (3.14 rad/s)
216 * INT32_RATE_FRAC = 12, CMD_SHIFT = 11
217 * fb cmd: 100 * 3.14 * 2^INT32_RATE_FRAC / 2^CMD_SHIFT = 628
218 * max possible command is 9600
219 */
220
221 /* compute feedback command */
222 /* attitude error */
223 struct Int32Eulers att_err;
224 struct Int32Eulers *ltp_to_body_euler = stateGetNedToBodyEulers_i();
225 EULERS_DIFF(att_err, att_ref_scaled, (*ltp_to_body_euler));
226 INT32_ANGLE_NORMALIZE(att_err.psi);
227
228 if (in_flight) {
229 /* update integrator */
230 EULERS_ADD(stabilization_att_sum_err, att_err);
231 EULERS_BOUND_CUBE(stabilization_att_sum_err, -MAX_SUM_ERR, MAX_SUM_ERR);
232 } else {
233 INT_EULERS_ZERO(stabilization_att_sum_err);
234 }
235
236 /* rate error */
237 struct Int32Rates rate_err;
238 struct Int32Rates *body_rate = stateGetBodyRates_i();
239 RATES_DIFF(rate_err, rate_ref_scaled, (*body_rate));
240
241 /* PID */
242 stabilization_att_fb_cmd[COMMAND_ROLL] =
243 stabilization_gains.p.x * att_err.phi +
244 stabilization_gains.d.x * rate_err.p +
245 OFFSET_AND_ROUND2((stabilization_gains.i.x * stabilization_att_sum_err.phi), 10);
246
247 stabilization_att_fb_cmd[COMMAND_PITCH] =
248 stabilization_gains.p.y * att_err.theta +
249 stabilization_gains.d.y * rate_err.q +
250 OFFSET_AND_ROUND2((stabilization_gains.i.y * stabilization_att_sum_err.theta), 10);
251
252 stabilization_att_fb_cmd[COMMAND_YAW] =
253 stabilization_gains.p.z * att_err.psi +
254 stabilization_gains.d.z * rate_err.r +
255 OFFSET_AND_ROUND2((stabilization_gains.i.z * stabilization_att_sum_err.psi), 10);
256
257
258 /* with P gain of 100, att_err of 180deg (3.14 rad)
259 * fb cmd: 100 * 3.14 * 2^12 / 2^CMD_SHIFT = 628
260 * max possible command is 9600
261 */
262#define CMD_SHIFT 11
263
264 /* sum feedforward and feedback */
265 cmd[COMMAND_ROLL] =
266 OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_ROLL] + stabilization_att_ff_cmd[COMMAND_ROLL]), CMD_SHIFT);
267
268 cmd[COMMAND_PITCH] =
269 OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_PITCH] + stabilization_att_ff_cmd[COMMAND_PITCH]), CMD_SHIFT);
270
271 cmd[COMMAND_YAW] =
272 OFFSET_AND_ROUND((stabilization_att_fb_cmd[COMMAND_YAW] + stabilization_att_ff_cmd[COMMAND_YAW]), CMD_SHIFT);
273
274 cmd[COMMAND_THRUST] = th_sp_to_thrust_i(thrust, 0, THRUST_AXIS_Z);
275
276 /* bound the result */
277 BoundAbs(cmd[COMMAND_ROLL], MAX_PPRZ);
278 BoundAbs(cmd[COMMAND_PITCH], MAX_PPRZ);
279 BoundAbs(cmd[COMMAND_YAW], MAX_PPRZ);
280 BoundAbs(cmd[COMMAND_THRUST], MAX_PPRZ);
281
282}
