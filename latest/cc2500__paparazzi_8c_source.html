63void radio_control_cc2500_event(void) {
64 if (rxRuntimeConfig.rcFrameStatusFn(&rxRuntimeConfig) & RX_FRAME_COMPLETE) {
65 rxRuntimeConfig.rcProcessFrameFn(&rxRuntimeConfig);
66 radio_control.frame_cpt++;
67 radio_control.time_since_last_frame = 0;
68 if (radio_control.radio_ok_cpt > 0) {
69 radio_control.radio_ok_cpt--;
70 } else {
71 radio_control.status = RC_OK;
72 for (int i = 0; i < RADIO_CTL_NB; ++i) {
73 frsky_raw[i] = rxRuntimeConfig.rcReadRawFn(&rxRuntimeConfig, i);
74 }
75 NormalizePpmIIR(frsky_raw, radio_control);
76 AbiSendMsgRADIO_CONTROL(RADIO_CONTROL_FRSKY_ID, &radio_control);
77 }
78 }
