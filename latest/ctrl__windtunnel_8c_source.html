94static void ctrl_module_run(bool in_flight __attribute__((unused)))
95{
96 bool done = false;
97 // Increase step in steptime
98 if (ctrl_windtunnel.rc_throttle > (MAX_PPRZ / 2) && (get_sys_time_float() - last_time) > ctrl_windtunnel_steptime) {
99 // Increase throttle step if flaps at the end
100 if ((ctrl_windtunnel_flaps.current + ctrl_windtunnel_flaps.step) > ctrl_windtunnel_flaps.max) {
101 // Only increase step if throttle is not at the end
102 if ((ctrl_windtunnel_throttle.current + ctrl_windtunnel_throttle.step) <= ctrl_windtunnel_throttle.max) {
103 ctrl_windtunnel_throttle.current += ctrl_windtunnel_throttle.step;
104 ctrl_windtunnel_flaps.current = ctrl_windtunnel_flaps.min;
105 last_time = get_sys_time_float();
106 } else {
107 // Finished
108 done = true;
109 }
110 } else {
111 // By default increase flaps
112 ctrl_windtunnel_flaps.current += ctrl_windtunnel_flaps.step;
113
114 // Double the amount of steptime during double transition
115 if ((ctrl_windtunnel_flaps.current == ctrl_windtunnel_flaps.min)) {
116 last_time = get_sys_time_float() + ctrl_windtunnel_steptime;
117 } else {
118 last_time = get_sys_time_float();
119 }
120 }
121 } else if (ctrl_windtunnel.rc_throttle < (MAX_PPRZ / 2)) {
122 // RESET
123 ctrl_windtunnel_throttle.current = ctrl_windtunnel_throttle.min;
124 ctrl_windtunnel_flaps.current = ctrl_windtunnel_flaps.min;
125 last_time = get_sys_time_float();
126 }
127
128 stabilization.cmd[COMMAND_ROLL] = 0;
129 stabilization.cmd[COMMAND_PITCH] = 0;
130 stabilization.cmd[COMMAND_YAW] = 0;
131 stabilization.cmd[COMMAND_THRUST] = (done) ? 0 : ctrl_windtunnel_throttle.current;
132 stabilization.cmd[COMMAND_FLAPS] = (done) ? 0 : ctrl_windtunnel_flaps.current;
133}
