64static inline void Matrix_Multiply(float a[3][3], float b[3][3], float mat[3][3])
65{
66 float op[3];
67 for (int x = 0; x < 3; x++) {
68 for (int y = 0; y < 3; y++) {
69 for (int w = 0; w < 3; w++) {
70 op[w] = a[x][w] * b[w][y];
71 }
72 mat[x][y] = op[0] + op[1] + op[2];
73 }
74 }
75}
