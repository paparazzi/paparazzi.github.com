70void usb_serial_parse_packet(int data)
71{
72 if (data == -1) { return; }
73 uint8_t c = (uint8_t)data;
74 VCOM_putchar(prompt);
75 VCOM_putchar(data);
76 VCOM_putchar('\r');
77 VCOM_putchar('\n');
78
79 if (c == 'S') {
80 run = false;
81 }
82 if (c == 'R') {
83 run = true;
84 }
85 VCOM_send_message();
86}
