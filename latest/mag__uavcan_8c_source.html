40static void mag_uavcan_cb(struct uavcan_iface_t *iface __attribute__((unused)), CanardRxTransfer *transfer) {
41 // current timestamp
42 uint32_t now_ts = get_sys_time_usec();
43
44 struct uavcan_equipment_ahrs_MagneticFieldStrength msg;
45 if(uavcan_equipment_ahrs_MagneticFieldStrength_decode(transfer, &msg)) {
46 return; // decode error
47 }
48
49 // The LIS3MDL module configure a +/- 4 gauss sensitivity, for 16bits data.
50 // (2**15)/4 = 8192
51 struct Int32Vect3 mag = {
52 msg.magnetic_field_ga[0] * 8192,
53 msg.magnetic_field_ga[1] * 8192,
54 msg.magnetic_field_ga[2] * 8192,
55 };
56
57 AbiSendMsgIMU_MAG_RAW(MAG_UAVCAN_SENDER_ID, now_ts, &mag);
58
59#if MAG_UAVCAN_SYNC_SEND
60 uint8_t abi_id = MAG_UAVCAN_SENDER_ID;
61 DOWNLINK_SEND_IMU_MAG_RAW(DefaultChannel, DefaultDevice, &abi_id, &mag.x, &mag.y, &mag.z);
62#endif
63
64}
