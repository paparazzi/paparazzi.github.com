56void nav_init_block(void)
57{
58 if (private_nav_block >= NB_BLOCK) {
59 private_nav_block = NB_BLOCK - 1;
60 nav_block = private_nav_block;
61 }
62 private_nav_stage = 0;
63 nav_stage = private_nav_stage;
64 block_time = 0;
65 InitStage();
66 nav_on_enter_block(private_nav_block); // generated by flight plan
67}

69void nav_goto_block(uint8_t b)
70{
71 if (b != private_nav_block) {
72 /* To avoid a loop in a the current block */
73 private_last_block = private_nav_block;
74 last_block = private_last_block;
75 private_last_stage = private_nav_stage;
76 last_stage = private_last_stage;
77 }
78 nav_on_exit_block(private_nav_block); // generated by flight plan
79 private_nav_block = b;
80 nav_block = private_nav_block;
81 nav_init_block();
82}

98void nav_return(uint8_t reset) {
99 nav_on_exit_block(private_nav_block); // generated by flight plan
100 private_nav_block = private_last_block;
101 nav_block = private_nav_block;
102 if (reset == 1) {
103 private_nav_stage = 0;
104 } else {
105 private_nav_stage = private_last_stage;
106 }
107 nav_stage = private_nav_stage;
108 block_time = 0;
109 InitStage();
110 nav_on_enter_block(private_nav_block); // generated by flight plan
111}
