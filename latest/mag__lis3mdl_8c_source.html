81void mag_lis3mdl_module_event(void)
82{
83 lis3mdl_event(&mag_lis3mdl);
84
85 if (mag_lis3mdl.data_available) {
86#if MODULE_LIS3MDL_UPDATE_AHRS
87 // current timestamp
88 uint32_t now_ts = get_sys_time_usec();
89
90 // set channel order
91 struct Int32Vect3 mag = {
92 LIS3MDL_CHAN_X_SIGN(int32_t)(mag_lis3mdl.data.value[LIS3MDL_CHAN_X]),
93 LIS3MDL_CHAN_Y_SIGN(int32_t)(mag_lis3mdl.data.value[LIS3MDL_CHAN_Y]),
94 LIS3MDL_CHAN_Z_SIGN(int32_t)(mag_lis3mdl.data.value[LIS3MDL_CHAN_Z])
95 };
96
97 AbiSendMsgIMU_MAG_RAW(MAG_LIS3MDL_SENDER_ID, now_ts, &mag);
98#endif
99#if MODULE_LIS3MDL_SYNC_SEND
100 mag_lis3mdl_report();
101#endif
102#if MODULE_LIS3MDL_UPDATE_AHRS || MODULE_LIS3MDL_SYNC_SEND
103 mag_lis3mdl.data_available = false;
104#endif
105 }
