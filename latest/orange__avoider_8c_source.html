96/*
97 * Function that checks it is safe to move forwards, and then moves a waypoint forward or changes the heading
98 */
99void orange_avoider_periodic(void)
100{
101 // only evaluate our state machine if we are flying
102 if(!autopilot_in_flight()){
103 return;
104 }
105
106 // compute current color thresholds
107 int32_t color_count_threshold = oa_color_count_frac * front_camera.output_size.w * front_camera.output_size.h;
108
109 VERBOSE_PRINT("Color_count: %d threshold: %d state: %d \n", color_count, color_count_threshold, navigation_state);
110
111 // update our safe confidence using color threshold
112 if(color_count < color_count_threshold){
113 obstacle_free_confidence++;
114 } else {
115 obstacle_free_confidence -= 2; // be more cautious with positive obstacle detections
116 }
117
118 // bound obstacle_free_confidence
119 Bound(obstacle_free_confidence, 0, max_trajectory_confidence);
120
121 float moveDistance = fminf(maxDistance, 0.2f * obstacle_free_confidence);
122
123 switch (navigation_state){
124 case SAFE:
125 // Move waypoint forward
126 moveWaypointForward(WP_TRAJECTORY, 1.5f * moveDistance);
127 if (!InsideObstacleZone(WaypointX(WP_TRAJECTORY),WaypointY(WP_TRAJECTORY))){
128 navigation_state = OUT_OF_BOUNDS;
129 } else if (obstacle_free_confidence == 0){
130 navigation_state = OBSTACLE_FOUND;
131 } else {
132 moveWaypointForward(WP_GOAL, moveDistance);
133 }
134
135 break;
136 case OBSTACLE_FOUND:
137 // stop
138 waypoint_move_here_2d(WP_GOAL);
139 waypoint_move_here_2d(WP_TRAJECTORY);
140
141 // randomly select new search direction
142 chooseRandomIncrementAvoidance();
143
144 navigation_state = SEARCH_FOR_SAFE_HEADING;
145
146 break;
147 case SEARCH_FOR_SAFE_HEADING:
148 increase_nav_heading(heading_increment);
149
150 // make sure we have a couple of good readings before declaring the way safe
151 if (obstacle_free_confidence >= 2){
152 navigation_state = SAFE;
153 }
154 break;
155 case OUT_OF_BOUNDS:
156 increase_nav_heading(heading_increment);
157 moveWaypointForward(WP_TRAJECTORY, 1.5f);
158
159 if (InsideObstacleZone(WaypointX(WP_TRAJECTORY),WaypointY(WP_TRAJECTORY))){
160 // add offset to head back into arena
161 increase_nav_heading(heading_increment);
162
163 // reset safe counter
164 obstacle_free_confidence = 0;
165
166 // ensure direction is safe before continuing
167 navigation_state = SEARCH_FOR_SAFE_HEADING;
168 }
169 break;
170 default:
171 break;
172 }
173 return;
174}

176/*
177 * Increases the NAV heading. Assumes heading is an INT32_ANGLE. It is bound in this function.
178 */
179uint8_t increase_nav_heading(float incrementDegrees)
180{
181 float new_heading = stateGetNedToBodyEulers_f()->psi + RadOfDeg(incrementDegrees);
182
183 // normalize heading to [-pi, pi]
184 FLOAT_ANGLE_NORMALIZE(new_heading);
185
186 // set heading, declared in firmwares/rotorcraft/navigation.h
187 nav.heading = new_heading;
188
189 VERBOSE_PRINT("Increasing heading to %f\n", DegOfRad(new_heading));
190 return false;
191}

204/*
205 * Calculates coordinates of a distance of 'distanceMeters' forward w.r.t. current position and heading
206 */
207uint8_t calculateForwards(struct EnuCoor_i *new_coor, float distanceMeters)
208{
209 float heading = stateGetNedToBodyEulers_f()->psi;
210
211 // Now determine where to place the waypoint you want to go to
212 new_coor->x = stateGetPositionEnu_i()->x + POS_BFP_OF_REAL(sinf(heading) * (distanceMeters));
213 new_coor->y = stateGetPositionEnu_i()->y + POS_BFP_OF_REAL(cosf(heading) * (distanceMeters));
214 VERBOSE_PRINT("Calculated %f m forward position. x: %f y: %f based on pos(%f, %f) and heading(%f)\n", distanceMeters,
215 POS_FLOAT_OF_BFP(new_coor->x), POS_FLOAT_OF_BFP(new_coor->y),
216 stateGetPositionEnu_f()->x, stateGetPositionEnu_f()->y, DegOfRad(heading));
217 return false;
218}
