102void stabilization_attitude_plane_pid_run(bool in_flight, struct StabilizationSetpoint *sp, struct ThrustSetpoint *thrust, int32_t *cmd)
103{
104
105 /* Compute feedback */
106 /* attitude error */
107 stab_att_sp_euler = stab_sp_to_eulers_f(sp);
108 struct FloatEulers *att_float = stateGetNedToBodyEulers_f();
109 struct FloatEulers att_err;
110 EULERS_DIFF(att_err, stab_att_sp_euler, *att_float);
111 att_err.psi = 0.f;
112
113 /* update integrator */
114 if (in_flight) {
115 EULERS_ADD(stab_plane_att_sum_err, att_err);
116 EULERS_BOUND_CUBE(stab_plane_att_sum_err, -MAX_SUM_ERR, MAX_SUM_ERR);
117 } else {
118 FLOAT_EULERS_ZERO(stab_plane_att_sum_err);
119 }
120
121 /* rate error */
122 struct FloatRates *rate_float = stateGetBodyRates_f();
123 struct FloatRates rate_sp, rate_err;
124 FLOAT_RATES_ZERO(rate_sp);
125 RATES_DIFF(rate_err, rate_sp, *rate_float);
126
127 /* PID */
128 stab_plane_att_cmd[COMMAND_ROLL] =
129 stab_plane_gains.p.x * att_err.phi +
130 stab_plane_gains.d.x * rate_err.p +
131 stab_plane_gains.i.x * stab_plane_att_sum_err.phi;
132
133 stab_plane_att_cmd[COMMAND_PITCH] =
134 stab_plane_gains.p.y * att_err.theta +
135 stab_plane_gains.d.y * rate_err.q +
136 stab_plane_gains.i.y * stab_plane_att_sum_err.theta;
137
138 cmd[COMMAND_ROLL] = (int32_t)stab_plane_att_cmd[COMMAND_ROLL];
139 cmd[COMMAND_PITCH] = (int32_t)stab_plane_att_cmd[COMMAND_PITCH];
140 cmd[COMMAND_YAW] = 0;
141 cmd[COMMAND_THRUST] = th_sp_to_thrust_i(thrust, 0, THRUST_AXIS_X);
142
143 /* bound the result */
144 BoundAbs(cmd[COMMAND_ROLL], MAX_PPRZ);
145 BoundAbs(cmd[COMMAND_PITCH], MAX_PPRZ);
146 BoundAbs(cmd[COMMAND_YAW], MAX_PPRZ);
147 BoundAbs(cmd[COMMAND_THRUST], MAX_PPRZ);
148}
