75void actuators_md25_init(void)
76{
77 actuators_md25.mode = ACTUATORS_MD25_MODE;
78 actuators_md25.bat = 0;
79 actuators_md25.cmds[0] = 0;
80 actuators_md25.cmds[1] = 0;
81 actuators_md25.encoders[0] = 0;
82 actuators_md25.encoders[1] = 0;
83 actuators_md25.mode = ACTUATORS_MD25_MODE;
84 actuators_md25.accel_rate = ACTUATORS_MD25_ACCEL_RATE;
85 actuators_md25.initialized = false;
86 actuators_md25.trans_cmd.status = I2CTransDone;
87 actuators_md25.trans_sensors.status = I2CTransDone;
88}

90void actuators_md25_periodic(void)
91{
92 if (actuators_md25.trans_sensors.status == I2CTransDone) {
93 if (!actuators_md25.initialized) {
94 // send accel rate and mode configuration
95 actuators_md25.trans_sensors.buf[0] = MD25_REG_ACCEL_RATE;
96 actuators_md25.trans_sensors.buf[1] = actuators_md25.accel_rate;
97 actuators_md25.trans_sensors.buf[2] = actuators_md25.mode;
98 i2c_transmit(&(ACTUATORS_MD25_DEV), &actuators_md25.trans_sensors, ACTUATORS_MD25_I2C_ADDR, 3);
99 }
100 else {
101 // read sensors: enc1, enc1, bat, cur1, cur2
102 actuators_md25.trans_sensors.buf[0] = MD25_REG_ENCODER1;
103 i2c_transceive(&(ACTUATORS_MD25_DEV), &actuators_md25.trans_sensors, ACTUATORS_MD25_I2C_ADDR, 1, 11);
104 }
105 }
106}

108void actuators_md25_set(void)
109{
110 if (actuators_md25.initialized && actuators_md25.trans_cmd.status == I2CTransDone) {
111 actuators_md25.trans_cmd.buf[0] = MD25_REG_SPEED1;
112 actuators_md25.trans_cmd.buf[1] = actuators_md25.cmds[0];
113 actuators_md25.trans_cmd.buf[2] = actuators_md25.cmds[1];
114 i2c_transmit(&(ACTUATORS_MD25_DEV), &actuators_md25.trans_cmd, ACTUATORS_MD25_I2C_ADDR, 3);
115 }

120void actuators_md25_event(void)
121{
122 // commands
123 if (actuators_md25.trans_cmd.status == I2CTransSuccess) {
124 actuators_md25.trans_cmd.status = I2CTransDone;
125 }
126 else if (actuators_md25.trans_cmd.status == I2CTransFailed) {
127 // TODO handle or report error
128 actuators_md25.trans_cmd.status = I2CTransDone;
129 }
130
131 // sensors
132 if (actuators_md25.trans_sensors.status == I2CTransSuccess) {
133 if (actuators_md25.initialized) {
134 actuators_md25.encoders[0] = Int32FromBuf(actuators_md25.trans_sensors.buf, 0);
135 actuators_md25.encoders[1] = Int32FromBuf(actuators_md25.trans_sensors.buf, 4);
136 actuators_md25.bat = actuators_md25.trans_sensors.buf[8];
137 actuators_md25.current[0] = actuators_md25.trans_sensors.buf[9];
138 actuators_md25.current[1] = actuators_md25.trans_sensors.buf[10];
139 actuators_md25.trans_sensors.status = I2CTransDone;
140 }
141 else {
142 actuators_md25.trans_sensors.status = I2CTransDone;
143 actuators_md25.initialized = true;
144 }
145 }
146 else if (actuators_md25.trans_sensors.status == I2CTransFailed) {
147 // TODO handle or report error
148 actuators_md25.trans_sensors.status = I2CTransDone;
149 }
150}
