82 void guidance_indi_calcg_wing(float Gmat[GUIDANCE_INDI_HYBRID_V][GUIDANCE_INDI_HYBRID_U], struct FloatVect3 a_diff, float body_v[GUIDANCE_INDI_HYBRID_V]) {
83  // Get attitude
84  struct FloatEulers eulers_zxy;
85  float_eulers_of_quat_zxy(&eulers_zxy, stateGetNedToBodyQuat_f());
86 
87  /*Pre-calculate sines and cosines*/
88  float sphi = sinf(eulers_zxy.phi);
89  float cphi = cosf(eulers_zxy.phi);
90  float stheta = sinf(eulers_zxy.theta);
91  float ctheta = cosf(eulers_zxy.theta);
92  float spsi = sinf(eulers_zxy.psi);
93  float cpsi = cosf(eulers_zxy.psi);
94 
95 #ifndef GUIDANCE_INDI_PITCH_EFF_SCALING
96 #define GUIDANCE_INDI_PITCH_EFF_SCALING 1.0
97 #endif
98 
99  /*Amount of lift produced by the wing*/
100  float lift_thrust_bz = accel_bodyz_filt.o[0]; // Sum of lift and thrust in boxy z axis (level flight)
101 
102  // get the derivative of the lift wrt to theta
103  float liftd = guidance_indi_get_liftd(0.0f, 0.0f);
104 
105  Gmat[0][0] = -sphi*stheta*lift_thrust_bz;
106  Gmat[1][0] = -cphi*lift_thrust_bz;
107  Gmat[2][0] = -sphi*ctheta*lift_thrust_bz;
108 
109  Gmat[0][1] = cphi*ctheta*lift_thrust_bz*GUIDANCE_INDI_PITCH_EFF_SCALING;
110  Gmat[1][1] = sphi*stheta*lift_thrust_bz*GUIDANCE_INDI_PITCH_EFF_SCALING - sphi*liftd;
111  Gmat[2][1] = -cphi*stheta*lift_thrust_bz*GUIDANCE_INDI_PITCH_EFF_SCALING + cphi*liftd;
112 
113  Gmat[0][2] = cphi*stheta;
114  Gmat[1][2] = -sphi;
115  Gmat[2][2] = cphi*ctheta;
116 
117  Gmat[0][3] = ctheta;
118  Gmat[1][3] = 0;
119  Gmat[2][3] = -stheta;
120 
121  // Convert acceleration error to body axis system
122  body_v[0] = cpsi * a_diff.x + spsi * a_diff.y;
123  body_v[1] = -spsi * a_diff.x + cpsi * a_diff.y;
124  body_v[2] = a_diff.z;
125 }

128 void WEAK guidance_indi_hybrid_set_wls_settings(float body_v[3], float roll_angle, float pitch_angle)
129 {
130  float roll_limit_rad = GUIDANCE_H_MAX_BANK;
131  float max_pitch_limit_rad = RadOfDeg(GUIDANCE_INDI_MAX_PITCH);
132  float min_pitch_limit_rad = RadOfDeg(GUIDANCE_INDI_MIN_PITCH);
133 
134  float pitch_pref_rad = RadOfDeg(guidance_indi_pitch_pref_deg);
135 
136  // Set lower limits
137  du_min_gih[0] = -roll_limit_rad - roll_angle; //roll
138  du_min_gih[1] = min_pitch_limit_rad - pitch_angle; // pitch
139  du_min_gih[2] = (MAX_PPRZ - stabilization.cmd[COMMAND_THRUST]) * guidance_indi_thrust_z_eff;
140  du_min_gih[3] = -stabilization.cmd[COMMAND_THRUST_X]*guidance_indi_thrust_x_eff;
141 
142  // Set upper limits limits
143  du_max_gih[0] = roll_limit_rad - roll_angle; //roll
144  du_max_gih[1] = max_pitch_limit_rad - pitch_angle; // pitch
145  du_max_gih[2] = -stabilization.cmd[COMMAND_THRUST] * guidance_indi_thrust_z_eff;
146  du_max_gih[3] = (MAX_PPRZ - stabilization.cmd[COMMAND_THRUST_X])*guidance_indi_thrust_x_eff;
147 
148  // Set prefered states
149  du_pref_gih[0] = -roll_angle; // prefered delta roll angle
150  du_pref_gih[1] = -pitch_angle + pitch_pref_rad;// prefered delta pitch angle
151  du_pref_gih[2] = du_max_gih[2]; // Low thrust better for efficiency
152  du_pref_gih[3] = body_v[0]; // solve the body acceleration
153 }
