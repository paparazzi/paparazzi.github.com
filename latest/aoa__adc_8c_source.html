67void aoa_adc_init(void)
68{
69 aoa_adc.offset = AOA_OFFSET;
70 aoa_adc.filter = AOA_FILTER;
71 aoa_adc.sens = AOA_SENS;
72 aoa_adc.angle = 0.0;
73 adc_buf_channel(ADC_CHANNEL_AOA, &aoa_adc.buf, ADC_CHANNEL_AOA_NB_SAMPLES);
74}
75
76void aoa_adc_update(void)
77{
78 static float prev_aoa = 0.0;
79
80 aoa_adc.raw = aoa_adc.buf.sum / aoa_adc.buf.av_nb_sample;
81
82 // PT1 filter and convert to rad
83 aoa_adc.angle = aoa_adc.filter * prev_aoa +
84 (1.0 - aoa_adc.filter) * (aoa_adc.raw * aoa_adc.sens - aoa_adc.offset);
85 prev_aoa = aoa_adc.angle;
86
87#ifdef USE_AOA
88 uint8_t flag = 1;
89 float foo = 0.f;
90 AbiSendMsgINCIDENCE(AOA_ADC_ID, flag, aoa_adc.angle, foo);
91 stateSetAngleOfAttack_f(MODULE_AOA_ADC_ID, aoa_adc.angle);
92#endif
93
94 RunOnceEvery(30, DOWNLINK_SEND_AOA(DefaultChannel, DefaultDevice, &aoa_adc.raw, &aoa_adc.angle));
95}
