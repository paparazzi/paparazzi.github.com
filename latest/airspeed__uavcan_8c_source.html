63 {

65  uint16_t raw = 0;

69  sign = -1.0f;

71  float airspeed = sqrt(airspeed_uavcan.diff_p * sign * 2.0f / 1.225f) * sign;
72  pprz_msg_send_AIRSPEED_RAW(trans,dev,AC_ID,
73  &dev_id,
74  &raw,
75  &offset,
76  &airspeed_uavcan.diff_p,
77  &airspeed_uavcan.temperature,
78  &airspeed);
