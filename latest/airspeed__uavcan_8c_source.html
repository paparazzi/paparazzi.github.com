63static void airspeed_uavcan_downlink(struct transport_tx *trans, struct link_device *dev)
64{
65 uint8_t dev_id = UAVCAN_SENDER_ID;
66 uint16_t raw = 0;
67 float offset = 0;
68 float sign = 1.0f;
69 if (airspeed_uavcan.diff_p < 0) {
70 sign = -1.0f;
71 }
72 float airspeed = sqrt(airspeed_uavcan.diff_p * sign * 2.0f / 1.225f) * sign;
73 pprz_msg_send_AIRSPEED_RAW(trans,dev,AC_ID,
74 &dev_id,
75 &raw,
76 &offset,
77 &airspeed_uavcan.diff_p,
78 &airspeed_uavcan.temperature,
79 &airspeed);
80}
81#endif /* PERIODIC_TELEMETRY */

83static void airspeed_uavcan_cb(struct uavcan_iface_t *iface __attribute__((unused)), CanardRxTransfer *transfer) {
84
85 struct uavcan_equipment_air_data_RawAirData msg;
86
87 if(uavcan_equipment_air_data_RawAirData_decode(transfer, &msg)) {
88 return; // decode error
89 }
90
91 float diff_p = msg.differential_pressure;
92 float static_air_temp = msg.static_air_temperature;
93
94
95 if(!isnan(diff_p)) {
96 // Remove the offset and apply a scaling factor
97 diff_p -= airspeed_uavcan.diff_p_offset;
98 diff_p *= airspeed_uavcan.diff_p_scale;
99
100 // Filtering
101#ifdef USE_AIRSPEED_UAVCAN_LOWPASS_FILTER
102 float diff_p_filt = update_butterworth_2_low_pass(&airspeed_filter, diff_p);
103 airspeed_uavcan.diff_p = diff_p_filt;
104#else
105 airspeed_uavcan.diff_p = diff_p;
106#endif
107
108 // Send the ABI message
109#if AIRSPEED_UAVCAN_SEND_ABI
110 AbiSendMsgBARO_DIFF(UAVCAN_SENDER_ID, airspeed_uavcan.diff_p);
111#endif
112 }
113
114 if(!isnan(static_air_temp)) {
115 airspeed_uavcan.temperature = static_air_temp;
116#if AIRSPEED_UAVCAN_SEND_ABI
117 AbiSendMsgTEMPERATURE(UAVCAN_SENDER_ID, airspeed_uavcan.temperature);
118#endif
119 }
120}

122void airspeed_uavcan_init(void)
123{
124 // Set the default values
125 airspeed_uavcan.diff_p_scale = AIRSPEED_UAVCAN_DIFF_P_SCALE;
126
127 // Setup the low pass filter
128#ifdef USE_AIRSPEED_UAVCAN_LOWPASS_FILTER
129 init_butterworth_2_low_pass(&airspeed_filter, AIRSPEED_UAVCAN_LOWPASS_TAU, AIRSPEED_UAVCAN_LOWPASS_PERIOD, 0);
130#endif
131
132 // Bind uavcan RAWAIRDATA message from EQUIPMENT.AIR_DATA
133 uavcan_bind(UAVCAN_EQUIPMENT_AIR_DATA_RAWAIRDATA_ID, UAVCAN_EQUIPMENT_AIR_DATA_RAWAIRDATA_SIGNATURE, &airspeed_uavcan_ev, &airspeed_uavcan_cb);
134
135#if PERIODIC_TELEMETRY
136 register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_AIRSPEED_RAW, airspeed_uavcan_downlink);
137#endif
138}
