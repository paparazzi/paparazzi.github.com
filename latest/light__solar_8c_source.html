62 void light_solar_periodic(void)
63 {
64  up[light_cnt] = buf_light_sol_up.sum / buf_light_sol_up.av_nb_sample;
65  dn[light_cnt] = buf_light_sol_dn.sum / buf_light_sol_dn.av_nb_sample;
66 
67  /* 10k/10k voltage divider, 10 bits adc, 3.3V max */
68 
69  if (++light_cnt >= LIGHT_NB) {
70  DOWNLINK_SEND_SOLAR_RADIATION(DefaultChannel, DefaultDevice,
