96 
97  switch (cmd) {

103  break;
104 #ifdef DC_ZOOM_IN_SERVO
105  case DC_TALLER:
106  DC_PUSH(DC_ZOOM_IN_SERVO);
107  break;
108 #endif
109 #ifdef DC_ZOOM_OUT_SERVO

112  break;

115  case DC_ON:

117  break;
118 #endif
119  default:
120  break;
121  }
