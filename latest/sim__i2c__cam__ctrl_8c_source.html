54 void atmega_i2c_cam_ctrl_send(uint8_t cmd)
55 {
56  static uint8_t zoom = 0;
57  static uint8_t mode = 0;
58  unsigned char cam_ret[1];
59 
60  if (cmd == DC_SHOOT) {
61  dc_send_shot_position();
62  } else if (cmd == DC_TALLER) {
63  zoom = 1;
64  } else if (cmd == DC_WIDER) {
65  zoom = 0;
66  } else if (cmd == DC_GET_STATUS) {
67  mode++;
68  if (mode > 15) {
69  mode = 0;
70  }
71  }
72 
73  cam_ret[0] = mode + zoom * 0x20;
74  RunOnceEvery(6, DOWNLINK_SEND_PAYLOAD(DefaultChannel, DefaultDevice, 1, cam_ret));
75 
76 }
