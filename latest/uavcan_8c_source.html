99static THD_FUNCTION(uavcan_rx, p)
100{
101 event_listener_t el;
102 CANRxFrame rx_msg;
103 CanardCANFrame rx_frame;
104 struct uavcan_iface_t *iface = (struct uavcan_iface_t *)p;
105
106 chRegSetThreadName("uavcan_rx");
107 chEvtRegister(&iface->can_driver->rxfull_event, &el, EVENT_MASK(0));
108 while (true) {
109 if (chEvtWaitAnyTimeout(ALL_EVENTS, TIME_MS2I(100)) == 0) {
110 continue;
111 }
112 chMtxLock(&iface->mutex);
113
114 // Wait until a CAN message is received
115 while (canReceive(iface->can_driver, CAN_ANY_MAILBOX, &rx_msg, TIME_IMMEDIATE) == MSG_OK) {
116 // Process message.
117 const uint32_t timestamp = TIME_I2US(chVTGetSystemTimeX());
118 memcpy(rx_frame.data, rx_msg.data8, 8);
119 rx_frame.data_len = rx_msg.DLC;
120#if defined(STM32_CAN_USE_FDCAN1) || defined(STM32_CAN_USE_FDCAN2)
121 if (rx_msg.common.XTD) {
122 rx_frame.id = CANARD_CAN_FRAME_EFF | rx_msg.ext.EID;
123 } else {
124 rx_frame.id = rx_msg.std.SID;
125 }
126#else
127 if (rx_msg.IDE) {
128 rx_frame.id = CANARD_CAN_FRAME_EFF | rx_msg.EID;
129 } else {
130 rx_frame.id = rx_msg.SID;
131 }
132#endif
133
134 // Let canard handle the frame
135 canardHandleRxFrame(&iface->canard, &rx_frame, timestamp);
136 }
137 chMtxUnlock(&iface->mutex);
138 }
139 chEvtUnregister(&iface->can_driver->rxfull_event, &el);
140}

142/*
143 * Transmitter thread.
144 */
145static THD_FUNCTION(uavcan_tx, p)
146{
147 event_listener_t txc, txe, txr;
148 struct uavcan_iface_t *iface = (struct uavcan_iface_t *)p;
149 uint8_t err_cnt = 0;
150
151 chRegSetThreadName("uavcan_tx");
152 chEvtRegister(&iface->can_driver->txempty_event, &txc, EVENT_MASK(0));
153 chEvtRegister(&iface->can_driver->error_event, &txe, EVENT_MASK(1));
154 chEvtRegister(&iface->tx_request, &txr, EVENT_MASK(2));
155
156 while (true) {
157 eventmask_t evts = chEvtWaitAnyTimeout(ALL_EVENTS, TIME_MS2I(100));
158 // Succesfull transmit
159 if (evts == 0) {
160 continue;
161 }
162
163 // Transmit error
164 if (evts & EVENT_MASK(1)) {
165 chEvtGetAndClearFlags(&txe);
166 continue;
167 }
168
169 chMtxLock(&iface->mutex);
170 for (const CanardCANFrame *txf = NULL; (txf = canardPeekTxQueue(&iface->canard)) != NULL;) {
171 CANTxFrame tx_msg;
172 memcpy(tx_msg.data8, txf->data, 8);
173#if defined(STM32_CAN_USE_FDCAN1) || defined(STM32_CAN_USE_FDCAN2)
174 tx_msg.DLC = txf->data_len; // TODO fixme for FDCAN (>8 bytes)
175 tx_msg.ext.EID = txf->id & CANARD_CAN_EXT_ID_MASK;
176 tx_msg.common.XTD = 1;
177 tx_msg.common.RTR = 0;
178#else
179 tx_msg.DLC = txf->data_len;
180 tx_msg.EID = txf->id & CANARD_CAN_EXT_ID_MASK;
181 tx_msg.IDE = CAN_IDE_EXT;
182 tx_msg.RTR = CAN_RTR_DATA;
183#endif
184 if (!canTryTransmitI(iface->can_driver, CAN_ANY_MAILBOX, &tx_msg)) {
185 err_cnt = 0;
186 canardPopTxQueue(&iface->canard);
187 } else {
188 // After 5 retries giveup and clean full queue
189 if (err_cnt >= 5) {
190 err_cnt = 0;
191 while (canardPeekTxQueue(&iface->canard)) { canardPopTxQueue(&iface->canard); }
192 continue;
193 }
194
195 // Timeout - just exit and try again later
196 chMtxUnlock(&iface->mutex);
197 chThdSleepMilliseconds(++err_cnt);
198 chMtxLock(&iface->mutex);
199 continue;
200 }
201 }
202 chMtxUnlock(&iface->mutex);
203 }

209static void onTransferReceived(CanardInstance *ins, CanardRxTransfer *transfer)
210{
211 struct uavcan_iface_t *iface = (struct uavcan_iface_t *)ins->user_reference;
212
213 // Go through all registered callbacks and call function callback if found
214 for (uavcan_event *ev = uavcan_event_hd; ev; ev = ev->next) {
215 if (transfer->data_type_id == ev->data_type_id) {
216 ev->cb(iface, transfer);
217 }
218 }

245static bool uavcanConfigureIface(struct uavcan_iface_t *iface)
246{
247 if (iface->can_baudrate < 1) {
248 return false;
249 }
250
251 // Hardware configurationn
252#if defined(STM32_CAN_USE_FDCAN1) || defined(STM32_CAN_USE_FDCAN2)
253 const uint32_t pclk = STM32_FDCANCLK;
254#else
255 const uint32_t pclk = STM32_PCLK1;
256#endif
257 static const int MaxBS1 = 16;
258 static const int MaxBS2 = 8;
259
260 /*
261 * Ref. "Automatic Baudrate Detection in CANopen Networks", U. Koppe, MicroControl GmbH & Co. KG
262 * CAN in Automation, 2003
263 *
264 * According to the source, optimal quanta per bit are:
265 * Bitrate Optimal Maximum
266 * 1000 kbps 8 10
267 * 500 kbps 16 17
268 * 250 kbps 16 17
269 * 125 kbps 16 17
270 */
271 const int max_quanta_per_bit = (iface->can_baudrate >= 1000000) ? 10 : 17;
272 static const int MaxSamplePointLocation = 900;
273
274 /*
275 * Computing (prescaler * BS):
276 * BITRATE = 1 / (PRESCALER * (1 / PCLK) * (1 + BS1 + BS2)) -- See the Reference Manual
277 * BITRATE = PCLK / (PRESCALER * (1 + BS1 + BS2)) -- Simplified
278 * let:
279 * BS = 1 + BS1 + BS2 -- Number of time quanta per bit
280 * PRESCALER_BS = PRESCALER * BS
281 * ==>
282 * PRESCALER_BS = PCLK / BITRATE
283 */
284 const uint32_t prescaler_bs = pclk / iface->can_baudrate;
285
286// Searching for such prescaler value so that the number of quanta per bit is highest.
287 uint8_t bs1_bs2_sum = max_quanta_per_bit - 1;
288 while ((prescaler_bs % (1 + bs1_bs2_sum)) != 0) {
289 if (bs1_bs2_sum <= 2) {
290 return false; // No solution
291 }
292 bs1_bs2_sum--;
293 }
294
295 const uint32_t prescaler = prescaler_bs / (1 + bs1_bs2_sum);
296 if ((prescaler < 1U) || (prescaler > 1024U)) {
297 return false; // No solution
298 }
299
300 /*
301 * Now we have a constraint: (BS1 + BS2) == bs1_bs2_sum.
302 * We need to find the values so that the sample point is as close as possible to the optimal value.
303 *
304 * Solve[(1 + bs1)/(1 + bs1 + bs2) == 7/8, bs2] (* Where 7/8 is 0.875, the recommended sample point location *)
305 * {{bs2 -> (1 + bs1)/7}}
306 *
307 * Hence:
308 * bs2 = (1 + bs1) / 7
309 * bs1 = (7 * bs1_bs2_sum - 1) / 8
310 *
311 * Sample point location can be computed as follows:
312 * Sample point location = (1 + bs1) / (1 + bs1 + bs2)
313 *
314 * Since the optimal solution is so close to the maximum, we prepare two solutions, and then pick the best one:
315 * - With rounding to nearest
316 * - With rounding to zero
317 */
318// First attempt with rounding to nearest
319 uint8_t bs1 = ((7 * bs1_bs2_sum - 1) + 4) / 8;
320 uint8_t bs2 = bs1_bs2_sum - bs1;
321 uint16_t sample_point_permill = 1000 * (1 + bs1) / (1 + bs1 + bs2);
322
323// Second attempt with rounding to zero
324 if (sample_point_permill > MaxSamplePointLocation) {
325 bs1 = (7 * bs1_bs2_sum - 1) / 8;
326 bs2 = bs1_bs2_sum - bs1;
327 sample_point_permill = 1000 * (1 + bs1) / (1 + bs1 + bs2);
328 }
329
330 /*
331 * Final validation
332 * Helpful Python:
333 * def sample_point_from_btr(x):
334 * assert 0b0011110010000000111111000000000 & x == 0
335 * ts2,ts1,brp = (x>>20)&7, (x>>16)&15, x&511
336 * return (1+ts1+1)/(1+ts1+1+ts2+1)
337 *
338 */
339 if ((iface->can_baudrate != (pclk / (prescaler * (1 + bs1 + bs2)))) || (bs1 < 1) || (bs1 > MaxBS1) || (bs2 < 1)
340 || (bs2 > MaxBS2)) {
341 return false;
342 }
343
344 // Configure the interface
345#if defined(STM32_CAN_USE_FDCAN1) || defined(STM32_CAN_USE_FDCAN2)
346 iface->can_cfg.NBTP = (0 << FDCAN_NBTP_NSJW_Pos) | ((bs1 - 1) << FDCAN_NBTP_NTSEG1_Pos) | ((
347 bs2 - 1) << FDCAN_NBTP_NTSEG2_Pos) | ((prescaler - 1) << FDCAN_NBTP_NBRP_Pos);
348 iface->can_cfg.CCCR = FDCAN_CCCR_FDOE | FDCAN_CCCR_BRSE;
349#else
350 iface->can_cfg.mcr = CAN_MCR_ABOM | CAN_MCR_AWUM | CAN_MCR_TXFP;
351 iface->can_cfg.btr = CAN_BTR_SJW(0) | CAN_BTR_TS1(bs1 - 1) | CAN_BTR_TS2(bs2 - 1) | CAN_BTR_BRP(prescaler - 1);
352#endif
353 return true;
354}

359static void uavcanInitIface(struct uavcan_iface_t *iface)
360{
361 // First try to configure abort if failed
362 if (!uavcanConfigureIface(iface)) {
363 return;
364 }
365
366 // Initialize mutexes/events for multithread locking
367 chMtxObjectInit(&iface->mutex);
368 chEvtObjectInit(&iface->tx_request);
369
370 // Initialize canard
371 canardInit(&iface->canard, iface->canard_memory_pool, sizeof(iface->canard_memory_pool),
372 onTransferReceived, shouldAcceptTransfer, iface);
373 // Update the uavcan node ID
374 canardSetLocalNodeID(&iface->canard, iface->node_id);
375
376 // Start the can interface
377 canStart(iface->can_driver, &iface->can_cfg);
378
379 // Start the receiver and transmitter thread
380 chThdCreateStatic(iface->thread_rx_wa, iface->thread_rx_wa_size, NORMALPRIO + 8, uavcan_rx, (void *)iface);
381 chThdCreateStatic(iface->thread_tx_wa, iface->thread_tx_wa_size, NORMALPRIO + 7, uavcan_tx, (void *)iface);
382 iface->initialized = true;
383}
