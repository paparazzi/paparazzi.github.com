160 void intermcu_parse(uint8_t c)
161 {
162  if (intermcu_data.status < LINK_MCU_GOT_PAYLOAD) {
163  intermcu_data.ck_a += c;
164  intermcu_data.ck_b += intermcu_data.ck_a;
165  }
166  switch (intermcu_data.status) {
167  case LINK_MCU_UNINIT:
168  if (c == INTERMCU_SYNC1) {
169  intermcu_data.status++;
170  }
171  break;
172  case LINK_MCU_GOT_SYNC1:
173  if (c != INTERMCU_SYNC2) {
174  goto error;
175  }
176  intermcu_data.ck_a = 0;
177  intermcu_data.ck_b = 0;
178  intermcu_data.status++;
179  break;
180  case LINK_MCU_GOT_SYNC2:
181  if (intermcu_data.msg_available) {
182  /* Previous message has not yet been parsed: discard this one */
183  goto error;
184  }
185  intermcu_data.msg_class = c;
186  intermcu_data.status++;
187  break;
188  case LINK_MCU_GOT_CLASS:
189  intermcu_data.msg_id = c;
190  intermcu_data.status++;
191  break;
192  case LINK_MCU_GOT_ID:
193  intermcu_data.len = c;
194  intermcu_data.status++;
195  break;
196  case LINK_MCU_GOT_LEN1:
197  intermcu_data.len |= (c << 8);
198  if (intermcu_data.len > INTERMCU_MAX_PAYLOAD) {
199  goto error;
200  }
201  intermcu_data.msg_idx = 0;
202  intermcu_data.status++;
203  break;
204  case LINK_MCU_GOT_LEN2:
205  intermcu_data.msg_buf[intermcu_data.msg_idx] = c;
206  intermcu_data.msg_idx++;
207  if (intermcu_data.msg_idx >= intermcu_data.len) {
208  intermcu_data.status++;
209  }
210  break;
211  case LINK_MCU_GOT_PAYLOAD:
212  if (c != intermcu_data.ck_a) {
213  goto error;
214  }
215  intermcu_data.status++;
216  break;
217  case LINK_MCU_GOT_CHECKSUM1:
218  if (c != intermcu_data.ck_b) {
219  goto error;
220  }
221  intermcu_data.msg_available = true;
222  goto restart;
223  break;
224  default:
225  goto error;
226  }
227  return;
228 error:
229  intermcu_data.error_cnt++;
230 restart:
231  intermcu_data.status = LINK_MCU_UNINIT;
232  return;
233 }

297 void parse_mavpilot_msg(void)
298 {
299  if (intermcu_data.msg_class == MSG_INTERMCU_ID) {
300  if (intermcu_data.msg_id == MSG_INTERMCU_COMMAND_ID) {
301 #if COMMANDS_NB > 8
302 #error "INTERMCU UART CAN ONLY SEND 8 COMMANDS OR THE UART WILL BE OVERFILLED"
303 #endif
304 
305  for (int i = 0; i < COMMANDS_NB; i++) {
306  ap_state->commands[i] = ((pprz_t)MSG_INTERMCU_COMMAND(intermcu_data.msg_buf, i));
307  }
308 
309 #ifdef LINK_MCU_LED
310  LED_TOGGLE(LINK_MCU_LED);
311 #endif
312  inter_mcu_received_ap = true;
313  } else if (intermcu_data.msg_id == MSG_INTERMCU_RADIO_ID) {
314 #if RADIO_CONTROL_NB_CHANNEL > 10
315 #error "INTERMCU UART CAN ONLY SEND 10 RADIO CHANNELS OR THE UART WILL BE OVERFILLED"
316 #endif
317 
318  for (int i = 0; i < RADIO_CONTROL_NB_CHANNEL; i++) {
319  fbw_state->channels[i] = ((pprz_t)MSG_INTERMCU_RADIO(intermcu_data.msg_buf, i));
320  }
321  } else if (intermcu_data.msg_id == MSG_INTERMCU_TRIM_ID) {
322  ap_state->command_roll_trim = ((pprz_t) MSG_INTERMCU_TRIM_ROLL(intermcu_data.msg_buf));
323  ap_state->command_pitch_trim = ((pprz_t) MSG_INTERMCU_TRIM_PITCH(intermcu_data.msg_buf));
324  } else if (intermcu_data.msg_id == MSG_INTERMCU_FBW_ID) {
325  fbw_state->ppm_cpt = MSG_INTERMCU_FBW_MOD(intermcu_data.msg_buf);
326  fbw_state->status = MSG_INTERMCU_FBW_STAT(intermcu_data.msg_buf);
327  fbw_state->nb_err = MSG_INTERMCU_FBW_ERR(intermcu_data.msg_buf);
328  fbw_state->electrical.vsupply = (float)(MSG_INTERMCU_FBW_VOLT(intermcu_data.msg_buf)) / 10.f;
329  fbw_state->electrical.current = (float)(MSG_INTERMCU_FBW_CURRENT(intermcu_data.msg_buf)) / 10.f;
330 
331 #ifdef LINK_MCU_LED
332  LED_TOGGLE(LINK_MCU_LED);
333 #endif
334  inter_mcu_received_fbw = true;
335  }
336  }
337 }
