82void sbus_dual_event(void)
83{
84 sbus_dual_decode_event();
85 if (sbus2.frame_available) {
86 radio_control.frame_cpt++;
87 radio_control.time_since_last_frame = 0;
88 if (radio_control.radio_ok_cpt > 0) {
89 radio_control.radio_ok_cpt--;
90 } else {
91 radio_control.status = RC_OK;
92 NormalizePpmIIR(sbus2.pulses, radio_control);
93 AbiSendMsgRADIO_CONTROL(RADIO_CONTROL_SBUS_DUAL_ID, &radio_control);
94 }
95 sbus2.frame_available = false;
96 }
97 if (sbus1.frame_available) {
98 radio_control.frame_cpt++;
99 radio_control.time_since_last_frame = 0;
100 if (radio_control.radio_ok_cpt > 0) {
101 radio_control.radio_ok_cpt--;
102 } else {
103 radio_control.status = RC_OK;
104 NormalizePpmIIR(sbus1.pulses, radio_control);
105 AbiSendMsgRADIO_CONTROL(RADIO_CONTROL_SBUS_DUAL_ID, &radio_control);
106 }
107 sbus1.frame_available = false;
108 }
