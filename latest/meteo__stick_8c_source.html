93static const float maxAdc = 8388608.0f; // 2 ** 23
94
95static inline float get_pressure(uint32_t raw)
96{
97 const float uncal_abs = ((raw / maxAdc) + 0.095f) / 0.0009f;
98
99#if USE_MS_EEPROM
100 return mtostk_get_calibrated_value(&meteo_stick.calib, MTOSTK_ABS_PRESS, uncal_abs, meteo_stick.current_temperature);
101#else
102 return (MS_PRESSURE_SCALE * uncal_abs) + MS_PRESSURE_OFFSET;
103#endif
104}
105
106static inline float get_temp(uint32_t raw)
107{
108 const float coeff_A = 3.9083e-3f;
109 const float coeff_B = -5.775e-7f;
110 const float R0 = 1000.0f;
111 const float gain_factor = Ads1220GainTable[ADS1220_GAIN_4];
112 const float pga_factor = 1.0f;
113 const float Rref = 6.8e3f;
114 const float Rt = (raw * Rref) / (gain_factor * pga_factor * maxAdc);
115 const float uncal_temp = ((-R0 * coeff_A) + (sqrtf(powf((R0 * coeff_A), 2.0f) +
116 (4.0f * R0 * coeff_B * (Rt - R0))))) / (2.0f * R0 * coeff_B);
117
118#if USE_MS_EEPROM
119 return mtostk_get_calibrated_value(&meteo_stick.calib, MTOSTK_TEMP, uncal_temp, 0.);
120#else
121 return (MS_TEMPERATURE_SCALE * uncal_temp) + MS_TEMPERATURE_OFFSET;
122#endif
123}

125static float pitot_offset;
126static int pitot_counter;
127// number of measures for averaging initial offset
128#define MS_PITOT_COUNTER 20
129// filtering coefficient (0 no filter, 1 constant value)
130#define MS_PITOT_FILTER 0.6
131
132static inline float get_diff(uint32_t raw)
133{
134 const float gain_factor = Ads1220GainTable[ADS1220_GAIN_2];
135 const uint32_t raw_diff = raw > pitot_offset ? raw - pitot_offset : 0;
136 const float uncal_diff = ((raw_diff / maxAdc) * 5000.f / gain_factor);
137
138#if USE_MS_EEPROM
139 return mtostk_get_calibrated_value(&meteo_stick.calib, MTOSTK_DIF_PRESS, uncal_diff, meteo_stick.current_temperature);
140#else
141 return (MS_DIFF_PRESSURE_SCALE * uncal_diff) + MS_DIFF_PRESSURE_OFFSET;
142#endif
143}
144
145static inline float get_pitot(uint32_t raw)
146{
147 return sqrtf((2.0f * get_diff(raw)) / 1.293f);
148}
149
150static inline float get_humidity(uint32_t raw)
151{
152 const float icu_freq = 42e6f; // Freq
153 const float Ra = 390e3f;
154 const float Rb = 680e3f;
155 const float S1 = 0.3e-12f; // pico farad by % of relative humidity
156 const float calib_raw_period = 17800.f;
157 const float calib_humidity = 45.f;
158 const float k = 2.f * logf(2.f) * (Ra + Rb + Rb);
159 const float uncal_hum = calib_humidity + ((raw - calib_raw_period) / (k * icu_freq * S1));
160
161#if USE_MS_EEPROM
162 return mtostk_get_calibrated_value(&meteo_stick.calib, MTOSTK_HUMIDITY, uncal_hum, meteo_stick.current_temperature);
163#else
164 return (MS_HUMIDTY_SCALE * uncal_hum) + MS_HUMIDTY_OFFSET;
165#endif
166}

300void meteo_stick_init(void)
301{
302#ifdef MS_PRESSURE_SLAVE_IDX
303 // Init absolute pressure
304 meteo_stick.pressure.config.mux = ADS1220_MUX_AIN0_AVSS;
305 meteo_stick.pressure.config.gain = ADS1220_GAIN_1;
306 meteo_stick.pressure.config.pga_bypass = true;
307 meteo_stick.pressure.config.rate = ADS1220_RATE_45_HZ;
308 meteo_stick.pressure.config.conv = ADS1220_CONTINIOUS_CONVERSION;
309 meteo_stick.pressure.config.vref = ADS1220_VREF_VDD;
310 meteo_stick.pressure.config.idac = ADS1220_IDAC_OFF;
311 meteo_stick.pressure.config.i1mux = ADS1220_IMUX_OFF;
312 meteo_stick.pressure.config.i2mux = ADS1220_IMUX_OFF;
313 ads1220_init(&meteo_stick.pressure, &(MS_SPI_DEV), MS_PRESSURE_SLAVE_IDX);
314#endif
315
316#ifdef MS_DIFF_PRESSURE_SLAVE_IDX
317 // Init differential pressure
318 meteo_stick.diff_pressure.config.mux = ADS1220_MUX_AIN0_AVSS;
319 meteo_stick.diff_pressure.config.gain = ADS1220_GAIN_2;
320 meteo_stick.diff_pressure.config.pga_bypass = true;
321 meteo_stick.diff_pressure.config.rate = ADS1220_RATE_45_HZ;
322 meteo_stick.diff_pressure.config.conv = ADS1220_CONTINIOUS_CONVERSION;
323 meteo_stick.diff_pressure.config.vref = ADS1220_VREF_VDD;
324 meteo_stick.diff_pressure.config.idac = ADS1220_IDAC_OFF;
325 meteo_stick.diff_pressure.config.i1mux = ADS1220_IMUX_OFF;
326 meteo_stick.diff_pressure.config.i2mux = ADS1220_IMUX_OFF;
327 ads1220_init(&meteo_stick.diff_pressure, &(MS_SPI_DEV), MS_DIFF_PRESSURE_SLAVE_IDX);
328#endif
329
330#ifdef MS_TEMPERATURE_SLAVE_IDX
331 // Init temperature
332 meteo_stick.temperature.config.mux = ADS1220_MUX_AIN0_AIN1;
333 meteo_stick.temperature.config.gain = ADS1220_GAIN_4;
334 meteo_stick.temperature.config.pga_bypass = true;
335 meteo_stick.temperature.config.rate = ADS1220_RATE_45_HZ;
336 meteo_stick.temperature.config.conv = ADS1220_CONTINIOUS_CONVERSION;
337 meteo_stick.temperature.config.vref = ADS1220_VREF_EXTERNAL_REF;
338 meteo_stick.temperature.config.idac = ADS1220_IDAC_250_uA;
339 meteo_stick.temperature.config.i1mux = ADS1220_IMUX_OFF;
340 meteo_stick.temperature.config.i2mux = ADS1220_IMUX_A0_RP1;
341 ads1220_init(&meteo_stick.temperature, &(MS_SPI_DEV), MS_TEMPERATURE_SLAVE_IDX);
342#endif
343
344 // Init humidity
345 meteo_stick.humidity_period = 0;
346
347 // Initial conditions
348 meteo_stick.current_pressure = 0.0f;
349 meteo_stick.current_temperature = 0.0f;
350 meteo_stick.current_humidity = 0.0f;
351 meteo_stick.current_airspeed = 0.0f;
352
353#if USE_MS_EEPROM
354 // Set number of calibration to 0 for all sensors
355 int i;
356 for (i = 0; i < MTOSTK_NUM_SENSORS; i++) {
357 meteo_stick.calib.params[i].num_temp = 0;
358 }
359 // Init eeprom
360 eeprom25AA256_init(&meteo_stick.eeprom, &(MS_SPI_DEV), MS_EEPROM_SLAVE_IDX);
361#endif
362
363 // Number of measurements before setting pitot offset
364 pitot_counter = MS_PITOT_COUNTER;
365 meteo_stick.reset_dp_offset = false;
366
367#if LOG_MS
368 log_ptu_started = false;
369#endif
370#if LOG_MS_FLIGHTRECORDER
371 log_tagged = false;
372#endif
373}

377void meteo_stick_periodic(void)
378{
379 // Read ADC
380#ifdef MS_PRESSURE_SLAVE_IDX
381 ads1220_periodic(&meteo_stick.pressure);
382#endif
383#ifdef MS_DIFF_PRESSURE_SLAVE_IDX
384 ads1220_periodic(&meteo_stick.diff_pressure);
385#endif
386#ifdef MS_TEMPERATURE_SLAVE_IDX
387 ads1220_periodic(&meteo_stick.temperature);
388#endif
389 // Read PWM
390#ifdef MS_HUMIDITY_PWM_INPUT
391 meteo_stick.humidity_period = pwm_input_period_tics[MS_HUMIDITY_PWM_INPUT];
392 meteo_stick.current_humidity = get_humidity(meteo_stick.humidity_period);
393#endif
394
395#if USE_MS_EEPROM
396 if (meteo_stick.eeprom.data_available) {
397 // Extract calibration data
398 if (!mtostk_populate_cal_from_buffer(&meteo_stick.calib, (uint8_t *)(meteo_stick.eeprom.rx_buf + 3))) {
399 // Extraction failed
400 // Force number of calibration to 0 for all sensors
401 int i;
402 for (i = 0; i < MTOSTK_NUM_SENSORS; i++) {
403 meteo_stick.calib.params[i].num_temp = 0;
404 }
405 }
406 } else if (meteo_stick.eeprom.spi_trans.status == SPITransDone) {
407 // Load reading request (reading 1Kb from address 0x0)
408 eeprom25AA256_read(&meteo_stick.eeprom, 0x0, 1024);
409 }
410#endif
411
412 // Log data
413#if LOG_MS
414 meteo_stick_log_data_ascii();
415#endif
416#if LOG_MS_FLIGHTRECORDER
417 meteo_stick_log_data_fr();
418#endif
419
420 // Send data synch to reading
421#if SEND_MS_SYNC
422 meteo_stick_send_data(&(DefaultChannel).trans_tx, &(DefaultDevice).device);
423#endif
424
425 // Check if DP offset reset is required
426 if (meteo_stick.reset_dp_offset) {
427 pitot_counter = MS_PITOT_COUNTER;
428 meteo_stick.reset_dp_offset = false;
429 }
430}

434void meteo_stick_event(void)
435{
436#ifdef MS_PRESSURE_SLAVE_IDX
437 ads1220_event(&meteo_stick.pressure);
438#endif
439#ifdef MS_DIFF_PRESSURE_SLAVE_IDX
440 ads1220_event(&meteo_stick.diff_pressure);
441#endif
442#ifdef MS_TEMPERATURE_SLAVE_IDX
443 ads1220_event(&meteo_stick.temperature);
444#endif
445
446#ifdef MS_PRESSURE_SLAVE_IDX
447 // send absolute pressure data over ABI as soon as available
448 if (meteo_stick.pressure.data_available) {
449 uint32_t now_ts = get_sys_time_usec();
450 meteo_stick.current_pressure = get_pressure(meteo_stick.pressure.data);
451#if USE_MS_PRESSURE
452 AbiSendMsgBARO_ABS(METEO_STICK_SENDER_ID, now_ts, meteo_stick.current_pressure);
453#endif
454 meteo_stick.pressure.data_available = false;
455 }
456#endif
457
458#ifdef MS_DIFF_PRESSURE_SLAVE_IDX
459 // send differential pressure data over ABI as soon as available
460 if (meteo_stick.diff_pressure.data_available) {
461 if (pitot_counter > 0) {
462 if (pitot_counter == MS_PITOT_COUNTER) {
463 // set initial value
464 pitot_offset = meteo_stick.diff_pressure.data;
465 } else {
466 pitot_offset = pitot_offset * MS_PITOT_FILTER + (1.0f - MS_PITOT_FILTER) * meteo_stick.diff_pressure.data;
467 }
468 pitot_counter--;
469 }
470 else {
471#if USE_MS_DIFF_PRESSURE
472 float diff = get_diff(meteo_stick.diff_pressure.data);
473 AbiSendMsgBARO_DIFF(METEO_STICK_SENDER_ID, diff);
474#endif
475 meteo_stick.current_airspeed = get_pitot(meteo_stick.diff_pressure.data);
476 meteo_stick.diff_pressure.data_available = false;
477 }
478 }
479#endif
480
481#ifdef MS_TEMPERATURE_SLAVE_IDX
482 // send temperature data over ABI as soon as available
483 if (meteo_stick.temperature.data_available) {
484 meteo_stick.current_temperature = get_temp(meteo_stick.temperature.data);
485#if USE_MS_TEMPERATURE
486 AbiSendMsgTEMPERATURE(METEO_STICK_SENDER_ID, meteo_stick.current_temperature);
487#endif
488 meteo_stick.temperature.data_available = false;
489 }
490#endif
491
492#if USE_MS_EEPROM
493 eeprom25AA256_event(&meteo_stick.eeprom);
494#endif
495}
