52void rssi_init()
53{
54 memset(rssi_acs_id, 0, NB_ACS_ID);
55
56 rssi_acs_id[0] = 0; // ground station
57 rssi_acs_id[AC_ID] = 1;
58 rssi_acs[rssi_acs_id[AC_ID]].ac_id = AC_ID;
59 rssi_acs_idx = 2;
60
61 /* register for rssi messages */
62 AbiBindMsgRSSI(ABI_BROADCAST, &ev, rssi_cb);
63}

65void parse_rssi_dl(uint8_t *buf)
66{
67 uint8_t sender_id = SenderIdOfPprzMsg(buf);
68 uint8_t msg_id = IdOfPprzMsg(buf);
69
70 if (sender_id > 0 && msg_id == DL_RSSI) {
71 set_rssi(sender_id,
72 DL_RSSI_tx_power(buf),
73 DL_RSSI_rssi(buf));
74 }
75}
76
77void set_rssi(uint8_t _ac_id, int8_t _tx_strength, int8_t _rssi)
78{
79 if (rssi_acs_idx < NB_ACS) {
80 if (_ac_id > 0 && rssi_acs_id[_ac_id] == 0) {
81 rssi_acs_id[_ac_id] = rssi_acs_idx++;
82 rssi_acs[rssi_acs_id[_ac_id]].ac_id = _ac_id;
83 }
84
85 rssi_acs[rssi_acs_id[_ac_id]].rssi = _rssi;
86 rssi_acs[rssi_acs_id[_ac_id]].tx_strength = _tx_strength;
87 }
88}

90struct rssi_info_ get_rssi(uint8_t _ac_id)
