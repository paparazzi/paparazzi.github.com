238 VL53L1X_ERROR VL53L1X_SensorInit(VL53L1_DEV dev)
239 {
240  VL53L1X_ERROR status = 0;
241  uint8_t Addr = 0x00, tmp;
242 
243  for (Addr = 0x2D; Addr <= 0x87; Addr++) {
244  status = VL53L1_WrByte(dev, Addr, VL51L1X_DEFAULT_CONFIGURATION[Addr - 0x2D]);
245  }
246  status = VL53L1X_StartRanging(dev);
247  tmp = 0;
248  while (tmp == 0) {
249  status = VL53L1X_CheckForDataReady(dev, &tmp);
250  }
251  status = VL53L1X_ClearInterrupt(dev);
252  status = VL53L1X_StopRanging(dev);
253  status = VL53L1_WrByte(dev, VL53L1_VHV_CONFIG__TIMEOUT_MACROP_LOOP_BOUND, 0x09); /* two bounds VHV */
254  status = VL53L1_WrByte(dev, 0x0B, 0); /* start VHV from the previous temperature */
255 #if VL53L1X_AUTO_INCR_ADDR
256  status = VL53L1X_SetI2CAddress(dev, auto_incr_addr);
257  auto_incr_addr += 2; // auto increment by 2 (+1 on 7 bits address)
258 #endif
259  return status;
260 }

487 {

489  VL53L1X_ERROR status = 0;
490 
491  status = VL53L1X_GetTimingBudgetInMs(dev, &TB);
492  if (status != 0) {
493  return 1;
494  }
495  switch (DM) {
496  case 1:
497  status = VL53L1_WrByte(dev, PHASECAL_CONFIG__TIMEOUT_MACROP, 0x14);
498  status = VL53L1_WrByte(dev, RANGE_CONFIG__VCSEL_PERIOD_A, 0x07);

502  status = VL53L1_WrWord(dev, SD_CONFIG__INITIAL_PHASE_SD0, 0x0606);
503  break;
504  case 2:
505  status = VL53L1_WrByte(dev, PHASECAL_CONFIG__TIMEOUT_MACROP, 0x0A);
506  status = VL53L1_WrByte(dev, RANGE_CONFIG__VCSEL_PERIOD_A, 0x0F);
507  status = VL53L1_WrByte(dev, RANGE_CONFIG__VCSEL_PERIOD_B, 0x0D);
508  status = VL53L1_WrByte(dev, RANGE_CONFIG__VALID_PHASE_HIGH, 0xB8);
509  status = VL53L1_WrWord(dev, SD_CONFIG__WOI_SD0, 0x0F0D);
510  status = VL53L1_WrWord(dev, SD_CONFIG__INITIAL_PHASE_SD0, 0x0E0E);
511  break;
512  default:
513  status = 1;
514  break;
515  }
