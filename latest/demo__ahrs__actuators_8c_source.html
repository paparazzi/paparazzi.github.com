112static inline void main_periodic_task(void)
113{
114 /* Simply set current roll/pitch as commands.
115 * Scale DEMO_MAX_ROLL/PITCH to MAX_PPRZ (the max commands)
116 */
117 commands[COMMAND_ROLL] = stateGetNedToBodyEulers_f()->phi * MAX_PPRZ / DEMO_MAX_ROLL;
118 commands[COMMAND_PITCH] = stateGetNedToBodyEulers_f()->theta * MAX_PPRZ / DEMO_MAX_ROLL;
119
120 /* generated macro from airframe file, seconds AP_MODE param not used */
121 SetActuatorsFromCommands(commands, 0);
122
123 if (sys_time.nb_sec > 1) {
124 modules_periodic_task();
125 }
126 RunOnceEvery(10, { LED_PERIODIC();});
127 RunOnceEvery(PERIODIC_FREQUENCY, { datalink_periodic(); });
128 periodic_telemetry_send_Main(DefaultPeriodic, &(DefaultChannel).trans_tx, &(DefaultDevice).device);
129
130 modules_periodic_task();
131}
