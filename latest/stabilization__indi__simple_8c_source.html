225 void indi_init_filters(void)
226 {
227  // tau = 1/(2*pi*Fc)
228  float tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF);
229  float tau_rdot = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_RDOT);
230  float tau_axis[3] = {tau, tau, tau_rdot};
231  float tau_est = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_ESTIMATION_FILT_CUTOFF);
232  float sample_time = 1.0 / PERIODIC_FREQUENCY;
233  // Filtering of gyroscope and actuators
234  for (int8_t i = 0; i < 3; i++) {
235  init_butterworth_2_low_pass(&indi.u[i], tau_axis[i], sample_time, 0.0);
236  init_butterworth_2_low_pass(&indi.rate[i], tau_axis[i], sample_time, 0.0);
237  init_butterworth_2_low_pass(&indi.est.u[i], tau_est, sample_time, 0.0);
238  init_butterworth_2_low_pass(&indi.est.rate[i], tau_est, sample_time, 0.0);
239  }
240 
241  // Init rate filter for feedback
242  float time_constants[3] = {1.0/(2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_P), 1.0/(2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_Q), 1.0/(2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_R)};
243 
244  init_first_order_low_pass(&rates_filt_fo[0], time_constants[0], sample_time, stateGetBodyRates_f()->p);
245  init_first_order_low_pass(&rates_filt_fo[1], time_constants[1], sample_time, stateGetBodyRates_f()->q);
246  init_first_order_low_pass(&rates_filt_fo[2], time_constants[2], sample_time, stateGetBodyRates_f()->r);
247 }

362 {
363  for (int8_t i = 0; i < 3; i++) {
364  output[i] = (filter[i].o[0] - filter[i].o[1]) * PERIODIC_FREQUENCY;

388 void stabilization_indi_rate_run(struct FloatRates rate_sp, bool in_flight __attribute__((unused)))
389 {
390  //Propagate input filters
391  //first order actuator dynamics
392  indi.u_act_dyn.p = indi.u_act_dyn.p + indi.act_dyn.p * (indi.u_in.p - indi.u_act_dyn.p);
393  indi.u_act_dyn.q = indi.u_act_dyn.q + indi.act_dyn.q * (indi.u_in.q - indi.u_act_dyn.q);
394  indi.u_act_dyn.r = indi.u_act_dyn.r + indi.act_dyn.r * (indi.u_in.r - indi.u_act_dyn.r);
395 
396  // Propagate the filter on the gyroscopes and actuators
397  struct FloatRates *body_rates = stateGetBodyRates_f();
398  filter_pqr(indi.u, &indi.u_act_dyn);
399  filter_pqr(indi.rate, body_rates);
400 
401  // Calculate the derivative of the rates
402  finite_difference_from_filter(indi.rate_d, indi.rate);
403 
404  //The rates used for feedback are by default the measured rates.
405  //If there is a lot of noise on the gyroscope, it might be good to use the filtered value for feedback.
406  //Note that due to the delay, the PD controller may need relaxed gains.
407  struct FloatRates rates_filt;
408 #if STABILIZATION_INDI_FILTER_ROLL_RATE
409  rates_filt.p = update_first_order_low_pass(&rates_filt_fo[0], body_rates->p);
410 #else
411  rates_filt.p = body_rates->p;
412 #endif
413 #if STABILIZATION_INDI_FILTER_PITCH_RATE
414  rates_filt.q = update_first_order_low_pass(&rates_filt_fo[1], body_rates->q);
415 #else
416  rates_filt.q = body_rates->q;
417 #endif
418 #if STABILIZATION_INDI_FILTER_YAW_RATE
419  rates_filt.r = update_first_order_low_pass(&rates_filt_fo[2], body_rates->r);
420 #else
421  rates_filt.r = body_rates->r;
422 #endif
423 
424  //This lets you impose a maximum yaw rate.
425  BoundAbs(rate_sp.r, indi.attitude_max_yaw_rate);
426 
427  // Compute reference angular acceleration:
428  indi.angular_accel_ref.p = (rate_sp.p - rates_filt.p) * indi.gains.rate.p;
429  indi.angular_accel_ref.q = (rate_sp.q - rates_filt.q) * indi.gains.rate.q;
430  indi.angular_accel_ref.r = (rate_sp.r - rates_filt.r) * indi.gains.rate.r;
431 
432  //Increment in angular acceleration requires increment in control input
433  //G1 is the control effectiveness. In the yaw axis, we need something additional: G2.
434  //It takes care of the angular acceleration caused by the change in rotation rate of the propellers
435  //(they have significant inertia, see the paper mentioned in the header for more explanation)
436  indi.du.p = 1.0 / indi.g1.p * (indi.angular_accel_ref.p - indi.rate_d[0]);
437  indi.du.q = 1.0 / indi.g1.q * (indi.angular_accel_ref.q - indi.rate_d[1]);
438  indi.du.r = 1.0 / (indi.g1.r + indi.g2) * (indi.angular_accel_ref.r - indi.rate_d[2] + indi.g2 * indi.du.r);
439 
440  //Don't increment if thrust is off and on the ground
441  //without this the inputs will increment to the maximum before even getting in the air.
442  if (stabilization_cmd[COMMAND_THRUST] < 300 && !in_flight) {
443  FLOAT_RATES_ZERO(indi.u_in);
444 
445  // If on the gournd, no increments, just proportional control
446  indi.u_in.p = indi.du.p;
447  indi.u_in.q = indi.du.q;
448  indi.u_in.r = indi.du.r;
449  } else {
450  //add the increment to the total control input
451  indi.u_in.p = indi.u[0].o[0] + indi.du.p;
452  indi.u_in.q = indi.u[1].o[0] + indi.du.q;
453  indi.u_in.r = indi.u[2].o[0] + indi.du.r;
454 
455  // only run the estimation if the commands are not zero.
456  lms_estimation();
457  }
458 
459  //bound the total control input
460 #if STABILIZATION_INDI_FULL_AUTHORITY
461  Bound(indi.u_in.p, -9600, 9600);
462  Bound(indi.u_in.q, -9600, 9600);
463  float rlim = 9600 - fabs(indi.u_in.q);
464  Bound(indi.u_in.r, -rlim, rlim);
465  Bound(indi.u_in.r, -9600, 9600);
466 #else
467  Bound(indi.u_in.p, -4500, 4500);
468  Bound(indi.u_in.q, -4500, 4500);
469  Bound(indi.u_in.r, -4500, 4500);
470 #endif
471 
472  /* INDI feedback */
473  stabilization_cmd[COMMAND_ROLL] = indi.u_in.p;
474  stabilization_cmd[COMMAND_PITCH] = indi.u_in.q;
475  stabilization_cmd[COMMAND_YAW] = indi.u_in.r;
476 
477  /* bound the result */
478  BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
479  BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
480  BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
481 }

489 void stabilization_indi_attitude_run(struct Int32Quat quat_sp, bool in_flight __attribute__((unused)))
490 {
491  /* attitude error */
492  struct FloatQuat att_err;
493  struct FloatQuat *att_quat = stateGetNedToBodyQuat_f();
494  struct FloatQuat quat_sp_f;
495 
496  QUAT_FLOAT_OF_BFP(quat_sp_f, quat_sp);
497  float_quat_inv_comp_norm_shortest(&att_err, att_quat, &quat_sp_f);
498 
499  struct FloatVect3 att_fb;
500 
501 #if TILT_TWIST_CTRL
502  struct FloatQuat tilt;
503  struct FloatQuat twist;
504  float_quat_tilt_twist(&tilt, &twist, &att_err);
505  att_fb.x = tilt.qx;
506  att_fb.y = tilt.qy;
507  att_fb.z = twist.qz;
508 #else
509  att_fb.x = att_err.qx;
510  att_fb.y = att_err.qy;
511  att_fb.z = att_err.qz;
512 #endif
513 
514  struct FloatRates rate_sp;
515  // Divide by rate gain to make it equivalent to a parallel structure
516  rate_sp.p = indi.gains.att.p * att_fb.x / indi.gains.rate.p;
517  rate_sp.q = indi.gains.att.q * att_fb.y / indi.gains.rate.q;
518  rate_sp.r = indi.gains.att.r * att_fb.z / indi.gains.rate.r;
519 
520  // Add feed-forward rates to the attitude feedback part
521  RATES_ADD(rate_sp, stab_att_ff_rates);
522 
523  /* compute the INDI command */
524  stabilization_indi_rate_run(rate_sp, in_flight);
525 }

549 static inline void lms_estimation(void)
550 {
551  static struct IndiEstimation *est = &indi.est;
552  // Only pass really low frequencies so you don't adapt to noise
553  struct FloatRates *body_rates = stateGetBodyRates_f();
554  filter_pqr(est->u, &indi.u_act_dyn);
555  filter_pqr(est->rate, body_rates);
556 
557  // Calculate the first and second derivatives of the rates and actuators
558  float rate_d_prev[3];
559  float u_d_prev[3];
560  float_vect_copy(rate_d_prev, est->rate_d, 3);
561  float_vect_copy(u_d_prev, est->u_d, 3);
562  finite_difference_from_filter(est->rate_d, est->rate);
563  finite_difference_from_filter(est->u_d, est->u);
564  finite_difference(est->rate_dd, est->rate_d, rate_d_prev);
565  finite_difference(est->u_dd, est->u_d, u_d_prev);
566 
567  // The inputs are scaled in order to avoid overflows
568  float du[3];
569  float_vect_copy(du, est->u_d, 3);
570  float_vect_scale(du, INDI_EST_SCALE, 3);
571  est->g1.p = est->g1.p - (est->g1.p * du[0] - est->rate_dd[0]) * du[0] * est->mu;
572  est->g1.q = est->g1.q - (est->g1.q * du[1] - est->rate_dd[1]) * du[1] * est->mu;
573  float ddu = est->u_dd[2] * INDI_EST_SCALE / PERIODIC_FREQUENCY;
574  float error = (est->g1.r * du[2] + est->g2 * ddu - est->rate_dd[2]);
575  est->g1.r = est->g1.r - error * du[2] * est->mu / 3;
576  est->g2 = est->g2 - error * 1000 * ddu * est->mu / 3;
577 
578  //the g values should be larger than zero, otherwise there is positive feedback, the command will go to max and there is nothing to learn anymore...
579  if (est->g1.p < 0.01) { est->g1.p = 0.01; }
580  if (est->g1.q < 0.01) { est->g1.q = 0.01; }
581  if (est->g1.r < 0.01) { est->g1.r = 0.01; }
582  if (est->g2 < 0.01) { est->g2 = 0.01; }
583 
584  if (indi.adaptive) {
585  //Commit the estimated G values and apply the scaling
586  indi.g1.p = est->g1.p * INDI_EST_SCALE;
587  indi.g1.q = est->g1.q * INDI_EST_SCALE;
588  indi.g1.r = est->g1.r * INDI_EST_SCALE;
589  indi.g2 = est->g2 * INDI_EST_SCALE;
590  }
