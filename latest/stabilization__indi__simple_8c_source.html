198 void indi_init_filters(void)
199 {
200  // tau = 1/(2*pi*Fc)
201  float tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF);
202  float tau_rdot = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_RDOT);
203  float tau_axis[3] = {tau, tau, tau_rdot};
204  float tau_est = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_ESTIMATION_FILT_CUTOFF);
205  float sample_time = 1.0 / PERIODIC_FREQUENCY;
206  // Filtering of gyroscope and actuators
207  for (int8_t i = 0; i < 3; i++) {
208  init_butterworth_2_low_pass(&indi.u[i], tau_axis[i], sample_time, 0.0);
209  init_butterworth_2_low_pass(&indi.rate[i], tau_axis[i], sample_time, 0.0);
210  init_butterworth_2_low_pass(&indi.est.u[i], tau_est, sample_time, 0.0);
211  init_butterworth_2_low_pass(&indi.est.rate[i], tau_est, sample_time, 0.0);
212  }
213 
214  // Init rate filter for feedback
215  float time_constants[3] = {1.0/(2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_P), 1.0/(2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_Q), 1.0/(2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_R)};
216 
217  init_first_order_low_pass(&rates_filt_fo[0], time_constants[0], sample_time, stateGetBodyRates_f()->p);
218  init_first_order_low_pass(&rates_filt_fo[1], time_constants[1], sample_time, stateGetBodyRates_f()->q);
219  init_first_order_low_pass(&rates_filt_fo[2], time_constants[2], sample_time, stateGetBodyRates_f()->r);
220 }

330 {

356 void stabilization_indi_rate_run(struct FloatRates rate_sp, bool in_flight __attribute__((unused)))
357 {
358  //Propagate input filters
359  //first order actuator dynamics
360  indi.u_act_dyn.p = indi.u_act_dyn.p + STABILIZATION_INDI_ACT_DYN_P * (indi.u_in.p - indi.u_act_dyn.p);
361  indi.u_act_dyn.q = indi.u_act_dyn.q + STABILIZATION_INDI_ACT_DYN_Q * (indi.u_in.q - indi.u_act_dyn.q);
362  indi.u_act_dyn.r = indi.u_act_dyn.r + STABILIZATION_INDI_ACT_DYN_R * (indi.u_in.r - indi.u_act_dyn.r);
363 
364  // Propagate the filter on the gyroscopes and actuators
365  struct FloatRates *body_rates = stateGetBodyRates_f();
366  filter_pqr(indi.u, &indi.u_act_dyn);
367  filter_pqr(indi.rate, body_rates);
368 
369  // Calculate the derivative of the rates
370  finite_difference_from_filter(indi.rate_d, indi.rate);
371 
372  //The rates used for feedback are by default the measured rates.
373  //If there is a lot of noise on the gyroscope, it might be good to use the filtered value for feedback.
374  //Note that due to the delay, the PD controller may need relaxed gains.
375  struct FloatRates rates_filt;
376 #if STABILIZATION_INDI_FILTER_ROLL_RATE
377  rates_filt.p = update_first_order_low_pass(&rates_filt_fo[0], body_rates->p);
378 #else
379  rates_filt.p = body_rates->p;
380 #endif
381 #if STABILIZATION_INDI_FILTER_PITCH_RATE
382  rates_filt.q = update_first_order_low_pass(&rates_filt_fo[1], body_rates->q);
383 #else
384  rates_filt.q = body_rates->q;
385 #endif
386 #if STABILIZATION_INDI_FILTER_YAW_RATE
387  rates_filt.r = update_first_order_low_pass(&rates_filt_fo[2], body_rates->r);
388 #else
389  rates_filt.r = body_rates->r;
390 #endif
391 
392  //This lets you impose a maximum yaw rate.
393  BoundAbs(rate_sp.r, indi.attitude_max_yaw_rate);
394 
395  // Compute reference angular acceleration:
396  indi.angular_accel_ref.p = (rate_sp.p - rates_filt.p) * indi.gains.rate.p;
397  indi.angular_accel_ref.q = (rate_sp.q - rates_filt.q) * indi.gains.rate.q;
398  indi.angular_accel_ref.r = (rate_sp.r - rates_filt.r) * indi.gains.rate.r;
399 
400  //Increment in angular acceleration requires increment in control input
401  //G1 is the control effectiveness. In the yaw axis, we need something additional: G2.
402  //It takes care of the angular acceleration caused by the change in rotation rate of the propellers
403  //(they have significant inertia, see the paper mentioned in the header for more explanation)
404  indi.du.p = 1.0 / indi.g1.p * (indi.angular_accel_ref.p - indi.rate_d[0]);
405  indi.du.q = 1.0 / indi.g1.q * (indi.angular_accel_ref.q - indi.rate_d[1]);
406  indi.du.r = 1.0 / (indi.g1.r + indi.g2) * (indi.angular_accel_ref.r - indi.rate_d[2] + indi.g2 * indi.du.r);
407 
408  //Don't increment if thrust is off and on the ground
409  //without this the inputs will increment to the maximum before even getting in the air.
410  if (stabilization_cmd[COMMAND_THRUST] < 300 && !in_flight) {
411  FLOAT_RATES_ZERO(indi.u_in);
412 
413  // If on the gournd, no increments, just proportional control
414  indi.u_in.p = indi.du.p;
415  indi.u_in.q = indi.du.q;
416  indi.u_in.r = indi.du.r;
417  } else {
418  //add the increment to the total control input
419  indi.u_in.p = indi.u[0].o[0] + indi.du.p;
420  indi.u_in.q = indi.u[1].o[0] + indi.du.q;
421  indi.u_in.r = indi.u[2].o[0] + indi.du.r;
422 
423  // only run the estimation if the commands are not zero.
424  lms_estimation();
425  }
426 
427  //bound the total control input
428 #if STABILIZATION_INDI_FULL_AUTHORITY
429  Bound(indi.u_in.p, -9600, 9600);
430  Bound(indi.u_in.q, -9600, 9600);
431  float rlim = 9600 - fabs(indi.u_in.q);
432  Bound(indi.u_in.r, -rlim, rlim);
433  Bound(indi.u_in.r, -9600, 9600);
434 #else
435  Bound(indi.u_in.p, -4500, 4500);
436  Bound(indi.u_in.q, -4500, 4500);
437  Bound(indi.u_in.r, -4500, 4500);
438 #endif
439 
440  /* INDI feedback */
441  stabilization_cmd[COMMAND_ROLL] = indi.u_in.p;
442  stabilization_cmd[COMMAND_PITCH] = indi.u_in.q;
443  stabilization_cmd[COMMAND_YAW] = indi.u_in.r;
444 
445  /* bound the result */
446  BoundAbs(stabilization_cmd[COMMAND_ROLL], MAX_PPRZ);
447  BoundAbs(stabilization_cmd[COMMAND_PITCH], MAX_PPRZ);
448  BoundAbs(stabilization_cmd[COMMAND_YAW], MAX_PPRZ);
449 }

457 void stabilization_indi_attitude_run(struct Int32Quat quat_sp, bool in_flight __attribute__((unused)))
458 {
459  /* attitude error */
460  struct FloatQuat att_err;
461  struct FloatQuat *att_quat = stateGetNedToBodyQuat_f();
462  struct FloatQuat quat_sp_f;
463 
464  QUAT_FLOAT_OF_BFP(quat_sp_f, quat_sp);
465  float_quat_inv_comp_norm_shortest(&att_err, att_quat, &quat_sp_f);
466 
467  struct FloatVect3 att_fb;
468 
469 #if TILT_TWIST_CTRL
470  struct FloatQuat tilt;
471  struct FloatQuat twist;
472  float_quat_tilt_twist(&tilt, &twist, &att_err);
473  att_fb.x = tilt.qx;
474  att_fb.y = tilt.qy;
475  att_fb.z = twist.qz;
476 #else
477  att_fb.x = att_err.qx;
478  att_fb.y = att_err.qy;
479  att_fb.z = att_err.qz;
480 #endif
481 
482  struct FloatRates rate_sp;
483  // Divide by rate gain to make it equivalent to a parallel structure
484  rate_sp.p = indi.gains.att.p * att_fb.x / indi.gains.rate.p;
485  rate_sp.q = indi.gains.att.q * att_fb.y / indi.gains.rate.q;
486  rate_sp.r = indi.gains.att.r * att_fb.z / indi.gains.rate.r;
487 
488  /* compute the INDI command */
489  stabilization_indi_rate_run(rate_sp, in_flight);
490 }

514 static inline void lms_estimation(void)
515 {
516  static struct IndiEstimation *est = &indi.est;
517  // Only pass really low frequencies so you don't adapt to noise
518  struct FloatRates *body_rates = stateGetBodyRates_f();
519  filter_pqr(est->u, &indi.u_act_dyn);
520  filter_pqr(est->rate, body_rates);
521 
522  // Calculate the first and second derivatives of the rates and actuators
523  float rate_d_prev[3];
524  float u_d_prev[3];
525  float_vect_copy(rate_d_prev, est->rate_d, 3);
526  float_vect_copy(u_d_prev, est->u_d, 3);
527  finite_difference_from_filter(est->rate_d, est->rate);
528  finite_difference_from_filter(est->u_d, est->u);
529  finite_difference(est->rate_dd, est->rate_d, rate_d_prev);
530  finite_difference(est->u_dd, est->u_d, u_d_prev);
531 
532  // The inputs are scaled in order to avoid overflows
533  float du[3];
534  float_vect_copy(du, est->u_d, 3);
535  float_vect_scale(du, INDI_EST_SCALE, 3);
536  est->g1.p = est->g1.p - (est->g1.p * du[0] - est->rate_dd[0]) * du[0] * est->mu;
537  est->g1.q = est->g1.q - (est->g1.q * du[1] - est->rate_dd[1]) * du[1] * est->mu;
538  float ddu = est->u_dd[2] * INDI_EST_SCALE / PERIODIC_FREQUENCY;
539  float error = (est->g1.r * du[2] + est->g2 * ddu - est->rate_dd[2]);
540  est->g1.r = est->g1.r - error * du[2] * est->mu / 3;
541  est->g2 = est->g2 - error * 1000 * ddu * est->mu / 3;
542 
543  //the g values should be larger than zero, otherwise there is positive feedback, the command will go to max and there is nothing to learn anymore...
544  if (est->g1.p < 0.01) { est->g1.p = 0.01; }
545  if (est->g1.q < 0.01) { est->g1.q = 0.01; }
546  if (est->g1.r < 0.01) { est->g1.r = 0.01; }
547  if (est->g2 < 0.01) { est->g2 = 0.01; }
548
