73 static float compute_pid(struct RoverGuidancePID *pid)
74 {
75  return pid->p * pid->err + pid->d * pid->d_err + pid->i * pid->sum_err;
76 }

83 void rover_guidance_run(float *heading_sp)
84 {
85  // compute position error
86  struct FloatVect2 pos_err;
87  VECT2_DIFF(pos_err, rover_guidance.sp.pos, *stateGetPositionEnu_f());
88  rover_guidance.speed_pid.err = float_vect2_norm(&pos_err);
89  BoundAbs(rover_guidance.speed_pid.err, MAX_POS_ERR);
90 
91  // speed and heading update when far enough
92  if (rover_guidance.speed_pid.err > PROXIMITY_DIST) {
93  // compute speed error
94  rover_guidance.speed_pid.d_err = - stateGetHorizontalSpeedNorm_f();
95  BoundAbs(rover_guidance.speed_pid.d_err, MAX_SPEED_ERR);
96  // integral
97  rover_guidance.speed_pid.sum_err = 0.f; // nothing for now
98  // run PID
99  rover_guidance.cmd.motor_speed = MAX_PPRZ * compute_pid(&rover_guidance.speed_pid);
100  rover_guidance.cmd.motor_speed = TRIM_PPRZ(rover_guidance.cmd.motor_speed);
101 
102  // if not close to WP, compute desired heading
103  rover_guidance.sp.heading = atan2f(pos_err.x, pos_err.y);
104  *heading_sp = rover_guidance.sp.heading; // update nav sp
105  }
106  else {
107  rover_guidance.cmd.motor_speed = 0.f;
108  // use nav heading ref and run angular control
109  rover_guidance.sp.heading = *heading_sp;
110  }
111 
112  // angular error
113  rover_guidance.turn_pid.err = rover_guidance.sp.heading - stateGetNedToBodyEulers_f()->psi;
114  NormRadAngle(rover_guidance.turn_pid.err);
115  // turn rate error
116  rover_guidance.turn_pid.d_err = - stateGetBodyRates_f()->r;
117  // integral
118  rover_guidance.turn_pid.sum_err = 0.f; // nothing for now
119  // run PID
120  rover_guidance.cmd.motor_turn = MAX_PPRZ * compute_pid(&rover_guidance.turn_pid);
121  rover_guidance.cmd.motor_turn = TRIM_PPRZ(rover_guidance.cmd.motor_turn);
122 }

221 {
222  float psi = stateGetNedToBodyEulers_f()->psi;
