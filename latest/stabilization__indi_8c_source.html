334 void stabilization_indi_init(void)
335 {

337  init_filters();

344  float_vect_zero(actuator_state_filt_vectd, INDI_NUM_ACT);
345  float_vect_zero(actuator_state_filt_vectdd, INDI_NUM_ACT);
346  float_vect_zero(estimation_rate_d, INDI_NUM_ACT);
347  float_vect_zero(estimation_rate_dd, INDI_NUM_ACT);
348  float_vect_zero(actuator_state_filt_vect, INDI_NUM_ACT);
349 
350  //Calculate G1G2
351  sum_g1_g2();
352 
353  // Do not compute if not needed
354 #if STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
355  //Calculate G1G2_PSEUDO_INVERSE
356  calc_g1g2_pseudo_inv();
357 #endif
358 
359  int8_t i;
360  // Initialize the array of pointers to the rows of g1g2
361  for (i = 0; i < INDI_OUTPUTS; i++) {
362  Bwls[i] = g1g2[i];
363  }
364 
365  // Initialize the estimator matrices
366  float_vect_copy(g1_est[0], g1[0], INDI_OUTPUTS * INDI_NUM_ACT);
367  float_vect_copy(g2_est, g2, INDI_NUM_ACT);
368  // Remember the initial matrices
369  float_vect_copy(g1_init[0], g1[0], INDI_OUTPUTS * INDI_NUM_ACT);
370  float_vect_copy(g2_init, g2, INDI_NUM_ACT);
371 
372  // Assume all non-servos are delivering thrust
373  num_thrusters = INDI_NUM_ACT;
374  num_thrusters_x = 0;
375  for (i = 0; i < INDI_NUM_ACT; i++) {
376  num_thrusters -= act_is_servo[i];
377  num_thrusters -= act_is_thruster_x[i];
378 
379  num_thrusters_x += act_is_thruster_x[i];
380 
381  act_is_thruster_z[i] = !act_is_servo[i] && !act_is_thruster_x[i];
382  }
383 
384 #if PERIODIC_TELEMETRY
385  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_INDI_G, send_indi_g);
386  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_AHRS_REF_QUAT, send_ahrs_ref_quat);
387  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_STAB_ATTITUDE_INDI, send_att_full_indi);
388 #endif
389 }

411 void init_filters(void)
412 {
413  // tau = 1/(2*pi*Fc)
414  float tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF);
415  float tau_est = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_ESTIMATION_FILT_CUTOFF);
416  float sample_time = 1.0 / PERIODIC_FREQUENCY;
417  // Filtering of the gyroscope
418  int8_t i;
419  for (i = 0; i < 3; i++) {
420  init_butterworth_2_low_pass(&measurement_lowpass_filters[i], tau, sample_time, 0.0);
421  init_butterworth_2_low_pass(&estimation_output_lowpass_filters[i], tau_est, sample_time, 0.0);
422  }
423 
424  // Filtering of the actuators
425  for (i = 0; i < INDI_NUM_ACT; i++) {
426  init_butterworth_2_low_pass(&actuator_lowpass_filters[i], tau, sample_time, 0.0);
427  init_butterworth_2_low_pass(&estimation_input_lowpass_filters[i], tau_est, sample_time, 0.0);
428  }
429 
430  // Filtering of the accel body z
431  init_butterworth_2_low_pass(&acceleration_lowpass_filter, tau_est, sample_time, 0.0);
432 
433 #if STABILIZATION_INDI_FILTER_RATES_SECOND_ORDER
434  tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_P);
435  init_butterworth_2_low_pass(&rates_filt_so[0], tau, sample_time, 0.0);
436  tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_Q);
437  init_butterworth_2_low_pass(&rates_filt_so[1], tau, sample_time, 0.0);
438  tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_R);
439  init_butterworth_2_low_pass(&rates_filt_so[2], tau, sample_time, 0.0);
440 #else
441  // Init rate filter for feedback
442  float time_constants[3] = {1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_P), 1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_Q), 1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_R)};
443 
444  init_first_order_low_pass(&rates_filt_fo[0], time_constants[0], sample_time, stateGetBodyRates_f()->p);
445  init_first_order_low_pass(&rates_filt_fo[1], time_constants[1], sample_time, stateGetBodyRates_f()->q);
446  init_first_order_low_pass(&rates_filt_fo[2], time_constants[2], sample_time, stateGetBodyRates_f()->r);
447 #endif
448 }

530 void stabilization_indi_rate_run(struct FloatRates rate_sp, bool in_flight)
531 {
532  /* Propagate the filter on the gyroscopes */
533  struct FloatRates *body_rates = stateGetBodyRates_f();
534  float rate_vect[3] = {body_rates->p, body_rates->q, body_rates->r};
535  int8_t i;
536  for (i = 0; i < 3; i++) {
537  update_butterworth_2_low_pass(&measurement_lowpass_filters[i], rate_vect[i]);
538  update_butterworth_2_low_pass(&estimation_output_lowpass_filters[i], rate_vect[i]);
539 
540  //Calculate the angular acceleration via finite difference
541  angular_acceleration[i] = (measurement_lowpass_filters[i].o[0]
542  - measurement_lowpass_filters[i].o[1]) * PERIODIC_FREQUENCY;
