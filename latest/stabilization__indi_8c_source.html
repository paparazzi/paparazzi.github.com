374 void stabilization_indi_init(void)
375 {
376  // Initialize filters
377  init_filters();
378 
379  int8_t i;
380 // If the deprecated STABILIZATION_INDI_ACT_DYN is used, convert it to the new FREQUENCY format
381 #ifdef STABILIZATION_INDI_ACT_FREQ
382  // Initialize the array of pointers to the rows of g1g2
383  for (i = 0; i < INDI_NUM_ACT; i++) {
384  act_dyn_discrete[i] = 1-exp(-act_first_order_cutoff[i]/PERIODIC_FREQUENCY);

386 #endif
387 
388 #if STABILIZATION_INDI_RPM_FEEDBACK
389  AbiBindMsgACT_FEEDBACK(STABILIZATION_INDI_ACT_FEEDBACK_ID, &act_feedback_ev, act_feedback_cb);

392  float_vect_zero(actuator_state_filt_vectd, INDI_NUM_ACT);
393  float_vect_zero(actuator_state_filt_vectdd, INDI_NUM_ACT);
394  float_vect_zero(estimation_rate_d, INDI_NUM_ACT);
395  float_vect_zero(estimation_rate_dd, INDI_NUM_ACT);
396  float_vect_zero(actuator_state_filt_vect, INDI_NUM_ACT);
397 
398  //Calculate G1G2
399  sum_g1_g2();
400 
401  // Do not compute if not needed
402 #if STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
403  //Calculate G1G2_PSEUDO_INVERSE
404  calc_g1g2_pseudo_inv();
405 #endif
406 
407  // Initialize the array of pointers to the rows of g1g2
408  for (i = 0; i < INDI_OUTPUTS; i++) {
409  Bwls[i] = g1g2[i];
410  }
411 
412  // Initialize the estimator matrices
413  float_vect_copy(g1_est[0], g1[0], INDI_OUTPUTS * INDI_NUM_ACT);
414  float_vect_copy(g2_est, g2, INDI_NUM_ACT);

417  float_vect_copy(g2_init, g2, INDI_NUM_ACT);
418 
419  // Assume all non-servos are delivering thrust
420  num_thrusters = INDI_NUM_ACT;
421  num_thrusters_x = 0;
422  for (i = 0; i < INDI_NUM_ACT; i++) {
423  num_thrusters -= act_is_servo[i];
424  num_thrusters -= act_is_thruster_x[i];
425 
426  num_thrusters_x += act_is_thruster_x[i];
427

429  }

431 #if PERIODIC_TELEMETRY

473 void init_filters(void)
474 {
475  // tau = 1/(2*pi*Fc)
476  float tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF);
477  float tau_est = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_ESTIMATION_FILT_CUTOFF);
478  float sample_time = 1.0 / PERIODIC_FREQUENCY;
479  // Filtering of the gyroscope
480  int8_t i;
481  for (i = 0; i < 3; i++) {
482  init_butterworth_2_low_pass(&measurement_lowpass_filters[i], tau, sample_time, 0.0);
483  init_butterworth_2_low_pass(&estimation_output_lowpass_filters[i], tau_est, sample_time, 0.0);
484  }
485 
486  // Filtering of the actuators
487  for (i = 0; i < INDI_NUM_ACT; i++) {
488  init_butterworth_2_low_pass(&actuator_lowpass_filters[i], tau, sample_time, 0.0);
489  init_butterworth_2_low_pass(&estimation_input_lowpass_filters[i], tau_est, sample_time, 0.0);
490  }
491 
492  // Filtering the bodyx acceleration with same cutoff as gyroscope
493  init_butterworth_2_low_pass(&acceleration_body_x_filter, tau, sample_time, 0.0);
494 
495  // Filtering of the accel body z
496  init_butterworth_2_low_pass(&acceleration_lowpass_filter, tau_est, sample_time, 0.0);
497 
498 #if STABILIZATION_INDI_FILTER_RATES_SECOND_ORDER
499  tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_P);
500  init_butterworth_2_low_pass(&rates_filt_so[0], tau, sample_time, 0.0);
501  tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_Q);
502  init_butterworth_2_low_pass(&rates_filt_so[1], tau, sample_time, 0.0);
503  tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_R);
504  init_butterworth_2_low_pass(&rates_filt_so[2], tau, sample_time, 0.0);
505 #else
506  // Init rate filter for feedback
507  float time_constants[3] = {1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_P), 1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_Q), 1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_R)};
508 
509  init_first_order_low_pass(&rates_filt_fo[0], time_constants[0], sample_time, stateGetBodyRates_f()->p);

523 void stabilization_indi_rate_run(bool in_flight, struct StabilizationSetpoint *sp, struct ThrustSetpoint *thrust, int32_t *cmd)
524 {

526  // Propagate actuator filters
527  get_actuator_state();
528  float actuator_state_filt_vect_prev[INDI_NUM_ACT];
529  for (int i = 0; i < INDI_NUM_ACT; i++) {
530  update_butterworth_2_low_pass(&actuator_lowpass_filters[i], actuator_state[i]);
531  update_butterworth_2_low_pass(&estimation_input_lowpass_filters[i], actuator_state[i]);
532  actuator_state_filt_vect[i] = actuator_lowpass_filters[i].o[0];
533  actuator_state_filt_vect_prev[i] = actuator_lowpass_filters[i].o[1];
534 
535  // calculate derivatives for estimation
536  float actuator_state_filt_vectd_prev = actuator_state_filt_vectd[i];
537  actuator_state_filt_vectd[i] = (estimation_input_lowpass_filters[i].o[0] - estimation_input_lowpass_filters[i].o[1]) *
538  PERIODIC_FREQUENCY;
539  actuator_state_filt_vectdd[i] = (actuator_state_filt_vectd[i] - actuator_state_filt_vectd_prev) * PERIODIC_FREQUENCY;
540  }
541 
542  // Use the last actuator state for this computation
543  float g2_times_u_act_filt = float_vect_dot_product(g2, actuator_state_filt_vect_prev, INDI_NUM_ACT)/INDI_G_SCALING;
544 
545  // Predict angular acceleration u*B
546  float angular_acc_prediction_filt[INDI_OUTPUTS];
547  float_mat_vect_mul(angular_acc_prediction_filt, Bwls, actuator_state_filt_vect, INDI_OUTPUTS, INDI_NUM_ACT);
548  angular_acc_prediction_filt[2] -= g2_times_u_act_filt;
549 
550  /* Propagate the filter on the gyroscopes */
551  struct FloatRates *body_rates = stateGetBodyRates_f();
552  float rate_vect[3] = {body_rates->p, body_rates->q, body_rates->r};
553 
554  // Get the acceleration in body axes
555  struct Int32Vect3 *body_accel_i;
556  body_accel_i = stateGetAccelBody_i();
557  ACCELS_FLOAT_OF_BFP(body_accel_f, *body_accel_i);
558 
559  int8_t i;
560  for (i = 0; i < 3; i++) {
561  update_butterworth_2_low_pass(&measurement_lowpass_filters[i], rate_vect[i]);
562  update_butterworth_2_low_pass(&estimation_output_lowpass_filters[i], rate_vect[i]);
563 
564  update_butterworth_2_low_pass(&acceleration_body_x_filter, body_accel_f.x);
565 
566  //Calculate the angular acceleration via finite difference
567  angular_acceleration[i] = (measurement_lowpass_filters[i].o[0]
568  - measurement_lowpass_filters[i].o[1]) * PERIODIC_FREQUENCY;
569 
570  // Calculate derivatives for estimation
571  float estimation_rate_d_prev = estimation_rate_d[i];
572  estimation_rate_d[i] = (estimation_output_lowpass_filters[i].o[0] - estimation_output_lowpass_filters[i].o[1]) *
573  PERIODIC_FREQUENCY;
574  estimation_rate_dd[i] = (estimation_rate_d[i] - estimation_rate_d_prev) * PERIODIC_FREQUENCY;
575  }
576 
577  // subtract u*B from angular acceleration
578  float angular_acc_disturbance_estimate[INDI_OUTPUTS];
579  float_vect_diff(angular_acc_disturbance_estimate, angular_acceleration, angular_acc_prediction_filt, 3);
580 
581  if (in_flight) {
582  // Limit the estimated disturbance in yaw for drones that are stable in sideslip
583  BoundAbs(angular_acc_disturbance_estimate[2], stablization_indi_yaw_dist_limit);
584  } else {
585  // Not in flight, so don't estimate disturbance
586  float_vect_zero(angular_acc_disturbance_estimate, INDI_OUTPUTS);
587  }
588 
589  //The rates used for feedback are by default the measured rates.
590  //If there is a lot of noise on the gyroscope, it might be good to use the filtered value for feedback.
591  //Note that due to the delay, the PD controller may need relaxed gains.
592  struct FloatRates rates_filt;
593 #if STABILIZATION_INDI_FILTER_ROLL_RATE
594 #if STABILIZATION_INDI_FILTER_RATES_SECOND_ORDER
595  rates_filt.p = update_butterworth_2_low_pass(&rates_filt_so[0], body_rates->p);
596 #else
597  rates_filt.p = update_first_order_low_pass(&rates_filt_fo[0], body_rates->p);
598 #endif
599 #else
600  rates_filt.p = body_rates->p;
601 #endif
602 #if STABILIZATION_INDI_FILTER_PITCH_RATE
603 #if STABILIZATION_INDI_FILTER_RATES_SECOND_ORDER
604  rates_filt.q = update_butterworth_2_low_pass(&rates_filt_so[1], body_rates->q);
605 #else
606  rates_filt.q = update_first_order_low_pass(&rates_filt_fo[1], body_rates->q);
607 #endif
608 #else
609  rates_filt.q = body_rates->q;
610 #endif
611 #if STABILIZATION_INDI_FILTER_YAW_RATE
612 #if STABILIZATION_INDI_FILTER_RATES_SECOND_ORDER
613  rates_filt.r = update_butterworth_2_low_pass(&rates_filt_so[2], body_rates->r);
614 #else
615  rates_filt.r = update_first_order_low_pass(&rates_filt_fo[2], body_rates->r);
616 #endif
617 #else
618  rates_filt.r = body_rates->r;
619 #endif
620 
621  // calculate the virtual control (reference acceleration) based on a PD controller
622  struct FloatRates rate_sp = stab_sp_to_rates_f(sp);
623  angular_accel_ref.p = (rate_sp.p - rates_filt.p) * indi_gains.rate.p;
624  angular_accel_ref.q = (rate_sp.q - rates_filt.q) * indi_gains.rate.q;
625  angular_accel_ref.r = (rate_sp.r - rates_filt.r) * indi_gains.rate.r;
626 
627  // compute virtual thrust
628  struct FloatVect3 v_thrust = { 0.f, 0.f, 0.f };
629  if (thrust->type == THRUST_INCR_SP) {
630  v_thrust.x = th_sp_to_incr_f(thrust, 0, THRUST_AXIS_X);
631  v_thrust.y = th_sp_to_incr_f(thrust, 0, THRUST_AXIS_Y);
632  v_thrust.z = th_sp_to_incr_f(thrust, 0, THRUST_AXIS_Z);
633 
634  // Compute estimated thrust
635  struct FloatVect3 thrust_filt = { 0.f, 0.f, 0.f };
636  for (i = 0; i < INDI_NUM_ACT; i++) {
637  thrust_filt.z += Bwls[3][i]* actuator_lowpass_filters[i].o[0] * (int32_t) act_is_thruster_z[i];
638 #if INDI_OUTPUTS == 5
639  thrust_filt.x += Bwls[4][i]* actuator_lowpass_filters[i].o[0] * (int32_t) act_is_thruster_x[i];
640 #endif
641  }
642  // Add the current estimated thrust to the increment
643  VECT3_ADD(v_thrust, thrust_filt);
644  } else {
645  // build incremental thrust
646  float th_cmd_z = (float)th_sp_to_thrust_i(thrust, 0, THRUST_AXIS_Z);
647  for (i = 0; i < INDI_NUM_ACT; i++) {
648  v_thrust.z += th_cmd_z * Bwls[3][i];
649 #if INDI_OUTPUTS == 5
650  // TODO set X thrust from RC in the thrust input setpoint
651  cmd[COMMAND_THRUST_X] = radio_control.values[RADIO_CONTROL_THRUST_X];
652  v_thrust.x += cmd[COMMAND_THRUST_X] * Bwls[4][i];
653 #endif
654  }
655  v_thrust.y = 0.f;
656  }
657 
658  // This term compensates for the spinup torque in the yaw axis
659  float g2_times_u = float_vect_dot_product(g2, indi_u, INDI_NUM_ACT)/INDI_G_SCALING;
660 
661  if (in_flight) {
662  // Limit the estimated disturbance in yaw for drones that are stable in sideslip
663  BoundAbs(angular_acc_disturbance_estimate[2], stablization_indi_yaw_dist_limit);
664  } else {
665  // Not in flight, so don't estimate disturbance
666  float_vect_zero(angular_acc_disturbance_estimate, INDI_OUTPUTS);
667  }
668 
669  // The control objective in array format
670  indi_v[0] = (angular_accel_ref.p - angular_acc_disturbance_estimate[0]);
671  indi_v[1] = (angular_accel_ref.q - angular_acc_disturbance_estimate[1]);
672  indi_v[2] = (angular_accel_ref.r - angular_acc_disturbance_estimate[2]) + g2_times_u;
673  indi_v[3] = v_thrust.z;
674 #if INDI_OUTPUTS == 5
675  indi_v[4] = v_thrust.x;
676 #endif
677 
678 #if STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
679  // Calculate the increment for each actuator
680  for (i = 0; i < INDI_NUM_ACT; i++) {
681  indi_u[i] = (g1g2_pseudo_inv[i][0] * indi_v[0])
682  + (g1g2_pseudo_inv[i][1] * indi_v[1])
683  + (g1g2_pseudo_inv[i][2] * indi_v[2])
684  + (g1g2_pseudo_inv[i][3] * indi_v[3]);
685  }
686 #else
687  stabilization_indi_set_wls_settings();
688 
689  // WLS Control Allocator
690  for (i = 0; i < INDI_OUTPUTS; i++) {
691  wls_stab_p.v[i] = indi_v[i];
692  }
693  wls_alloc(&wls_stab_p, Bwls, 0, 0, 10);
694  for (i = 0; i < INDI_NUM_ACT; i++) {
695  indi_u [i] = wls_stab_p.u[i];
696  }
697 #endif
698 
699  // Bound the inputs to the actuators
700  for (i = 0; i < INDI_NUM_ACT; i++) {
701  if (act_is_servo[i]) {
702  BoundAbs(indi_u[i], MAX_PPRZ);
703  } else {
704  if (autopilot_get_motors_on()) {
705  Bound(indi_u[i], 0, MAX_PPRZ);
706  } else {
707  indi_u[i] = -MAX_PPRZ;
708  }
709  }
710  }
711 
712  // Use online effectiveness estimation only when flying
713  if (in_flight && indi_use_adaptive) {
714  lms_estimation();
715  }
716 
717  /*Commit the actuator command*/
718  for (i = 0; i < INDI_NUM_ACT; i++) {
719  actuators_pprz[i] = (int16_t) indi_u[i];
720  }
721 
722  //update thrust command such that the current is correctly estimated
723  cmd[COMMAND_THRUST] = 0;
724  for (i = 0; i < INDI_NUM_ACT; i++) {
725  cmd[COMMAND_THRUST] += actuator_state[i] * (int32_t) act_is_thruster_z[i];
726  }
727  cmd[COMMAND_THRUST] /= num_thrusters;
728

882 void lms_estimation(void)
883 {
884 
885  // Get the acceleration in body axes
886  struct Int32Vect3 *body_accel_i;
887  body_accel_i = stateGetAccelBody_i();
888  ACCELS_FLOAT_OF_BFP(body_accel_f, *body_accel_i);
889 
890  // Filter the acceleration in z axis
891  update_butterworth_2_low_pass(&acceleration_lowpass_filter, body_accel_f.z);
892 
893  // Calculate the derivative of the acceleration via finite difference
894  float indi_accel_d = (acceleration_lowpass_filter.o[0]
895  - acceleration_lowpass_filter.o[1]) * PERIODIC_FREQUENCY;

901 #else

903 #endif
904

907  float_vect_smul(ddu_estimation, actuator_state_filt_vectdd, adaptive_mu_lr / PERIODIC_FREQUENCY, INDI_NUM_ACT);
908

915  for (i = 0; i < INDI_OUTPUTS; i++) {
916  // Calculate the error between prediction and measurement

918  int8_t j;

931  }
932  } else if (i == 3) {

935  ddx_error = 0.0;

938

942  }
