252  float_vect_zero(estimation_rate_dd, INDI_NUM_ACT);
253  float_vect_zero(actuator_state_filt_vect, INDI_NUM_ACT);

255  //Calculate G1G2_PSEUDO_INVERSE
256  sum_g1_g2();
257  calc_g1g2_pseudo_inv();
258 
259  int8_t i;
260  // Initialize the array of pointers to the rows of g1g2
261  for (i = 0; i < INDI_OUTPUTS; i++) {
262  Bwls[i] = g1g2[i];
263  }
264 
265  // Initialize the estimator matrices
266  float_vect_copy(g1_est[0], g1[0], INDI_OUTPUTS * INDI_NUM_ACT);
267  float_vect_copy(g2_est, g2, INDI_NUM_ACT);
268  // Remember the initial matrices
269  float_vect_copy(g1_init[0], g1[0], INDI_OUTPUTS * INDI_NUM_ACT);
270  float_vect_copy(g2_init, g2, INDI_NUM_ACT);

275  num_thrusters -= act_is_servo[i];

304 void init_filters(void)
305 {
306  // tau = 1/(2*pi*Fc)
307  float tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF);
308  float tau_est = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_ESTIMATION_FILT_CUTOFF);
309  float sample_time = 1.0 / PERIODIC_FREQUENCY;
310  // Filtering of the gyroscope
311  int8_t i;
312  for (i = 0; i < 3; i++) {
313  init_butterworth_2_low_pass(&measurement_lowpass_filters[i], tau, sample_time, 0.0);
314  init_butterworth_2_low_pass(&estimation_output_lowpass_filters[i], tau_est, sample_time, 0.0);
315  }
316 
317  // Filtering of the actuators
318  for (i = 0; i < INDI_NUM_ACT; i++) {
319  init_butterworth_2_low_pass(&actuator_lowpass_filters[i], tau, sample_time, 0.0);
320  init_butterworth_2_low_pass(&estimation_input_lowpass_filters[i], tau_est, sample_time, 0.0);
321  }
322 
323  // Filtering of the accel body z
324  init_butterworth_2_low_pass(&acceleration_lowpass_filter, tau_est, sample_time, 0.0);
325 
326  // Init rate filter for feedback
327  float time_constants[3] = {1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_P), 1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_Q), 1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_R)};
328 
329  init_first_order_low_pass(&rates_filt_fo[0], time_constants[0], sample_time, stateGetBodyRates_f()->p);
330  init_first_order_low_pass(&rates_filt_fo[1], time_constants[1], sample_time, stateGetBodyRates_f()->q);
331  init_first_order_low_pass(&rates_filt_fo[2], time_constants[2], sample_time, stateGetBodyRates_f()->r);
332 }

410 void stabilization_indi_rate_run(struct FloatRates rate_sp, bool in_flight)
411 {
412  /* Propagate the filter on the gyroscopes */
413  struct FloatRates *body_rates = stateGetBodyRates_f();
414  float rate_vect[3] = {body_rates->p, body_rates->q, body_rates->r};
415  int8_t i;
416  for (i = 0; i < 3; i++) {
417  update_butterworth_2_low_pass(&measurement_lowpass_filters[i], rate_vect[i]);
418  update_butterworth_2_low_pass(&estimation_output_lowpass_filters[i], rate_vect[i]);
419 
420  //Calculate the angular acceleration via finite difference
421  angular_acceleration[i] = (measurement_lowpass_filters[i].o[0]
422  - measurement_lowpass_filters[i].o[1]) * PERIODIC_FREQUENCY;
423 
424  // Calculate derivatives for estimation
425  float estimation_rate_d_prev = estimation_rate_d[i];
426  estimation_rate_d[i] = (estimation_output_lowpass_filters[i].o[0] - estimation_output_lowpass_filters[i].o[1]) *
427  PERIODIC_FREQUENCY;
428  estimation_rate_dd[i] = (estimation_rate_d[i] - estimation_rate_d_prev) * PERIODIC_FREQUENCY;
429  }
430 
431  //The rates used for feedback are by default the measured rates.
432  //If there is a lot of noise on the gyroscope, it might be good to use the filtered value for feedback.
433  //Note that due to the delay, the PD controller may need relaxed gains.
434  struct FloatRates rates_filt;
435 #if STABILIZATION_INDI_FILTER_ROLL_RATE
436  rates_filt.p = update_first_order_low_pass(&rates_filt_fo[0], body_rates->p);
437 #else
438  rates_filt.p = body_rates->p;
439 #endif
440 #if STABILIZATION_INDI_FILTER_PITCH_RATE
441  rates_filt.q = update_first_order_low_pass(&rates_filt_fo[1], body_rates->q);
442 #else
443  rates_filt.q = body_rates->q;
444 #endif
445 #if STABILIZATION_INDI_FILTER_YAW_RATE
446  rates_filt.r = update_first_order_low_pass(&rates_filt_fo[2], body_rates->r);
447 #else
448  rates_filt.r = body_rates->r;
449 #endif
450 
451  //calculate the virtual control (reference acceleration) based on a PD controller
452  angular_accel_ref.p = (rate_sp.p - rates_filt.p) * indi_gains.rate.p;
453  angular_accel_ref.q = (rate_sp.q - rates_filt.q) * indi_gains.rate.q;
454  angular_accel_ref.r = (rate_sp.r - rates_filt.r) * indi_gains.rate.r;
455 
456  g2_times_du = 0.0;
457  for (i = 0; i < INDI_NUM_ACT; i++) {
458  g2_times_du += g2[i] * indi_du[i];
459  }
460  //G2 is scaled by INDI_G_SCALING to make it readable
461  g2_times_du = g2_times_du / INDI_G_SCALING;
462 
463  float use_increment = 0.0;
464  if(in_flight) {
465  use_increment = 1.0;
466  }
467 
468  float v_thrust = 0.0;
469  if (indi_thrust_increment_set) {
470  v_thrust = indi_thrust_increment;
471 
472  //update thrust command such that the current is correctly estimated
473  stabilization_cmd[COMMAND_THRUST] = 0;
474  for (i = 0; i < INDI_NUM_ACT; i++) {
475  stabilization_cmd[COMMAND_THRUST] += actuator_state[i] * -((int32_t) act_is_servo[i] - 1);
476  }
477  stabilization_cmd[COMMAND_THRUST] /= num_thrusters;
478 
479  } else {
480  // incremental thrust
481  for (i = 0; i < INDI_NUM_ACT; i++) {
482  v_thrust +=
483  (stabilization_cmd[COMMAND_THRUST] - use_increment*actuator_state_filt_vect[i]) * Bwls[3][i];
484  }
485  }
486 
487  // The control objective in array format
488  indi_v[0] = (angular_accel_ref.p - use_increment*angular_acceleration[0]);
489  indi_v[1] = (angular_accel_ref.q - use_increment*angular_acceleration[1]);
490  indi_v[2] = (angular_accel_ref.r - use_increment*angular_acceleration[2] + g2_times_du);
491  indi_v[3] = v_thrust;
492 
493 #if STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
494  // Calculate the increment for each actuator
495  for (i = 0; i < INDI_NUM_ACT; i++) {
496  indi_du[i] = (g1g2_pseudo_inv[i][0] * indi_v[0])
497  + (g1g2_pseudo_inv[i][1] * indi_v[1])
498  + (g1g2_pseudo_inv[i][2] * indi_v[2])
499  + (g1g2_pseudo_inv[i][3] * indi_v[3]);
500  }
501 #else
502  // Calculate the min and max increments
503  for (i = 0; i < INDI_NUM_ACT; i++) {
504  du_min[i] = -MAX_PPRZ * act_is_servo[i] - use_increment*actuator_state_filt_vect[i];
505  du_max[i] = MAX_PPRZ - use_increment*actuator_state_filt_vect[i];
506  du_pref[i] = act_pref[i] - use_increment*actuator_state_filt_vect[i];
507 
508 #ifdef GUIDANCE_INDI_MIN_THROTTLE
509  float airspeed = stateGetAirspeed_f();
510  //limit minimum thrust ap can give
511  if (!act_is_servo[i]) {
512  if ((guidance_h.mode == GUIDANCE_H_MODE_HOVER) || (guidance_h.mode == GUIDANCE_H_MODE_NAV)) {
513  if (airspeed < INDI_HROTTLE_LIMIT_AIRSPEED_FWD) {
514  du_min[i] = GUIDANCE_INDI_MIN_THROTTLE - use_increment*actuator_state_filt_vect[i];
515  } else {
516  du_min[i] = GUIDANCE_INDI_MIN_THROTTLE_FWD - use_increment*actuator_state_filt_vect[i];
517  }
518  }
519  }
520 #endif
521  }
522 
523  // WLS Control Allocator
524  num_iter =
525  wls_alloc(indi_du, indi_v, du_min, du_max, Bwls, 0, 0, Wv, indi_Wu, du_pref, 10000, 10);
526 #endif
527 
528  if (in_flight) {
529  // Add the increments to the actuators
530  float_vect_sum(indi_u, actuator_state_filt_vect, indi_du, INDI_NUM_ACT);
531  } else {
532  // Not in flight, so don't increment
533  float_vect_copy(indi_u, indi_du, INDI_NUM_ACT);
534  }
535 
536  // Bound the inputs to the actuators
537  for (i = 0; i < INDI_NUM_ACT; i++) {
538  if (act_is_servo[i]) {
539  BoundAbs(indi_u[i], MAX_PPRZ);
540  } else {
541  if (autopilot_get_motors_on()) {
542  Bound(indi_u[i], 0, MAX_PPRZ);
543  }

708 void lms_estimation(void)
709 {
710 
711  // Get the acceleration in body axes
712  struct Int32Vect3 *body_accel_i;
713  body_accel_i = stateGetAccelBody_i();
714  ACCELS_FLOAT_OF_BFP(body_accel_f, *body_accel_i);
715 
716  // Filter the acceleration in z axis
717  update_butterworth_2_low_pass(&acceleration_lowpass_filter, body_accel_f.z);
718 
719  // Calculate the derivative of the acceleration via finite difference
720  float indi_accel_d = (acceleration_lowpass_filter.o[0]
721  - acceleration_lowpass_filter.o[1]) * PERIODIC_FREQUENCY;
722 
723  // Use xml setting for adaptive mu for lms
724  // Set default value if not defined
725  #ifndef STABILIZATION_INDI_ADAPTIVE_MU
726  float adaptive_mu_lr = 0.001;
727  #else
728  float adaptive_mu_lr = STABILIZATION_INDI_ADAPTIVE_MU;
729  #endif
730 
731  // scale the inputs to avoid numerical errors
732  float_vect_smul(du_estimation, actuator_state_filt_vectd, adaptive_mu_lr, INDI_NUM_ACT);
733  float_vect_smul(ddu_estimation, actuator_state_filt_vectdd, adaptive_mu_lr / PERIODIC_FREQUENCY, INDI_NUM_ACT);
734 
735  float ddx_estimation[INDI_OUTPUTS] = {estimation_rate_dd[0], estimation_rate_dd[1], estimation_rate_dd[2], indi_accel_d};
736 
737  //Estimation of G
738  // TODO: only estimate when du_norm2 is large enough (enough input)
739  /*float du_norm2 = du_estimation[0]*du_estimation[0] + du_estimation[1]*du_estimation[1] +du_estimation[2]*du_estimation[2] + du_estimation[3]*du_estimation[3];*/
740  int8_t i;
741  for (i = 0; i < INDI_OUTPUTS; i++) {
742  // Calculate the error between prediction and measurement
743  float ddx_error = - ddx_estimation[i];
744  int8_t j;
745  for (j = 0; j < INDI_NUM_ACT; j++) {
746  ddx_error += g1_est[i][j] * du_estimation[j];
747  if (i == 2) {
748  // Changing the momentum of the rotors gives a counter torque
749  ddx_error += g2_est[j] * ddu_estimation[j];
750  }
751  }
752 
753  // when doing the yaw axis, also use G2
754  if (i == 2) {
755  for (j = 0; j < INDI_NUM_ACT; j++) {
756  calc_g2_element(ddx_error, j, mu2);
757  }
758  } else if (i == 3) {
759  // If the acceleration change is very large (rough landing), don't adapt
760  if (fabs(indi_accel_d) > 60.0) {
761  ddx_error = 0.0;
762  }
763  }
764 
765  // Calculate the row of the G1 matrix corresponding to this axis
766  for (j = 0; j < INDI_NUM_ACT; j++) {
767  calc_g1_element(ddx_error, i, j, mu1[i]);
768  }
769  }
770 
771  bound_g_mat();
772 
773  // Save the calculated matrix to G1 and G2
774  // until thrust is included, first part of the array
775  float_vect_copy(g1[0], g1_est[0], INDI_OUTPUTS * INDI_NUM_ACT);
776  float_vect_copy(g2, g2_est, INDI_NUM_ACT);
777 
778  // Calculate sum of G1 and G2 for Bwls
779  sum_g1_g2();
780 
781 #if STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
782  // Calculate the inverse of (G1+G2)
783  calc_g1g2_pseudo_inv();
784 #endif
785 }

810 {

813  float element = 0;
