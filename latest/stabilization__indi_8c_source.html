403void stabilization_indi_init(void)
404{
405 // Initialize filters
406 init_filters();
407
408 int8_t i;
409// If the deprecated STABILIZATION_INDI_ACT_DYN is used, convert it to the new FREQUENCY format
410#ifdef STABILIZATION_INDI_ACT_FREQ
411 // Initialize the array of pointers to the rows of g1g2
412 for (i = 0; i < INDI_NUM_ACT; i++) {
413 act_dyn_discrete[i] = 1-exp(-act_first_order_cutoff[i]/PERIODIC_FREQUENCY);
414 }
415#endif
416
417#if STABILIZATION_INDI_RPM_FEEDBACK
418 AbiBindMsgACT_FEEDBACK(STABILIZATION_INDI_ACT_FEEDBACK_ID, &act_feedback_ev, act_feedback_cb);
419#endif
420
421 float_vect_zero(actuator_state_filt_vectd, INDI_NUM_ACT);
422 float_vect_zero(actuator_state_filt_vectdd, INDI_NUM_ACT);
423 float_vect_zero(estimation_rate_d, INDI_NUM_ACT);
424 float_vect_zero(estimation_rate_dd, INDI_NUM_ACT);
425 float_vect_zero(actuator_state_filt_vect, INDI_NUM_ACT);
426
427 //Calculate G1G2
428 sum_g1_g2();
429
430 // Do not compute if not needed
431#if STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
432 //Calculate G1G2_PSEUDO_INVERSE
433 calc_g1g2_pseudo_inv();
434#endif
435
436 // Initialize the array of pointers to the rows of g1g2
437 for (i = 0; i < INDI_OUTPUTS; i++) {
438 Bwls[i] = g1g2[i];
439 }
440
441 // Initialize the estimator matrices
442 float_vect_copy(g1_est[0], g1[0], INDI_OUTPUTS * INDI_NUM_ACT);
443 float_vect_copy(g2_est, g2, INDI_NUM_ACT);
444 // Remember the initial matrices
445 float_vect_copy(g1_init[0], g1[0], INDI_OUTPUTS * INDI_NUM_ACT);
446 float_vect_copy(g2_init, g2, INDI_NUM_ACT);
447
448 num_thrusters = 0; // sum of Z thrusters
449 for (i = 0; i < INDI_NUM_ACT; i++) {
450#ifndef STABILIZATION_INDI_ACT_IS_THRUSTER_Z
451 // Assume non-servos and non thrust-x/y motors, not already set to true, are delivering (Z) thrust
452 act_thrust_mat[2][i] = act_thrust_mat[2][i] || (!act_thrust_mat[0][i] && !act_thrust_mat[1][i] && !act_is_servo[i]);
453#endif
454 if (act_thrust_mat[2][i]) {
455 num_thrusters++;
456 }
457 }
458
459#if PERIODIC_TELEMETRY
460 register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_EFF_MAT_STAB, send_eff_mat_g_indi);
461 register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_AHRS_REF_QUAT, send_ahrs_ref_quat);
462 register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_STAB_ATTITUDE, send_att_full_indi);
463 #if !STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
464 register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_WLS_V, send_wls_v_stab);
465 register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_WLS_U, send_wls_u_stab);
466 #endif
467#endif
468}

503static void init_filters(void)
504{
505 // tau = 1/(2*pi*Fc)
506 float tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF);
507 float tau_est = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_ESTIMATION_FILT_CUTOFF);
508 float sample_time = 1.0 / PERIODIC_FREQUENCY;
509 // Filtering of the gyroscope
510 int8_t i;
511 for (i = 0; i < 3; i++) {
512 init_butterworth_2_low_pass(&measurement_lowpass_filters[i], tau, sample_time, 0.0);
513 init_butterworth_2_low_pass(&estimation_output_lowpass_filters[i], tau_est, sample_time, 0.0);
514 }
515
516 // Filtering of the actuators
517 for (i = 0; i < INDI_NUM_ACT; i++) {
518 init_butterworth_2_low_pass(&actuator_lowpass_filters[i], tau, sample_time, 0.0);
519 init_butterworth_2_low_pass(&estimation_input_lowpass_filters[i], tau_est, sample_time, 0.0);
520 }
521
522 // Filtering of the accel body z
523 init_butterworth_2_low_pass(&acceleration_lowpass_filter, tau_est, sample_time, 0.0);
524
525#if STABILIZATION_INDI_FILTER_RATES_SECOND_ORDER
526 tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_P);
527 init_butterworth_2_low_pass(&rates_filt_so[0], tau, sample_time, 0.0);
528 tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_Q);
529 init_butterworth_2_low_pass(&rates_filt_so[1], tau, sample_time, 0.0);
530 tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_R);
531 init_butterworth_2_low_pass(&rates_filt_so[2], tau, sample_time, 0.0);
532#else
533 // Init rate filter for feedback
534 float time_constants[3] = {1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_P), 1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_Q), 1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_R)};
535
536 init_first_order_low_pass(&rates_filt_fo[0], time_constants[0], sample_time, stateGetBodyRates_f()->p);
537 init_first_order_low_pass(&rates_filt_fo[1], time_constants[1], sample_time, stateGetBodyRates_f()->q);
538 init_first_order_low_pass(&rates_filt_fo[2], time_constants[2], sample_time, stateGetBodyRates_f()->r);
539#endif
540}

583void stabilization_indi_rate_run(bool in_flight, struct StabilizationSetpoint *sp, struct ThrustSetpoint *thrust, int32_t *cmd)
584{
585
586 // Propagate actuator filters
587 get_actuator_state();
588 float actuator_state_filt_vect_prev[INDI_NUM_ACT];
589 for (int i = 0; i < INDI_NUM_ACT; i++) {
590 update_butterworth_2_low_pass(&actuator_lowpass_filters[i], actuator_state[i]);
591 update_butterworth_2_low_pass(&estimation_input_lowpass_filters[i], actuator_state[i]);
592 actuator_state_filt_vect[i] = actuator_lowpass_filters[i].o[0];
593 actuator_state_filt_vect_prev[i] = actuator_lowpass_filters[i].o[1];
594
595 // calculate derivatives for estimation
596 float actuator_state_filt_vectd_prev = actuator_state_filt_vectd[i];
597 actuator_state_filt_vectd[i] = (estimation_input_lowpass_filters[i].o[0] - estimation_input_lowpass_filters[i].o[1]) *
598 PERIODIC_FREQUENCY;
599 actuator_state_filt_vectdd[i] = (actuator_state_filt_vectd[i] - actuator_state_filt_vectd_prev) * PERIODIC_FREQUENCY;
600 }
601
602 // Use the last actuator state for this computation
603 float g2_times_u_act_filt = float_vect_dot_product(g2, actuator_state_filt_vect_prev, INDI_NUM_ACT)/INDI_G_SCALING;
604
605 // Predict angular acceleration u*B
606 float angular_acc_prediction_filt[INDI_OUTPUTS];
607 float_mat_vect_mul(angular_acc_prediction_filt, Bwls, actuator_state_filt_vect, INDI_OUTPUTS, INDI_NUM_ACT);
608 angular_acc_prediction_filt[2] -= g2_times_u_act_filt;
609
610 /* Propagate the filter on the gyroscopes */
611 struct FloatRates *body_rates = stateGetBodyRates_f();
612 float rate_vect[3] = {body_rates->p, body_rates->q, body_rates->r};
613
614 // Get the acceleration in body axes
615 struct Int32Vect3 *body_accel_i;
616 body_accel_i = stateGetAccelBody_i();
617 ACCELS_FLOAT_OF_BFP(body_accel_f, *body_accel_i);
618
619 int8_t i;
620 for (i = 0; i < 3; i++) {
621 update_butterworth_2_low_pass(&measurement_lowpass_filters[i], rate_vect[i]);
622 update_butterworth_2_low_pass(&estimation_output_lowpass_filters[i], rate_vect[i]);
623
624 // Calculate the angular acceleration via finite difference
625 angular_acceleration[i] = (measurement_lowpass_filters[i].o[0]
626 - measurement_lowpass_filters[i].o[1]) * PERIODIC_FREQUENCY;
627
628 // Calculate derivatives for estimation
629 float estimation_rate_d_prev = estimation_rate_d[i];
630 estimation_rate_d[i] = (estimation_output_lowpass_filters[i].o[0] - estimation_output_lowpass_filters[i].o[1]) *
631 PERIODIC_FREQUENCY;
632 estimation_rate_dd[i] = (estimation_rate_d[i] - estimation_rate_d_prev) * PERIODIC_FREQUENCY;
633 }
634
635 // subtract u*B from angular acceleration
636 float angular_acc_disturbance_estimate[INDI_OUTPUTS];
637 float_vect_diff(angular_acc_disturbance_estimate, angular_acceleration, angular_acc_prediction_filt, 3);
638
639 if (in_flight) {
640 // Limit the estimated disturbance in yaw for drones that are stable in sideslip
641 BoundAbs(angular_acc_disturbance_estimate[2], stablization_indi_yaw_dist_limit);
642 } else {
643 // Not in flight, so don't estimate disturbance
644 float_vect_zero(angular_acc_disturbance_estimate, INDI_OUTPUTS);
645 }
646
647 // The rates used for feedback are by default the measured rates.
648 // If there is a lot of noise on the gyroscope, it might be good to use the filtered value for feedback.
649 // Note that due to the delay, the controller may need relaxed gains.
650 struct FloatRates rates_filt = filter_rates(body_rates);
651
652 // calculate the virtual control (reference acceleration) based on a controller
653 struct FloatRates rates_sp = stab_sp_to_rates_f(sp);
654 angular_accel_ref = stabilization_indi_rate_controller(rates_filt, rates_sp);
655
656 // compute virtual thrust
657 struct FloatVect3 v_thrust = { 0.f, 0.f, 0.f };
658 if (thrust->type == THRUST_INCR_SP) {
659 v_thrust.x = th_sp_to_incr_f(thrust, 0, THRUST_AXIS_X);
660 v_thrust.y = th_sp_to_incr_f(thrust, 0, THRUST_AXIS_Y);
661 v_thrust.z = th_sp_to_incr_f(thrust, 0, THRUST_AXIS_Z);
662
663 // Compute estimated thrust
664 FLOAT_VECT3_ZERO(stab_thrust_filt);
665 for (i = 0; i < INDI_NUM_ACT; i++) {
666#if INDI_OUTPUTS == 4
667 stab_thrust_filt.z += Bwls[3][i] * actuator_lowpass_filters[i].o[0] * (int32_t) act_thrust_mat[2][i];
668#endif
669#if INDI_OUTPUTS == 5 // FIXME change order of Z and X, or better detect that automatically ?
670 stab_thrust_filt.x += Bwls[4][i] * actuator_lowpass_filters[i].o[0] * (int32_t) act_thrust_mat[0][i];
671 stab_thrust_filt.z += Bwls[3][i] * actuator_lowpass_filters[i].o[0] * (int32_t) act_thrust_mat[2][i];
672#endif
673#if INDI_OUTPUTS == 6
674 stab_thrust_filt.x += Bwls[3][i] * actuator_lowpass_filters[i].o[0] * (int32_t) act_thrust_mat[0][i];
675 stab_thrust_filt.y += Bwls[4][i] * actuator_lowpass_filters[i].o[0] * (int32_t) act_thrust_mat[1][i];
676 stab_thrust_filt.z += Bwls[5][i] * actuator_lowpass_filters[i].o[0] * (int32_t) act_thrust_mat[2][i];
677#endif
678 }
679 // Add the current estimated thrust to the increment
680 VECT3_ADD(v_thrust, stab_thrust_filt);
681 } else {
682 // build incremental thrust
683 struct FloatVect3 th_cmd;
684 th_cmd.x = (float)th_sp_to_thrust_i(thrust, 0, THRUST_AXIS_X);
685 th_cmd.y = (float)th_sp_to_thrust_i(thrust, 0, THRUST_AXIS_Y);
686 th_cmd.z = (float)th_sp_to_thrust_i(thrust, 0, THRUST_AXIS_Z);
687#if (INDI_OUTPUTS == 5) && (defined RADIO_CONTROL_THRUST_X) && (defined COMMAND_THRUST_X)
688 // TODO set X thrust from RC in the thrust input setpoint
689 cmd[COMMAND_THRUST_X] = radio_control.values[RADIO_CONTROL_THRUST_X];
690 th_cmd.x = (float)cmd[COMMAND_THRUST_X];
691#endif
692 for (i = 0; i < INDI_NUM_ACT; i++) {
693#if INDI_OUTPUTS == 4
694 v_thrust.z += th_cmd.z * Bwls[3][i] * (int32_t) act_thrust_mat[2][i];
695#endif
696#if INDI_OUTPUTS == 5 // FIXME change order of Z and X, or better detect that automatically ?
697 v_thrust.x += th_cmd.x * Bwls[4][i] * (int32_t) act_thrust_mat[0][i];
698 v_thrust.z += th_cmd.z * Bwls[3][i] * (int32_t) act_thrust_mat[2][i];
699#endif
700#if INDI_OUTPUTS == 6
701 v_thrust.x += th_cmd.x * Bwls[3][i] * (int32_t) act_thrust_mat[0][i];
702 v_thrust.y += th_cmd.y * Bwls[4][i] * (int32_t) act_thrust_mat[1][i];
703 v_thrust.z += th_cmd.z * Bwls[5][i] * (int32_t) act_thrust_mat[2][i];
704#endif
705 }
706 // store estimated thrust
707 stab_thrust_filt = v_thrust;
708 }
709
710 // This term compensates for the spinup torque in the yaw axis
711 float g2_times_u = float_vect_dot_product(g2, indi_u, INDI_NUM_ACT)/INDI_G_SCALING;
712
713 if (in_flight) {
714 // Limit the estimated disturbance in yaw for drones that are stable in sideslip
715 BoundAbs(angular_acc_disturbance_estimate[2], stablization_indi_yaw_dist_limit);
716 } else {
717 // Not in flight, so don't estimate disturbance
718 float_vect_zero(angular_acc_disturbance_estimate, INDI_OUTPUTS);
719 }
720
721 // The control objective in array format
722 indi_v[0] = (angular_accel_ref.p - angular_acc_disturbance_estimate[0]);
723 indi_v[1] = (angular_accel_ref.q - angular_acc_disturbance_estimate[1]);
724 indi_v[2] = (angular_accel_ref.r - angular_acc_disturbance_estimate[2]) + g2_times_u;
725#if INDI_OUTPUTS == 4
726 indi_v[3] = v_thrust.z;
727#endif
728#if INDI_OUTPUTS == 5 // FIXME order of Z/X is reversed
729 indi_v[3] = v_thrust.z;
730 indi_v[4] = v_thrust.x;
731#endif
732#if INDI_OUTPUTS == 6
733 indi_v[3] = v_thrust.x;
734 indi_v[4] = v_thrust.y;
735 indi_v[5] = v_thrust.z;
736#endif
737
738#if STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
739 // Calculate the increment for each actuator
740 for (i = 0; i < INDI_NUM_ACT; i++) {
741 indi_u[i] = (g1g2_pseudo_inv[i][0] * indi_v[0])
742 + (g1g2_pseudo_inv[i][1] * indi_v[1])
743 + (g1g2_pseudo_inv[i][2] * indi_v[2])
744 + (g1g2_pseudo_inv[i][3] * indi_v[3]);
745 }
746#else
747 stabilization_indi_set_wls_settings();
748
749 // WLS Control Allocator
750 for (i = 0; i < INDI_OUTPUTS; i++) {
751 wls_stab_p.v[i] = indi_v[i];
752 }
753 wls_alloc(&wls_stab_p, Bwls, 0, 0, 10);
754 for (i = 0; i < INDI_NUM_ACT; i++) {
755 indi_u[i] = wls_stab_p.u[i];
756 }
757#endif
758
759 // Use online effectiveness estimation only when flying
760 if (in_flight && indi_use_adaptive) {
761 lms_estimation();
762 }
763
764 // update actuators and thrust commands
765 cmd[COMMAND_THRUST] = 0;
766 for (i = 0; i < INDI_NUM_ACT; i++) {
767 // bound actuator command
768 if (act_is_servo[i]) {
769 BoundAbs(indi_u[i], MAX_PPRZ);
770 } else {
771 if (autopilot_get_motors_on()) {
772 Bound(indi_u[i], 0, MAX_PPRZ);
773 } else {
774 indi_u[i] = -MAX_PPRZ;
775 }
776 }
777
778 // commit actuator command
779 actuators_pprz[i] = (int16_t) indi_u[i];
780#ifdef STABILIZATION_INDI_COMMANDS
781 // copy to actuators to specified commands
782 cmd[act_to_commands[i]] = actuators_pprz[i];
783#endif
784
785 // update thrust command such that the current is correctly estimated
786 cmd[COMMAND_THRUST] += actuator_state[i] * (int32_t) act_thrust_mat[2][i];
787 }
788 cmd[COMMAND_THRUST] /= num_thrusters;
789}

821struct FloatRates WEAK stabilization_indi_attitude_controller(struct FloatQuat att, struct FloatQuat att_sp, struct FloatRates rates_ff)
822{
823 /* attitude error */
824 struct FloatQuat att_err;
825 float_quat_inv_comp_norm_shortest(&att_err, &att, &att_sp);
826
827 struct FloatVect3 att_fb;
828#if TILT_TWIST_CTRL
829 struct FloatQuat tilt;
830 struct FloatQuat twist;
831 float_quat_tilt_twist(&tilt, &twist, &att_err);
832 att_fb.x = tilt.qx;
833 att_fb.y = tilt.qy;
834 att_fb.z = twist.qz;
835#else
836 att_fb.x = att_err.qx;
837 att_fb.y = att_err.qy;
838 att_fb.z = att_err.qz;
839#endif
840
841 struct FloatRates rates_ref;
842 rates_ref.p = indi_gains.att.p * att_fb.x / indi_gains.rate.p;
843 rates_ref.q = indi_gains.att.q * att_fb.y / indi_gains.rate.q;
844 rates_ref.r = indi_gains.att.r * att_fb.z / indi_gains.rate.r;
845 // add feed-forward term
846 RATES_ADD(rates_ref, rates_ff);
847 return rates_ref;
848}

984void lms_estimation(void)
985{
986
987 // Get the acceleration in body axes
988 struct Int32Vect3 *body_accel_i;
989 body_accel_i = stateGetAccelBody_i();
990 ACCELS_FLOAT_OF_BFP(body_accel_f, *body_accel_i);
991
992 // Filter the acceleration in z axis
993 update_butterworth_2_low_pass(&acceleration_lowpass_filter, body_accel_f.z);
994
995 // Calculate the derivative of the acceleration via finite difference
996 float indi_accel_d = (acceleration_lowpass_filter.o[0]
997 - acceleration_lowpass_filter.o[1]) * PERIODIC_FREQUENCY;
998
999 // scale the inputs to avoid numerical errors
1000 float_vect_smul(du_estimation, actuator_state_filt_vectd, STABILIZATION_INDI_ADAPTIVE_MU, INDI_NUM_ACT);
1001 float_vect_smul(ddu_estimation, actuator_state_filt_vectdd, STABILIZATION_INDI_ADAPTIVE_MU / PERIODIC_FREQUENCY, INDI_NUM_ACT);
1002
1003 float ddx_estimation[INDI_OUTPUTS] = {estimation_rate_dd[0], estimation_rate_dd[1], estimation_rate_dd[2], indi_accel_d};
1004
1005 //Estimation of G
1006 // TODO: only estimate when du_norm2 is large enough (enough input)
1007 /*float du_norm2 = du_estimation[0]*du_estimation[0] + du_estimation[1]*du_estimation[1] +du_estimation[2]*du_estimation[2] + du_estimation[3]*du_estimation[3];*/
1008 int8_t i;
1009 for (i = 0; i < INDI_OUTPUTS; i++) {
1010 // Calculate the error between prediction and measurement
1011 float ddx_error = - ddx_estimation[i];
1012 int8_t j;
1013 for (j = 0; j < INDI_NUM_ACT; j++) {
1014 ddx_error += g1_est[i][j] * du_estimation[j];
1015 if (i == 2) {
1016 // Changing the momentum of the rotors gives a counter torque
1017 ddx_error += g2_est[j] * ddu_estimation[j];
1018 }
1019 }
1020
1021 // when doing the yaw axis, also use G2
1022 if (i == 2) {
1023 for (j = 0; j < INDI_NUM_ACT; j++) {
1024 calc_g2_element(ddx_error, j, mu2);
1025 }
1026 } else if (i == 3) {
1027 // If the acceleration change is very large (rough landing), don't adapt
1028 if (fabs(indi_accel_d) > 60.0) {
1029 ddx_error = 0.0;
1030 }
1031 }
1032
1033 // Calculate the row of the G1 matrix corresponding to this axis
1034 for (j = 0; j < INDI_NUM_ACT; j++) {
1035 calc_g1_element(ddx_error, i, j, mu1[i]);
1036 }
1037 }
1038
1039 bound_g_mat();
1040
1041 // Save the calculated matrix to G1 and G2
1042 // until thrust is included, first part of the array
1043 float_vect_copy(g1[0], g1_est[0], INDI_OUTPUTS * INDI_NUM_ACT);
1044 float_vect_copy(g2, g2_est, INDI_NUM_ACT);
1045
1046 // Calculate sum of G1 and G2 for Bwls
1047 sum_g1_g2();
1048
1049#if STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
1050 // Calculate the inverse of (G1+G2)
1051 calc_g1g2_pseudo_inv();
1052#endif
1053}

1059void sum_g1_g2(void)
1060{
1061 int8_t i;
1062 int8_t j;
1063 for (i = 0; i < INDI_OUTPUTS; i++) {
1064 for (j = 0; j < INDI_NUM_ACT; j++) {
1065 if (i != 2) {
1066 g1g2[i][j] = g1[i][j] / INDI_G_SCALING;
1067 } else {
1068 g1g2[i][j] = (g1[i][j] + g2[j]) / INDI_G_SCALING;
1069 }
1070 }
1071 }
1072}

1074#if STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
1079void calc_g1g2_pseudo_inv(void)
1080{
