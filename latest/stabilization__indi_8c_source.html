347 void stabilization_indi_init(void)
348 {
349  // Initialize filters
350  init_filters();
351 
352 #ifdef STABILIZATION_INDI_ACT_FREQ
353  int8_t i;
354  // Initialize the array of pointers to the rows of g1g2
355  for (i = 0; i < INDI_NUM_ACT; i++) {
356  act_dyn_discrete[i] = 1-exp(-act_first_order_cutoff[i]/PERIODIC_FREQUENCY);
357  }
358 #endif
359 
360 #if STABILIZATION_INDI_RPM_FEEDBACK
361  AbiBindMsgACT_FEEDBACK(STABILIZATION_INDI_ACT_FEEDBACK_ID, &act_feedback_ev, act_feedback_cb);
362 #endif
363  AbiBindMsgTHRUST(THRUST_INCREMENT_ID, &thrust_ev, thrust_cb);
364 
365  float_vect_zero(actuator_state_filt_vectd, INDI_NUM_ACT);
366  float_vect_zero(actuator_state_filt_vectdd, INDI_NUM_ACT);
367  float_vect_zero(estimation_rate_d, INDI_NUM_ACT);
368  float_vect_zero(estimation_rate_dd, INDI_NUM_ACT);
369  float_vect_zero(actuator_state_filt_vect, INDI_NUM_ACT);
370 
371  //Calculate G1G2
372  sum_g1_g2();
373 
374  // Do not compute if not needed
375 #if STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
376  //Calculate G1G2_PSEUDO_INVERSE
377  calc_g1g2_pseudo_inv();
378 #endif
379 
380  // Initialize the array of pointers to the rows of g1g2
381  for (i = 0; i < INDI_OUTPUTS; i++) {
382  Bwls[i] = g1g2[i];
383  }
384 
385  // Initialize the estimator matrices
386  float_vect_copy(g1_est[0], g1[0], INDI_OUTPUTS * INDI_NUM_ACT);
387  float_vect_copy(g2_est, g2, INDI_NUM_ACT);
388  // Remember the initial matrices
389  float_vect_copy(g1_init[0], g1[0], INDI_OUTPUTS * INDI_NUM_ACT);
390  float_vect_copy(g2_init, g2, INDI_NUM_ACT);
391 
392  // Assume all non-servos are delivering thrust
393  num_thrusters = INDI_NUM_ACT;
394  num_thrusters_x = 0;
395  for (i = 0; i < INDI_NUM_ACT; i++) {
396  num_thrusters -= act_is_servo[i];
397  num_thrusters -= act_is_thruster_x[i];
398 
399  num_thrusters_x += act_is_thruster_x[i];
400 
401  act_is_thruster_z[i] = !act_is_servo[i] && !act_is_thruster_x[i];
402  }
403 
404 #if PERIODIC_TELEMETRY
405  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_EFF_MAT_G, send_eff_mat_g_indi);
406  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_AHRS_REF_QUAT, send_ahrs_ref_quat);
407  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_STAB_ATTITUDE, send_att_full_indi);
408 #endif
409 }

431 void init_filters(void)
432 {
433  // tau = 1/(2*pi*Fc)
434  float tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF);
435  float tau_est = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_ESTIMATION_FILT_CUTOFF);
436  float sample_time = 1.0 / PERIODIC_FREQUENCY;
437  // Filtering of the gyroscope
438  int8_t i;
439  for (i = 0; i < 3; i++) {
440  init_butterworth_2_low_pass(&measurement_lowpass_filters[i], tau, sample_time, 0.0);
441  init_butterworth_2_low_pass(&estimation_output_lowpass_filters[i], tau_est, sample_time, 0.0);
442  }
443 
444  // Filtering of the actuators
445  for (i = 0; i < INDI_NUM_ACT; i++) {
446  init_butterworth_2_low_pass(&actuator_lowpass_filters[i], tau, sample_time, 0.0);
447  init_butterworth_2_low_pass(&estimation_input_lowpass_filters[i], tau_est, sample_time, 0.0);
448  }
449 
450  // Filtering of the accel body z
451  init_butterworth_2_low_pass(&acceleration_lowpass_filter, tau_est, sample_time, 0.0);
452 
453 #if STABILIZATION_INDI_FILTER_RATES_SECOND_ORDER
454  tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_P);
455  init_butterworth_2_low_pass(&rates_filt_so[0], tau, sample_time, 0.0);
456  tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_Q);
457  init_butterworth_2_low_pass(&rates_filt_so[1], tau, sample_time, 0.0);
458  tau = 1.0 / (2.0 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_R);
459  init_butterworth_2_low_pass(&rates_filt_so[2], tau, sample_time, 0.0);
460 #else
461  // Init rate filter for feedback
462  float time_constants[3] = {1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_P), 1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_Q), 1.0 / (2 * M_PI * STABILIZATION_INDI_FILT_CUTOFF_R)};
463 
464  init_first_order_low_pass(&rates_filt_fo[0], time_constants[0], sample_time, stateGetBodyRates_f()->p);
465  init_first_order_low_pass(&rates_filt_fo[1], time_constants[1], sample_time, stateGetBodyRates_f()->q);
466  init_first_order_low_pass(&rates_filt_fo[2], time_constants[2], sample_time, stateGetBodyRates_f()->r);
467 #endif
468 }

513 void stabilization_indi_set_earth_cmd_i(struct Int32Vect2 *cmd, int32_t heading)
514 {
515  // stab_att_sp_euler.psi still used in ref..
516  stab_att_sp_euler.psi = heading;
517 
518  // compute sp_euler phi/theta for debugging/telemetry
519  /* Rotate horizontal commands to body frame by psi */
520  int32_t psi = stateGetNedToBodyEulers_i()->psi;
521  int32_t s_psi, c_psi;
522  PPRZ_ITRIG_SIN(s_psi, psi);
523  PPRZ_ITRIG_COS(c_psi, psi);
524  stab_att_sp_euler.phi = (-s_psi * cmd->x + c_psi * cmd->y) >> INT32_TRIG_FRAC;
525  stab_att_sp_euler.theta = -(c_psi * cmd->x + s_psi * cmd->y) >> INT32_TRIG_FRAC;
526 
527  quat_from_earth_cmd_i(&stab_att_sp_quat, cmd, heading);
528  FLOAT_RATES_ZERO(stab_att_ff_rates);
529 }

550 void stabilization_indi_rate_run(struct FloatRates rate_sp, bool in_flight)
551 {
552  /* Propagate the filter on the gyroscopes */
553  struct FloatRates *body_rates = stateGetBodyRates_f();
554  float rate_vect[3] = {body_rates->p, body_rates->q, body_rates->r};
555  int8_t i;
556  for (i = 0; i < 3; i++) {
557  update_butterworth_2_low_pass(&measurement_lowpass_filters[i], rate_vect[i]);
558  update_butterworth_2_low_pass(&estimation_output_lowpass_filters[i], rate_vect[i]);
559 
560  //Calculate the angular acceleration via finite difference
561  angular_acceleration[i] = (measurement_lowpass_filters[i].o[0]
562  - measurement_lowpass_filters[i].o[1]) * PERIODIC_FREQUENCY;
563 
564  // Calculate derivatives for estimation
565  float estimation_rate_d_prev = estimation_rate_d[i];
566  estimation_rate_d[i] = (estimation_output_lowpass_filters[i].o[0] - estimation_output_lowpass_filters[i].o[1]) *
567  PERIODIC_FREQUENCY;
568  estimation_rate_dd[i] = (estimation_rate_d[i] - estimation_rate_d_prev) * PERIODIC_FREQUENCY;
569  }
570 
571  //The rates used for feedback are by default the measured rates.
572  //If there is a lot of noise on the gyroscope, it might be good to use the filtered value for feedback.
573  //Note that due to the delay, the PD controller may need relaxed gains.
574  struct FloatRates rates_filt;
575 #if STABILIZATION_INDI_FILTER_ROLL_RATE
576 #if STABILIZATION_INDI_FILTER_RATES_SECOND_ORDER
577  rates_filt.p = update_butterworth_2_low_pass(&rates_filt_so[0], body_rates->p);
578 #else
579  rates_filt.p = update_first_order_low_pass(&rates_filt_fo[0], body_rates->p);
580 #endif
581 #else
582  rates_filt.p = body_rates->p;
583 #endif
584 #if STABILIZATION_INDI_FILTER_PITCH_RATE
585 #if STABILIZATION_INDI_FILTER_RATES_SECOND_ORDER
586  rates_filt.q = update_butterworth_2_low_pass(&rates_filt_so[1], body_rates->q);
587 #else
588  rates_filt.q = update_first_order_low_pass(&rates_filt_fo[1], body_rates->q);
589 #endif
590 #else
591  rates_filt.q = body_rates->q;
592 #endif
593 #if STABILIZATION_INDI_FILTER_YAW_RATE
594 #if STABILIZATION_INDI_FILTER_RATES_SECOND_ORDER
595  rates_filt.r = update_butterworth_2_low_pass(&rates_filt_so[2], body_rates->r);
596 #else
597  rates_filt.r = update_first_order_low_pass(&rates_filt_fo[2], body_rates->r);
598 #endif
599 #else
600  rates_filt.r = body_rates->r;
601 #endif
602 
603  //calculate the virtual control (reference acceleration) based on a PD controller
604  angular_accel_ref.p = (rate_sp.p - rates_filt.p) * indi_gains.rate.p;
605  angular_accel_ref.q = (rate_sp.q - rates_filt.q) * indi_gains.rate.q;
606  angular_accel_ref.r = (rate_sp.r - rates_filt.r) * indi_gains.rate.r;
607 
608  g2_times_du = 0.0;
609  for (i = 0; i < INDI_NUM_ACT; i++) {
610  g2_times_du += g2[i] * indi_du[i];
611  }
612  //G2 is scaled by INDI_G_SCALING to make it readable
613  g2_times_du = g2_times_du / INDI_G_SCALING;
614 
615  float use_increment = 0.0;
616  if (in_flight) {
617  use_increment = 1.0;
618  }
619 
620  struct FloatVect3 v_thrust;
621  v_thrust.x = 0.0;
622  v_thrust.y = 0.0;
623  v_thrust.z = 0.0;
624  if (indi_thrust_increment_set) {
625  v_thrust = indi_thrust_increment;
626 
627  //update thrust command such that the current is correctly estimated
628  stabilization_cmd[COMMAND_THRUST] = 0;
629  for (i = 0; i < INDI_NUM_ACT; i++) {
630  stabilization_cmd[COMMAND_THRUST] += actuator_state[i] * (int32_t) act_is_thruster_z[i];
631  }
632  stabilization_cmd[COMMAND_THRUST] /= num_thrusters;
633 
634 #if INDI_OUTPUTS == 5
635  stabilization_cmd[COMMAND_THRUST_X] = 0;
636  for (i = 0; i < INDI_NUM_ACT; i++) {
637  stabilization_cmd[COMMAND_THRUST_X] += actuator_state[i] * (int32_t) act_is_thruster_x[i];
638  }
639  stabilization_cmd[COMMAND_THRUST_X] /= num_thrusters_x;
640 #endif
641 
642  } else {
643  // incremental thrust
644  for (i = 0; i < INDI_NUM_ACT; i++) {
645  v_thrust.z +=
646  (stabilization_cmd[COMMAND_THRUST] - use_increment * actuator_state_filt_vect[i]) * Bwls[3][i];
647 #if INDI_OUTPUTS == 5
648  stabilization_cmd[COMMAND_THRUST_X] = radio_control.values[RADIO_CONTROL_THRUST_X];
649  v_thrust.x +=
650  (stabilization_cmd[COMMAND_THRUST_X] - use_increment * actuator_state_filt_vect[i]) * Bwls[4][i];
651 #endif
652  }
653  }
654 
655  // The control objective in array format
656  indi_v[0] = (angular_accel_ref.p - use_increment * angular_acceleration[0]);
657  indi_v[1] = (angular_accel_ref.q - use_increment * angular_acceleration[1]);
658  indi_v[2] = (angular_accel_ref.r - use_increment * angular_acceleration[2] + g2_times_du);
659  indi_v[3] = v_thrust.z;
660 #if INDI_OUTPUTS == 5
661  indi_v[4] = v_thrust.x;
662 #endif
663 
664 
665 #if STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
666  // Calculate the increment for each actuator
667  for (i = 0; i < INDI_NUM_ACT; i++) {
668  indi_du[i] = (g1g2_pseudo_inv[i][0] * indi_v[0])
669  + (g1g2_pseudo_inv[i][1] * indi_v[1])
670  + (g1g2_pseudo_inv[i][2] * indi_v[2])
671  + (g1g2_pseudo_inv[i][3] * indi_v[3]);
672  }
673 #else
674  stabilization_indi_set_wls_settings(use_increment);
675 
676 
677  // WLS Control Allocator
678  num_iter =
679  wls_alloc(indi_du, indi_v, du_min_stab_indi, du_max_stab_indi, Bwls, 0, 0, Wv, indi_Wu, du_pref_stab_indi, 10000, 10,
680  INDI_NUM_ACT, INDI_OUTPUTS);
681 #endif
682 
683  if (in_flight) {
684  // Add the increments to the actuators
685  float_vect_sum(indi_u, actuator_state_filt_vect, indi_du, INDI_NUM_ACT);
686  } else {
687  // Not in flight, so don't increment
688  float_vect_copy(indi_u, indi_du, INDI_NUM_ACT);
689  }
690 
691  // Bound the inputs to the actuators
692  for (i = 0; i < INDI_NUM_ACT; i++) {
693  if (act_is_servo[i]) {
694  BoundAbs(indi_u[i], MAX_PPRZ);
695  } else {
696  if (autopilot_get_motors_on()) {
697  Bound(indi_u[i], 0, MAX_PPRZ);
698  } else {
699  indi_u[i] = -MAX_PPRZ;
700  }
701  }
702  }
703 
704  // Propagate actuator filters
705  get_actuator_state();
706  for (i = 0; i < INDI_NUM_ACT; i++) {
707  update_butterworth_2_low_pass(&actuator_lowpass_filters[i], actuator_state[i]);
708  update_butterworth_2_low_pass(&estimation_input_lowpass_filters[i], actuator_state[i]);
709  actuator_state_filt_vect[i] = actuator_lowpass_filters[i].o[0];
710 
711  // calculate derivatives for estimation
712  float actuator_state_filt_vectd_prev = actuator_state_filt_vectd[i];
713  actuator_state_filt_vectd[i] = (estimation_input_lowpass_filters[i].o[0] - estimation_input_lowpass_filters[i].o[1]) *
714  PERIODIC_FREQUENCY;
715  actuator_state_filt_vectdd[i] = (actuator_state_filt_vectd[i] - actuator_state_filt_vectd_prev) * PERIODIC_FREQUENCY;
716  }
717 
718  // Use online effectiveness estimation only when flying
719  if (in_flight && indi_use_adaptive) {
720  lms_estimation();
721  }
722 
723  /*Commit the actuator command*/
724  for (i = 0; i < INDI_NUM_ACT; i++) {
725  actuators_pprz[i] = (int16_t) indi_u[i];
726  }
727 
728  // Set the stab_cmd to 42 to indicate that it is not used
729  stabilization_cmd[COMMAND_ROLL] = 42;
730  stabilization_cmd[COMMAND_PITCH] = 42;
731  stabilization_cmd[COMMAND_YAW] = 42;
732 }

899 void lms_estimation(void)
900 {
901 
902  // Get the acceleration in body axes
903  struct Int32Vect3 *body_accel_i;
904  body_accel_i = stateGetAccelBody_i();
905  ACCELS_FLOAT_OF_BFP(body_accel_f, *body_accel_i);
906 
907  // Filter the acceleration in z axis
908  update_butterworth_2_low_pass(&acceleration_lowpass_filter, body_accel_f.z);
909 
910  // Calculate the derivative of the acceleration via finite difference
911  float indi_accel_d = (acceleration_lowpass_filter.o[0]
912  - acceleration_lowpass_filter.o[1]) * PERIODIC_FREQUENCY;
913 
914  // Use xml setting for adaptive mu for lms
915  // Set default value if not defined
916 #ifndef STABILIZATION_INDI_ADAPTIVE_MU
917  float adaptive_mu_lr = 0.001;
918 #else
919  float adaptive_mu_lr = STABILIZATION_INDI_ADAPTIVE_MU;
920 #endif
921 
922  // scale the inputs to avoid numerical errors
923  float_vect_smul(du_estimation, actuator_state_filt_vectd, adaptive_mu_lr, INDI_NUM_ACT);
924  float_vect_smul(ddu_estimation, actuator_state_filt_vectdd, adaptive_mu_lr / PERIODIC_FREQUENCY, INDI_NUM_ACT);
925 
926  float ddx_estimation[INDI_OUTPUTS] = {estimation_rate_dd[0], estimation_rate_dd[1], estimation_rate_dd[2], indi_accel_d};
927 
928  //Estimation of G
929  // TODO: only estimate when du_norm2 is large enough (enough input)
930  /*float du_norm2 = du_estimation[0]*du_estimation[0] + du_estimation[1]*du_estimation[1] +du_estimation[2]*du_estimation[2] + du_estimation[3]*du_estimation[3];*/
931  int8_t i;
932  for (i = 0; i < INDI_OUTPUTS; i++) {
933  // Calculate the error between prediction and measurement
934  float ddx_error = - ddx_estimation[i];
935  int8_t j;
936  for (j = 0; j < INDI_NUM_ACT; j++) {
937  ddx_error += g1_est[i][j] * du_estimation[j];
938  if (i == 2) {
939  // Changing the momentum of the rotors gives a counter torque
940  ddx_error += g2_est[j] * ddu_estimation[j];
941  }
942  }
943 
944  // when doing the yaw axis, also use G2
945  if (i == 2) {
946  for (j = 0; j < INDI_NUM_ACT; j++) {
947  calc_g2_element(ddx_error, j, mu2);
948  }
949  } else if (i == 3) {
950  // If the acceleration change is very large (rough landing), don't adapt
951  if (fabs(indi_accel_d) > 60.0) {
952  ddx_error = 0.0;
953  }
954  }
955 
956  // Calculate the row of the G1 matrix corresponding to this axis
957  for (j = 0; j < INDI_NUM_ACT; j++) {
958  calc_g1_element(ddx_error, i, j, mu1[i]);
959  }
960  }
961 
962  bound_g_mat();
963 
964  // Save the calculated matrix to G1 and G2
965  // until thrust is included, first part of the array
966  float_vect_copy(g1[0], g1_est[0], INDI_OUTPUTS * INDI_NUM_ACT);
967  float_vect_copy(g2, g2_est, INDI_NUM_ACT);
968 
969  // Calculate sum of G1 and G2 for Bwls
970  sum_g1_g2();
971 
972 #if STABILIZATION_INDI_ALLOCATION_PSEUDO_INVERSE
973  // Calculate the inverse of (G1+G2)
974  calc_g1g2_pseudo_inv();
975 #endif
976 }
