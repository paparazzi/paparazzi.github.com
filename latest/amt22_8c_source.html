91static bool amt22_checkbit(uint8_t p0, uint8_t p1) {
92 uint16_t data = ((p0 << 8 | p1) & 0x3fff) >> 2;
93 uint8_t odd = 0;
94 uint8_t even = 0;
95 while (data)

97 even ^= data & 1;

103 odd = !odd;
