138void multi_ranger_init(void)
139{
140 multi_ranger.status = MULTI_RANGER_UNINIT;
141
142 // init I/O expander
143 pca95xx_init(&multi_ranger.pca, &(MULTI_RANGER_I2C_DEV), PCA95XX_DEFAULT_ADDRESS);
144#if MULTI_RANGER_EARLY_INIT
145 pca95xx_configure(&multi_ranger.pca, ~(MULTI_RANGER_PIN_ALL), true); // configure output
146 pca95xx_set_output(&multi_ranger.pca, ~(MULTI_RANGER_PIN_ALL), true); // select none
147#endif
148
149 // init vl53l1x array
150 for (uint8_t i = 0; i < MULTI_RANGER_NB; i++) {
151 multi_ranger.ranger[i].dev.i2c_p = &(MULTI_RANGER_I2C_DEV);
152 multi_ranger.ranger[i].dev.i2c_trans.slave_addr = VL53L1_DEFAULT_ADDRESS;
153 multi_ranger.ranger[i].dev.read_status = VL53L1_READ_IDLE;
154 multi_ranger.ranger[i].distance = 0.f;
155 multi_ranger.ranger[i].azimuth = multi_ranger_array_orientation[i][0];
156 multi_ranger.ranger[i].bearing = multi_ranger_array_orientation[i][1];
157 }
158}

164static bool multi_ranger_read(struct SingleRanger *ranger UNUSED)
165{
166#ifndef SITL
167 uint16_t range_mm;
168 bool new_data = false;
169 bool ret = VL53L1X_NonBlocking_ReadDataEvent(&ranger->dev, &range_mm, &new_data);
170 if (new_data) {
171 ranger->distance = range_mm / 1000.f;
172 AbiSendMsgOBSTACLE_DETECTION(OBS_DETECTION_MULTI_RANGER_DECK_ID, ranger->distance, ranger->azimuth, ranger->bearing);
173 }
174 return ret;
175#endif
176}

181void multi_ranger_periodic(void)
182{
183 switch (multi_ranger.status) {
184 case MULTI_RANGER_UNINIT:
185 pca95xx_configure(&multi_ranger.pca, ~(MULTI_RANGER_PIN_ALL), false); // configure output
186 multi_ranger.status++;
187 break;
188 case MULTI_RANGER_CONF_IO:
189 pca95xx_set_output(&multi_ranger.pca, MULTI_RANGER_PIN_FRONT, false); // select front
190 multi_ranger.status++;
191 break;
192 case MULTI_RANGER_CONF_FRONT:
193 multi_ranger_boot_device(&multi_ranger.ranger[MULTI_RANGER_FRONT].dev);
194 pca95xx_set_output(&multi_ranger.pca, MULTI_RANGER_PIN_FRONT | MULTI_RANGER_PIN_BACK, false); // select back
195 multi_ranger.status++;
196 break;
197 case MULTI_RANGER_CONF_BACK:
198 multi_ranger_boot_device(&multi_ranger.ranger[MULTI_RANGER_BACK].dev);
199 pca95xx_set_output(&multi_ranger.pca, MULTI_RANGER_PIN_FRONT | MULTI_RANGER_PIN_BACK | MULTI_RANGER_PIN_RIGHT, false); // select right
200 multi_ranger.status++;
201 break;
202 case MULTI_RANGER_CONF_RIGHT:
203 multi_ranger_boot_device(&multi_ranger.ranger[MULTI_RANGER_RIGHT].dev);
204 pca95xx_set_output(&multi_ranger.pca, MULTI_RANGER_PIN_FRONT | MULTI_RANGER_PIN_BACK | MULTI_RANGER_PIN_RIGHT | MULTI_RANGER_PIN_LEFT, false); // select left
205 multi_ranger.status++;
206 break;
207 case MULTI_RANGER_CONF_LEFT:
208 multi_ranger_boot_device(&multi_ranger.ranger[MULTI_RANGER_LEFT].dev);
209 pca95xx_set_output(&multi_ranger.pca, MULTI_RANGER_PIN_ALL, false); // select up
210 multi_ranger.status++;
211 break;
212 case MULTI_RANGER_CONF_UP:
213 multi_ranger_boot_device(&multi_ranger.ranger[MULTI_RANGER_UP].dev);
214 multi_ranger.status++;
215 break;
216 case MULTI_RANGER_READ_FRONT:
217 if (VL53L1X_NonBlocking_IsIdle(&multi_ranger.ranger[MULTI_RANGER_FRONT].dev)) {
218 VL53L1X_NonBlocking_RequestData(&multi_ranger.ranger[MULTI_RANGER_FRONT].dev);
219 multi_ranger.status++;
220 }
221 break;
222 case MULTI_RANGER_READ_BACK:
223 if (VL53L1X_NonBlocking_IsIdle(&multi_ranger.ranger[MULTI_RANGER_BACK].dev)) {
224 VL53L1X_NonBlocking_RequestData(&multi_ranger.ranger[MULTI_RANGER_BACK].dev);
225 multi_ranger.status++;
226 }
227 break;
228 case MULTI_RANGER_READ_RIGHT:
229 if (VL53L1X_NonBlocking_IsIdle(&multi_ranger.ranger[MULTI_RANGER_RIGHT].dev)) {
230 VL53L1X_NonBlocking_RequestData(&multi_ranger.ranger[MULTI_RANGER_RIGHT].dev);
231 multi_ranger.status++;
232 }
233 break;
234 case MULTI_RANGER_READ_LEFT:
235 if (VL53L1X_NonBlocking_IsIdle(&multi_ranger.ranger[MULTI_RANGER_LEFT].dev)) {
236 VL53L1X_NonBlocking_RequestData(&multi_ranger.ranger[MULTI_RANGER_LEFT].dev);
237 multi_ranger.status++;
238 }
239 break;
240 case MULTI_RANGER_READ_UP:
241 if (VL53L1X_NonBlocking_IsIdle(&multi_ranger.ranger[MULTI_RANGER_UP].dev)) {
242 VL53L1X_NonBlocking_RequestData(&multi_ranger.ranger[MULTI_RANGER_UP].dev);
243 multi_ranger.status = MULTI_RANGER_READ_FRONT;
244 }
245 break;
246 default:
247 break;
248 }
249}

251void multi_ranger_event(void)
252{
253 // call non blocking read/event functions
254 multi_ranger_read(&multi_ranger.ranger[MULTI_RANGER_FRONT]);
255 multi_ranger_read(&multi_ranger.ranger[MULTI_RANGER_BACK]);
256 multi_ranger_read(&multi_ranger.ranger[MULTI_RANGER_RIGHT]);
257 multi_ranger_read(&multi_ranger.ranger[MULTI_RANGER_LEFT]);
258 multi_ranger_read(&multi_ranger.ranger[MULTI_RANGER_UP]);
259}

261void multi_ranger_report(void)
262{
263 float dist_array[MULTI_RANGER_NB];
264 for (int i = 0; i < MULTI_RANGER_NB; i++) {
265 dist_array[i] = multi_ranger.ranger[i].distance;
