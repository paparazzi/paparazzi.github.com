65static bool nav_takeoff_mission(uint8_t nb UNUSED, float *params UNUSED, enum MissionRunFlag flag)
66{
67 if (flag == MissionInit && nb == 1) {
68 float height = params[0];
69 takeoff.status = NAV_TAKEOFF_INIT;
70 return nav_takeoff_from_here(height);
71 }
72 else if (flag == MissionRun) {
73 return nav_takeoff_run();
74 }
75 return false; // not a valid case
76}
77
78static bool nav_land_mission(uint8_t nb, float *params, enum MissionRunFlag flag)
79{
80 if (flag == MissionInit && nb == 1) {
81 float td_alt = params[0];
82 landing.status = NAV_LANDING_INIT;
83 return nav_land_here(td_alt, 0.f);
84 } else if (flag == MissionInit && nb == 2) {
85 uint8_t td_id = (uint8_t)(params[0]);
86 uint8_t af_id = (uint8_t)(params[1]);
87 return nav_land_at_wp(td_id, af_id, 0.f);
88 } else if (flag == MissionInit && nb == 5) {
89 float td_alt = params[0];
90 float lat = params[1];
91 float lon = params[2];
92 float dir = params[3];
93 float dist = params[4];
94 return nav_land_at_loc(td_alt, lat, lon, dir, dist, 0.f);
95 } else if (flag == MissionRun) {
96 return nav_land_run();
97 }
98 return false; // not a valid case
99}

136static bool nav_takeoff_run(void) {
137 static int start_motor_counter = 0;
138
139 switch (takeoff.status) {
140 case NAV_TAKEOFF_INIT:
141 takeoff.status = NAV_TAKEOFF_START_MOTOR;
142 start_motor_counter = 0;
143 break;
144 case NAV_TAKEOFF_START_MOTOR:
145 NavResurrect();
146 start_motor_counter++;
147 if (start_motor_counter == NAVIGATION_FREQUENCY) {
148 start_motor_counter = 0;
149 if (autopilot_get_motors_on()) {
150 takeoff.status = NAV_TAKEOFF_CLIMB; // next step after 1s
151 } else {
152 takeoff.status = NAV_TAKEOFF_DONE; // failed to start motors, end now
153 }
154 }
155 break;
156 case NAV_TAKEOFF_CLIMB:
157 // call vertical climb from nav/guidance
158 autopilot_set_in_flight(true);
159 NavGotoWaypoint(takeoff.climb_id);
160 NavVerticalClimbMode(NAV_TAKEOFF_CLIMB_SPEED);
161 if (stateGetPositionEnu_f()->z - takeoff.start_pos.z > takeoff.climb_pos.z) {
162 // end when takeoff height is reached
163 takeoff.status = NAV_TAKEOFF_DONE;
164 }
165 break;
166 case NAV_TAKEOFF_DONE:
167 default:
168 takeoff.status = NAV_TAKEOFF_INIT;
169 return false;
170 }
171 takeoff.timeout = false;
172 return true;
173}

175bool nav_takeoff_from_wp(uint8_t wp_id, float height)
176{
177 if (takeoff.status == NAV_TAKEOFF_INIT) {
178 takeoff.climb_id = wp_id;
179 waypoint_set_here_2d(wp_id);
180 takeoff.climb_pos = *waypoint_get_enu_f(wp_id);
181 takeoff.start_pos = *stateGetPositionEnu_f();
182 if (height < 0.f) {
183 takeoff.climb_pos.z = takeoff.start_pos.z + NAV_TAKEOFF_HEIGHT; // default value
184 } else {
185 takeoff.climb_pos.z = takeoff.start_pos.z + height;
186 }
187 }
188
189 return nav_takeoff_run();
190}

197bool nav_takeoff_from_here(float height)
198{
199 if (takeoff.status == NAV_TAKEOFF_INIT) {
200 takeoff.climb_id = 0; // use dummy hidden WP
201 takeoff.start_pos = *stateGetPositionEnu_f();
202 takeoff.climb_pos = takeoff.start_pos;
203 if (height < 0.f) {
204 takeoff.climb_pos.z += NAV_TAKEOFF_HEIGHT; // default value
205 } else {
206 takeoff.climb_pos.z += height;
207 }
208 waypoint_set_enu(0, &takeoff.climb_pos);
209 }
210
211 return nav_takeoff_run();
212}

215static bool nav_land_run(void)
216{
217 float dx, dy, dist2;
218 switch (landing.status) {
219 case NAV_LANDING_INIT:
220 landing.status = NAV_LANDING_REACH_AF;
221 break;
222 case NAV_LANDING_REACH_AF:
223 nav.nav_goto(&landing.af_pos);
224 NavVerticalAltitudeMode(landing.af_pos.z, 0.f);
225 if (nav.nav_approaching(&landing.af_pos, NULL, 0.f)) {
226 landing.status = NAV_LANDING_DESCENT;
227 }
228 break;
229 case NAV_LANDING_DESCENT:
230 // test if glide is needed or not
231 dx = landing.af_pos.x - landing.td_pos.x;
232 dy = landing.af_pos.y - landing.td_pos.y;
233 dist2 = dx*dx + dy*dy;
234 if (dist2 > 1.f) {
235 nav.nav_route(&landing.af_pos, &landing.td_pos);
236 nav_glide_points(&landing.af_pos, &landing.td_pos);
237 } else {
238 nav.nav_goto(&landing.td_pos);
239 NavVerticalClimbMode(NAV_LANDING_DESCEND_SPEED);
240 }
241 if (nav.nav_approaching(&landing.td_pos, &landing.af_pos, 0.f)
242 || (stateGetPositionEnu_f()->z < landing.td_pos.z)) {
243 landing.status = NAV_LANDING_FLARE;
244 }
245 break;
246 case NAV_LANDING_FLARE:
247 nav.nav_goto(&landing.td_pos);
248 NavVerticalClimbMode(NAV_LANDING_DESCEND_SPEED);
249 if (!nav_is_in_flight()) {
250 landing.status = NAV_LANDING_DONE;
251 }
252 break;
253 case NAV_LANDING_DONE:
254 default:
255 NavKillThrottle();
256 landing.status = NAV_LANDING_INIT;
257 return false;
258 }
259 landing.timeout = false;
260 return true;
261}

263bool nav_land_at_wp(uint8_t td_id, uint8_t af_id, float radius UNUSED)
264{
265 if (landing.status == NAV_LANDING_INIT) {
266 landing.td_id = td_id;
267 landing.af_id = af_id;
268 landing.td_pos = *waypoint_get_enu_f(td_id);
269 landing.td_pos.z = landing.td_pos.z + NAV_LANDING_FLARE_HEIGHT;
270 landing.af_pos = *waypoint_get_enu_f(af_id);
271 landing.af_pos.z = landing.td_pos.z + NAV_LANDING_AF_HEIGHT;
272 }
273
274 return nav_land_run();
275}

277bool nav_land_at_loc(float td_alt, float lat, float lon, float dir, float dist, float radius UNUSED)
278{
279 if (landing.status == NAV_LANDING_INIT) {
280 landing.td_id = 0;
281 landing.af_id = 0;
282 struct LlaCoor_f lla = { RadOfDeg(lat), RadOfDeg(lon), stateGetLlaOrigin_f().alt + td_alt };
283 enu_of_lla_point_f(&landing.td_pos, stateGetNedOrigin_f(), &lla);
284 landing.td_pos.z = landing.td_pos.z + NAV_LANDING_FLARE_HEIGHT;
285 landing.af_pos.x = landing.td_pos.x + dist * sinf(RadOfDeg(dir));
286 landing.af_pos.y = landing.td_pos.y + dist * cosf(RadOfDeg(dir));
287 landing.af_pos.z = landing.td_pos.z + NAV_LANDING_AF_HEIGHT;
288 }
289
290 return nav_land_run();
291}
