85static bool pdec_send_frame(pdec_t *pdec, const uint8_t *payload, uint8_t payload_len)
86{
87 uint8_t frame[PDEC_FRAME_OVERHEAD + PDEC_MAX_PAYLOAD_LEN];
88 uint8_t checksum = payload_len;
89 uint16_t frame_len = (uint16_t)payload_len + PDEC_FRAME_OVERHEAD;
90 long fd = 0;
91
92 if (payload_len == 0) {
93 pdec->counters.tx_drops++;
94 return false;
95 }
96
97 frame[0] = PDEC_SYNC_BYTE;
98 frame[1] = PDEC_SYNC_BYTE;
99 frame[2] = payload_len;
100 for (uint8_t i = 0; i < payload_len; i++) {
101 frame[3 + i] = payload[i];
102 checksum ^= payload[i];
103 }
104 frame[3 + payload_len] = checksum;
105
106 uart_put_buffer(pdec->dev, fd, frame, frame_len);
107 pdec->counters.tx_frames++;
108 return true;
109}

120static void pdec_dispatch_frame(pdec_t *pdec)
121{
122 const uint8_t *payload = pdec->rx_payload;
123 uint8_t len = pdec->rx_len;
124 uint8_t msg_id;
125
126 if (len == 0) {
127 pdec->counters.rx_length_errors++;
128 return;
129 }
130
131 msg_id = payload[0];
132 switch (msg_id) {
133 case PDEC_MSG_ERROR: {
134 if (len != sizeof(struct pdec_error_payload)) {
135 pdec->counters.rx_length_errors++;
136 return;
137 }
138 const struct pdec_error_payload *msg = (const struct pdec_error_payload *)payload;
139 pdec->last_error.command = msg->command;
140 pdec->last_error.status = (enum pdec_status)msg->status;
141 pdec->last_error.updated = true;
142 break;
143 }
144
145 case PDEC_MSG_RANGE_RESULT:
146 if (len != sizeof(struct pdec_distance_payload)) {
147 pdec->counters.rx_length_errors++;
148 return;
149 }
150 pdec_parse_distance_result(&pdec->last_range, (const struct pdec_distance_payload *)payload);
151 break;
152
153 case PDEC_MSG_DISTANCE:
154 if (len != sizeof(struct pdec_distance_payload)) {
155 pdec->counters.rx_length_errors++;
156 return;
157 }
158 pdec_parse_distance_result(&pdec->last_distance, (const struct pdec_distance_payload *)payload);
159 break;
160
161 case PDEC_MSG_DEVICE_LIST: {
162 if (len < sizeof(struct pdec_device_list_payload)) {
163 pdec->counters.rx_length_errors++;
164 return;
165 }
166 const struct pdec_device_list_payload *msg = (const struct pdec_device_list_payload *)payload;
167 uint8_t total_count = msg->count;
168 if (total_count > ((PDEC_MAX_PAYLOAD_LEN - sizeof(struct pdec_device_list_payload)) / sizeof(msg->device_ids[0])) ||
169 len != (uint8_t)(sizeof(struct pdec_device_list_payload) + (total_count * sizeof(msg->device_ids[0])))) {
170 pdec->counters.rx_length_errors++;
171 return;
172 }
173 pdec->devices.total_count = total_count;
174 pdec->devices.count = total_count > PDEC_MAX_DEVICES ? PDEC_MAX_DEVICES : total_count;
175 pdec->devices.truncated = pdec->devices.total_count > pdec->devices.count;
176 for (uint8_t i = 0; i < pdec->devices.count; i++) {
177 pdec->devices.device_ids[i] = msg->device_ids[i];
178 }
179 pdec->devices.updated = true;
180 break;
181 }
182
183 case PDEC_MSG_SEND_DATA_RESULT: {
184 if (len != sizeof(struct pdec_send_data_result_payload)) {
185 pdec->counters.rx_length_errors++;
186 return;
187 }
188 const struct pdec_send_data_result_payload *msg = (const struct pdec_send_data_result_payload *)payload;
189 pdec->last_send_data.dst_id = msg->dst_id;
190 pdec->last_send_data.status = (enum pdec_status)msg->status;
191 pdec->last_send_data.sent_len = msg->sent_len;
192 pdec->last_send_data.updated = true;
193 break;
194 }
195
196 case PDEC_MSG_RANGING_EVENT: {
197 if (len != sizeof(struct pdec_ranging_event_payload)) {
198 pdec->counters.rx_length_errors++;
199 return;
200 }
201 const struct pdec_ranging_event_payload *msg = (const struct pdec_ranging_event_payload *)payload;
202 pdec->last_ranging_event.kind = (enum pdec_event_kind)msg->kind;
203 pdec->last_ranging_event.src_id = msg->src_id;
204 pdec->last_ranging_event.dst_id = msg->dst_id;
205 pdec->last_ranging_event.distance = msg->distance;
206 pdec->last_ranging_event.updated = true;
207 AbiSendMsgUWB_RANGING(UWB_PDEC_ID,
208 get_sys_time_usec(),
209 msg->src_id,
210 msg->dst_id,
211 msg->distance
212 );
213 break;
214 }
215
216 case PDEC_MSG_TDOA_REPORT_EVENT: {
217 if (len < sizeof(struct pdec_tdoa_report_event_payload)) {
218 pdec->counters.rx_length_errors++;
219 return;
220 }
221 const struct pdec_tdoa_report_event_payload *msg = (const struct pdec_tdoa_report_event_payload *)payload;
222 uint8_t total_count = msg->count;
223 if (total_count > ((PDEC_MAX_PAYLOAD_LEN - sizeof(struct pdec_tdoa_report_event_payload)) /
224 sizeof(struct pdec_tdoa_report_payload)) ||
225 len != (uint8_t)(sizeof(struct pdec_tdoa_report_event_payload) +
226 (total_count * sizeof(struct pdec_tdoa_report_payload)))) {
227 pdec->counters.rx_length_errors++;
228 return;
229 }
230 pdec->last_tdoa_report_event.blink_id = msg->blink_id;
231 pdec->last_tdoa_report_event.expected_count = msg->expected_count;
232 pdec->last_tdoa_report_event.total_count = total_count;
233 pdec->last_tdoa_report_event.count = total_count > PDEC_MAX_TDOA_REPORTS ? PDEC_MAX_TDOA_REPORTS : total_count;
234 pdec->last_tdoa_report_event.timed_out = msg->timed_out != 0;
235 pdec->last_tdoa_report_event.truncated = pdec->last_tdoa_report_event.total_count > pdec->last_tdoa_report_event.count;
236 uint16_t src_id[PDEC_MAX_TDOA_REPORTS];
237 float range_diff[PDEC_MAX_TDOA_REPORTS];
238 for (uint8_t i = 0; i < pdec->last_tdoa_report_event.count; i++) {
239 src_id[i] = msg->reports[i].reporter_id;
240 range_diff[i] = msg->reports[i].distance;
241 pdec->last_tdoa_report_event.reports[i].reporter_id = src_id[i];
242 pdec->last_tdoa_report_event.reports[i].distance = range_diff[i];
243 }
244 pdec->last_tdoa_report_event.updated = true;
245 AbiSendMsgUWB_TDOA(UWB_PDEC_ID,
246 get_sys_time_usec(),
247 (uint16_t)msg->blink_id,
248 pdec->last_tdoa_report_event.count,
249 src_id,
250 range_diff
251 );
252 break;
253 }
254
255 default:
256 pdec->counters.rx_unknown_messages++;
257 return;
258 }
259
260 pdec->counters.rx_frames++;
261}

263static void pdec_parse_byte(pdec_t *pdec, uint8_t byte)
264{
265 switch (pdec->rx_state) {
266 case PDEC_RX_SYNC_1:
267 if (byte == PDEC_SYNC_BYTE) {
268 pdec->rx_state = PDEC_RX_SYNC_2;
269 }
270 break;
271
272 case PDEC_RX_SYNC_2:
273 if (byte == PDEC_SYNC_BYTE) {
274 pdec->rx_state = PDEC_RX_LEN;
275 } else {
276 pdec->rx_state = PDEC_RX_SYNC_1;
277 }
278 break;
279
280 case PDEC_RX_LEN:
281 pdec->rx_len = byte;
282 pdec->rx_idx = 0;
283 pdec->rx_checksum = byte;
284 pdec->rx_state = byte == 0 ? PDEC_RX_CHECKSUM : PDEC_RX_PAYLOAD;
285 break;
286
287 case PDEC_RX_PAYLOAD:
288 pdec->rx_payload[pdec->rx_idx++] = byte;
289 pdec->rx_checksum ^= byte;
290 if (pdec->rx_idx >= pdec->rx_len) {
291 pdec->rx_state = PDEC_RX_CHECKSUM;
292 }
293 break;
294
295 case PDEC_RX_CHECKSUM:
296 if (pdec->rx_checksum == byte) {
297 pdec_dispatch_frame(pdec);
298 } else {
299 pdec->counters.rx_checksum_errors++;
300 }
301 pdec->rx_state = PDEC_RX_SYNC_1;
302 break;
303
304 default:
305 pdec->rx_state = PDEC_RX_SYNC_1;
306 break;
307 }
308}

394bool pdec_send_data(uint16_t dst_id, const uint8_t *data, uint8_t len)
395{
396 uint8_t payload[sizeof(struct pdec_target_cmd_payload) + PDEC_MAX_DATA_LEN];
397 struct pdec_target_cmd_payload header = {
398 .id = PDEC_CMD_SEND_DATA,
399 .dst_id = dst_id,
400 };
401
402 if (len > PDEC_MAX_DATA_LEN || (len > 0 && data == NULL)) {
403 pdec.counters.tx_drops++;
404 return false;
405 }
406
407 memcpy(payload, &header, sizeof(header));
408 if (len > 0) {
409 memcpy(&payload[sizeof(header)], data, len);
410 }
411
412 return pdec_send_frame(&pdec, payload, (uint8_t)(sizeof(header) + len));
413}
