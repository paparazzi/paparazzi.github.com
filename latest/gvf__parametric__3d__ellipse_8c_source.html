72 void gvf_parametric_3d_ellipse_info(float *f1, float *f2, float *f3, float *f1d, float *f2d, float *f3d,
73  float *f1dd, float *f2dd, float *f3dd)
74 {
75  float xo = gvf_parametric_trajectory.p_parametric[0];
76  float yo = gvf_parametric_trajectory.p_parametric[1];
77  float r = gvf_parametric_trajectory.p_parametric[2];
78  float zl = gvf_parametric_trajectory.p_parametric[3];
79  float zh = gvf_parametric_trajectory.p_parametric[4];
80  float alpha_rad = gvf_parametric_trajectory.p_parametric[5]*M_PI/180;
81 
82  float w = gvf_parametric_control.w;
83  float wb = w * gvf_parametric_control.beta * gvf_parametric_control.s;
84 
85  // Parametric equations of the trajectory and the partial derivatives w.r.t. 'w'
86  *f1 = r * cosf(wb) + xo;
87  *f2 = r * sinf(wb) + yo;
88  *f3 = 0.5 * (zh + zl + (zl - zh) * sinf(alpha_rad - wb));
89 
90  *f1d = -r * sinf(wb);
91  *f2d = r * cosf(wb);
92  *f3d = -0.5 * (zl - zh) * cosf(alpha_rad - wb);
93 
94  *f1dd = -r * cosf(wb);
95  *f2dd = -r * sinf(wb);
96  *f3dd = -0.5 * (zl - zh) * sinf(alpha_rad - wb);
97 }
