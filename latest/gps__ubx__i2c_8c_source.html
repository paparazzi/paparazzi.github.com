170void gps_ubx_i2c_periodic(void)
171{
172 if (gps_i2c.trans.status == I2CTransDone)
173 {
174 switch(gps_i2c.write_state)
175 {
176 case gps_i2c_write_standby:
177 if (!gps_i2c_char_available(&gps_i2c))
178 {
179 if (gps_ubx_i2c_bytes_to_read > GPS_I2C_BUF_SIZE || gps_ubx_i2c_ucenter_done)
180 {
181 gps_i2c.write_state = gps_i2c_write_request_size;
182 } else {
183 gps_i2c.tx_rdy = TRUE;
184 }
185 }
186 break;
187
188 case gps_i2c_write_request_size:
189 gps_i2c.trans.buf[0] = GPS_I2C_ADDR_NB_AVAIL_BYTES;
190 i2c_transceive(&GPS_UBX_I2C_DEV, &gps_i2c.trans, GPS_I2C_SLAVE_ADDR, 1, 2);
191 gps_i2c.write_state = gps_i2c_write_standby;
192 gps_i2c.read_state = gps_i2c_read_sizeof;
193 break;
194
195 case gps_i2c_write_cfg:
196 memcpy((uint8_t *) & (gps_i2c.trans.buf), gps_i2c.tx_buf, gps_i2c.tx_buf_idx);
197 i2c_transmit(&GPS_UBX_I2C_DEV, &gps_i2c.trans, GPS_I2C_SLAVE_ADDR, gps_i2c.tx_buf_idx);
198 gps_i2c.tx_buf_idx = 0;
199 gps_i2c.read_state = gps_i2c_read_standby;
200 gps_i2c.write_state = gps_i2c_write_request_size;
201 break;
202
203 default:
204 break;
205 }
206 }

209void gps_ubx_i2c_read_event(void)
210{
211 switch(gps_i2c.read_state)
212 {
213 case gps_i2c_read_standby:
214 break;
215 break;
216
217 // how many bytes are available
218 case gps_i2c_read_sizeof:
219 gps_ubx_i2c_bytes_to_read = ((uint16_t)(gps_i2c.trans.buf[0]) << 7) | (uint16_t)(gps_i2c.trans.buf[1]);
220 gps_i2c.trans.status = I2CTransDone;
221 if (gps_ubx_i2c_bytes_to_read == 0xFFFF || gps_ubx_i2c_bytes_to_read == 0x0000)
222 {
223 gps_i2c.write_state = gps_i2c_write_request_size;
224 return;
225 } else if (gps_ubx_i2c_bytes_to_read > GPS_I2C_BUF_SIZE)
226 {
227 gps_i2c.trans.buf[0] = GPS_I2C_ADDR_DATA;
228 i2c_transceive(&GPS_UBX_I2C_DEV, &gps_i2c.trans, GPS_I2C_SLAVE_ADDR, 1, GPS_I2C_BUF_SIZE);
229 gps_i2c.read_state = gps_i2c_read_data;
230 } else
231 {
232 gps_i2c.trans.buf[0] = GPS_I2C_ADDR_DATA;
233 i2c_transceive(&GPS_UBX_I2C_DEV, &gps_i2c.trans, GPS_I2C_SLAVE_ADDR, 1, gps_ubx_i2c_bytes_to_read);
234 gps_i2c.read_state = gps_i2c_read_data;
235 return;
236 }
237 break;
238 case gps_i2c_read_data:
239 if (gps_ubx_i2c_bytes_to_read > GPS_I2C_BUF_SIZE)
240 {
241 memcpy(&gps_i2c.rx_buf, (uint8_t *) & (gps_i2c.trans.buf), GPS_I2C_BUF_SIZE);
242 gps_i2c.rx_buf_idx = 0;
243 gps_i2c.rx_buf_avail = GPS_I2C_BUF_SIZE;
244 } else
245 {
246 memcpy(&gps_i2c.rx_buf, (uint8_t *) & (gps_i2c.trans.buf), gps_ubx_i2c_bytes_to_read);
247 gps_i2c.rx_buf_idx = 0;
248 gps_i2c.rx_buf_avail = gps_ubx_i2c_bytes_to_read;
249 }
250
251 gps_i2c.write_state = gps_i2c_write_standby;
252 gps_i2c.read_state = gps_i2c_read_standby;
253 break;
254
255 default:
256 break;
257 }
258
259 // Transaction has been read
260 gps_i2c.trans.status = I2CTransDone;
