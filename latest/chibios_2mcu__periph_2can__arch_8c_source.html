81static void can_thd_rx(void* arg) {
82 struct can_periph* canp = (struct can_periph*)arg;
83 struct can_arch_periph* cas = (struct can_arch_periph*)canp->arch_struct;
84 char thd_name[10];
85 snprintf(thd_name, 10, "can%d_rx", cas->if_index);
86 chRegSetThreadName(thd_name);
87
88 event_listener_t rxe;
89 chEvtRegister(&cas->cand->error_event, &rxe, EVENT_MASK(1));
90
91
92 struct pprzaddr_can addr = {
93 .can_ifindex = cas->if_index
94 };
95
96 while(!chThdShouldTerminateX()) {
97
98 eventmask_t evts = chEvtWaitAnyTimeout(ALL_EVENTS, TIME_IMMEDIATE);
99 // receive error
100 if (evts & EVENT_MASK(1)) {
101 chEvtGetAndClearFlags(&rxe);
102 canp->nb_errors++;
103 }
104
105 CANRxFrame rx_frame;
106 msg_t status = canReceiveTimeout(cas->cand, CAN_ANY_MAILBOX, &rx_frame, chTimeMS2I(50));
107 if(status == MSG_OK) {
108 uint32_t id = 0;
109 if(rx_frame.common.XTD) {
110 id = rx_frame.ext.EID | CAN_FRAME_EFF;
111 } else {
112 id = rx_frame.std.SID;
113 }
114 if(rx_frame.common.RTR) {
115 id |= CAN_FRAME_RTR;
116 }
117 if(rx_frame.common.ESI) {
118 id |= CAN_FRAME_ERR;
119 }
120
121 struct pprzcan_frame pprz_frame = {
122 .can_id = id,
123 .len = can_dlc_to_len(rx_frame.DLC),
124 .flags = 0,
125 .timestamp = TIME_I2US(chVTGetSystemTimeX())
126 };
127
128 if(rx_frame.FDF) {
129 pprz_frame.flags |= CANFD_FDF;
130 }
131 if(rx_frame.common.ESI) {
132 pprz_frame.flags |= CANFD_ESI;
133 }
134
135
136
137 memcpy(pprz_frame.data, rx_frame.data8, pprz_frame.len);
138
139 for(int i=0; i<CAN_NB_CALLBACKS_MAX; i++) {
140 if(canp->callbacks[i] != NULL) {
141 canp->callbacks[i](&pprz_frame, &addr, canp->callback_user_data[i]);
142 }
143 }
144 }
145 }
146
147}

149int can_transmit_frame(struct pprzcan_frame* txframe, struct pprzaddr_can* addr) {
150 CANTxFrame frame;
151 frame.DLC = can_len_to_dlc(txframe->len);
152 if(txframe->can_id & CAN_FRAME_RTR) {
153 frame.common.RTR = 1;
154 }
155 if(txframe->can_id & CAN_FRAME_EFF) {
156 frame.common.XTD = 1;
157 frame.ext.EID = txframe->can_id & CAN_EID_MASK;
158 } else {
159 frame.std.SID = txframe->can_id & CAN_SID_MASK;
160 }
161 memcpy(frame.data8, txframe->data, txframe->len);
162
163 #if USE_CAN1
164 if(addr->can_ifindex == 1 || addr->can_ifindex == 0) {
165 msg_t ret = canTransmitTimeout(&CAND1, CAN_ANY_MAILBOX, &frame, TIME_IMMEDIATE);
166 if(ret != MSG_OK) {
167 return ret;
168 }
169 }
170 #endif
171
172 #if USE_CAN2
173 if(addr->can_ifindex == 2 || addr->can_ifindex == 0) {
174 msg_t ret = canTransmitTimeout(&CAND2, CAN_ANY_MAILBOX, &frame, TIME_IMMEDIATE);
175 if(ret != MSG_OK) {
176 return ret;
177 }
178 }
179 #endif
180
181 return 0;
182}

184static void can_start(struct can_periph* canp) {
185 struct can_arch_periph* cas = (struct can_arch_periph*)canp->arch_struct;
186
187 #if defined(STM32_CAN_USE_FDCAN1) || defined(STM32_CAN_USE_FDCAN2)
188 // Configure the RAM
189 can1_arch_s.cfg.RXF0C = (32 << FDCAN_RXF0C_F0S_Pos) | (0 << FDCAN_RXF0C_F0SA_Pos);
190 can1_arch_s.cfg.RXF1C = (32 << FDCAN_RXF1C_F1S_Pos) | (128 << FDCAN_RXF1C_F1SA_Pos);
191 can1_arch_s.cfg.TXBC = (32 << FDCAN_TXBC_TFQS_Pos) | (256 << FDCAN_TXBC_TBSA_Pos);
192 can1_arch_s.cfg.TXESC = 0x000; // 8 Byte mode only (4 words per message)
193 can1_arch_s.cfg.RXESC = 0x000; // 8 Byte mode only (4 words per message)
194 #endif
195 if (!canConfigureIface(cas)) {
196 return;
197 }
198
199
200 canStart(cas->cand, &can1_arch_s.cfg);
201 chThdCreateStatic(cas->thread_rx_wa, cas->thread_rx_wa_size,
202 NORMALPRIO + 8, can_thd_rx, canp);
203}

209static bool canConfigureIface(struct can_arch_periph* cas)
210{
211 if (cas->can_baudrate < 1) {
212 return false;
213 }
214
215 // Hardware configurationn
216#if defined(STM32_CAN_USE_FDCAN1) || defined(STM32_CAN_USE_FDCAN2)
217 const uint32_t pclk = STM32_FDCANCLK;
218#else
219 const uint32_t pclk = STM32_PCLK1;
220#endif
221 static const int MaxBS1 = 16;
222 static const int MaxBS2 = 8;
223
224 /*
225 * Ref. "Automatic Baudrate Detection in CANopen Networks", U. Koppe, MicroControl GmbH & Co. KG
226 * CAN in Automation, 2003
227 *
228 * According to the source, optimal quanta per bit are:
229 * Bitrate Optimal Maximum
230 * 1000 kbps 8 10
231 * 500 kbps 16 17
232 * 250 kbps 16 17
233 * 125 kbps 16 17
234 */
235 const int max_quanta_per_bit = (cas->can_baudrate >= 1000000) ? 10 : 17;
236 static const int MaxSamplePointLocation = 900;
237
238 /*
239 * Computing (prescaler * BS):
240 * BITRATE = 1 / (PRESCALER * (1 / PCLK) * (1 + BS1 + BS2)) -- See the Reference Manual
241 * BITRATE = PCLK / (PRESCALER * (1 + BS1 + BS2)) -- Simplified
242 * let:
243 * BS = 1 + BS1 + BS2 -- Number of time quanta per bit
244 * PRESCALER_BS = PRESCALER * BS
245 * ==>
246 * PRESCALER_BS = PCLK / BITRATE
247 */
248 const uint32_t prescaler_bs = pclk / cas->can_baudrate;
249
250// Searching for such prescaler value so that the number of quanta per bit is highest.
251 uint8_t bs1_bs2_sum = max_quanta_per_bit - 1;
252 while ((prescaler_bs % (1 + bs1_bs2_sum)) != 0) {
253 if (bs1_bs2_sum <= 2) {
254 return false; // No solution
255 }
256 bs1_bs2_sum--;
257 }
258
259 const uint32_t prescaler = prescaler_bs / (1 + bs1_bs2_sum);
260 if ((prescaler < 1U) || (prescaler > 1024U)) {
261 return false; // No solution
262 }
263
264 /*
265 * Now we have a constraint: (BS1 + BS2) == bs1_bs2_sum.
266 * We need to find the values so that the sample point is as close as possible to the optimal value.
267 *
268 * Solve[(1 + bs1)/(1 + bs1 + bs2) == 7/8, bs2] (* Where 7/8 is 0.875, the recommended sample point location *)
269 * {{bs2 -> (1 + bs1)/7}}
270 *
271 * Hence:
272 * bs2 = (1 + bs1) / 7
273 * bs1 = (7 * bs1_bs2_sum - 1) / 8
274 *
275 * Sample point location can be computed as follows:
276 * Sample point location = (1 + bs1) / (1 + bs1 + bs2)
277 *
278 * Since the optimal solution is so close to the maximum, we prepare two solutions, and then pick the best one:
279 * - With rounding to nearest
280 * - With rounding to zero
281 */
282// First attempt with rounding to nearest
283 uint8_t bs1 = ((7 * bs1_bs2_sum - 1) + 4) / 8;
284 uint8_t bs2 = bs1_bs2_sum - bs1;
285 uint16_t sample_point_permill = 1000 * (1 + bs1) / (1 + bs1 + bs2);
286
287// Second attempt with rounding to zero
288 if (sample_point_permill > MaxSamplePointLocation) {
289 bs1 = (7 * bs1_bs2_sum - 1) / 8;
290 bs2 = bs1_bs2_sum - bs1;
291 sample_point_permill = 1000 * (1 + bs1) / (1 + bs1 + bs2);
292 }
293
294 /*
295 * Final validation
296 * Helpful Python:
297 * def sample_point_from_btr(x):
298 * assert 0b0011110010000000111111000000000 & x == 0
299 * ts2,ts1,brp = (x>>20)&7, (x>>16)&15, x&511
300 * return (1+ts1+1)/(1+ts1+1+ts2+1)
301 *
302 */
303 if ((cas->can_baudrate != (pclk / (prescaler * (1 + bs1 + bs2)))) || (bs1 < 1) || (bs1 > MaxBS1) || (bs2 < 1)
304 || (bs2 > MaxBS2)) {
305 return false;
306 }
307
308 // Configure the interface
309#if defined(STM32_CAN_USE_FDCAN1) || defined(STM32_CAN_USE_FDCAN2)
310 cas->cfg.NBTP = (0 << FDCAN_NBTP_NSJW_Pos) | ((bs1 - 1) << FDCAN_NBTP_NTSEG1_Pos) | ((
311 bs2 - 1) << FDCAN_NBTP_NTSEG2_Pos) | ((prescaler - 1) << FDCAN_NBTP_NBRP_Pos);
312 #if USE_CANFD
313 cas->cfg.CCCR = FDCAN_CCCR_FDOE | FDCAN_CCCR_BRSE;
314 #else
315 cas->cfg.CCCR = 0;
316 #endif
317
318#else
319 cas->cfg.mcr = CAN_MCR_ABOM | CAN_MCR_AWUM | CAN_MCR_TXFP;
320 cas->cfg.btr = CAN_BTR_SJW(0) | CAN_BTR_TS1(bs1 - 1) | CAN_BTR_TS2(bs2 - 1) | CAN_BTR_BRP(prescaler - 1);
321#endif
322 return true;
323}
