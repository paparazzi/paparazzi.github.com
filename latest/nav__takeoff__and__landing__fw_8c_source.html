77static bool nav_takeoff_mission(uint8_t nb UNUSED, float *params UNUSED, enum MissionRunFlag flag)
78{
79 if (flag == MissionInit && nb == 1) {
80 float height = params[0];
81 takeoff.status = NAV_TAKEOFF_INIT;
82 return nav_takeoff_from_here(height);
83 }
84 else if (flag == MissionRun) {
85 return nav_takeoff_run();
86 }
87 return false; // not a valid case
88}

90static bool nav_land_mission(uint8_t nb, float *params, enum MissionRunFlag flag)
91{
92 if (flag == MissionInit && nb == 2) {
93 float td_alt = params[0];
94 float radius = params[1];
95 landing.status = NAV_LANDING_INIT;
96 return nav_land_here(td_alt, radius);
97 } else if (flag == MissionInit && nb == 3) {
98 uint8_t td_id = (uint8_t)(params[0]);
99 uint8_t af_id = (uint8_t)(params[1]);
100 float radius = params[2];
101 return nav_land_at_wp(td_id, af_id, radius);
102 } else if (flag == MissionInit && nb == 6) {
103 float td_alt = params[0];
104 float lat = params[1];
105 float lon = params[2];
106 float dir = params[3];
107 float dist = params[4];
108 float radius = params[5];
109 return nav_land_at_loc(td_alt, lat, lon, dir, dist, radius);
110 } else if (flag == MissionRun) {
111 return nav_land_run();
112 }
113 return false; // not a valid case
114}

150static bool nav_takeoff_run(void) {
151 static int start_motor_counter = 0;
152
153 switch (takeoff.status) {
154 case NAV_TAKEOFF_INIT:
155 takeoff.status = NAV_TAKEOFF_START_MOTOR;
156 start_motor_counter = 0;
157 break;
158 case NAV_TAKEOFF_START_MOTOR:
159 autopilot_set_motors_on(true);
160 start_motor_counter++;
161 if (start_motor_counter == NAVIGATION_FREQUENCY) {
162 start_motor_counter = 0;
163 if (autopilot_get_motors_on()) {
164#if NAV_TAKEOFF_AUTO_LAUNCH
165 autopilot.launch = true;
166#endif
167 takeoff.status = NAV_TAKEOFF_CLIMB; // next step after 1s
168 } else {
169 takeoff.status = NAV_TAKEOFF_DONE; // failed to start motors, end now
170 }
171 }
172 break;
173 case NAV_TAKEOFF_CLIMB:
174 // climb towards point at specified pitch and throttle
175 NavGotoPoint(takeoff.climb_pos);
176 NavVerticalAutoThrottleMode(RadOfDeg(NAV_TAKEOFF_PITCH));
177 NavVerticalThrottleMode(TRIM_UPPRZ(MAX_PPRZ*NAV_TAKEOFF_THROTTLE));
178 if (nav_approaching_xy(takeoff.climb_pos.x, takeoff.climb_pos.y, takeoff.start_pos.x, takeoff.start_pos.y, CARROT)
179 || (stateGetPositionEnu_f()->z - takeoff.start_pos.z > takeoff.climb_pos.z)) {
180 // end when climb point or target alt is reached
181 takeoff.status = NAV_TAKEOFF_DONE;
182 }
183 break;
184 default:
185 takeoff.status = NAV_TAKEOFF_INIT;
186 return false;
187 }
188 takeoff.timeout = false;
189 return true;
190}

192bool nav_takeoff_from_wp(uint8_t wp_id, float height)
193{
194 if (takeoff.status == NAV_TAKEOFF_INIT) {
195 takeoff.start_pos = *stateGetPositionEnu_f();
196 takeoff.climb_id = wp_id;
197 takeoff.climb_pos.x = WaypointX(wp_id);
198 takeoff.climb_pos.y = WaypointY(wp_id);
199 if (height < 0.f) {
200 takeoff.climb_pos.z = takeoff.start_pos.z + NAV_TAKEOFF_HEIGHT;
201 } else {
202 takeoff.climb_pos.z = takeoff.start_pos.z + height;
203 }
204 }
205
206 return nav_takeoff_run();
207}

209bool nav_takeoff_from_loc(float lat, float lon, float height)
210{
211 if (takeoff.status == NAV_TAKEOFF_INIT) {
212 takeoff.start_pos = *stateGetPositionEnu_f();
213 float alt = stateGetPositionLla_f()->alt;
214 if (height < 0.f) {
215 alt += NAV_TAKEOFF_HEIGHT;
216 } else {
217 alt += height;
218 }
219 struct LlaCoor_f lla = { RadOfDeg(lat), RadOfDeg(lon), alt };
220 struct UtmCoor_f utm;
221 utm_of_lla_f(&utm, &lla);
222 ENU_OF_UTM_DIFF(takeoff.climb_pos, utm, *stateGetUtmOrigin_f());
223 }
224 return nav_takeoff_run();
225}

227bool nav_takeoff_from_here(float height)
228{
229 if (takeoff.status == NAV_TAKEOFF_INIT) {
230 takeoff.climb_id = 0;
231 takeoff.start_pos = *stateGetPositionEnu_f();
232 takeoff.climb_pos.x = takeoff.start_pos.x + NAV_TAKEOFF_DIST * sinf(RadOfDeg(nav_takeoff_direction));
233 takeoff.climb_pos.y = takeoff.start_pos.y + NAV_TAKEOFF_DIST * cosf(RadOfDeg(nav_takeoff_direction));
234 if (height < 0.f) {
235 takeoff.climb_pos.z = takeoff.start_pos.z + NAV_TAKEOFF_HEIGHT;
236 } else {
237 takeoff.climb_pos.z = takeoff.start_pos.z + height;
238 }
239 }
240
241 return nav_takeoff_run();

245static bool nav_land_run(void)
246{
247 static float baseleg_out_qdr = 0.f;
248 float qdr = 0.f;
249
250 // test if glide is needed or not
251 float dx = landing.td_pos.x - landing.af_pos.x;
252 float dy = landing.td_pos.y - landing.af_pos.y;
253 float dist = sqrtf(dx*dx + dy*dy);
254 bool glide = false;
255 float x_unit = 0.f;
256 float y_unit = 0.f;
257 if (dist > 1.f) {
258 glide = true;
259 x_unit = dx / dist;
260 y_unit = dy / dist;
261 }
262
263 switch (landing.status) {
264 case NAV_LANDING_INIT:
265 if (glide) {
266 // compute baseleg circle center, store in dummy WP 0
267 waypoints[0].x = landing.af_pos.x + y_unit * landing.radius;
268 waypoints[0].y = landing.af_pos.y - x_unit * landing.radius;
269 waypoints[0].a = landing.af_pos.z + GetAltRef();
270 baseleg_out_qdr = M_PI - atan2f(-y_unit, -x_unit);
271 if (landing.radius < 0.f) {
272 baseleg_out_qdr += M_PI;
273 }
274 } else {
275 // turn around AF point
276 waypoints[0].x = landing.af_pos.x;
277 waypoints[0].y = landing.af_pos.y;
278 waypoints[0].a = landing.af_pos.z + GetAltRef();
279 baseleg_out_qdr = 0.f;
280 }
281 landing.status = NAV_LANDING_REACH_AF;
282 break;
283 case NAV_LANDING_REACH_AF:
284 NavVerticalAutoThrottleMode(0.f);
285 NavVerticalAltitudeMode(waypoints[0].a, 0.f);
286 NavCircleWaypoint(0, landing.radius);
287 if (NavCircleCount() > 0.25 && (fabsf(GetPosAlt() - WaypointAlt(0)) < 10.f)) {
288 if (glide) {
289 // if final glide, direction should be correct
290 qdr = M_PI_2 - nav_circle_trigo_qdr;
291 NormCourseRad(qdr);
292 if (CloseRadAngles(baseleg_out_qdr, qdr)) {
293 landing.status = NAV_LANDING_DESCENT;
294 }
295 } else {
296 // no final glide, start from correct altitude
297 landing.status = NAV_LANDING_DESCENT;
298 }
299 }
300 break;
301 case NAV_LANDING_DESCENT:
302 if (glide) {
303 nav_route_xy(landing.af_pos.x, landing.af_pos.y, landing.td_pos.x, landing.td_pos.y);
304 NavVerticalAutoThrottleMode(0.f);
305 nav_glide_alt(landing.af_pos.z + GetAltRef(), landing.td_pos.z + GetAltRef());
306 if (nav_approaching_xy(landing.td_pos.x, landing.td_pos.y, landing.af_pos.x, landing.af_pos.y, CARROT)
307 || (GetPosAlt() < landing.td_pos.z + NAV_LANDING_FLARE_HEIGHT + GetAltRef())) {
308 landing.status = NAV_LANDING_FLARE;
309 }
310 } else {
311 NavCircleWaypoint(0, landing.radius);
312 NavVerticalAutoThrottleMode(0.f);
313 NavVerticalClimbMode(NAV_LANDING_DESCEND_SPEED);
314 if (GetPosAlt() < landing.td_pos.z + NAV_LANDING_FLARE_HEIGHT + GetAltRef()) {
315 landing.status = NAV_LANDING_FLARE;
316 }
317 }
318 break;
319 case NAV_LANDING_FLARE:
320 NavVerticalAutoThrottleMode(0.f);
321 NavVerticalThrottleMode(0.f);
322 if (glide) {
323 nav_route_xy(landing.af_pos.x, landing.af_pos.y, landing.td_pos.x, landing.td_pos.y);
324 if (nav_approaching_xy(landing.td_pos.x, landing.td_pos.y, landing.af_pos.x, landing.af_pos.y, 0.f)) {
325 landing.status = NAV_LANDING_DONE;
326 }
327 } else {
328 NavCircleWaypoint(0, landing.radius);
329 // stay on circle, no end
330 }
331 break;
332 case NAV_LANDING_DONE:
333 default:
334 NavKillThrottle();
335 landing.status = NAV_LANDING_INIT;
336 return false;
337 }
338 landing.timeout = false;
339 return true;
340}

342bool nav_land_at_wp(uint8_t td_id, uint8_t af_id, float radius)
343{
344 if (landing.status == NAV_LANDING_INIT) {
345 landing.td_id = td_id;
346 landing.af_id = af_id;
347 landing.td_pos.x = WaypointX(td_id);
348 landing.td_pos.y = WaypointY(td_id);
349 landing.td_pos.z = WaypointAlt(td_id) - GetAltRef();
350 landing.af_pos.x = WaypointX(af_id);
351 landing.af_pos.y = WaypointY(af_id);
352 landing.af_pos.z = landing.td_pos.z + NAV_LANDING_AF_HEIGHT;
353 landing.radius = radius;
354 }
355
356 return nav_land_run();
357}

359bool nav_land_at_loc(float td_alt, float lat, float lon, float dir, float dist, float radius)
360{
361 if (landing.status == NAV_LANDING_INIT) {
362 landing.td_id = 0;
363 landing.af_id = 0;
364 struct LlaCoor_f lla = { RadOfDeg(lat), RadOfDeg(lon), GetAltRef() + td_alt };
365 struct UtmCoor_f utm = {0};
366 utm_of_lla_f(&utm, &lla);
367 ENU_OF_UTM_DIFF(landing.td_pos, utm, *stateGetUtmOrigin_f());
368 landing.af_pos.x = landing.td_pos.x + dist * sinf(RadOfDeg(dir));
369 landing.af_pos.y = landing.td_pos.y + dist * cosf(RadOfDeg(dir));
370 landing.af_pos.z = landing.td_pos.z + NAV_LANDING_AF_HEIGHT;
371 landing.radius = radius;
372 }
373
374 return nav_land_run();
375}
