48int pprz_bsem_wait_timeout(pprz_bsem_t* bsem, float timeout) {
49 struct timespec ts;
50 clock_gettime(CLOCK_REALTIME, &ts);
51 time_t timeout_secs = (time_t)timeout;
52 long timeout_nsecs = (timeout - timeout_secs) * 1e9;
53 ts.tv_sec += timeout_secs;
54 ts.tv_nsec += timeout_nsecs;
55 return sem_timedwait(&bsem->sem, &ts);
56}

58void pprz_bsem_signal(pprz_bsem_t* bsem) {
59 int val;
60 pthread_mutex_lock(&bsem->mtx);
61 if(!sem_getvalue(&bsem->sem, &val)) {
62 if(val < 1) {
63 sem_post(&bsem->sem);
64 }
65 }
66 pthread_mutex_unlock(&bsem->mtx);
67}

86int pprz_thread_create(pprz_thread_t* thread, size_t size, const char *name, uint8_t prio, void (*func)(void*), void* arg) {
87 (void)size;
88 (void)prio;
89 struct stub_arg *sa = malloc(sizeof(struct stub_arg)); // allocate a stub arg. Must be allocated on the heap
90 sa->func = func;
91 sa->arg = arg;
92 int ret = pthread_create(thread, NULL, stub, (void*)sa); // launch the stub in a new thread
93 if(ret) {return ret;}
