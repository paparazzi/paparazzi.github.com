122 static void send_fp(struct transport_tx *trans, struct link_device *dev)
123 {
124  int32_t carrot_up = -guidance_v.z_sp;
125  int32_t carrot_heading = ANGLE_BFP_OF_REAL(guidance_h.sp.heading);
126  int32_t thrust = (int32_t)autopilot.throttle;
127  struct EnuCoor_i *pos = stateGetPositionEnu_i();
128 #if GUIDANCE_INDI_HYBRID
129  struct FloatEulers eulers_zxy;
130  float_eulers_of_quat_zxy(&eulers_zxy, stateGetNedToBodyQuat_f());
131  struct Int32Eulers att;
132  EULERS_BFP_OF_REAL(att, eulers_zxy);
133 #else
134  struct Int32Eulers att = *stateGetNedToBodyEulers_i();
135 #endif
136  pprz_msg_send_ROTORCRAFT_FP(trans, dev, AC_ID,
137  &pos->x,
138  &pos->y,
139  &pos->z,
140  &(stateGetSpeedEnu_i()->x),
141  &(stateGetSpeedEnu_i()->y),
142  &(stateGetSpeedEnu_i()->z),
143  &att.phi,
144  &att.theta,
145  &att.psi,
146  &guidance_h.sp.pos.y,
147  &guidance_h.sp.pos.x,
148  &carrot_up,
149  &carrot_heading,
150  &thrust,
151  &autopilot.flight_time);

230 void autopilot_event(void)
231 {
232  if (autopilot.detect_ground_once
233 #ifdef AP_MODE_FAILSAFE
234  || autopilot.mode == AP_MODE_FAILSAFE
235 #endif
236  ) {
237  struct NedCoor_f *accel = stateGetAccelNed_f();
238  if (accel->z < -THRESHOLD_GROUND_DETECT ||
239  accel->z > THRESHOLD_GROUND_DETECT) {
240  autopilot.ground_detected = true;
241  autopilot.detect_ground_once = false;
242  }
243  }
244 }

255 void autopilot_check_in_flight(bool motors_on)
256 {
257  if (autopilot.in_flight) {
258  if (autopilot_in_flight_counter > 0) {
259  /* probably in_flight if thrust, speed and accel above IN_FLIGHT_MIN thresholds */
260  if ((stabilization_cmd[COMMAND_THRUST] <= AUTOPILOT_IN_FLIGHT_MIN_THRUST) &&
261  (fabsf(stateGetSpeedNed_f()->z) < AUTOPILOT_IN_FLIGHT_MIN_SPEED) &&
262  (fabsf(stateGetAccelNed_f()->z) < AUTOPILOT_IN_FLIGHT_MIN_ACCEL)) {
263  autopilot_in_flight_counter--;
264  if (autopilot_in_flight_counter == 0) {
265  autopilot.in_flight = false;
266  }
267  } else { /* thrust, speed or accel not above min threshold, reset counter */
268  autopilot_in_flight_counter = AUTOPILOT_IN_FLIGHT_TIME;
269  }
270  }
271  } else { /* currently not in flight */
272  if (autopilot_in_flight_counter < AUTOPILOT_IN_FLIGHT_TIME &&
273  motors_on) {
274  /* if thrust above min threshold, assume in_flight.
275  * Don't check for velocity and acceleration above threshold here...
276  */
277  if (stabilization_cmd[COMMAND_THRUST] > AUTOPILOT_IN_FLIGHT_MIN_THRUST) {
278  autopilot_in_flight_counter++;
279  if (autopilot_in_flight_counter == AUTOPILOT_IN_FLIGHT_TIME) {
280  autopilot.in_flight = true;
281  }
282  } else { /* currently not in_flight and thrust below threshold, reset counter */
283  autopilot_in_flight_counter = 0;
284  }
285  }
286  }
