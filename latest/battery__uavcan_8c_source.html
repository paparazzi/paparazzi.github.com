57 static void battery_uavcan_cb(struct uavcan_iface_t *iface __attribute__((unused)), CanardRxTransfer *transfer)
58 {
59  uint16_t tmp_float = 0;
60 
61  /* Decode the message */
62  canardDecodeScalar(transfer, (uint32_t)0, 16, true, (void *)&tmp_float);
63  float temperature = canardConvertFloat16ToNativeFloat(tmp_float);
64  canardDecodeScalar(transfer, (uint32_t)16, 16, true, (void *)&tmp_float);
65  float voltage = canardConvertFloat16ToNativeFloat(tmp_float);
66  canardDecodeScalar(transfer, (uint32_t)32, 16, true, (void *)&tmp_float);
67  float current = canardConvertFloat16ToNativeFloat(tmp_float);
68  canardDecodeScalar(transfer, (uint32_t)48, 16, true, (void *)&tmp_float);
69  float average_power_10sec = canardConvertFloat16ToNativeFloat(tmp_float);
70  canardDecodeScalar(transfer, (uint32_t)64, 16, true, (void *)&tmp_float);
71  float remaining_capacity_wh = canardConvertFloat16ToNativeFloat(tmp_float);
72  canardDecodeScalar(transfer, (uint32_t)80, 16, true, (void *)&tmp_float);
73  float full_charge_capacity_wh = canardConvertFloat16ToNativeFloat(tmp_float);
74  canardDecodeScalar(transfer, (uint32_t)96, 16, true, (void *)&tmp_float);
75  float hours_to_full_charge = canardConvertFloat16ToNativeFloat(tmp_float);
76  uint16_t status_flags = 0;
77  canardDecodeScalar(transfer, (uint32_t)112, 11, false, (void *)&status_flags);
78  uint8_t state_of_health_pct = 0;
79  canardDecodeScalar(transfer, (uint32_t)123, 7, false, (void *)&state_of_health_pct);
80  uint8_t state_of_charge_pct = 0;
81  canardDecodeScalar(transfer, (uint32_t)130, 7, false, (void *)&state_of_charge_pct);
82  uint8_t state_of_charge_pct_stdev = 0;
83  canardDecodeScalar(transfer, (uint32_t)137, 7, false, (void *)&state_of_charge_pct_stdev);
84  uint8_t battery_id = 0;
85  canardDecodeScalar(transfer, (uint32_t)144, 8, false, (void *)&battery_id);
86  uint32_t model_instance_id = 0;
87  canardDecodeScalar(transfer, (uint32_t)152, 32, false, (void *)&model_instance_id);
88 
89  // Set the battery info
90  if (battery_id < 3) {
91  batteries[battery_id].set = true;
92  batteries[battery_id].temperature = temperature;
93  batteries[battery_id].voltage = voltage;
94  batteries[battery_id].current = current;
95  batteries[battery_id].average_power_10sec = average_power_10sec;
96  batteries[battery_id].remaining_capacity_wh = remaining_capacity_wh;
97  batteries[battery_id].full_charge_capacity_wh = full_charge_capacity_wh;
98  batteries[battery_id].hours_to_full_charge = hours_to_full_charge;
99  batteries[battery_id].status_flags = status_flags;
100  batteries[battery_id].state_of_health_pct = state_of_health_pct;
101  batteries[battery_id].state_of_charge_pct = state_of_charge_pct;
102  batteries[battery_id].state_of_charge_pct_stdev = state_of_charge_pct_stdev;
103  batteries[battery_id].battery_id = battery_id;
104  batteries[battery_id].model_instance_id = model_instance_id;
105  }
106 
107  // Sum the battery currents
108  float current_sum = 0;
109  for (uint8_t i = 0; i < 3; i++) {
110  if (batteries[i].set) {
111  current_sum += batteries[i].current;
112  }
113  }
114  electrical.current = current_sum;
115  if (voltage > 0) {
116  electrical.vsupply = voltage;
117  }
118 }
