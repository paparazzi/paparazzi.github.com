116 void imu_mpu9250_init(void)
117 {
118  /* MPU9250 */
119  mpu9250_spi_init(&imu_mpu9250.mpu, &(IMU_MPU9250_SPI_DEV), IMU_MPU9250_SPI_SLAVE_IDX);
120  // change the default configuration
121  imu_mpu9250.mpu.config.smplrt_div = IMU_MPU9250_SMPLRT_DIV;
122  imu_mpu9250.mpu.config.dlpf_gyro_cfg = IMU_MPU9250_GYRO_LOWPASS_FILTER;
123  imu_mpu9250.mpu.config.dlpf_accel_cfg = IMU_MPU9250_ACCEL_LOWPASS_FILTER;
124  imu_mpu9250.mpu.config.gyro_range = IMU_MPU9250_GYRO_RANGE;
125  imu_mpu9250.mpu.config.accel_range = IMU_MPU9250_ACCEL_RANGE;
126 
127  // Set the default scaling
128  imu_set_defaults_gyro(IMU_MPU9250_ID, NULL, NULL, MPU9250_GYRO_SENS_FRAC[IMU_MPU9250_GYRO_RANGE]);
129  imu_set_defaults_accel(IMU_MPU9250_ID, NULL, NULL, MPU9250_ACCEL_SENS_FRAC[IMU_MPU9250_ACCEL_RANGE]);
130 
131  /* "internal" ak8963 magnetometer as I2C slave */
132 #if IMU_MPU9250_READ_MAG
133  /* read 15 bytes for status, accel, gyro + 7 bytes for mag slave */
134  imu_mpu9250.mpu.config.nb_bytes = 22;
135  imu_mpu9250.mpu.config.nb_slaves = 1;
136 #endif
137  /* set callback function to configure mag */
138  imu_mpu9250.mpu.config.slaves[0].configure = &imu_mpu9250_configure_mag_slave;
139 
140  /* Set MPU I2C master clock */
141  imu_mpu9250.mpu.config.i2c_mst_clk = MPU9250_MST_CLK_400KHZ;
142  /* Enable I2C slave0 delayed sample rate */
143  imu_mpu9250.mpu.config.i2c_mst_delay = 1;
144 
145 
146  /* configure spi transaction for wait_slave4 */
147  imu_mpu9250.wait_slave4_trans.cpol = SPICpolIdleHigh;
148  imu_mpu9250.wait_slave4_trans.cpha = SPICphaEdge2;
149  imu_mpu9250.wait_slave4_trans.dss = SPIDss8bit;
150  imu_mpu9250.wait_slave4_trans.bitorder = SPIMSBFirst;
151  imu_mpu9250.wait_slave4_trans.cdiv = SPIDiv64;
152 
153  imu_mpu9250.wait_slave4_trans.select = SPISelectUnselect;
154  imu_mpu9250.wait_slave4_trans.slave_idx = IMU_MPU9250_SPI_SLAVE_IDX;
155  imu_mpu9250.wait_slave4_trans.output_length = 1;
156  imu_mpu9250.wait_slave4_trans.input_length = 2;
157  imu_mpu9250.wait_slave4_trans.before_cb = NULL;
158  imu_mpu9250.wait_slave4_trans.after_cb = mpu_wait_slave4_ready_cb;
159  imu_mpu9250.wait_slave4_trans.input_buf = &(imu_mpu9250.wait_slave4_rx_buf[0]);
160  imu_mpu9250.wait_slave4_trans.output_buf = &(imu_mpu9250.wait_slave4_tx_buf[0]);
161 
162  imu_mpu9250.wait_slave4_trans.status = SPITransDone;
163  imu_mpu9250.slave4_ready = false;
164 }

171 #define Int16FromBuf(_buf,_idx) ((int16_t)(_buf[_idx] | (_buf[_idx+1] << 8)))
172 void imu_mpu9250_event(void)
173 {
174  uint32_t now_ts = get_sys_time_usec();
175 
176  // If the MPU9250 SPI transaction has succeeded: convert the data
177  mpu9250_spi_event(&imu_mpu9250.mpu);
178 
179  if (imu_mpu9250.mpu.data_available) {
180  // set channel order
181  struct Int32Vect3 accel = {
182  IMU_MPU9250_X_SIGN * (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_X]),
183  IMU_MPU9250_Y_SIGN * (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Y]),
184  IMU_MPU9250_Z_SIGN * (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Z])
185  };
186  struct Int32Rates rates = {
187  IMU_MPU9250_X_SIGN * (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_X]),
188  IMU_MPU9250_Y_SIGN * (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Y]),
189  IMU_MPU9250_Z_SIGN * (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Z])
190  };
191 
192 #if IMU_MPU9250_READ_MAG
193  if (!bit_is_set(imu_mpu9250.mpu.data_ext[6], 3)) { //mag valid just HOFL == 0
195  struct Int32Vect3 mag;
196  mag.x = (IMU_MPU9250_X_SIGN) * Int16FromBuf(imu_mpu9250.mpu.data_ext, 2 * IMU_MPU9250_CHAN_Y);
197  mag.y = (IMU_MPU9250_Y_SIGN) * Int16FromBuf(imu_mpu9250.mpu.data_ext, 2 * IMU_MPU9250_CHAN_X);
198  mag.z = -(IMU_MPU9250_Z_SIGN) * Int16FromBuf(imu_mpu9250.mpu.data_ext, 2 * IMU_MPU9250_CHAN_Z);
199  AbiSendMsgIMU_MAG_RAW(IMU_MPU9250_ID, now_ts, &mag);
200  }
201 #endif
202 
203  imu_mpu9250.mpu.data_available = false;
204  AbiSendMsgIMU_GYRO_RAW(IMU_MPU9250_ID, now_ts, &rates, 1, IMU_MPU9250_PERIODIC_FREQ, NAN);
205  AbiSendMsgIMU_ACCEL_RAW(IMU_MPU9250_ID, now_ts, &accel, 1, IMU_MPU9250_PERIODIC_FREQ, NAN);
206  }
207 
208 }

220 bool imu_mpu9250_configure_mag_slave(Mpu9250ConfigSet mpu_set, void *mpu)
221 {
222  // wait before starting the configuration of the mag
223  // doing to early may void the mode configuration
224  if (get_sys_time_float() < IMU_MPU9250_MAG_STARTUP_DELAY) {
225  return false;
226  }
227 
228  //config AK8963 soft reset
229  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_ADDR, (MPU9250_MAG_ADDR >> 1));
230  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_REG, AK8963_REG_CNTL2);
231  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_DO, 1);
232  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_CTRL, (1 << 7)); // Slave 4 enable
233 
234  mpu_wait_slave4_ready();
235 
236  // Set it to continious measuring mode 2
237  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_ADDR, (MPU9250_MAG_ADDR >> 1));
238  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_REG, AK8963_REG_CNTL1);
239  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_DO, AK8963_CNTL1_CM_2);
240  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_CTRL, (1 << 7)); // Slave 4 enable
241 
242  mpu_wait_slave4_ready();
243 
244  //Config SLV0 for continus Read
245  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV0_ADDR, (MPU9250_MAG_ADDR >> 1) | MPU9250_SPI_READ);
246  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV0_REG, AK8963_REG_HXL);
247  // Put the enable command as last.
248  mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV0_CTRL,
249  (1 << 7) | // Slave 0 enable
250  (7 << 0)); // Read 7 bytes (mag x,y,z + status)
251 
252  return true;
253 }

255 void mpu_wait_slave4_ready(void)
256 {
257  while (!imu_mpu9250.slave4_ready) {
258  if (imu_mpu9250.wait_slave4_trans.status == SPITransDone) {
259  imu_mpu9250.wait_slave4_tx_buf[0] = MPU9250_REG_I2C_MST_STATUS | MPU9250_SPI_READ;
260  spi_submit(imu_mpu9250.mpu.spi_p, &(imu_mpu9250.wait_slave4_trans));
261  }
262  }
263 }

265 void mpu_wait_slave4_ready_cb(struct spi_transaction *t)
266 {
267  if (bit_is_set(t->input_buf[1], MPU9250_I2C_SLV4_DONE)) {
268  imu_mpu9250.slave4_ready = true;
269  } else {
270  imu_mpu9250.slave4_ready = false;
271  }
272  t->status = SPITransDone;
273 }
