117void imu_mpu9250_init(void)
118{
119 /* MPU9250 */
120 mpu9250_spi_init(&imu_mpu9250.mpu, &(IMU_MPU9250_SPI_DEV), IMU_MPU9250_SPI_SLAVE_IDX);
121 // change the default configuration
122 imu_mpu9250.mpu.config.smplrt_div = IMU_MPU9250_SMPLRT_DIV;
123 imu_mpu9250.mpu.config.dlpf_gyro_cfg = IMU_MPU9250_GYRO_LOWPASS_FILTER;
124 imu_mpu9250.mpu.config.dlpf_accel_cfg = IMU_MPU9250_ACCEL_LOWPASS_FILTER;
125 imu_mpu9250.mpu.config.gyro_range = IMU_MPU9250_GYRO_RANGE;
126 imu_mpu9250.mpu.config.accel_range = IMU_MPU9250_ACCEL_RANGE;
127
128 // Set the default scaling
129 imu_set_defaults_gyro(IMU_MPU9250_ID, NULL, NULL, &MPU9250_GYRO_SENS_F[IMU_MPU9250_GYRO_RANGE]);
130 imu_set_defaults_accel(IMU_MPU9250_ID, NULL, NULL, &MPU9250_ACCEL_SENS_F[IMU_MPU9250_ACCEL_RANGE]);
131
132 /* "internal" ak8963 magnetometer as I2C slave */
133#if IMU_MPU9250_READ_MAG
134 /* read 15 bytes for status, accel, gyro + 7 bytes for mag slave */
135 imu_mpu9250.mpu.config.nb_bytes = 22;
136 imu_mpu9250.mpu.config.nb_slaves = 1;
137#endif
138 /* set callback function to configure mag */
139 imu_mpu9250.mpu.config.slaves[0].configure = &imu_mpu9250_configure_mag_slave;
140
141 /* Set MPU I2C master clock */
142 imu_mpu9250.mpu.config.i2c_mst_clk = MPU9250_MST_CLK_400KHZ;
143 /* Enable I2C slave0 delayed sample rate */
144 imu_mpu9250.mpu.config.i2c_mst_delay = 1;
145
146
147 /* configure spi transaction for wait_slave4 */
148 imu_mpu9250.wait_slave4_trans.cpol = SPICpolIdleHigh;
149 imu_mpu9250.wait_slave4_trans.cpha = SPICphaEdge2;
150 imu_mpu9250.wait_slave4_trans.dss = SPIDss8bit;
151 imu_mpu9250.wait_slave4_trans.bitorder = SPIMSBFirst;
152 imu_mpu9250.wait_slave4_trans.cdiv = SPIDiv64;
153
154 imu_mpu9250.wait_slave4_trans.select = SPISelectUnselect;
155 imu_mpu9250.wait_slave4_trans.slave_idx = IMU_MPU9250_SPI_SLAVE_IDX;
156 imu_mpu9250.wait_slave4_trans.output_length = 1;
157 imu_mpu9250.wait_slave4_trans.input_length = 2;
158 imu_mpu9250.wait_slave4_trans.before_cb = NULL;
159 imu_mpu9250.wait_slave4_trans.after_cb = mpu_wait_slave4_ready_cb;
160 imu_mpu9250.wait_slave4_trans.input_buf = &(imu_mpu9250.wait_slave4_rx_buf[0]);
161 imu_mpu9250.wait_slave4_trans.output_buf = &(imu_mpu9250.wait_slave4_tx_buf[0]);
162
163 imu_mpu9250.wait_slave4_trans.status = SPITransDone;
164 imu_mpu9250.slave4_ready = false;
165}

172#define Int16FromBuf(_buf,_idx) ((int16_t)(_buf[_idx] | (_buf[_idx+1] << 8)))
173void imu_mpu9250_event(void)
174{
175 uint32_t now_ts = get_sys_time_usec();
176
177 // If the MPU9250 SPI transaction has succeeded: convert the data
178 mpu9250_spi_event(&imu_mpu9250.mpu);
179
180 if (imu_mpu9250.mpu.data_available) {
181 // set channel order
182 struct Int32Vect3 accel = {
183 IMU_MPU9250_X_SIGN * (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_X]),
184 IMU_MPU9250_Y_SIGN * (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Y]),
185 IMU_MPU9250_Z_SIGN * (int32_t)(imu_mpu9250.mpu.data_accel.value[IMU_MPU9250_CHAN_Z])
186 };
187 struct Int32Rates rates = {
188 IMU_MPU9250_X_SIGN * (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_X]),
189 IMU_MPU9250_Y_SIGN * (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Y]),
190 IMU_MPU9250_Z_SIGN * (int32_t)(imu_mpu9250.mpu.data_rates.value[IMU_MPU9250_CHAN_Z])
191 };
192
193#if IMU_MPU9250_READ_MAG
194 if (!bit_is_set(imu_mpu9250.mpu.data_ext[6], 3)) { //mag valid just HOFL == 0
196 struct Int32Vect3 mag;
197 mag.x = (IMU_MPU9250_X_SIGN) * Int16FromBuf(imu_mpu9250.mpu.data_ext, 2 * IMU_MPU9250_CHAN_Y);
198 mag.y = (IMU_MPU9250_Y_SIGN) * Int16FromBuf(imu_mpu9250.mpu.data_ext, 2 * IMU_MPU9250_CHAN_X);
199 mag.z = -(IMU_MPU9250_Z_SIGN) * Int16FromBuf(imu_mpu9250.mpu.data_ext, 2 * IMU_MPU9250_CHAN_Z);
200 AbiSendMsgIMU_MAG_RAW(IMU_MPU9250_ID, now_ts, &mag);
201 }
202#endif
203
204 imu_mpu9250.mpu.data_available = false;
205 AbiSendMsgIMU_GYRO_RAW(IMU_MPU9250_ID, now_ts, &rates, 1, IMU_MPU9250_PERIODIC_FREQ, NAN);
206 AbiSendMsgIMU_ACCEL_RAW(IMU_MPU9250_ID, now_ts, &accel, 1, IMU_MPU9250_PERIODIC_FREQ, NAN);
207 }
208
209}

221bool imu_mpu9250_configure_mag_slave(Mpu9250ConfigSet mpu_set, void *mpu)
222{
223 // wait before starting the configuration of the mag
224 // doing to early may void the mode configuration
225 if (get_sys_time_float() < IMU_MPU9250_MAG_STARTUP_DELAY) {
226 return false;
227 }
228
229 //config AK8963 soft reset
230 mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_ADDR, (MPU9250_MAG_ADDR >> 1));
231 mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_REG, AK8963_REG_CNTL2);
232 mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_DO, 1);
233 mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_CTRL, (1 << 7)); // Slave 4 enable
234
235 mpu_wait_slave4_ready();
236
237 // Set it to continious measuring mode 2
238 mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_ADDR, (MPU9250_MAG_ADDR >> 1));
239 mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_REG, AK8963_REG_CNTL1);
240 mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_DO, AK8963_CNTL1_CM_2);
241 mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV4_CTRL, (1 << 7)); // Slave 4 enable
242
243 mpu_wait_slave4_ready();
244
245 //Config SLV0 for continus Read
246 mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV0_ADDR, (MPU9250_MAG_ADDR >> 1) | MPU9250_SPI_READ);
247 mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV0_REG, AK8963_REG_HXL);
248 // Put the enable command as last.
249 mpu_set_and_wait(mpu_set, mpu, MPU9250_REG_I2C_SLV0_CTRL,
250 (1 << 7) | // Slave 0 enable
251 (7 << 0)); // Read 7 bytes (mag x,y,z + status)
252
253 return true;
254}

256void mpu_wait_slave4_ready(void)
257{
258 while (!imu_mpu9250.slave4_ready) {
259 if (imu_mpu9250.wait_slave4_trans.status == SPITransDone) {
260 imu_mpu9250.wait_slave4_tx_buf[0] = MPU9250_REG_I2C_MST_STATUS | MPU9250_SPI_READ;
261 spi_submit(imu_mpu9250.mpu.spi_p, &(imu_mpu9250.wait_slave4_trans));
262 }
263 }
264}
265
266void mpu_wait_slave4_ready_cb(struct spi_transaction *t)
267{
268 if (bit_is_set(t->input_buf[1], MPU9250_I2C_SLV4_DONE)) {
269 imu_mpu9250.slave4_ready = true;
270 } else {
271 imu_mpu9250.slave4_ready = false;
272 }
273 t->status = SPITransDone;
274}
