197void ins_ext_pose_msg_update(uint8_t *buf)
198{
199 if (DL_EXTERNAL_POSE_ac_id(buf) != AC_ID) { return; } // not for this aircraft
200
201 float enu_x = DL_EXTERNAL_POSE_enu_x(buf);
202 float enu_y = DL_EXTERNAL_POSE_enu_y(buf);
203 float enu_z = DL_EXTERNAL_POSE_enu_z(buf);
204 float enu_xd = DL_EXTERNAL_POSE_enu_xd(buf);
205 float enu_yd = DL_EXTERNAL_POSE_enu_yd(buf);
206 float enu_zd = DL_EXTERNAL_POSE_enu_zd(buf);
207 float quat_i = DL_EXTERNAL_POSE_body_qi(buf);
208 float quat_x = DL_EXTERNAL_POSE_body_qx(buf);
209 float quat_y = DL_EXTERNAL_POSE_body_qy(buf);
210 float quat_z = DL_EXTERNAL_POSE_body_qz(buf);
211
212 DEBUG_PRINT("EXT_UPDATE\n");
213
214 struct FloatQuat orient;
215 struct FloatEulers orient_eulers;
216
217 // Transformation of External Pose. Optitrack motive 2.X Yup
218 orient.qi = quat_i ;
219 orient.qx = quat_y ;
220 orient.qy = quat_x ;
221 orient.qz = -quat_z;
222
223#ifdef INS_EXT_VISION_ROTATION
224 // Rotate the quaternion
225 struct FloatQuat rot_q;
226 float_quat_comp(&rot_q, &orient, &ins_ext_vision_rot);
227 orient.qi = rot_q.qi;
228 orient.qx = rot_q.qx;
229 orient.qy = rot_q.qy;
230 orient.qz = rot_q.qz;
231#endif
232
233 float_eulers_of_quat(&orient_eulers, &orient);
234
235 ins_ext_pos.ev_time = get_sys_time_usec();
236 ins_ext_pos.ev_pos.x = enu_y;
237 ins_ext_pos.ev_pos.y = enu_x;
238 ins_ext_pos.ev_pos.z = -enu_z;
239 ins_ext_pos.ev_vel.x = enu_yd;
240 ins_ext_pos.ev_vel.y = enu_xd;
241 ins_ext_pos.ev_vel.z = -enu_zd;
242 ins_ext_pos.ev_att.phi = orient_eulers.phi;
243 ins_ext_pos.ev_att.theta = orient_eulers.theta;
244 ins_ext_pos.ev_att.psi = orient_eulers.psi;
245 ins_ext_pos.ev_quat.qi = orient.qi;
246 ins_ext_pos.ev_quat.qx = orient.qx;
247 ins_ext_pos.ev_quat.qy = orient.qy;
248 ins_ext_pos.ev_quat.qz = orient.qz;
249
250 ins_ext_pos.has_new_ext_pose = true;
251
252 DEBUG_PRINT("Att = %f %f %f \n", ins_ext_pos.ev_att.phi, ins_ext_pos.ev_att.theta, ins_ext_pos.ev_att.psi);
253}

800 float k3[EKF_NUM_STATES];
801 float k4[EKF_NUM_STATES];
802
803 float Xtmp[EKF_NUM_STATES];
804
805 // k1 = f(X,U)
806 ekf_f(X, U, k1);
807
808 // Xtmp = X+dt*k1/2
809 float_vect_smul(Xtmp, k1, dt / 2, EKF_NUM_STATES);

844static inline void ekf_step(const float U[EKF_NUM_INPUTS], const float Z[EKF_NUM_OUTPUTS], const float dt)
845{
846 // [1] Predicted (a priori) state estimate:
847 float Xkk_1[EKF_NUM_STATES];
848 ekf_f_rk4(ekf_X, U, dt, Xkk_1);
849
850
851 // [2] Get matrices
852 float F[EKF_NUM_STATES][EKF_NUM_STATES];
853 float L[EKF_NUM_STATES][EKF_NUM_INPUTS];
854 ekf_F(ekf_X, U, F);
855 ekf_L(ekf_X, U, L);
856
857
858 // [3] Continuous to discrete
859 // Fd = eye(N) + F*dt + F*F*dt**2/2 = I + [I+F*dt/2]*F*dt
860 // Ld = L*dt+F*L*dt**2/2 = [I+F*dt/2]*L*dt
861 float Fd[EKF_NUM_STATES][EKF_NUM_STATES];
862 float Ld[EKF_NUM_STATES][EKF_NUM_INPUTS];
863 float tmp[EKF_NUM_STATES][EKF_NUM_STATES];
864
865 MAKE_MATRIX_PTR(F_, F, EKF_NUM_STATES);
866 MAKE_MATRIX_PTR(L_, L, EKF_NUM_STATES);
867 MAKE_MATRIX_PTR(Fd_, Fd, EKF_NUM_STATES);
868 MAKE_MATRIX_PTR(Ld_, Ld, EKF_NUM_STATES);
869 MAKE_MATRIX_PTR(tmp_, tmp, EKF_NUM_STATES);
870
871 // tmp = I+F*dt/2
872 float_mat_diagonal_scal(tmp_, 1, EKF_NUM_STATES);
873 float_mat_sum_scaled(tmp_, F_, dt / 2, EKF_NUM_STATES, EKF_NUM_STATES);
874
875 // Ld = tmp*L*dt
876 float_mat_mul(Ld_, tmp_, L_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_INPUTS);
877 float_mat_scale(Ld_, dt, EKF_NUM_STATES, EKF_NUM_INPUTS);
878
879 // Fd = tmp*F*dt
880 float_mat_mul(Fd_, tmp_, F_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_STATES);
881 float_mat_scale(Fd_, dt, EKF_NUM_STATES, EKF_NUM_STATES);
882
883 // Fd += I
884 int i;
885 for (i = 0; i < EKF_NUM_STATES; i++) {
886 Fd[i][i] += 1;
887 }
888
889
890 // [4] Predicted covariance estimate:
891 // Pkk_1 = Fd*P*Fd.T + Ld*Q*Ld.T
892 float Pkk_1[EKF_NUM_STATES][EKF_NUM_STATES];
893 float LdT[EKF_NUM_INPUTS][EKF_NUM_STATES];
894 float QLdT[EKF_NUM_INPUTS][EKF_NUM_STATES];
895
896 MAKE_MATRIX_PTR(Pkk_1_, Pkk_1, EKF_NUM_STATES);
897 MAKE_MATRIX_PTR(ekf_P_, ekf_P, EKF_NUM_STATES);
898 MAKE_MATRIX_PTR(ekf_Q_, ekf_Q, EKF_NUM_STATES);
899 MAKE_MATRIX_PTR(LdT_, LdT, EKF_NUM_INPUTS);
900 MAKE_MATRIX_PTR(QLdT_, QLdT, EKF_NUM_INPUTS);
901
902 // Fd = Fd.T
903 float_mat_transpose_square(Fd_, EKF_NUM_STATES);
904
905 // tmp = P*Fd
906 float_mat_mul(tmp_, ekf_P_, Fd_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_STATES);
907
908 // Fd = Fd.T
909 float_mat_transpose_square(Fd_, EKF_NUM_STATES);
910
911 // Pkk_1 = Fd*tmp
912 float_mat_mul(Pkk_1_, Fd_, tmp_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_STATES);
913
914 // LdT = Ld.T
915 float_mat_transpose(LdT_, Ld_, EKF_NUM_STATES, EKF_NUM_INPUTS);
916
917 // QLdT = Q*LdT
918 float_mat_mul(QLdT_, ekf_Q_, LdT_, EKF_NUM_INPUTS, EKF_NUM_INPUTS, EKF_NUM_STATES);
919
920 // tmp = Ld*QLdT
921 float_mat_mul(tmp_, Ld_, QLdT_, EKF_NUM_STATES, EKF_NUM_INPUTS, EKF_NUM_STATES);
922
923 // Pkk_1 += tmp
924 float_mat_sum_scaled(Pkk_1_, tmp_, 1, EKF_NUM_STATES, EKF_NUM_STATES);
925
926
927 // [5] Measurement residual:
928 // yk = Z - H*Xkk_1
929 float yk[EKF_NUM_OUTPUTS];
930
931 MAKE_MATRIX_PTR(ekf_H_, ekf_H, EKF_NUM_OUTPUTS);
932
933 float_mat_vect_mul(yk, ekf_H_, Xkk_1, EKF_NUM_OUTPUTS, EKF_NUM_STATES);
934 float_vect_scale(yk, -1, EKF_NUM_OUTPUTS);
935 float_vect_add(yk, Z, EKF_NUM_OUTPUTS);
936
937
938 // [6] Residual covariance:
939 // Sk = H*Pkk_1*H.T + R
940 float Sk[EKF_NUM_OUTPUTS][EKF_NUM_OUTPUTS];
941 float PHT[EKF_NUM_STATES][EKF_NUM_OUTPUTS];
942
943 MAKE_MATRIX_PTR(Sk_, Sk, EKF_NUM_OUTPUTS);
944 MAKE_MATRIX_PTR(PHT_, PHT, EKF_NUM_STATES);
945 MAKE_MATRIX_PTR(ekf_R_, ekf_R, EKF_NUM_OUTPUTS);
946
947 // PHT = Pkk_1*H.T
948 float_mat_transpose(PHT_, ekf_H_, EKF_NUM_OUTPUTS, EKF_NUM_STATES);
949 float_mat_mul_copy(PHT_, Pkk_1_, PHT_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_OUTPUTS);
950
951 // Sk = H*PHT
952 float_mat_mul(Sk_, ekf_H_, PHT_, EKF_NUM_OUTPUTS, EKF_NUM_STATES, EKF_NUM_OUTPUTS);
953
954 // Sk += R
955 float_mat_sum_scaled(Sk_, ekf_R_, 1, EKF_NUM_OUTPUTS, EKF_NUM_OUTPUTS);
956
957
958 // [7] Near-optimal Kalman gain:
959 // K = Pkk_1*H.T*inv(Sk)
960 float Sk_inv[EKF_NUM_OUTPUTS][EKF_NUM_OUTPUTS];
961 float K[EKF_NUM_STATES][EKF_NUM_OUTPUTS];
962
963 MAKE_MATRIX_PTR(Sk_inv_, Sk_inv, EKF_NUM_OUTPUTS);
964 MAKE_MATRIX_PTR(K_, K, EKF_NUM_STATES);
965
966 // Sk_inv = inv(Sk)
967 float_mat_invert(Sk_inv_, Sk_, EKF_NUM_OUTPUTS);
968
969 // K = PHT*Sk_inv
970 float_mat_mul(K_, PHT_, Sk_inv_, EKF_NUM_STATES, EKF_NUM_OUTPUTS, EKF_NUM_OUTPUTS);
971
972
973 // [8] Updated state estimate
974 // Xkk = Xkk_1 + K*yk
975 float_mat_vect_mul(ekf_X, K_, yk, EKF_NUM_STATES, EKF_NUM_OUTPUTS);
976 float_vect_add(ekf_X, Xkk_1, EKF_NUM_STATES);
977
978
979 // [9] Updated covariance estimate:
980 // Pkk = (I - K*H)*Pkk_1
981
982 // tmp = K*H
983 float_mat_mul(tmp_, K_, ekf_H_, EKF_NUM_STATES, EKF_NUM_OUTPUTS, EKF_NUM_STATES);
984
985 // tmp *= -1
986 float_mat_scale(tmp_, -1, EKF_NUM_STATES, EKF_NUM_STATES);
987
988 // tmp += I
989 for (i = 0; i < EKF_NUM_STATES; i++) {
990 tmp_[i][i] += 1;
991 }
992 // P = tmp*Pkk_1
993 float_mat_mul(ekf_P_, tmp_, Pkk_1_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_STATES);
994}

1103 float PHT[EKF_NUM_STATES][EKF_NUM_OUTPUTS];
1104

1106 MAKE_MATRIX_PTR(PHT_, PHT, EKF_NUM_STATES);

1108

1111 float_mat_mul_copy(PHT_, Pkk_1_, PHT_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_OUTPUTS);
1112
1113 // Sk = H*PHT
1114 float_mat_mul(Sk_, ekf_H_, PHT_, EKF_NUM_OUTPUTS, EKF_NUM_STATES, EKF_NUM_OUTPUTS);
1115
1116 // Sk += R
1117 float_mat_sum_scaled(Sk_, ekf_R_, 1, EKF_NUM_OUTPUTS, EKF_NUM_OUTPUTS);
1118
1119

1121 // K = Pkk_1*H.T*inv(Sk)

1123 float K[EKF_NUM_STATES][EKF_NUM_OUTPUTS];

1129 float_mat_invert(Sk_inv_, Sk_, EKF_NUM_OUTPUTS);
1130
1131 // K = PHT*Sk_inv
1132 float_mat_mul(K_, PHT_, Sk_inv_, EKF_NUM_STATES, EKF_NUM_OUTPUTS, EKF_NUM_OUTPUTS);
1133
1134
1135 // [8] Updated state estimate
1136 // Xkk = Xkk_1 + K*yk
1137 float_mat_vect_mul(ekf_X, K_, yk, EKF_NUM_STATES, EKF_NUM_OUTPUTS);
1138 float_vect_add(ekf_X, Xkk_1, EKF_NUM_STATES);
1139
1140
1141 // [9] Updated covariance estimate:
1142 // Pkk = (I - K*H)*Pkk_1
1143 float tmp[EKF_NUM_STATES][EKF_NUM_STATES];
1144 MAKE_MATRIX_PTR(tmp_, tmp, EKF_NUM_STATES);
1145
1146 // tmp = K*H
1147 float_mat_mul(tmp_, K_, ekf_H_, EKF_NUM_STATES, EKF_NUM_OUTPUTS, EKF_NUM_STATES);
1148
1149 // tmp *= -1
1150 float_mat_scale(tmp_, -1, EKF_NUM_STATES, EKF_NUM_STATES);
1151
1152 // tmp += I
1153 int i;
1154 for (i = 0; i < EKF_NUM_STATES; i++) {
1155 tmp_[i][i] += 1;
1156 }
1157 // P = tmp*Pkk_1
1158 float_mat_mul(ekf_P_, tmp_, Pkk_1_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_STATES);
1159}
