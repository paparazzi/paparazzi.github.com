226void ins_ext_pose_msg_update(uint8_t *buf)
227{
228 if (DL_EXTERNAL_POSE_ac_id(buf) != AC_ID) { return; } // not for this aircraft
229
230 struct EnuCoor_f enu_pos, enu_vel;
231 enu_pos.x = DL_EXTERNAL_POSE_enu_x(buf);
232 enu_pos.y = DL_EXTERNAL_POSE_enu_y(buf);
233 enu_pos.z = DL_EXTERNAL_POSE_enu_z(buf);
234 enu_vel.x = DL_EXTERNAL_POSE_enu_xd(buf);
235 enu_vel.y = DL_EXTERNAL_POSE_enu_yd(buf);
236 enu_vel.z = DL_EXTERNAL_POSE_enu_zd(buf);
237 float quat_i = DL_EXTERNAL_POSE_body_qi(buf);
238 float quat_x = DL_EXTERNAL_POSE_body_qx(buf);
239 float quat_y = DL_EXTERNAL_POSE_body_qy(buf);
240 float quat_z = DL_EXTERNAL_POSE_body_qz(buf);
241
242 DEBUG_PRINT("EXT_UPDATE\n");
243
244 struct FloatQuat orient;
245 struct FloatEulers orient_eulers;
246
247 // Transformation of External Pose. Optitrack motive 2.X Yup
248 orient.qi = quat_i ;
249 orient.qx = quat_y ;
250 orient.qy = quat_x ;
251 orient.qz = -quat_z;
252
253#ifdef INS_EXT_VISION_ROTATION
254 // Rotate the quaternion
255 struct FloatQuat rot_q;
256 float_quat_comp(&rot_q, &orient, &ins_ext_vision_rot);
257 QUAT_COPY(orient, rot_q);
258#endif
259
260 float_eulers_of_quat(&orient_eulers, &orient);
261
262 ins_ext_pose.ev_time = get_sys_time_usec();
263 ENU_OF_TO_NED(ins_ext_pose.ev_pos, enu_pos);
264 ENU_OF_TO_NED(ins_ext_pose.ev_vel, enu_vel);
265 EULERS_COPY(ins_ext_pose.ev_att, orient_eulers);
266 QUAT_COPY(ins_ext_pose.ev_quat, orient);
267
268 ins_ext_pose.has_new_ext_pose = true;
269
270 DEBUG_PRINT("Att = %f %f %f \n", ins_ext_pose.ev_att.phi, ins_ext_pose.ev_att.theta, ins_ext_pose.ev_att.psi);
271}

499 out[EKF_X_VEL_Z][EKF_X_A_BIAS_X] = x4;

505 out[EKF_X_PHI][EKF_X_G_BIAS_Q] = -x32;
506 out[EKF_X_PHI][EKF_X_G_BIAS_R] = -x30;
507 out[EKF_X_THETA][EKF_X_PHI] = -x34 + x5 * (-U[EKF_U_GYRO_R] + X[EKF_X_G_BIAS_R]);
508 out[EKF_X_THETA][EKF_X_G_BIAS_Q] = -x5;
509 out[EKF_X_THETA][EKF_X_G_BIAS_R] = x1;
510 out[EKF_X_PSI][EKF_X_PHI] = x28 * x36 - x31 * x37;
511 out[EKF_X_PSI][EKF_X_THETA] = x27 * x31 * x38 + x34 * x38 * x4;
512 out[EKF_X_PSI][EKF_X_G_BIAS_Q] = -x37;
513 out[EKF_X_PSI][EKF_X_G_BIAS_R] = -x36;

516static void ekf_L(const float X[EKF_NUM_STATES], __attribute__((unused)) const float U[EKF_NUM_INPUTS],
517 float out[EKF_NUM_STATES][EKF_NUM_INPUTS])
518{
519 const float x0 = cosf(X[EKF_X_THETA]);
520 const float x1 = cosf(X[EKF_X_PSI]);
521 const float x2 = sinf(X[EKF_X_PSI]);
522 const float x3 = cosf(X[EKF_X_PHI]);
523 const float x4 = x2 * x3;
524 const float x5 = sinf(X[EKF_X_THETA]);
525 const float x6 = sinf(X[EKF_X_PHI]);
526 const float x7 = x1 * x6;
527 const float x8 = x2 * x6;
528 const float x9 = x1 * x3;
529 const float x10 = tanf(X[EKF_X_THETA]);
530 const float x11 = 1.f / x0; // FIXME protect against div by 0
531
532 memset(out, 0, (EKF_NUM_STATES*EKF_NUM_INPUTS) * (sizeof **out));
533 out[EKF_X_VEL_X][EKF_Z_POS_X] = -x0 * x1;
534 out[EKF_X_VEL_X][EKF_Z_POS_Y] = x4 - x5 * x7;
535 out[EKF_X_VEL_X][EKF_Z_POS_Z] = -x5 * x9 - x8;
536 out[EKF_X_VEL_Y][EKF_Z_POS_X] = -x0 * x2;
537 out[EKF_X_VEL_Y][EKF_Z_POS_Y] = -x5 * x8 - x9;
538 out[EKF_X_VEL_Y][EKF_Z_POS_Z] = -x4 * x5 + x7;
539 out[EKF_X_VEL_Z][EKF_Z_POS_X] = x5;
540 out[EKF_X_VEL_Z][EKF_Z_POS_Y] = -x0 * x6;

542 out[EKF_X_PHI][EKF_Z_PHI] = -1.f;
543 out[EKF_X_PHI][EKF_Z_THETA] = -x10 * x6;
544 out[EKF_X_PHI][EKF_Z_PSI] = -x10 * x3;
545 out[EKF_X_THETA][EKF_Z_THETA] = -x3;
546 out[EKF_X_THETA][EKF_Z_PSI] = x6;
547 out[EKF_X_PSI][EKF_Z_THETA] = -x11 * x6;
548 out[EKF_X_PSI][EKF_Z_PSI] = -x11 * x3;

554static void ekf_f_rk4(const float X[EKF_NUM_STATES], const float U[EKF_NUM_INPUTS], const float dt,
555 float out[EKF_NUM_STATES])
556{
557 float k1[EKF_NUM_STATES];
558 float k2[EKF_NUM_STATES];
559 float k3[EKF_NUM_STATES];
560 float k4[EKF_NUM_STATES];
561
562 float Xtmp[EKF_NUM_STATES];
563
564 // k1 = f(X,U)
565 ekf_f(X, U, k1);
566
567 // Xtmp = X+dt*k1/2
568 float_vect_smul(Xtmp, k1, dt / 2.f, EKF_NUM_STATES);
569 float_vect_add(Xtmp, X, EKF_NUM_STATES);
570
571 // k2 = f(Xtmp,U)
572 ekf_f(Xtmp, U, k2);
573
574 // Xtmp = X+dt*k2/2
575 float_vect_smul(Xtmp, k2, dt / 2.f, EKF_NUM_STATES);
576 float_vect_add(Xtmp, X, EKF_NUM_STATES);
577
578 // k3 = f(Xtmp,U)
579 ekf_f(Xtmp, U, k3);
580
581 // Xtmp = X+dt*k3
582 float_vect_smul(Xtmp, k3, dt, EKF_NUM_STATES);
583 float_vect_add(Xtmp, X, EKF_NUM_STATES);
584
585 // k4 = f(Xtmp,U)
586 ekf_f(Xtmp, U, k4);
587
588 // out = k2+k3
589 float_vect_sum(out, k2, k3, EKF_NUM_STATES);
590 // out *= 2
591 float_vect_scale(out, 2.f, EKF_NUM_STATES);
592 // out += k1
593 float_vect_add(out, k1, EKF_NUM_STATES);
594 // out += k4
595 float_vect_add(out, k4, EKF_NUM_STATES);
596 // out *= dt/6
597 float_vect_scale(out, dt / 6.f, EKF_NUM_STATES);
598 // out += X
599 float_vect_add(out, X, EKF_NUM_STATES);
600}

604static void ekf_prediction_step(const float U[EKF_NUM_INPUTS], const float dt)
605{
606 float tmp[EKF_NUM_STATES][EKF_NUM_STATES];
607 MAKE_MATRIX_PTR(tmp_, tmp, EKF_NUM_STATES);
608
609#if INS_EXT_POSE_USE_RK4
610 // [1] Predicted (a priori) state estimate:
611 float Xkk_1[EKF_NUM_STATES];
612 ekf_f_rk4(ekf_X, U, dt, Xkk_1);
613
614 // [2] Get matrices
615 float F[EKF_NUM_STATES][EKF_NUM_STATES];
616 float L[EKF_NUM_STATES][EKF_NUM_INPUTS];
617 ekf_F(ekf_X, U, F);
618 ekf_L(ekf_X, U, L);
619
620
621 // [3] Continuous to discrete
622 // Fd = eye(N) + F*dt + F*F*dt**2/2 = I + [I+F*dt/2]*F*dt
623 // Ld = L*dt+F*L*dt**2/2 = [I+F*dt/2]*L*dt
624 float Fd[EKF_NUM_STATES][EKF_NUM_STATES];
625 float Ld[EKF_NUM_STATES][EKF_NUM_INPUTS];
626
627 MAKE_MATRIX_PTR(F_, F, EKF_NUM_STATES);
628 MAKE_MATRIX_PTR(L_, L, EKF_NUM_STATES);
629 MAKE_MATRIX_PTR(Fd_, Fd, EKF_NUM_STATES);
630 MAKE_MATRIX_PTR(Ld_, Ld, EKF_NUM_STATES);
631
632 // tmp = I+F*dt/2
633 float_mat_diagonal_scal(tmp_, 1.f, EKF_NUM_STATES);
634 float_mat_sum_scaled(tmp_, F_, dt / 2.f, EKF_NUM_STATES, EKF_NUM_STATES);
635
636 // Ld = tmp*L*dt
637 float_mat_mul(Ld_, tmp_, L_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_INPUTS);
638 float_mat_scale(Ld_, dt, EKF_NUM_STATES, EKF_NUM_INPUTS);
639
640 // Fd = tmp*F*dt
641 float_mat_mul(Fd_, tmp_, F_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_STATES);
642 float_mat_scale(Fd_, dt, EKF_NUM_STATES, EKF_NUM_STATES);
643
644 // Fd += I
645 int i;
646 for (i = 0; i < EKF_NUM_STATES; i++) {
647 Fd[i][i] += 1.f;
648 }
649
650#else
651
652 // [1] Predicted (a priori) state estimate:
653 float Xkk_1[EKF_NUM_STATES];
654 // Xkk_1 = f(X,U)
655 ekf_f(ekf_X, U, Xkk_1);
656 // Xkk_1 *= dt
657 float_vect_scale(Xkk_1, dt, EKF_NUM_STATES);
658 // Xkk_1 += X
659 float_vect_add(Xkk_1, ekf_X, EKF_NUM_STATES);
660
661
662 // [2] Get matrices
663 float F[EKF_NUM_STATES][EKF_NUM_STATES];
664 float Ld[EKF_NUM_STATES][EKF_NUM_INPUTS];
665 ekf_F(ekf_X, U, F);
666 ekf_L(ekf_X, U, Ld);
667
668
669 // [3] Continuous to discrete
670 // Fd = eye(N) + F*dt
671 // Ld = L*dt
672 float Fd[EKF_NUM_STATES][EKF_NUM_STATES];
673
674 MAKE_MATRIX_PTR(F_, F, EKF_NUM_STATES);
675 MAKE_MATRIX_PTR(Fd_, Fd, EKF_NUM_STATES);
676 MAKE_MATRIX_PTR(Ld_, Ld, EKF_NUM_STATES);
677
678 // Fd = I+F*dt
679 float_mat_diagonal_scal(Fd_, 1.f, EKF_NUM_STATES);
680 float_mat_sum_scaled(Fd_, F_, dt, EKF_NUM_STATES, EKF_NUM_STATES);
681
682 // Ld = Ld*dt
683 float_mat_scale(Ld_, dt, EKF_NUM_STATES, EKF_NUM_INPUTS);
684#endif
685
686 // [4] Predicted covariance estimate:
687 // Pkk_1 = Fd*P*Fd.T + Ld*Q*Ld.T
688 float Pkk_1[EKF_NUM_STATES][EKF_NUM_STATES];
689 float LdT[EKF_NUM_INPUTS][EKF_NUM_STATES];
690 float QLdT[EKF_NUM_INPUTS][EKF_NUM_STATES];
691
692 MAKE_MATRIX_PTR(Pkk_1_, Pkk_1, EKF_NUM_STATES);
693 MAKE_MATRIX_PTR(ekf_P_, ekf_P, EKF_NUM_STATES);
694 MAKE_MATRIX_PTR(ekf_Q_, ekf_Q, EKF_NUM_STATES);
695 MAKE_MATRIX_PTR(LdT_, LdT, EKF_NUM_INPUTS);
696 MAKE_MATRIX_PTR(QLdT_, QLdT, EKF_NUM_INPUTS);
697
698 // Fd = Fd.T
699 float_mat_transpose_square(Fd_, EKF_NUM_STATES);
700
701 // tmp = P*Fd
702 float_mat_mul(tmp_, ekf_P_, Fd_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_STATES);
703
704 // Fd = Fd.T
705 float_mat_transpose_square(Fd_, EKF_NUM_STATES);
706
707 // Pkk_1 = Fd*tmp
708 float_mat_mul(Pkk_1_, Fd_, tmp_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_STATES);
709
710 // LdT = Ld.T
711 float_mat_transpose(LdT_, Ld_, EKF_NUM_STATES, EKF_NUM_INPUTS);
712
713 // QLdT = Q*LdT
714 float_mat_mul(QLdT_, ekf_Q_, LdT_, EKF_NUM_INPUTS, EKF_NUM_INPUTS, EKF_NUM_STATES);
715
716 // tmp = Ld*QLdT
717 float_mat_mul(tmp_, Ld_, QLdT_, EKF_NUM_STATES, EKF_NUM_INPUTS, EKF_NUM_STATES);
718
719 // Pkk_1 += tmp
720 float_mat_sum_scaled(Pkk_1_, tmp_, 1, EKF_NUM_STATES, EKF_NUM_STATES);
721
722 // Store state and covariance
723
724 // X = Xkk_1
725 float_vect_copy(ekf_X, Xkk_1, EKF_NUM_STATES);
726
727 // P = Pkk_1
728 float_mat_copy(ekf_P_, Pkk_1_, EKF_NUM_STATES, EKF_NUM_STATES);
729}

731static void ekf_measurement_step(const float Z[EKF_NUM_OUTPUTS])
732{
733 // Xkk_1 = X
734 float Xkk_1[EKF_NUM_STATES];
735 float_vect_copy(Xkk_1, ekf_X, EKF_NUM_STATES);
736
737 // Pkk_1 = P
738 float Pkk_1[EKF_NUM_STATES][EKF_NUM_STATES];
739 MAKE_MATRIX_PTR(Pkk_1_, Pkk_1, EKF_NUM_STATES);
740 MAKE_MATRIX_PTR(ekf_P_, ekf_P, EKF_NUM_STATES);
741 float_mat_copy(Pkk_1_, ekf_P_, EKF_NUM_STATES, EKF_NUM_STATES);
742
743 // [5] Measurement residual:
744 // yk = Z - H*Xkk_1
745 float yk[EKF_NUM_OUTPUTS];
746
747 MAKE_MATRIX_PTR(ekf_H_, ekf_H, EKF_NUM_OUTPUTS);
748
749 float_mat_vect_mul(yk, ekf_H_, Xkk_1, EKF_NUM_OUTPUTS, EKF_NUM_STATES);
750 float_vect_scale(yk, -1.f, EKF_NUM_OUTPUTS);
751 float_vect_add(yk, Z, EKF_NUM_OUTPUTS);
752
753
754 // [6] Residual covariance:
755 // Sk = H*Pkk_1*H.T + R
756 float Sk[EKF_NUM_OUTPUTS][EKF_NUM_OUTPUTS];
757 float PHT[EKF_NUM_STATES][EKF_NUM_OUTPUTS];
758
759 MAKE_MATRIX_PTR(Sk_, Sk, EKF_NUM_OUTPUTS);
760 MAKE_MATRIX_PTR(PHT_, PHT, EKF_NUM_STATES);
761 MAKE_MATRIX_PTR(ekf_R_, ekf_R, EKF_NUM_OUTPUTS);
762
763 // PHT = Pkk_1*H.T
764 float_mat_transpose(PHT_, ekf_H_, EKF_NUM_OUTPUTS, EKF_NUM_STATES);
765 float_mat_mul_copy(PHT_, Pkk_1_, PHT_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_OUTPUTS);
766
767 // Sk = H*PHT
768 float_mat_mul(Sk_, ekf_H_, PHT_, EKF_NUM_OUTPUTS, EKF_NUM_STATES, EKF_NUM_OUTPUTS);
769
770 // Sk += R
771 float_mat_sum_scaled(Sk_, ekf_R_, 1.f, EKF_NUM_OUTPUTS, EKF_NUM_OUTPUTS);
772
773
774 // [7] Near-optimal Kalman gain:
775 // K = Pkk_1*H.T*inv(Sk)
776 float Sk_inv[EKF_NUM_OUTPUTS][EKF_NUM_OUTPUTS];
777 float K[EKF_NUM_STATES][EKF_NUM_OUTPUTS];
778
779 MAKE_MATRIX_PTR(Sk_inv_, Sk_inv, EKF_NUM_OUTPUTS);
780 MAKE_MATRIX_PTR(K_, K, EKF_NUM_STATES);
781
782 // Sk_inv = inv(Sk)
783 float_mat_invert(Sk_inv_, Sk_, EKF_NUM_OUTPUTS);
784
785 // K = PHT*Sk_inv
786 float_mat_mul(K_, PHT_, Sk_inv_, EKF_NUM_STATES, EKF_NUM_OUTPUTS, EKF_NUM_OUTPUTS);
787
788
789 // [8] Updated state estimate
790 // Xkk = Xkk_1 + K*yk
791 float_mat_vect_mul(ekf_X, K_, yk, EKF_NUM_STATES, EKF_NUM_OUTPUTS);
792 float_vect_add(ekf_X, Xkk_1, EKF_NUM_STATES);
793
794
795 // [9] Updated covariance estimate:
796 // Pkk = (I - K*H)*Pkk_1
797 float tmp[EKF_NUM_STATES][EKF_NUM_STATES];
798 MAKE_MATRIX_PTR(tmp_, tmp, EKF_NUM_STATES);
799
800 // tmp = K*H
801 float_mat_mul(tmp_, K_, ekf_H_, EKF_NUM_STATES, EKF_NUM_OUTPUTS, EKF_NUM_STATES);
802
803 // tmp *= -1
804 float_mat_scale(tmp_, -1.f, EKF_NUM_STATES, EKF_NUM_STATES);
805
806 // tmp += I
807 int i;
808 for (i = 0; i < EKF_NUM_STATES; i++) {
809 tmp_[i][i] += 1.f;
810 }
811 // P = tmp*Pkk_1
812 float_mat_mul(ekf_P_, tmp_, Pkk_1_, EKF_NUM_STATES, EKF_NUM_STATES, EKF_NUM_STATES);
813}

816static void ekf_run(void)
817{
818 // Time
819 t1 = get_sys_time_float();
820 float dt = t1 - t0;
821 t0 = t1;
822
823 // Only Start If External Pose is Available
824 if (!ins_ext_pose.started) {
825 // ekf starts at the first ev update
826 if (ins_ext_pose.has_new_ext_pose) {
827 ins_ext_pose.started = true;
828
829 // initial guess
830 ekf_X[EKF_X_POS_X] = ins_ext_pose.ev_pos.x;
831 ekf_X[EKF_X_POS_Y] = ins_ext_pose.ev_pos.y;
832 ekf_X[EKF_X_POS_Z] = ins_ext_pose.ev_pos.z;
833 ekf_X[EKF_X_PHI] = ins_ext_pose.ev_att.phi;
834 ekf_X[EKF_X_THETA] = ins_ext_pose.ev_att.theta;
835 ekf_X[EKF_X_PSI] = ins_ext_pose.ev_att.psi;
836 }
837 }
838
839 // set input values
840 if (ins_ext_pose.has_new_acc) {
841 ekf_U[EKF_U_ACC_X] = ins_ext_pose.accels_f.x;
842 ekf_U[EKF_U_ACC_Y] = ins_ext_pose.accels_f.y;
843 ekf_U[EKF_U_ACC_Z] = ins_ext_pose.accels_f.z;
844 ins_ext_pose.has_new_acc = false;
845 } else {
846 DEBUG_PRINT("ekf missing acc\n");
847 }
848 if (ins_ext_pose.has_new_gyro) {
849 ekf_U[EKF_U_GYRO_P] = ins_ext_pose.gyros_f.p;
850 ekf_U[EKF_U_GYRO_Q] = ins_ext_pose.gyros_f.q;
851 ekf_U[EKF_U_GYRO_R] = ins_ext_pose.gyros_f.r;
852 ins_ext_pose.has_new_gyro = false;
853 } else {
854 DEBUG_PRINT("ekf missing gyro\n");
855 }
856
857 if (ins_ext_pose.started) {
858
859 // prediction step
860 ekf_prediction_step(ekf_U, dt);
861
862 DEBUG_PRINT("ekf prediction step U = %f, %f, %f, %f, %f, %f dt = %f \n",
863 ekf_U[0], ekf_U[1], ekf_U[2], ekf_U[3], ekf_U[4], ekf_U[5], dt);
864
865 // measurement step
866 if (ins_ext_pose.has_new_ext_pose) {
867
868 //fix psi
869 static float last_psi = 0;
870 float delta_psi = ins_ext_pose.ev_att.psi - last_psi;
871 last_psi = ins_ext_pose.ev_att.psi;
872
873 if (delta_psi > M_PI) {
874 delta_psi -= 2.f * M_PI;
875 } else if (delta_psi < -M_PI) {
876 delta_psi += 2.f * M_PI;
877 }
878
879 ekf_Z[EKF_Z_POS_X] = ins_ext_pose.ev_pos.x;
880 ekf_Z[EKF_Z_POS_Y] = ins_ext_pose.ev_pos.y;
881 ekf_Z[EKF_Z_POS_Z] = ins_ext_pose.ev_pos.z;
882 ekf_Z[EKF_Z_PHI] = ins_ext_pose.ev_att.phi;
883 ekf_Z[EKF_Z_THETA] = ins_ext_pose.ev_att.theta;
884 ekf_Z[EKF_Z_PSI] += delta_psi;
885 ins_ext_pose.has_new_ext_pose = false;
886
887 ekf_measurement_step(ekf_Z);
888
889 DEBUG_PRINT("ekf measurement step Z = %f, %f, %f, %f \n",
890 ekf_Z[0], ekf_Z[1], ekf_Z[2], ekf_Z[3]);
891 }
892 }
