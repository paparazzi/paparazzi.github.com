73static float transition_ratio;
74static const float transition_increment = 1.f / (CMH_TRANSITION_TIME * PERIODIC_FREQUENCY);
75
76static void transition_run(bool to_forward) {
77 if (to_forward && transition_ratio < 1.f) {
78 if (stateIsAirspeedValid()) {
79 if ((stateGetAirspeed_f() < CMH_TRANSITION_AIRSPEED && transition_ratio < 0.4f)
80 || stateGetAirspeed_f() >= CMH_TRANSITION_AIRSPEED) {
81 transition_ratio += transition_increment;
82 }
83 else if (stateGetAirspeed_f() < CMH_TRANSITION_AIRSPEED && transition_ratio > 0.4f) {
84 transition_ratio -= transition_increment;
85 }
86 }
87 } else if (!to_forward && transition_ratio > 0.f) {
88 transition_ratio = 0.f; // immediately switch back to hover
89 }
90 Bound(transition_ratio, 0.f, 1.f);
91}
92
93static int32_t command_from_transition(int32_t hover_cmd, int32_t forward_cmd) {
94 Bound(transition_ratio, 0.f, 1.f);
95 return (int32_t) (hover_cmd + transition_ratio * (forward_cmd - hover_cmd));
96}

103void control_mixing_heewing_manual(void)
104{
105 transition_run(TRANSITION_TO_FORWARD);
106 commands[COMMAND_ROLL] = radio_control_get(RADIO_ROLL);
107 commands[COMMAND_PITCH] = radio_control_get(RADIO_PITCH);
108 commands[COMMAND_YAW] = 0;
109 commands[COMMAND_TILT] = CMH_TILT_FORWARD;
110 commands[COMMAND_MOTOR_RIGHT] = radio_control_get(RADIO_THROTTLE);
111 commands[COMMAND_MOTOR_LEFT] = radio_control_get(RADIO_THROTTLE);
112 commands[COMMAND_MOTOR_TAIL] = MIN_PPRZ;
113 commands[COMMAND_THRUST] = (commands[COMMAND_MOTOR_RIGHT] + commands[COMMAND_MOTOR_LEFT]) / 2;
114 autopilot.throttle = commands[COMMAND_THRUST];
115}

117void control_mixing_heewing_attitude_direct(void)
118{
119 transition_run(TRANSITION_TO_HOVER);
120 commands[COMMAND_TILT] = CMH_TILT_VERTICAL;
121 commands[COMMAND_ROLL] = 0;
122 commands[COMMAND_PITCH] = 0;
123 struct ThrustSetpoint th_sp = guidance_v_run(autopilot_in_flight());
124 stabilization_run(autopilot_in_flight(), &stabilization.rc_sp, &th_sp, stabilization.cmd);
125 if (autopilot_get_motors_on()) {
126 commands[COMMAND_MOTOR_RIGHT] = actuators_pprz[CMH_ACT_MOTOR_RIGHT];
127 commands[COMMAND_MOTOR_LEFT] = actuators_pprz[CMH_ACT_MOTOR_LEFT];
128 commands[COMMAND_MOTOR_TAIL] = actuators_pprz[CMH_ACT_MOTOR_TAIL];
129 if (autopilot_in_flight()) {
130 commands[COMMAND_YAW] = actuators_pprz[CMH_ACT_YAW];
131 } else {
132 commands[CMH_ACT_YAW] = 0;
133 }
134 commands[COMMAND_THRUST] = stabilization.cmd[COMMAND_THRUST];
135 } else {
136 commands[COMMAND_MOTOR_RIGHT] = -MAX_PPRZ;
137 commands[COMMAND_MOTOR_LEFT] = -MAX_PPRZ;
138 commands[COMMAND_MOTOR_TAIL] = -MAX_PPRZ;
139 commands[COMMAND_YAW] = 0;
140 commands[COMMAND_THRUST] = 0;
141 }
142 autopilot.throttle = commands[COMMAND_THRUST];
143}

182void control_mixing_heewing_attitude_plane(void)
183{
184 transition_run(TRANSITION_TO_FORWARD);
185
186 if (transition_ratio < 0.5f) {
187 struct ThrustSetpoint hover_th_sp = guidance_v_run(autopilot_in_flight());
188 stabilization_run(autopilot_in_flight(), &stabilization.rc_sp, &hover_th_sp, stabilization.cmd);
189 }
190 struct ThrustSetpoint th_sp = th_sp_from_thrust_i(radio_control_get(RADIO_THROTTLE), THRUST_AXIS_X);
191 struct FloatEulers rc_sp = {
192 .phi = STABILIZATION_ATTITUDE_SP_MAX_PHI * radio_control_get(RADIO_ROLL) / MAX_PPRZ,
193 .theta = STABILIZATION_ATTITUDE_SP_MAX_THETA * radio_control_get(RADIO_PITCH) / MAX_PPRZ,
194 .psi = 0.f
195 };
196 struct StabilizationSetpoint stab_sp = stab_sp_from_eulers_f(&rc_sp);
197 stabilization_attitude_plane_pid_run(transition_ratio < 0.8 ? false : autopilot_in_flight(), &stab_sp, &th_sp, stabilization.cmd);
198
199 commands[COMMAND_ROLL] = stabilization.cmd[COMMAND_ROLL];
200 commands[COMMAND_PITCH] = stabilization.cmd[COMMAND_PITCH];
201 commands[COMMAND_THRUST] = stabilization.cmd[COMMAND_THRUST];
202 if (autopilot_in_flight()) {
203 if (transition_ratio < 0.5f) {
204 // only hover stabilization
205 commands[COMMAND_MOTOR_RIGHT] = actuators_pprz[CMH_ACT_MOTOR_RIGHT];
206 commands[COMMAND_MOTOR_LEFT] = actuators_pprz[CMH_ACT_MOTOR_LEFT];
207 commands[COMMAND_MOTOR_TAIL] = actuators_pprz[CMH_ACT_MOTOR_TAIL];
208 } else {
209 // blend with plane control
210 commands[COMMAND_MOTOR_RIGHT] = command_from_transition(actuators_pprz[CMH_ACT_MOTOR_RIGHT], stabilization.cmd[COMMAND_THRUST]);
211 commands[COMMAND_MOTOR_LEFT] = command_from_transition(actuators_pprz[CMH_ACT_MOTOR_LEFT], stabilization.cmd[COMMAND_THRUST]);
212 commands[COMMAND_MOTOR_TAIL] = command_from_transition(actuators_pprz[CMH_ACT_MOTOR_TAIL], 0);
213 }
214 } else {
215 commands[COMMAND_MOTOR_RIGHT] = stabilization.cmd[COMMAND_THRUST];
216 commands[COMMAND_MOTOR_LEFT] = stabilization.cmd[COMMAND_THRUST];
217 commands[COMMAND_MOTOR_TAIL] = 0;
218 }
219 commands[COMMAND_TILT] = command_from_transition(CMH_TILT_VERTICAL, CMH_TILT_FORWARD);
220 commands[COMMAND_YAW] = 0;
221 autopilot.throttle = commands[COMMAND_THRUST];
222
223}

232void control_mixing_heewing_nav_run(void)
233{
234 if (nav.horizontal_mode == NAV_HORIZONTAL_MODE_ROUTE ||
235 nav.horizontal_mode == NAV_HORIZONTAL_MODE_CIRCLE) {
236 transition_run(TRANSITION_TO_FORWARD);
237
238 struct ThrustSetpoint th_sp = guidance_plane_thrust_from_nav(transition_ratio < 0.8f ? false : autopilot_in_flight());
239 if (transition_ratio < 0.8f) {
240 guidance_plane.pitch_cmd = 0.f;
241 }
242 struct StabilizationSetpoint stab_sp = guidance_plane_attitude_from_nav(autopilot_in_flight());
243 if (transition_ratio < 0.5f) {
244 stabilization_attitude_plane_pid_run(transition_ratio < 0.8f ? false : autopilot_in_flight(), &stab_sp, &th_sp, stabilization.cmd);
245 struct ThrustSetpoint hover_th_sp = guidance_v_run(autopilot_in_flight());
246 struct FloatEulers eulers_sp = { .phi = 0.f , .theta = 0.f, .psi = stateGetNedToBodyEulers_f()->psi };
247 struct StabilizationSetpoint hover_stab_sp = stab_sp_from_eulers_f(&eulers_sp);
248 stabilization_run(autopilot_in_flight(), &hover_stab_sp, &hover_th_sp, stabilization.cmd); // will overwrite COMMAND_THRUST
249 } else {
250 stabilization_attitude_plane_pid_run(transition_ratio < 0.8f ? false : autopilot_in_flight(), &stab_sp, &th_sp, stabilization.cmd);
251 }
252 nav.heading = stateGetNedToBodyEulers_f()->psi; // overwrite nav heading to avoid problems when transition to hover
253
254 commands[COMMAND_TILT] = command_from_transition(CMH_TILT_VERTICAL, CMH_TILT_FORWARD);
255 commands[COMMAND_ROLL] = stabilization.cmd[COMMAND_ROLL];
256 commands[COMMAND_PITCH] = stabilization.cmd[COMMAND_PITCH];
257 commands[COMMAND_YAW] = command_from_transition(actuators_pprz[CMH_ACT_YAW], 0);
258 if (autopilot_get_motors_on()) {
259 if (transition_ratio < 0.5f) {
260 // only hover stabilization
261 commands[COMMAND_MOTOR_RIGHT] = actuators_pprz[CMH_ACT_MOTOR_RIGHT];
262 commands[COMMAND_MOTOR_LEFT] = actuators_pprz[CMH_ACT_MOTOR_LEFT];
263 commands[COMMAND_MOTOR_TAIL] = actuators_pprz[CMH_ACT_MOTOR_TAIL];
264 } else {
265 // blend with plane control
266 commands[COMMAND_MOTOR_RIGHT] = command_from_transition(actuators_pprz[CMH_ACT_MOTOR_RIGHT], stabilization.cmd[COMMAND_THRUST]);
267 commands[COMMAND_MOTOR_LEFT] = command_from_transition(actuators_pprz[CMH_ACT_MOTOR_LEFT], stabilization.cmd[COMMAND_THRUST]);
268 commands[COMMAND_MOTOR_TAIL] = command_from_transition(actuators_pprz[CMH_ACT_MOTOR_TAIL], 0);
269 }
270 commands[COMMAND_THRUST] = stabilization.cmd[COMMAND_THRUST];
271 } else {
272 commands[COMMAND_THRUST] = 0;
273 commands[COMMAND_MOTOR_RIGHT] = MIN_PPRZ;
274 commands[COMMAND_MOTOR_LEFT] = MIN_PPRZ;
275 commands[COMMAND_MOTOR_TAIL] = 0;
276 }
277
278 } else {
279 // all other nav modes are in rotorcraft flight mode
280 transition_run(TRANSITION_TO_HOVER);
281 commands[COMMAND_TILT] = command_from_transition(CMH_TILT_VERTICAL, CMH_TILT_FORWARD);
282 commands[COMMAND_ROLL] = 0;
283 commands[COMMAND_PITCH] = 0;
284
285 struct ThrustSetpoint th_sp = guidance_v_run(autopilot_in_flight());
286 struct StabilizationSetpoint stab_sp = guidance_h_run(autopilot_in_flight());
287 stabilization_run(autopilot_in_flight(), &stab_sp, &th_sp, stabilization.cmd);
288
289 if (autopilot_get_motors_on()) {
290 commands[COMMAND_MOTOR_RIGHT] = actuators_pprz[CMH_ACT_MOTOR_RIGHT];
291 commands[COMMAND_MOTOR_LEFT] = actuators_pprz[CMH_ACT_MOTOR_LEFT];
292 commands[COMMAND_MOTOR_TAIL] = command_from_transition(actuators_pprz[CMH_ACT_MOTOR_TAIL], 0);
293 if (autopilot_in_flight()) {
294 commands[COMMAND_YAW] = actuators_pprz[CMH_ACT_YAW];
295 } else {
296 commands[CMH_ACT_YAW] = 0;
297 }
298 commands[COMMAND_THRUST] = stabilization.cmd[COMMAND_THRUST];
299 } else {
300 commands[COMMAND_MOTOR_RIGHT] = -MAX_PPRZ;
301 commands[COMMAND_MOTOR_LEFT] = -MAX_PPRZ;
302 commands[COMMAND_MOTOR_TAIL] = -MAX_PPRZ;
303 commands[COMMAND_YAW] = 0;
304 commands[COMMAND_THRUST] = 0;
305 }
306 }
307 autopilot.throttle = commands[COMMAND_THRUST];
308}
