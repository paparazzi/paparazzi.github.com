101 #if MS45XX_OUTPUT_TYPE == 0
102 /* Offset and scaling for OUTPUT TYPE A:
103  * p_raw = (0.8*16383)/ (Pmax - Pmin) * (pressure - Pmin) + 0.1*16383
104  * For differential sensors Pmax = MS45XX_PRESSURE_RANGE = -Pmin.
105  *
106  * p_diff = (p_raw - 0.1*16383) * 2*RANGE/(0.8*16383) - RANGE
107  * p_diff = p_raw * 2*RANGE/(0.8*16383) - (RANGE + (0.1 * 16383) * 2*RANGE/(0.8*16383)
108  * p_diff = p_raw * 2*RANGE/(0.8*16383) - (1.25 * RANGE)
109  * p_diff = (p_raw - offset) * scale
110  * then convert to Pascal
111  */
112 #ifndef MS45XX_PRESSURE_SCALE
113 #define MS45XX_PRESSURE_SCALE (2 * MS45XX_PRESSURE_RANGE / (0.8 * 16383) * OutputPressureToPa)
114 #endif
115 #ifndef MS45XX_PRESSURE_OFFSET
116 #define MS45XX_PRESSURE_OFFSET (16383 / 2)
117 #endif
118 #else /* Can still be improved using another if statment with MS45XX_PRESSURE_TYPE etc. */
119 /* Offset and scaling for OUTPUT TYPE B:
120  * p_raw = (0.9*16383)/ (Pmax - Pmin) * (pressure - Pmin) + 0.05*16383
121  */
122 #ifndef MS45XX_PRESSURE_SCALE
123 #define MS45XX_PRESSURE_SCALE (MS45XX_PRESSURE_RANGE/(0.9*16383)*OutputPressureToPa)
124 #endif
125 #ifndef MS45XX_PRESSURE_OFFSET
126 #define MS45XX_PRESSURE_OFFSET (0.05*16383)
127 #endif
128 #endif

227 void ms45xx_i2c_event(void)
228 {
229  static int autoset_nb = 0;
230  static float autoset_offset = 0.f;
231 
232  /* Check if transaction is succesfull */
233  if (ms45xx_trans.status == I2CTransSuccess) {
234 
235  /* 2 MSB of data are status bits, 0 = good data, 2 = already fetched, 3 = fault */
236  uint8_t status = (0xC0 & ms45xx_trans.buf[0]) >> 6;
237 
238  if (status == 0) {
239  /* 14bit raw pressure */
240  uint16_t p_raw = 0x3FFF & (((uint16_t)(ms45xx_trans.buf[0]) << 8) | (uint16_t)(ms45xx_trans.buf[1]));
241  ms45xx.raw_p = p_raw;
242 
243  /* 11bit raw temperature, 5 LSB bits not used */
244  uint16_t temp_raw = 0xFFE0 & (((uint16_t)(ms45xx_trans.buf[2]) << 8) |
245  (uint16_t)(ms45xx_trans.buf[3]));
246  temp_raw = temp_raw >> 5;
247 
248  /* Reject any values that are the absolute minimum or maximums these
249  can happen due to gnd lifts or communication errors on the bus */
250  if(p_raw == 0x3FFF || p_raw == 0 || temp_raw == 0x7FF || temp_raw == 0) {
251  ms45xx_trans.status = I2CTransDone;
252  return;
253  }
254 
255  /* For type Diff
256  * Output is proportional to the difference between Port 1 and Port 2. Output
257  * swings positive when Port 1> Port 2. Output is 50% of total counts
258  * when Port 1=Port 2.
259  * For type Gauge
260  * p_out = (p_raw - offset) * scale
261  */
262 
263  float p_out = (p_raw - ms45xx.pressure_offset) * ms45xx.pressure_scale;
264 #ifdef USE_AIRSPEED_LOWPASS_FILTER
265  ms45xx.pressure = update_butterworth_2_low_pass(&ms45xx_filter, p_out);
266 #else
267  ms45xx.pressure = p_out;
268 #endif
269 
270  if (ms45xx.autoset_offset) {
271  if (autoset_nb < AUTOSET_NB_MAX) {
272  autoset_offset += p_raw;
273  autoset_nb++;
274  } else {
275  ms45xx.pressure_offset = autoset_offset / (float)autoset_nb;
276  autoset_offset = 0.f;
277  autoset_nb = 0;
278  ms45xx.autoset_offset = false;
279  ms45xx.offset_set = true;
280  }
281  }
282 
283  /* 0 = -50degC, 20147 = 150degC
284  * ms45xx_temperature in 0.1 deg Celcius
285  */
286  ms45xx.temperature = ((uint32_t)temp_raw * 2000) / 2047 - 500;
287 
288  // Send (differential) pressure via ABI
289  AbiSendMsgBARO_DIFF(MS45XX_SENDER_ID, ms45xx.pressure);
290  // Send temperature as float in deg Celcius via ABI
291  float temp = ms45xx.temperature / 10.0f;
292  AbiSendMsgTEMPERATURE(MS45XX_SENDER_ID, temp);
293  // Compute airspeed
294  float sign = 1.0f;
295  if (ms45xx.pressure < 0.0f) {
296  sign = -1.0f;
297  }
298  ms45xx.airspeed = sqrtf(ms45xx.pressure * sign * MS45XX_RHO_DIV_2) * sign;
299 
300  }
301 
302  // Set to done
303  ms45xx_trans.status = I2CTransDone;
304  } else if (ms45xx_trans.status == I2CTransFailed) {
305  // Just retry if failed
306  ms45xx_trans.status = I2CTransDone;
307  }
308 }
