119static void cc2500_persistent_write(void) {
120 cc2500_settings_persistent.bindVars =
121 cc2500spiconfig.bindTxId[0] |
122 cc2500spiconfig.bindTxId[1] << 8 |
123 cc2500spiconfig.bindOffset << 16 |
124 cc2500spiconfig.rxNum << 24;
125 for (int i = 0; i < 48; i += 4) {
126 cc2500_settings_persistent.bindHopData[i / 4] =
127 cc2500spiconfig.bindHopData[i] |
128 cc2500spiconfig.bindHopData[i + 1] << 8 |
129 cc2500spiconfig.bindHopData[i + 2] << 16 |
130 cc2500spiconfig.bindHopData[i + 3] << 24;
131 }
132 cc2500_settings_persistent.bindHopData[12] =
133 cc2500spiconfig.bindHopData[48] |
134 cc2500spiconfig.bindHopData[49] << 8 |
135 0xFF << 24;
136}
137
138static void cc2500_persistent_read(void) {
139 // Check that persistent data is loaded
140 // highest bindHopData byte is initialized 0 at boot
141 if (cc2500_settings_persistent.bindHopData[12] >> 24) {
142 cc2500spiconfig.bindTxId[0] = cc2500_settings_persistent.bindVars & 0xFF;
143 cc2500spiconfig.bindTxId[1] = (cc2500_settings_persistent.bindVars >> 8) & 0xFF;
144 cc2500spiconfig.bindOffset = (cc2500_settings_persistent.bindVars >> 16) & 0xFF;
145 cc2500spiconfig.rxNum = (cc2500_settings_persistent.bindVars >> 24) & 0xFF;
146 }
147 for (int i = 0; i < 48; i += 4) {
148 cc2500spiconfig.bindHopData[i] = cc2500_settings_persistent.bindHopData[i / 4] & 0xFF;
149 cc2500spiconfig.bindHopData[i + 1] = (cc2500_settings_persistent.bindHopData[i / 4] >> 8) & 0xFF;
150 cc2500spiconfig.bindHopData[i + 2] = (cc2500_settings_persistent.bindHopData[i / 4] >> 16) & 0xFF;
151 cc2500spiconfig.bindHopData[i + 3] = (cc2500_settings_persistent.bindHopData[i / 4] >> 24) & 0xFF;
152 }
153 cc2500spiconfig.bindHopData[48] = cc2500_settings_persistent.bindHopData[12] & 0xFF;
154 cc2500spiconfig.bindHopData[49] = (cc2500_settings_persistent.bindHopData[12] >> 8) & 0xFF;
155}

159void cc2500_settings_init(void) {
160 // rxConfig
161 rxconfig.rssi_channel = 0;
162 rxconfig.midrc = 1500;
163 rxconfig.max_aux_channel = 0; // TODO
164 rxconfig.rssi_src_frame_lpf_period = 30;
165
166 // rxSpiConfig
167 spiconfig.rx_spi_protocol = CC2500_RX_SPI_PROTOCOL;
168 extiIo.port = CC2500_GDO0_GPIO;
169 extiIo.pin = CC2500_GDO0_PIN;
170 spiconfig.extiIoTag = &extiIo;
171
172#ifdef CC2500_RX_LED
173 ledIo.port = LED_GPIO(CC2500_RX_LED);
174 ledIo.pin = LED_GPIO_PIN(CC2500_RX_LED);
175 ledIo.hi = LED_GPIO_ON(CC2500_RX_LED);
176 ledIo.lo = LED_GPIO_OFF(CC2500_RX_LED);
177 spiconfig.ledIoTag = &ledIo;
178#else
179 (void) ledIo;
180 spiconfig.ledIoTag = NULL;
181#endif
182 spiconfig.ledInversion = FALSE; // Handled by paparazzi LED_X_GPIO_ON|_OFF
183
184#ifdef CC2500_BIND_BUTTON
185 bindIo.port = BUTTON_GPIO(CC2500_BIND_BUTTON);
186 bindIo.pin = BUTTON_PIN(CC2500_BIND_BUTTON);
187 spiconfig.bindIoTag = &bindIo;
188#else
189 (void) bindIo;
190 spiconfig.bindIoTag = NULL;
191#endif
192
193 // rxCc2500SpiConfig
194 cc2500spiconfig.autoBind = CC2500_AUTOBIND;
195 cc2500spiconfig.bindTxId[0] = 0;
196 cc2500spiconfig.bindTxId[1] = 0;
197 cc2500spiconfig.bindOffset = 0;
198 memset(cc2500spiconfig.bindHopData, 0, sizeof(cc2500spiconfig.bindHopData));
199 cc2500spiconfig.rxNum = 0;
200 cc2500spiconfig.a1Source = FRSKY_SPI_A1_SOURCE_VBAT;
201 cc2500spiconfig.chipDetectEnabled = TRUE;
202
203 settings_init();
204 cc2500_persistent_read();
205
206 // telemetryConfig
207 telemetryconfig.pidValuesAsTelemetry = FALSE;
208 telemetryconfig.report_cell_voltage = FALSE;
209}
