182void v_ctl_guidance_loop(void)
183{
184 if (v_ctl_mode == V_CTL_MODE_AUTO_ALT) {
185 v_ctl_altitude_loop();
186 }
187#if CTRL_VERTICAL_LANDING
188 if (v_ctl_mode == V_CTL_MODE_LANDING) {
189 v_ctl_landing_loop();
190 } else {
191#endif
192 if (v_ctl_mode == V_CTL_MODE_AUTO_THROTTLE) {
193 v_ctl_throttle_setpoint = nav_throttle_setpoint;
194 v_ctl_pitch_setpoint = nav_pitch;
195 } else {
196 if (v_ctl_mode >= V_CTL_MODE_AUTO_CLIMB) {
197 v_ctl_climb_loop();
198 } /* v_ctl_mode >= V_CTL_MODE_AUTO_CLIMB */
199 } /* v_ctl_mode == V_CTL_MODE_AUTO_THROTTLE */
200#if CTRL_VERTICAL_LANDING
201 } /* v_ctl_mode == V_CTL_MODE_LANDING */
202#endif
203
204#if defined V_CTL_THROTTLE_IDLE
205 Bound(v_ctl_throttle_setpoint, TRIM_PPRZ(V_CTL_THROTTLE_IDLE * MAX_PPRZ), MAX_PPRZ);
206#endif
207
208#ifdef V_CTL_POWER_CTL_BAT_NOMINAL
209 if (electrical.vsupply > 0.) {
210 v_ctl_throttle_setpoint *= V_CTL_POWER_CTL_BAT_NOMINAL / electrical.vsupply;
211 v_ctl_throttle_setpoint = TRIM_UPPRZ(v_ctl_throttle_setpoint);
212 }
213#endif
214
215 if (autopilot.kill_throttle || (!autopilot.flight_time && !autopilot.launch)) {
216 v_ctl_throttle_setpoint = 0;
217 }
218}

253static inline void v_ctl_set_pitch(void)
254{
255 static float last_err = 0.;
256
257 if (autopilot_get_mode() == AP_MODE_MANUAL || autopilot.launch == false) {
258 v_ctl_auto_pitch_sum_err = 0;
259 }
260
261 // Compute errors
262 float err = v_ctl_climb_setpoint - stateGetSpeedEnu_f()->z;
263 float d_err = err - last_err;
264 last_err = err;
265
266 if (v_ctl_auto_pitch_igain > 0.) {
267 v_ctl_auto_pitch_sum_err += err * (1. / 60.);
268 BoundAbs(v_ctl_auto_pitch_sum_err, V_CTL_AUTO_PITCH_MAX_SUM_ERR / v_ctl_auto_pitch_igain);
269 }
270
271 // PI loop + feedforward ctl
272 v_ctl_pitch_setpoint = nav_pitch
273 + v_ctl_pitch_trim
274 + v_ctl_auto_throttle_pitch_of_vz_pgain * v_ctl_climb_setpoint
275 + v_ctl_auto_pitch_pgain * err
276 + v_ctl_auto_pitch_dgain * d_err
277 + v_ctl_auto_pitch_igain * v_ctl_auto_pitch_sum_err;
278
279}

281static inline void v_ctl_set_throttle(void)
282{
283 static float last_err = 0.;
284
285 if (autopilot_get_mode() == AP_MODE_MANUAL || autopilot.launch == false) {
286 v_ctl_auto_throttle_sum_err = 0;
287 }
288
289 // Compute errors

292 last_err = err;

294 if (v_ctl_auto_throttle_igain > 0.) {

302 + v_ctl_auto_throttle_pgain * err

311// Airspeed control loop (input: [airspeed controlled, climb_setpoint], output: [throttle controlled, pitch setpoint])
312static inline void v_ctl_set_airspeed(void)
313{
314 static float last_err_vz = 0.;
315 static float last_err_as = 0.;
316
317 // Bound airspeed setpoint
318 Bound(v_ctl_auto_airspeed_setpoint, V_CTL_AIRSPEED_MIN, V_CTL_AIRSPEED_MAX);
319
320 // Compute errors
321 float err_vz = v_ctl_climb_setpoint - stateGetSpeedEnu_f()->z;
322 float d_err_vz = (err_vz - last_err_vz) * AIRSPEED_LOOP_PERIOD;
323 last_err_vz = err_vz;
324 if (v_ctl_auto_throttle_igain > 0.) {
325 v_ctl_auto_throttle_sum_err += err_vz * AIRSPEED_LOOP_PERIOD;
326 BoundAbs(v_ctl_auto_throttle_sum_err, V_CTL_AUTO_THROTTLE_MAX_SUM_ERR / v_ctl_auto_throttle_igain);
327 }
328 if (v_ctl_auto_pitch_igain > 0.) {
329 v_ctl_auto_pitch_sum_err += err_vz * AIRSPEED_LOOP_PERIOD;
330 BoundAbs(v_ctl_auto_pitch_sum_err, V_CTL_AUTO_PITCH_MAX_SUM_ERR / v_ctl_auto_pitch_igain);
331 }
332
333 float err_airspeed = v_ctl_auto_airspeed_setpoint - stateGetAirspeed_f();
334 float d_err_airspeed = (err_airspeed - last_err_as) * AIRSPEED_LOOP_PERIOD;
335 last_err_as = err_airspeed;
336 if (v_ctl_auto_airspeed_throttle_igain > 0.) {
337 v_ctl_auto_airspeed_throttle_sum_err += err_airspeed * AIRSPEED_LOOP_PERIOD;
338 BoundAbs(v_ctl_auto_airspeed_throttle_sum_err,
339 V_CTL_AUTO_AIRSPEED_THROTTLE_MAX_SUM_ERR / v_ctl_auto_airspeed_throttle_igain);
340 }
341 if (v_ctl_auto_airspeed_pitch_igain > 0.) {
342 v_ctl_auto_airspeed_pitch_sum_err += err_airspeed * AIRSPEED_LOOP_PERIOD;
343 BoundAbs(v_ctl_auto_airspeed_pitch_sum_err, V_CTL_AUTO_AIRSPEED_PITCH_MAX_SUM_ERR / v_ctl_auto_airspeed_pitch_igain);
344 }
345
346
347 // Reset integrators in manual or before flight
348 if (autopilot_get_mode() == AP_MODE_MANUAL || autopilot.launch == false) {
349 v_ctl_auto_throttle_sum_err = 0.;
350 v_ctl_auto_pitch_sum_err = 0.;
351 v_ctl_auto_airspeed_throttle_sum_err = 0.;
352 v_ctl_auto_airspeed_pitch_sum_err = 0.;
353 }
354
355 // Pitch loop
356 v_ctl_pitch_setpoint =
357 v_ctl_auto_throttle_pitch_of_vz_pgain * v_ctl_climb_setpoint
358 + v_ctl_pitch_trim
359 + v_ctl_auto_pitch_pgain * err_vz
360 + v_ctl_auto_pitch_dgain * d_err_vz
361 + v_ctl_auto_pitch_igain * v_ctl_auto_pitch_sum_err
362 - v_ctl_auto_airspeed_pitch_pgain * err_airspeed
363 - v_ctl_auto_airspeed_pitch_dgain * d_err_airspeed
364 - v_ctl_auto_airspeed_pitch_igain * v_ctl_auto_airspeed_pitch_sum_err;
365
366 // Throttle loop
367 controlled_throttle = v_ctl_auto_throttle_cruise_throttle
368 + v_ctl_auto_throttle_climb_throttle_increment * v_ctl_climb_setpoint
369 + v_ctl_auto_throttle_pgain * err_vz
370 + v_ctl_auto_throttle_dgain * d_err_vz
371 + v_ctl_auto_throttle_igain * v_ctl_auto_throttle_sum_err
372 + v_ctl_auto_airspeed_throttle_pgain * err_airspeed
373 + v_ctl_auto_airspeed_throttle_dgain * d_err_airspeed
374 + v_ctl_auto_airspeed_throttle_igain * v_ctl_auto_airspeed_throttle_sum_err;
375
376}

390void v_ctl_climb_loop(void)
391{
392
393 switch (v_ctl_speed_mode) {
394 case V_CTL_SPEED_THROTTLE:
395 // Set pitch
396 v_ctl_set_pitch();
397 // Set throttle
398 v_ctl_set_throttle();
399 break;
400#if USE_AIRSPEED
401 case V_CTL_SPEED_AIRSPEED:
402 v_ctl_set_airspeed();
403 break;
404 case V_CTL_SPEED_GROUNDSPEED:
405 v_ctl_set_groundspeed();
406 v_ctl_set_airspeed();
407 break;
408#endif
409 default:
410 controlled_throttle = v_ctl_auto_throttle_cruise_throttle; // ???
411 break;
412 }
413
414 // Set Pitch output
415 Bound(v_ctl_pitch_setpoint, V_CTL_AUTO_PITCH_MIN_PITCH, V_CTL_AUTO_PITCH_MAX_PITCH);
416 // Set Throttle output
417 v_ctl_throttle_setpoint = TRIM_UPPRZ(controlled_throttle * MAX_PPRZ);
418
419}
