93 int ahrs_switch(uint8_t idx)
94 {
95  if (idx >= AHRS_NB_IMPL) { return -1; }
96  if (ahrs_impls[idx].enable == NULL) { return -1; }
97  /* first disable other AHRS output */

103  }
