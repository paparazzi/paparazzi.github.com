139static void send_i2c_err(struct transport_tx *trans __attribute__((unused)),
140 struct link_device *dev __attribute__((unused)))
141{
142 static uint8_t _i2c_nb_cnt = 0;
143 switch (_i2c_nb_cnt) {
144 case 0:
145#if USE_I2C0
146 send_i2cx_err(trans, dev, &i2c0);
147#endif
148 break;
149 case 1:
150#if USE_I2C1
151 send_i2cx_err(trans, dev, &i2c1);
152#endif
153 break;
154 case 2:
155#if USE_I2C2
156 send_i2cx_err(trans, dev, &i2c2);
157#endif
158 break;
159 case 3:
160#if USE_I2C3
161 send_i2cx_err(trans, dev, &i2c3);
162#endif
163 break;
164 case 4:
165#if USE_I2C4
166 send_i2cx_err(trans, dev, &i2c4);
167#endif
168 case 5:
169#if USE_SOFTI2C0
170 send_softi2c0_err(trans, dev);
171#endif
172 case 6:
173#if USE_SOFTI2C1
174 send_softi2c1_err(trans, dev);
175#endif
176 break;
177 default:
178 break;
179 }
180 _i2c_nb_cnt++;
181 if (_i2c_nb_cnt == 7) {
182 _i2c_nb_cnt = 0;
183 }
184}

232static enum I2CTransactionStatus i2c_blocking_submit(struct i2c_periph *p, struct i2c_transaction *t, float timeout) {
233 if (!i2c_submit(p, t)) {
234 return I2CTransFailed;
235 }
236
237 // Wait for transaction to complete
238 if(pprz_bsem_wait_timeout(&t->bsem, timeout) == 0) {
239 return t->status;
240 } else {
241 return I2CTransFailed;
242 }
243}
