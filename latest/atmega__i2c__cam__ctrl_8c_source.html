80void atmega_i2c_cam_ctrl_send(uint8_t cmd)
81{
82 atmega_i2c_cam_ctrl_just_sent_command = 1;
83
84 // Send Command
85 atmega_i2c_cam_ctrl_trans.buf[0] = cmd;
86 i2c_transceive(&ATMEGA_I2C_DEV, &atmega_i2c_cam_ctrl_trans, ATMEGA_SLAVE_ADDR, 1, 1);
87
88 if (cmd == DC_SHOOT) {
89 dc_send_shot_position();
90 }
91}

93void atmega_i2c_cam_ctrl_event(void)
94{
95 if (atmega_i2c_cam_ctrl_trans.status == I2CTransSuccess) {
96 unsigned char cam_ret[1];
97 cam_ret[0] = atmega_i2c_cam_ctrl_trans.buf[0];
98 RunOnceEvery(6, DOWNLINK_SEND_PAYLOAD(DefaultChannel, DefaultDevice, 1, cam_ret));
99 atmega_i2c_cam_ctrl_trans.status = I2CTransDone;
100 }
