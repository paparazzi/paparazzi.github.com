65void opa_controller_periodic(void) {
66 static uint16_t off_cnt = 0;
67 static uint16_t arming_cnt = 0;
68
69 /* Check E-Stop and power off Main power if pressed */
70 if(!gpio_get(BTN_ESTOP, BTN_ESTOP_PIN)) {
71 AP_PWR_OFF(AP_PWR, AP_PWR_PIN);
72 MAIN_PWR_OFF(MAIN_PWR, MAIN_PWR_PIN);
73 BAL_PWR_OFF(BAL_PWR, BAL_PWR_PIN);
74 RADIO_CONTROL_POWER_OFF(RADIO_CONTROL_POWER_PORT,RADIO_CONTROL_POWER_PIN);
75 MCU_PWR_OFF(MCU_PWR, MCU_PWR_PIN);
76 arming_led = false;
77 return;
78 }
79
80 /* AP soft disarm */
81 if (INTERMCU_GET_CMD_STATUS(INTERMCU_CMD_DISARM)) {
82 MAIN_PWR_OFF(MAIN_PWR, MAIN_PWR_PIN);
83 BAL_PWR_OFF(BAL_PWR, BAL_PWR_PIN);
84 arming_led = false;
85 INTERMCU_CLR_CMD_STATUS(INTERMCU_CMD_DISARM);
86 }
87
88 /* Check On/Off button and disable if pressed for 3 seconds */
89 if(gpio_get(BTN_ON, BTN_ON_PIN)) {
90 if(off_cnt >= OFF_TIMER) {
91 MAIN_PWR_OFF(MAIN_PWR, MAIN_PWR_PIN);
92 BAL_PWR_OFF(BAL_PWR, BAL_PWR_PIN);
93 arming_led = false;
94 } else {
95 off_cnt++;
96 }
97 }
98 else {
99 off_cnt = 0;
100 }
101
102 /* Check Arming button and set LED */
103 if(!gpio_get(BTN_ARMING, BTN_ARMING_PIN)) {
104 if (arming_cnt >= MIN_ARMING_TRIG) {
105 arming_led = true;
106 MAIN_PWR_ON(MAIN_PWR, MAIN_PWR_PIN);
107 BAL_PWR_ON(BAL_PWR, BAL_PWR_PIN);
108 //AP_PWR_ON(AP_PWR, AP_PWR_PIN);
109 //RADIO_CONTROL_POWER_ON(RADIO_CONTROL_POWER_PORT,RADIO_CONTROL_POWER_PIN);
110 } else {
111 arming_cnt++;
112 }
113 }
114 else {
115 arming_cnt = 0;
116 }
117
118#if defined ARMING_LED
119 /* Update Arming LED */
120 if (arming_led) {
121 LED_OFF(ARMING_LED);
122 } else {
123 LED_ON(ARMING_LED);
124 }
125 // autopilot_motors_on;
126#endif
127}
