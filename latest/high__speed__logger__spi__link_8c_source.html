59void high_speed_logger_spi_link_periodic(void)
60{
61 // Static counter to identify missing samples
62 static int32_t counter = 0;
63
64 // count all periodic steps
65 counter ++;
66
67 // only send a new message if the previous was completely sent
68 if (high_speed_logger_spi_link_ready) {
69 // copy the counter into the SPI datablock
70 high_speed_logger_spi_link_data.id = counter;
71
72 if(hsl_gyro == NULL)
73 hsl_gyro = imu_get_gyro(ABI_BROADCAST, false);
74 if(hsl_accel == NULL)
75 hsl_accel = imu_get_accel(ABI_BROADCAST, false);
76 if(hsl_mag == NULL)
77 hsl_mag = imu_get_mag(ABI_BROADCAST, false);
78
79 if(hsl_gyro == NULL || hsl_accel == NULL || hsl_mag == NULL)
80 return;
81
82 high_speed_logger_spi_link_ready = false;
83 high_speed_logger_spi_link_data.gyro_p = hsl_gyro->unscaled.p;
84 high_speed_logger_spi_link_data.gyro_q = hsl_gyro->unscaled.q;
85 high_speed_logger_spi_link_data.gyro_r = hsl_gyro->unscaled.r;
86 high_speed_logger_spi_link_data.acc_x = hsl_accel->unscaled.x;
87 high_speed_logger_spi_link_data.acc_y = hsl_accel->unscaled.y;
88 high_speed_logger_spi_link_data.acc_z = hsl_accel->unscaled.z;
89 high_speed_logger_spi_link_data.mag_x = hsl_mag->unscaled.x;
90 high_speed_logger_spi_link_data.mag_y = hsl_mag->unscaled.y;
91 high_speed_logger_spi_link_data.mag_z = hsl_mag->unscaled.z;
92
93 spi_submit(&(HIGH_SPEED_LOGGER_SPI_LINK_DEVICE), &high_speed_logger_spi_link_transaction);
94 }
95}
