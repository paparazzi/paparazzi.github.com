40extern void max1168_init(void)
41{
42
43 max1168_arch_init();
44
45 uint8_t i;
46 for (i = 0; i < MAX1168_NB_CHAN; i++) {
47 max1168_values[i] = 0;
48 }
49
50 // init spi transaction parameters
51 max1168_req_trans.cpol = SPICpolIdleLow;
52 max1168_req_trans.cpha = SPICphaEdge1;
53 max1168_req_trans.dss = SPIDss16bit;
54 max1168_req_trans.bitorder = SPIMSBFirst;
55 max1168_req_trans.cdiv = SPIDiv64;
56
57 max1168_req_trans.slave_idx = MAX1168_SLAVE_IDX;
58 max1168_req_trans.select = SPISelect;
59 max1168_conv_req = MAX1168_CONF_CR << 8;
60 max1168_req_trans.output_buf = (uint8_t *)(&max1168_conv_req);
61 max1168_req_trans.output_length = 1;
62 max1168_req_trans.input_buf = NULL;
63 max1168_req_trans.input_length = 0;
64 max1168_req_trans.after_cb = max1168_lock_cb;
65 max1168_req_trans.status = SPITransDone;
66
67 max1168_read_trans.cpol = SPICpolIdleLow;
68 max1168_read_trans.cpha = SPICphaEdge1;
69 max1168_read_trans.dss = SPIDss16bit;
70 max1168_read_trans.bitorder = SPIMSBFirst;
71 max1168_read_trans.cdiv = SPIDiv64;
72
73 max1168_read_trans.slave_idx = MAX1168_SLAVE_IDX;
74 max1168_read_trans.select = SPIUnselect;
75 // read 8 16bit frames
76 // FIXME should be function of control register options
77 max1168_read_trans.output_buf = NULL;
78 max1168_read_trans.output_length = 0;
79 max1168_read_trans.input_buf = (uint8_t *)max1168_values;
80 max1168_read_trans.input_length = 8;
81 max1168_read_trans.status = SPITransDone;
82
83 max1168_status = MAX1168_IDLE;
84}

87void max1168_read(void)
88{
89 //ASSERT((max1168_status == MAX1168_IDLE), DEBUG_MAX_1168, MAX1168_ERR_READ_OVERUN);

91 /* set SPI transaction */
92 /* SPI is locked between the two transactions (callback) */
93 /* SPI is unlocked when EOC is received */
94
95 spi_submit(&(MAX1168_SPI_DEV), &max1168_req_trans);

98 max1168_status = MAX1168_SENDING_REQ;

101void max1168_event(void)
102{
103 // handle request transaction
104 if (max1168_req_trans.status == SPITransSuccess) {
105 max1168_req_trans.status = SPITransDone;
106 } else if (max1168_req_trans.status == SPITransFailed) {
107 max1168_status = MAX1168_IDLE;
108 spi_slave_unselect(MAX1168_SLAVE_IDX);
109 spi_resume(&(MAX1168_SPI_DEV), MAX1168_SLAVE_IDX);
110 max1168_req_trans.status = SPITransDone;
111 }
112
113 // handle reading transaction
114 if (max1168_read_trans.status == SPITransSuccess) {
115 if (max1168_status == MAX1168_READING_RES) {
116 // result was already written to max1168_values by DMA
117 max1168_status = MAX1168_DATA_AVAILABLE;
118 max1168_read_trans.status = SPITransDone;
119 }
120 } else if (max1168_read_trans.status == SPITransFailed) {
121 max1168_status = MAX1168_IDLE;
122 spi_slave_unselect(MAX1168_SLAVE_IDX);
123 max1168_read_trans.status = SPITransDone;
124 }
125 // Waiting for EOC
126 // FIXME possible race condition, should suspend external int ?
127 if (max1168_status == MAX1168_GOT_EOC) {
128 // eoc occurs, unlock SPI
129 spi_resume(&(MAX1168_SPI_DEV), MAX1168_SLAVE_IDX);
130 max1168_status = MAX1168_READING_RES;
131 }
132}
