56static int findFreeNodeID(const uint8_t preferred)
57{
58 // Search up
59 int candidate = (preferred > 0) ? preferred : 125;
60 while (candidate <= 125) {
61 if (!uavcan_get_node_id_mapping(candidate)) {
62 return candidate;
63 }
64 candidate++;
65 }
66 // Search down
67 candidate = (preferred > 0) ? preferred : 125;
68 while (candidate > 0) {
69 if (!uavcan_get_node_id_mapping(candidate)) {
70 return candidate;
71 }
72 candidate--;
73 }
74 // Not found
75 return -1;
76}

79static int detectRequestStage(struct uavcan_protocol_dynamic_node_id_Allocation* msg) {
80 if(msg->first_part_of_unique_id) {
81 return 1;
82 }
83 if(msg->unique_id.len == UAVCAN_PROTOCOL_DYNAMIC_NODE_ID_ALLOCATION_MAX_LENGTH_OF_UNIQUE_ID_IN_REQUEST) {
84 return 2;

110static void handleAllocationRequest(struct uavcan_iface_t *iface, uint8_t preferred_node_id) {
111 uint8_t allocated_id = 0;
112
113 for(int i=0; i<UAVCAN_MAX_NODES; i++) {
114
115 if(uavcan_node_ids[i].allocated_id != 0) {
116 // allocation slot taken, check if unique id matches
117 if(unique_id_identical(i)) {
118 allocated_id = uavcan_node_ids[i].allocated_id;
119 }
120 }
121 else {
122 // free allocation slot
123
124 // copy unique id
125 for(int k=0; k<current_unique_id.len; k++) {
126 uavcan_node_ids[i].unique_id.data[k] = current_unique_id.data[k];
127 }
128 uavcan_node_ids[i].unique_id.len = current_unique_id.len;
129
130 int free_id = findFreeNodeID(preferred_node_id);
131
132 if(free_id != -1) {
133 uavcan_node_ids[i].allocated_id = free_id;
134 allocated_id = free_id;
135 break;
136 } else {
137 // error: not more free IDs
138 return;
139 }
140
141 }
142 }
143
144 struct uavcan_protocol_dynamic_node_id_Allocation response_msg;
145 // copy unique id
146 for(int k=0; k<current_unique_id.len; k++) {
147 response_msg.unique_id.data[k] = current_unique_id.data[k];
148 }
149 response_msg.unique_id.len = current_unique_id.len;
150 response_msg.node_id = allocated_id;
151
152 uint8_t msg_buffer[50];
153
154 uint32_t total_size = uavcan_protocol_dynamic_node_id_Allocation_encode(&response_msg, msg_buffer);
155
156 uavcan_broadcast(
157 iface,
158 UAVCAN_PROTOCOL_DYNAMIC_NODE_ID_ALLOCATION_SIGNATURE, UAVCAN_PROTOCOL_DYNAMIC_NODE_ID_ALLOCATION_ID,
159 CANARD_TRANSFER_PRIORITY_HIGH, msg_buffer, total_size);
160}

163static void id_alloc_uavcan_cb(struct uavcan_iface_t *iface, CanardRxTransfer *transfer)
164{
165 struct uavcan_protocol_dynamic_node_id_Allocation msg;
166 if(uavcan_protocol_dynamic_node_id_Allocation_decode(transfer, &msg)) {
167 return; // decode error
168 }
169
170 const int unique_id_capacity = sizeof(msg.unique_id.data);
171
172 // only process anonymous transfers
173 if(transfer->source_node_id != 0) {
174 return;
175 }
176
177 uint32_t timestamp = transfer->timestamp_usec / 1000;
178
179 // Reset the expected stage on timeout
180 if(timestamp > last_message_timestamp + UAVCAN_PROTOCOL_DYNAMIC_NODE_ID_ALLOCATION_FOLLOWUP_TIMEOUT_MS) {
181 current_unique_id.len = 0;
182 }
183
184 // Checking if request stage matches the expected stage
185 int request_stage = detectRequestStage(&msg);
186 if(request_stage == INVALID_STAGE) { return; }
187 if(request_stage != getExpectedStage()) { return; }
188 if (msg.unique_id.len > (unique_id_capacity - current_unique_id.len)) { return; }
189
190 // Updating the local state
191 for(int i=0; i<msg.unique_id.len; i++) {
192 current_unique_id.data[current_unique_id.len] = msg.unique_id.data[i];
193 current_unique_id.len += 1;
194 }
195
196 if(current_unique_id.len == unique_id_capacity) {
197 // Proceeding with allocation.
198 handleAllocationRequest(iface, msg.node_id);
199 current_unique_id.len = 0;
200 } else {
201 struct uavcan_protocol_dynamic_node_id_Allocation response_msg;
202 // copy unique id
203 for(int k=0; k<current_unique_id.len; k++) {
204 response_msg.unique_id.data[k] = current_unique_id.data[k];
205 }
206 response_msg.unique_id.len = current_unique_id.len;
207
208 uint8_t msg_buffer[50];
209
210 uint32_t total_size = uavcan_protocol_dynamic_node_id_Allocation_encode(&response_msg, msg_buffer);
211
212 uavcan_broadcast(
213 iface,
214 UAVCAN_PROTOCOL_DYNAMIC_NODE_ID_ALLOCATION_SIGNATURE, UAVCAN_PROTOCOL_DYNAMIC_NODE_ID_ALLOCATION_ID,
215 CANARD_TRANSFER_PRIORITY_HIGH, msg_buffer, total_size);
216 }
217
218 // It is important to update the timestamp only if the request has been processed successfully.
219 last_message_timestamp = timestamp;
220}

222static void node_info_resp_cb(struct uavcan_iface_t *iface, CanardRxTransfer *transfer) {
223 (void)iface;
224 struct uavcan_protocol_GetNodeInfoResponse msg;
225 if(uavcan_protocol_GetNodeInfoResponse_decode(transfer, &msg)) {
226 return; // decode error
227 }
228
229 struct uavcan_node_mapping_t* mapping = uavcan_get_node_id_mapping(transfer->source_node_id);
230
231 if(!mapping) {
232 mapping = get_free_id_mapping();
233 }
234
235
236 mapping->allocated_id = transfer->source_node_id;
237 for(int i=0; i<16; i++) {
238 mapping->unique_id.data[i] = msg.hardware_version.unique_id[i];
239 }
240 mapping->unique_id.len = 16;
241
242}
