55 static void ctrl_module_run(bool in_flight)
56 {
57  if (!in_flight) {
58  // Reset integrators
59  stabilization.cmd[COMMAND_ROLL] = 0;
60  stabilization.cmd[COMMAND_PITCH] = 0;
61  stabilization.cmd[COMMAND_YAW] = 0;
62  stabilization.cmd[COMMAND_THRUST] = 0;
63  } else {
64  stabilization.cmd[COMMAND_ROLL] = ctrl_module_demo.rc_x * ctrl_module_demo_pr_ff_gain -
65  stateGetBodyRates_i()->p * ctrl_module_demo_pr_d_gain;
66  stabilization.cmd[COMMAND_PITCH] = ctrl_module_demo.rc_y * ctrl_module_demo_pr_ff_gain -
67  stateGetBodyRates_i()->q * ctrl_module_demo_pr_d_gain;
68  stabilization.cmd[COMMAND_YAW] = ctrl_module_demo.rc_z * ctrl_module_demo_y_ff_gain -
69  stateGetBodyRates_i()->r * ctrl_module_demo_y_d_gain;
70  stabilization.cmd[COMMAND_THRUST] = ctrl_module_demo.rc_t;
71  }
72 }
