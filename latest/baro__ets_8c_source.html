139void baro_ets_read_periodic(void)
140{
141 // Initiate next read
142 if (!baro_ets_delay_done) {
143 if (SysTimeTimer(baro_ets_delay_time) < USEC_OF_SEC(BARO_ETS_START_DELAY)) { return; }
144 else { baro_ets_delay_done = true; }
145 }
146 if (baro_ets_i2c_trans.status == I2CTransDone) {
147 i2c_receive(&BARO_ETS_I2C_DEV, &baro_ets_i2c_trans, BARO_ETS_ADDR, 2);
148 }
149}

151void baro_ets_read_event(void)
152{
153 // Get raw altimeter from buffer
154 baro_ets_adc = ((uint16_t)(baro_ets_i2c_trans.buf[1]) << 8) | (uint16_t)(baro_ets_i2c_trans.buf[0]);
155 // Check if this is valid altimeter
156 if (baro_ets_adc == 0) {
157 baro_ets_valid = false;
158 } else {
159 baro_ets_valid = true;
160 }
161
162 // Continue only if a new altimeter value was received
163 if (baro_ets_valid) {
164 // Calculate offset average if not done already
165 if (!baro_ets_offset_init) {
166 --baro_ets_cnt;
167 // Check if averaging completed
168 if (baro_ets_cnt == 0) {
169 // Calculate average
170 baro_ets_offset = (uint16_t)(baro_ets_offset_tmp / BARO_ETS_OFFSET_NBSAMPLES_AVRG);
171 // Limit offset
172 if (baro_ets_offset < BARO_ETS_OFFSET_MIN) {
173 baro_ets_offset = BARO_ETS_OFFSET_MIN;
174 }
175 if (baro_ets_offset > BARO_ETS_OFFSET_MAX) {
176 baro_ets_offset = BARO_ETS_OFFSET_MAX;
177 }
178 baro_ets_offset_init = true;
179 }
180 // Check if averaging needs to continue
181 else if (baro_ets_cnt <= BARO_ETS_OFFSET_NBSAMPLES_AVRG) {
182 baro_ets_offset_tmp += baro_ets_adc;
183 }
184 }
185 // Convert raw to m/s
186 if (baro_ets_offset_init) {
187 baro_ets_altitude = ground_alt + BARO_ETS_ALT_SCALE * (float)(baro_ets_offset - baro_ets_adc);
188 // New value available
189 uint32_t now_ts = get_sys_time_usec();
190 float pressure = BARO_ETS_SCALE * (float) baro_ets_adc + BARO_ETS_PRESSURE_OFFSET;
191 AbiSendMsgBARO_ABS(BARO_ETS_SENDER_ID, now_ts, pressure);
192#ifdef BARO_ETS_SYNC_SEND
193 DOWNLINK_SEND_BARO_ETS(DefaultChannel, DefaultDevice, &baro_ets_adc, &baro_ets_offset, &baro_ets_altitude);
194#endif
195 } else {
196 baro_ets_altitude = 0.0;
197 }
198 } else {
199 baro_ets_altitude = 0.0;
200 }
201
202 // Transaction has been read
203 baro_ets_i2c_trans.status = I2CTransDone;
204}
