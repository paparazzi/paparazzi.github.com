73 void actuators_bebop_commit(void)
74 {
75  // Receive the status
76  actuators_bebop.i2c_trans.buf[0] = ACTUATORS_BEBOP_GET_OBS_DATA;
77  i2c_blocking_transceive(&i2c1, &actuators_bebop.i2c_trans, actuators_bebop.i2c_trans.slave_addr, 1, 13);
78 
79  // Update status
80  electrical.vsupply = (float)(actuators_bebop.i2c_trans.buf[9] + (actuators_bebop.i2c_trans.buf[8] << 8)) / 1000.f;
81  // The 15th bit contains saturation information, so it needs to be removed to get the rpm
82  actuators_bebop.rpm_obs[0] = (actuators_bebop.i2c_trans.buf[1] + (actuators_bebop.i2c_trans.buf[0] << 8)) & ~(1<<15);
83  actuators_bebop.rpm_obs[1] = (actuators_bebop.i2c_trans.buf[3] + (actuators_bebop.i2c_trans.buf[2] << 8)) & ~(1<<15);
84  actuators_bebop.rpm_obs[2] = (actuators_bebop.i2c_trans.buf[5] + (actuators_bebop.i2c_trans.buf[4] << 8)) & ~(1<<15);
85  actuators_bebop.rpm_obs[3] = (actuators_bebop.i2c_trans.buf[7] + (actuators_bebop.i2c_trans.buf[6] << 8)) & ~(1<<15);
86 
87  // When detected a suicide
88  actuators_bebop.i2c_trans.buf[10] = actuators_bebop.i2c_trans.buf[10] & 0x7;
89  if (actuators_bebop.i2c_trans.buf[11] == 2 && actuators_bebop.i2c_trans.buf[10] != 1) {
90  autopilot_set_motors_on(FALSE);
91  }
92 
93  // Start the motors
94  if (actuators_bebop.i2c_trans.buf[10] != 4 && actuators_bebop.i2c_trans.buf[10] != 2 && autopilot_get_motors_on()) {
95  // Reset the error
96  actuators_bebop.i2c_trans.buf[0] = ACTUATORS_BEBOP_CLEAR_ERROR;
97  i2c_blocking_transmit(&i2c1, &actuators_bebop.i2c_trans, actuators_bebop.i2c_trans.slave_addr, 1);
98 
99  // Start the motors
100  actuators_bebop.i2c_trans.buf[0] = ACTUATORS_BEBOP_START_PROP;
101 #if BEBOP_VERSION2
102  // For Bebop version 2 some motors are reversed (FIXME: test final version)
103  actuators_bebop.i2c_trans.buf[1] = 0b00001010;
104 #else
105  actuators_bebop.i2c_trans.buf[1] = 0b00000101;
106 #endif
107  i2c_blocking_transmit(&i2c1, &actuators_bebop.i2c_trans, actuators_bebop.i2c_trans.slave_addr, 2);
108  }
109  // Stop the motors
110  else if (actuators_bebop.i2c_trans.buf[10] == 4 && !autopilot_get_motors_on()) {
111  actuators_bebop.i2c_trans.buf[0] = ACTUATORS_BEBOP_STOP_PROP;
112  i2c_blocking_transmit(&i2c1, &actuators_bebop.i2c_trans, actuators_bebop.i2c_trans.slave_addr, 1);
113  } else if (actuators_bebop.i2c_trans.buf[10] == 4 && autopilot_get_motors_on()) {
114  // Send the commands
115  actuators_bebop.i2c_trans.buf[0] = ACTUATORS_BEBOP_SET_REF_SPEED;
116  actuators_bebop.i2c_trans.buf[1] = actuators_bebop.rpm_ref[0] >> 8;
117  actuators_bebop.i2c_trans.buf[2] = actuators_bebop.rpm_ref[0] & 0xFF;
118  actuators_bebop.i2c_trans.buf[3] = actuators_bebop.rpm_ref[1] >> 8;
119  actuators_bebop.i2c_trans.buf[4] = actuators_bebop.rpm_ref[1] & 0xFF;
120  actuators_bebop.i2c_trans.buf[5] = actuators_bebop.rpm_ref[2] >> 8;
121  actuators_bebop.i2c_trans.buf[6] = actuators_bebop.rpm_ref[2] & 0xFF;
122  actuators_bebop.i2c_trans.buf[7] = actuators_bebop.rpm_ref[3] >> 8;
123  actuators_bebop.i2c_trans.buf[8] = actuators_bebop.rpm_ref[3] & 0xFF;
124  actuators_bebop.i2c_trans.buf[9] = 0x00; //UNK enable security?
125 #pragma GCC diagnostic push
126 #pragma GCC diagnostic ignored "-Wcast-qual"
127  actuators_bebop.i2c_trans.buf[10] = actuators_bebop_checksum((uint8_t *)actuators_bebop.i2c_trans.buf, 9);
128 #pragma GCC diagnostic pop
129  i2c_blocking_transmit(&i2c1, &actuators_bebop.i2c_trans, actuators_bebop.i2c_trans.slave_addr, 11);
130  }
131 
132  // Update the LEDs
133  if (actuators_bebop.led != (led_hw_values & 0x3)) {
134  actuators_bebop.i2c_trans.buf[0] = ACTUATORS_BEBOP_TOGGLE_GPIO;
135  actuators_bebop.i2c_trans.buf[1] = (led_hw_values & 0x3);
136  i2c_blocking_transmit(&i2c1, &actuators_bebop.i2c_trans, actuators_bebop.i2c_trans.slave_addr, 2);
137 
138  actuators_bebop.led = led_hw_values & 0x3;
139  }
140  // Send ABI message
141  struct rpm_act_t rpm_message[4];
142  for (int i=0;i<4;i++) {
143 #ifdef SERVOS_BEBOP_OFFSET
144  rpm_message[i].actuator_idx = SERVOS_BEBOP_OFFSET + i;
145 #else
146  rpm_message[i].actuator_idx = SERVOS_DEFAULT_OFFSET + i;
147 #endif
148  rpm_message[i].rpm = actuators_bebop.rpm_obs[i];
149  }
150  AbiSendMsgRPM(RPM_SENSOR_ID, rpm_message, 4);
151 }

153 static uint8_t actuators_bebop_checksum(uint8_t *bytes, uint8_t size)
154 {
155  uint8_t checksum = 0;
156  for (int i = 0; i < size; i++) {
157  checksum = checksum ^ bytes[i];
158  }
159 
160  return checksum;
161 }
