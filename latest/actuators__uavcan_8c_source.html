104 static uint8_t old_idx = 0;
105 static uint8_t esc_idx = 0;
106 static struct actuators_uavcan_telem_t *actuators_uavcan_next_telem(void) {
107  // Randomness added for multiple transport devices
108  uint8_t add_idx = 0;
109  if (rand_uniform() > 0.02) {
110  add_idx = 1;
111  }
112 
113  // Find the next set telemetry
114  uint8_t offset = 0;
115 #ifdef SERVOS_UAVCAN1_NB
116  for(uint8_t i = esc_idx - offset; i < SERVOS_UAVCAN1_NB; i++) {
117  if(uavcan1_telem[i].set) {
118  old_idx = i + offset;
119  esc_idx = i + offset + add_idx;
120  return &uavcan1_telem[i];
121  } else {
122  esc_idx = i + offset + 1;
123  }
124  }
125  offset += SERVOS_UAVCAN1_NB;
126 #endif
127 #ifdef SERVOS_UAVCAN2_NB
128  for(uint8_t i = esc_idx - offset; i < SERVOS_UAVCAN2_NB; i++) {
129  if(uavcan2_telem[i].set) {
130  old_idx = i + offset;
131  esc_idx = i + offset + add_idx;
132  return &uavcan2_telem[i];
133  } else {
134  esc_idx = i + offset + 1;
135  }
136  }
137  offset += SERVOS_UAVCAN2_NB;
138 #endif
139 
140  // Going round or no telemetry found
141  esc_idx = 0;
142  return NULL;
143 }

145 static void actuators_uavcan_send_esc(struct transport_tx *trans, struct link_device *dev)
146 {
147  // Find the correct telemetry (Try twice if going round)
148  struct actuators_uavcan_telem_t *telem = actuators_uavcan_next_telem();
149  if(telem == NULL) {
150  telem = actuators_uavcan_next_telem();
151  }
152 
153  // Safety check (no telemetry received 2 times)
154  if (telem == NULL) {
155  return;
156  }
157 
158  float power = telem->current * telem->voltage;
159  float rpm = telem->rpm;
160  float energy = telem->energy;
161  pprz_msg_send_ESC(trans, dev, AC_ID, &telem->current, &electrical.vsupply, &power,
162  &rpm, &telem->voltage, &energy, &telem->temperature, &telem->temperature_dev, &telem->node_id, &old_idx);
163 }

169 static void actuators_uavcan_esc_status_cb(struct uavcan_iface_t *iface, CanardRxTransfer *transfer)
170 {
171  uint8_t esc_idx;
172  uint16_t tmp_float;
173 
174  struct actuators_uavcan_telem_t *telem = NULL;
175  uint8_t max_id = 0;
176 #ifdef SERVOS_UAVCAN1_NB
177  if (iface == &uavcan1) {
178  telem = uavcan1_telem;
179  max_id = SERVOS_UAVCAN1_NB;
180  }
181 #endif
182 #ifdef SERVOS_UAVCAN2_NB
183  if (iface == &uavcan2) {
184  telem = uavcan2_telem;
185  max_id = SERVOS_UAVCAN2_NB;
186  }
187 #endif
188 
189  canardDecodeScalar(transfer, 105, 5, false, (void *)&esc_idx);
190  //Could not find the right interface
191  if (esc_idx >= max_id || telem == NULL || max_id == 0) {
192  return;
193  }
194  telem[esc_idx].set = true;
195  telem[esc_idx].node_id = transfer->source_node_id;
196  telem[esc_idx].timestamp = get_sys_time_float();
197  canardDecodeScalar(transfer, 0, 32, false, (void *)&telem[esc_idx].energy);
198  canardDecodeScalar(transfer, 32, 16, true, (void *)&tmp_float);
199  telem[esc_idx].voltage = canardConvertFloat16ToNativeFloat(tmp_float);
200  canardDecodeScalar(transfer, 48, 16, true, (void *)&tmp_float);
201  telem[esc_idx].current = canardConvertFloat16ToNativeFloat(tmp_float);
202  canardDecodeScalar(transfer, 64, 16, true, (void *)&tmp_float);
203  telem[esc_idx].temperature = canardConvertFloat16ToNativeFloat(tmp_float) - 273.15;
204  canardDecodeScalar(transfer, 80, 18, true, (void *)&telem[esc_idx].rpm);
205 
206 #if UAVCAN_ACTUATORS_USE_CURRENT
207  // Update total current
208  electrical.current = 0;
209 #ifdef SERVOS_UAVCAN1_NB
210  for (uint8_t i = 0; i < SERVOS_UAVCAN1_NB; ++i) {
211  electrical.current += uavcan1_telem[i].current;
212  }
213 #endif
214 #ifdef SERVOS_UAVCAN2_NB
215  for (uint8_t i = 0; i < SERVOS_UAVCAN2_NB; ++i) {
216  electrical.current += uavcan2_telem[i].current;
217  }
218 #endif
219 #endif
220 
221  // Feedback ABI RPM messages
222 #ifdef SERVOS_UAVCAN1_NB
223  if (iface == &uavcan1) {
224  struct act_feedback_t feedback;
225  feedback.idx = SERVOS_UAVCAN1_OFFSET + esc_idx;
226  feedback.rpm = telem[esc_idx].rpm;
227  feedback.set.rpm = true;
228 
229  // Send ABI message
230  AbiSendMsgACT_FEEDBACK(ACT_FEEDBACK_UAVCAN_ID, &feedback, 1);
231  }
232 #endif
233 #ifdef SERVOS_UAVCAN2_NB
234  if (iface == &uavcan2) {
235  struct act_feedback_t feedback;
236  feedback.idx = SERVOS_UAVCAN2_OFFSET + esc_idx;
237  feedback.rpm = telem[esc_idx].rpm;
238  feedback.set.rpm = true;
239 
240  // Send ABI message
241  AbiSendMsgACT_FEEDBACK(ACT_FEEDBACK_UAVCAN_ID, &feedback, 1);
242  }
243 #endif
244 }

343 void actuators_uavcan_init(struct uavcan_iface_t *iface __attribute__((unused)))
344 {
345  // Check if not already initialized (for multiple interfaces, needs only 1)
346  if (actuators_uavcan_initialized) { return; }
347 
348  // Bind uavcan ESC_STATUS message from EQUIPMENT
349  uavcan_bind(UAVCAN_EQUIPMENT_ESC_STATUS_ID, UAVCAN_EQUIPMENT_ESC_STATUS_SIGNATURE, &esc_status_ev,
350  &actuators_uavcan_esc_status_cb);
351  // Bind uavcan ACTUATOR_STATUS message from EQUIPMENT
352  uavcan_bind(UAVCAN_EQUIPMENT_ACTUATOR_STATUS_ID, UAVCAN_EQUIPMENT_ACTUATOR_STATUS_SIGNATURE, &actuator_status_ev,
353  &actuators_uavcan_actuator_status_cb);
354  // Bind uavcan DEVICE_TEMPERATURE message from EQUIPMENT
355  uavcan_bind(UAVCAN_EQUIPMENT_DEVICE_TEMPERATURE_ID, UAVCAN_EQUIPMENT_DEVICE_TEMPERATURE_SIGNATURE, &device_temperature_ev,
356  &actuators_uavcan_device_temperature_cb);
357 
358  // Configure telemetry
359 #if PERIODIC_TELEMETRY
360  register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_ESC, actuators_uavcan_send_esc);
361 #endif
362 
363  // Set default to not set
364 #ifdef SERVOS_UAVCAN1CMD_NB
365  for(uint8_t i = 0; i < SERVOS_UAVCAN1CMD_NB; i++)
366  actuators_uavcan1cmd_values[i] = UAVCAN_CMD_UNUSED;
367 #endif
368 #ifdef SERVOS_UAVCAN2CMD_NB
369  for(uint8_t i = 0; i < SERVOS_UAVCAN2CMD_NB; i++)
370  actuators_uavcan2cmd_values[i] = UAVCAN_CMD_UNUSED;
371 #endif
372 
373  // Set initialization
374  actuators_uavcan_initialized = true;
375 
376  // Initialize Random (for telemetry)
377  init_random();
378 }

383 void actuators_uavcan_commit(struct uavcan_iface_t *iface, int16_t *values, uint8_t nb)
384 {
385  uint8_t buffer[UAVCAN_EQUIPMENT_ESC_RAWCOMMAND_MAX_SIZE];
386  uint32_t offset = 0;
387 
388  // Encode the values as 14-bit signed integers
389  for (uint8_t i = 0; i < nb; i++) {
390  canardEncodeScalar(buffer, offset, 14, (void *)&values[i]);
391  offset += 14;
392  }
393 
394  // Broadcast the raw command message on the interface
395  uavcan_broadcast(iface, UAVCAN_EQUIPMENT_ESC_RAWCOMMAND_SIGNATURE, UAVCAN_EQUIPMENT_ESC_RAWCOMMAND_ID,
396  CANARD_TRANSFER_PRIORITY_HIGH, buffer, (offset + 7) / 8);
397 }

402 void actuators_uavcan_cmd_commit(struct uavcan_iface_t *iface, int16_t *values, uint8_t nb)
403 {
404  uint8_t buffer[UAVCAN_EQUIPMENT_ACTUATOR_ARRAYCOMMAND_MAX_SIZE];
405  uint32_t offset = 0;
406  uint8_t command_type = 0; // 0:UNITLESS, 1:meter or radian, 2:N or Nm, 3:m/s or rad/s
407 
408  // Encode the values for each command
409  for (uint8_t i = 0; i < nb; i++) {
410  // Skip unused commands
411  if(values[i] == UAVCAN_CMD_UNUSED || values[i] < MIN_PPRZ || values[i] > MAX_PPRZ)
412  continue;
413 
414  // Set the command id
415  canardEncodeScalar(buffer, offset, 8, (void*)&i); // 255
416  offset += 8;
417 
418  // Set the command type
419  canardEncodeScalar(buffer, offset, 8, (void*)&command_type); // 255
420  offset += 8;
421 
422  // Set the command value
423  uint16_t tmp_float = canardConvertNativeFloatToFloat16((float)values[i] / (float)MAX_PPRZ);
424  canardEncodeScalar(buffer, offset, 16, (void*)&tmp_float); // 32767
425  offset += 16;
426  }
427 
428  // Broadcast the raw command message on the interface
429  uavcan_broadcast(iface, UAVCAN_EQUIPMENT_ACTUATOR_ARRAYCOMMAND_SIGNATURE, UAVCAN_EQUIPMENT_ACTUATOR_ARRAYCOMMAND_ID,
430  CANARD_TRANSFER_PRIORITY_HIGH, buffer, (offset + 7) / 8);
431 }
