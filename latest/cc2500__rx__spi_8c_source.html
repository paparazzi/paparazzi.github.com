181/*
182 * Returns true if the RX has received new data.
183 * Called from updateRx in rx.c, updateRx called from taskUpdateRxCheck.
184 * If taskUpdateRxCheck returns true, then taskUpdateRxMain will shortly be called.
185 */
186static uint8_t rxSpiFrameStatus(rxRuntimeConfig_t *rxRuntimeConfig)
187{
188 UNUSED(rxRuntimeConfig);
189
190 uint8_t status = RX_FRAME_PENDING;
191
192 rx_spi_received_e result = protocolDataReceived(rxSpiPayload);
193
194 if (result & RX_SPI_RECEIVED_DATA) {
195 rxSpiNewPacketAvailable = true;
196 status = RX_FRAME_COMPLETE;
197 }
198
199 if (result & RX_SPI_ROCESSING_REQUIRED) {
200 status |= RX_FRAME_PROCESSING_REQUIRED;
201 }
202
203 return status;
204}
205
206static bool rxSpiProcessFrame(const rxRuntimeConfig_t *rxRuntimeConfig)
207{
208 UNUSED(rxRuntimeConfig);
209
210 if (protocolProcessFrame) {
211 rx_spi_received_e result = protocolProcessFrame(rxSpiPayload);
212
213 if (result & RX_SPI_RECEIVED_DATA) {
214 rxSpiNewPacketAvailable = true;
215 }
216
217 if (result & RX_SPI_ROCESSING_REQUIRED) {
218 return false;
219 }
220 }
221
222 return true;
223}
