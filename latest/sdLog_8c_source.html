191 SdioError sdLogInit(uint32_t *freeSpaceInKo)
192 {
193  DWORD clusters = 0;
194  FATFS *fsp = NULL;
195  nbBytesWritten = 0;
196 
197  // if init is already done, return ERROR
198  if (sdLogThd != NULL) {
199  *freeSpaceInKo = 0;
200  return storageStatus = SDLOG_WAS_LAUNCHED;
201  }
202 
203 #ifdef SDLOG_NEED_QUEUE
204  msgqueue_init(&messagesQueue, &HEAP_DEFAULT, queMbBuffer, SDLOG_QUEUE_BUCKETS);
205 #endif
206 
207  if (!sdc_lld_is_card_inserted(NULL)) {
208  return storageStatus = SDLOG_NOCARD;
209  }
210 
211 
212  sdio_connect();
213  chThdSleepMilliseconds(10);
214  sdio_disconnect();
215 
216  if (sdio_connect() == FALSE) {
217  return storageStatus = SDLOG_NOCARD;
218  }
219 
220 #if FFCONF_DEF < 8000
221  FRESULT rc = f_mount(0, &fatfs);
222 #else
223  FRESULT rc = f_mount(&fatfs, "/", 1);
224 #endif
225 
226  if (rc != FR_OK) {
227  return storageStatus = SDLOG_FATFS_ERROR;
228  }
229 
230  if (freeSpaceInKo != NULL) {
231  f_getfree("/", &clusters, &fsp);
232  *freeSpaceInKo = clusters * (uint32_t)fatfs.csize / 2;
233  }
234 
235 #ifdef SDLOG_NEED_QUEUE
236  for (uint8_t i = 0; i < SDLOG_NUM_FILES; i++) {
237  fileDes[i].inUse = fileDes[i].tagAtClose = false;
238  fileDes[i].writeByteCache = NULL;
239  fileDes[i].writeByteSeek = 0;
240  }
241 
242  storageStatus = sdLoglaunchThread();
243 #else
244  storageStatus = SDLOG_OK;
245 #endif
246  return storageStatus;
247 
248 }

272 SdioError sdLogOpenLog(FileDes *fd, const char *directoryName, const char *prefix,
273  const uint32_t autoFlushPeriod, const bool appendTagAtClose,
274  const size_t sizeInMo, const bool preallocate, char *fileName, const size_t nameLength)
275 {
276  FRESULT rc; /* fatfs result code */
277  SdioError sde = SDLOG_OK; /* sdio result code */
278  //DIR dir; /* Directory object */
279  //FILINFO fno; /* File information object */
280 
281  /* local file descriptor
282  using fd is a bad idea since fd is set before fatfs objets are coherents
283  in a multithreaded application where sdLogXXX are done before sdLogWriteLog is done
284  we can have a race condition. setting fd only when fatfs files are opened resolve the problem
285  */
286  FileDes ldf;
287 
288  sde = getNextFIL(&ldf);
289  if (sde != SDLOG_OK) {
290  return storageStatus = sde;
291  }
292 
293  sde = getFileName(prefix, directoryName, fileName, nameLength, +1);
294  if (sde != SDLOG_OK) {
295  // sd card is not inserted, so logging task can be deleted
296  fileDes[ldf].inUse = false;
297  return storageStatus = SDLOG_FATFS_ERROR;
298  }
299 
300 
301  rc = f_open(&fileDes[ldf].fil, fileName, FA_WRITE | FA_CREATE_ALWAYS);
302  if (rc) {
303  fileDes[ldf].inUse = false;
304  return storageStatus = SDLOG_FATFS_ERROR;
305  } else {
306  fileDes[ldf].tagAtClose = appendTagAtClose;
307  fileDes[ldf].autoFlushPeriod = autoFlushPeriod;
308  fileDes[ldf].lastFlushTs = 0;
309  sde = sdLogExpandLogFile(ldf, sizeInMo, preallocate);
310  }
311 
312  *fd = ldf;
313  return storageStatus = sde;
314 }

316 SdioError sdLogCloseAllLogs(bool flush)
317 {
318  FRESULT rc = 0; /* Result code */
319 
320  // do not flush what is in ram, close as soon as possible
321  if (flush == false) {
322  UINT bw;
323  // stop worker thread then close file
324  cleanQueue(true);
325  sdLogStopThread();
326 
327  for (FileDes fd = 0; fd < SDLOG_NUM_FILES; fd++) {
328  if (fileDes[fd].inUse) {
329  FIL *fo = &fileDes[fd].fil;
330  if (fileDes[fd].tagAtClose) {
331  f_write(fo, "\r\nEND_OF_LOG\r\n", 14, &bw);
332  nbBytesWritten += bw;
333  }
334  FRESULT trc = f_close(fo);
335  fileDes[fd].inUse = false;
336  if (!rc) {
337  rc = trc;
338  }
339  }
340  }
341 
342  if (rc) {
343  return storageStatus = SDLOG_FATFS_ERROR;
344  }
345 
346  // flush ram buffer then close
347  } else { // flush == true
348  if (sdLogThd == NULL) {
349  // something goes wrong, log thread is no more working
350  return storageStatus = SDLOG_NOTHREAD;
351  }
352 
353  // queue flush + close order, then stop worker thread
354  for (FileDes fd = 0; fd < SDLOG_NUM_FILES; fd++) {
355  if (fileDes[fd].inUse) {
356  flushWriteByteBuffer(fd);
357  sdLogCloseLog(fd);
358  }
359  }
360 
361  LogMessage *lm = tlsf_malloc_r(&HEAP_DEFAULT, sizeof(LogMessage));
362  if (lm == NULL) {
363  return storageStatus = SDLOG_MEMFULL;
364  }
365 
366  lm->op.fcntl = FCNTL_EXIT;
367 
368  if (msgqueue_send(&messagesQueue, lm, sizeof(LogMessage), MsgQueue_REGULAR) < 0) {
369  return storageStatus = SDLOG_QUEUEFULL;
370  } else {
371  chThdWait(sdLogThd);
372  sdLogThd = NULL;
373  }
374 
375  }
376  return storageStatus = SDLOG_OK;
377 }

416 SdioError sdLogWriteLog(const FileDes fd, const char *fmt, ...)
417 {
418  FD_CHECK(fd);
419 
420  const SdioError status = flushWriteByteBuffer(fd);
421  storageStatus = status;
422  if (status != SDLOG_OK) {
423  return status;
424  }
425 
426  va_list ap;
427  va_start(ap, fmt);
428 
429  LogMessage *lm = tlsf_malloc_r(&HEAP_DEFAULT, LOG_MESSAGE_PREBUF_LEN);
430  if (lm == NULL) {
431  va_end(ap);
432  return storageStatus = SDLOG_MEMFULL;
433  }
434 
435  lm->op.fcntl = FCNTL_WRITE;
436  lm->op.fd = fd & 0x1f;
437 
438  chvsnprintf(lm->mess, SDLOG_MAX_MESSAGE_LEN - 1, fmt, ap);
439  lm->mess[SDLOG_MAX_MESSAGE_LEN - 1] = 0;
440  va_end(ap);
441 
442  const size_t msgLen = logMessageLen(lm);
443  lm = tlsf_realloc_r(&HEAP_DEFAULT, lm, msgLen);
444  if (lm == NULL) {
445  return storageStatus = SDLOG_MEMFULL;
446  }
447 
448  if (msgqueue_send(&messagesQueue, lm, msgLen, MsgQueue_REGULAR) < 0) {
449  return storageStatus = SDLOG_QUEUEFULL;
450  }
451 
452  return SDLOG_OK;
453 }

624 SdioError sdLogWriteByte(const FileDes fd, const uint8_t value)
625 {
626  FD_CHECK(fd);
627  LogMessage *lm;
628 
629  if (fileDes[fd].writeByteCache == NULL) {
630  lm = tlsf_malloc_r(&HEAP_DEFAULT, sizeof(LogMessage) + WRITE_BYTE_CACHE_SIZE);
631  if (lm == NULL) {
632  return storageStatus = SDLOG_MEMFULL;
633  }
634 
635  lm->op.fcntl = FCNTL_WRITE;
636  lm->op.fd = fd & 0x1f;
637 
638  fileDes[fd].writeByteCache = lm;
639  fileDes[fd].writeByteSeek = 0;
640  } else {
641  lm = fileDes[fd].writeByteCache;
642  }
643 
644  lm->mess[fileDes[fd].writeByteSeek++] = value;
645 
646  if (fileDes[fd].writeByteSeek == WRITE_BYTE_CACHE_SIZE) {
647  const SdioError status = flushWriteByteBuffer(fd);
648  // message is not sent so allocated buffer will not be released by receiver side
649  // instead of freeing buffer, we just reset cache seek.
650  if (status == SDLOG_QUEUEFULL) {
651  fileDes[fd].writeByteSeek = 0;
652  return storageStatus = SDLOG_QUEUEFULL;
653  }
654  }
655  return storageStatus = SDLOG_OK;
656 }
