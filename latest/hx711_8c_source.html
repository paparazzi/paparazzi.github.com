110void hx711_init(void)
111{
112 /*----------------
113 * Configure GPIO
114 *----------------*/
115 gpio_setup_pin_af(HX711_PWM_PORT, HX711_PWM_PIN, HX711_PWM_AF, true);
116 pwmcfg.channels[HX711_PWM_CHANNEL].mode = PWM_OUTPUT_ACTIVE_HIGH;
117 pwmcfg.channels[HX711_PWM_CHANNEL].callback = pwmpcb;
118
119 for(uint8_t i = 0; i < HX711_DEVICES_NB; i++) {
120 gpio_setup_input(hx711.devices[i].data_port, hx711.devices[i].data_pin);
121 }
122
123 /*---------------
124 * Configure PWM
125 *---------------*/
126 pwmStart(&HX711_PWM_DRIVER, &pwmcfg);
127 hx711.busy = false;
128 hx711.measurement_ready = false;
129 hx711.read_bit_idx = 0;
130}

135void hx711_event(void)
136{
137 // If busy reading data, return
138 if(hx711.busy)
139 return;
140
141 // Check if we have a measurement to read
142 if(hx711.measurement_ready) {
143 int32_t values[HX711_DEVICES_NB];
144 for(uint8_t i = 0; i < HX711_DEVICES_NB; i++) {
145 values[i] = hx711.devices[i].measurement;
146 }
147
148 AbiSendMsgFORCE_SENSOR(HX711_FORCE_SENSOR_ID, get_sys_time_usec(), HX711_DEVICES_NB, values);
149
150 // Send down for debug
151 float raw_val[HX711_DEVICES_NB];
152 for(uint8_t i = 0; i < HX711_DEVICES_NB; i++) {
153 raw_val[i] = values[i];
154 }
155
156#if HX711_DEBUG
157 float current_time = get_sys_time_float();
158 float freq = 1 / (current_time - hx711_meas_time);
159 hx711_meas_time = current_time;
160
161 char hx711_str[10];
162 char hx711_freq_str[15];
163 int hx711_len = snprintf(hx711_str, sizeof(hx711_str), "HX711");
164 int hx711_freq_len = snprintf(hx711_freq_str, sizeof(hx711_freq_str), "HX711 freq");
165 RunOnceEvery(10, {
166 DOWNLINK_SEND_DEBUG_VECT(DefaultChannel, DefaultDevice, hx711_len, hx711_str, HX711_DEVICES_NB, raw_val);
167 });
168 RunOnceEvery(15, {
169 DOWNLINK_SEND_DEBUG_VECT(DefaultChannel, DefaultDevice, hx711_freq_len, hx711_freq_str, 1, &freq);
170 });
171 pprz_msg_send_DEBUG_VECT(&pprzlog_tp.trans_tx, &flightrecorder_sdlog.device, AC_ID, hx711_len, hx711_str, HX711_DEVICES_NB, raw_val);
172 pprz_msg_send_DEBUG_VECT(&pprzlog_tp.trans_tx, &flightrecorder_sdlog.device, AC_ID, hx711_freq_len, hx711_freq_str, 1, &freq);
173#endif
174
175 hx711.measurement_ready = false;
176 }
177
178 // Check if all data pins are low, ready to read data
179 for(uint8_t i = 0; i < HX711_DEVICES_NB; i++) {
180 if(gpio_get(hx711.devices[i].data_port, hx711.devices[i].data_pin) != 0)
181 return;
182 }
183
184 // Start reading data
185 chSysLock();
186 hx711.busy = true;
187 hx711.measurement_ready = false;
188 hx711.read_bit_idx = 0;
189 pwmEnableChannel(&HX711_PWM_DRIVER, HX711_PWM_CHANNEL, HX711_PERIOD/2); // 50% duty cycle
190 pwmEnableChannelNotification(&HX711_PWM_DRIVER, HX711_PWM_CHANNEL);
191 chSysUnlock();
192}

197static void pwmpcb(PWMDriver *pwmp __attribute__((unused))) {
198 chSysLockFromISR();
199
200 /* In case we where not reading just stop */
201 if(!hx711.busy) {
202 pwmDisableChannelNotificationI(&HX711_PWM_DRIVER, HX711_PWM_CHANNEL);
203 pwmDisableChannelI(&HX711_PWM_DRIVER, HX711_PWM_CHANNEL);
204 chSysUnlockFromISR();
205 return;
206 }
207
208 /* Parse the measurements for all devices */
209 if(hx711.read_bit_idx < 24) {
210 for(uint8_t i = 0; i < HX711_DEVICES_NB; i++) {
211 volatile struct hx711_dev_t *dev = &hx711.devices[i];
212
213 // Reset or bitshift
214 if(hx711.read_bit_idx == 0)
215 dev->measurement = 0;
216 else
217 dev->measurement <<= 1;
218
219 // Read the bit
220 if(gpio_get(dev->data_port, dev->data_pin) != 0)
221 dev->measurement |= 1;
222
223 // Fix 2s complement for 32 bit signed integer from 24 bit
224 if(dev->measurement & (1 << 23)) {
225 dev->measurement |= 0xFF000000;
226 }
227 }
228 }
229 hx711.read_bit_idx++;
230
231 /* Send gain and finish reading data */
232 if(hx711.read_bit_idx >= (24 + HX711_GAIN)) {
233 pwmDisableChannelNotificationI(&HX711_PWM_DRIVER, HX711_PWM_CHANNEL);
234 pwmDisableChannelI(&HX711_PWM_DRIVER, HX711_PWM_CHANNEL);
235 hx711.busy = false;
236 hx711.measurement_ready = true;
237 chSysUnlockFromISR();
238 return;
239 }
240
241 chSysUnlockFromISR();
242}
