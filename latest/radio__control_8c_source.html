62void radio_control_periodic_task(void)
63{
64 static uint8_t _1Hz;
65 _1Hz++;
66
67 if (_1Hz >= 60) {
68 _1Hz = 0;
69 radio_control.frame_rate = radio_control.frame_cpt;
70 radio_control.frame_cpt = 0;
71 }
72
73 if (radio_control.time_since_last_frame >= RC_REALLY_LOST_TIME) {
74 radio_control.status = RC_REALLY_LOST;
75 } else {
76 if (radio_control.time_since_last_frame >= RC_LOST_TIME) {
77 radio_control.status = RC_LOST;
78 radio_control.radio_ok_cpt = RC_OK_CPT;
79 }
80 radio_control.time_since_last_frame++;
81 }
82
83#if defined(RADIO_CONTROL_LED)
84 if (radio_control.status == RC_OK) {
85 LED_ON(RADIO_CONTROL_LED);
86 } else {
87 LED_OFF(RADIO_CONTROL_LED);
88 }
89#endif
