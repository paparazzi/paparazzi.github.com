72void baro_event(void)
73{
74 bmp085_event(&baro_bmp085);
75
76 if (baro_bmp085.data_available) {
77 uint32_t now_ts = get_sys_time_usec();
78 float pressure = (float)baro_bmp085.pressure;
79 AbiSendMsgBARO_ABS(BARO_BOARD_SENDER_ID, now_ts, pressure);
80 float temp = baro_bmp085.temperature / 10.0f;
81 AbiSendMsgTEMPERATURE(BARO_BOARD_SENDER_ID, temp);
82 baro_bmp085.data_available = false;
83#ifdef BARO_LED
84 RunOnceEvery(10, LED_TOGGLE(BARO_LED));
85#endif
86 }
