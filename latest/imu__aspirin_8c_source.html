68 void imu_aspirin_init(void)
69 {
70  imu_aspirin.accel_valid = false;
71  imu_aspirin.gyro_valid = false;
72  imu_aspirin.mag_valid = false;
73 
74  /* Set accel configuration */
75  adxl345_spi_init(&imu_aspirin.acc_adxl, &(ASPIRIN_SPI_DEV), ASPIRIN_SPI_SLAVE_IDX);
76  // set the data rate
77  imu_aspirin.acc_adxl.config.rate = ASPIRIN_ACCEL_RATE;
79  //imu_aspirin.acc_adxl.config.drdy_int_enable = true;
80 
81  /* Gyro configuration and initalization */
82  itg3200_init(&imu_aspirin.gyro_itg, &(ASPIRIN_I2C_DEV), ITG3200_ADDR);
83  /* change the default config */
84  // Aspirin sample rate divider defaults to 533Hz
85  imu_aspirin.gyro_itg.config.smplrt_div = ASPIRIN_GYRO_SMPLRT_DIV;
86  // aspirin defaults to 8kHz internal with 256Hz low pass
87  imu_aspirin.gyro_itg.config.dlpf_cfg = ASPIRIN_GYRO_LOWPASS;
88 
90  /* interrupt on data ready, idle high, latch until read any register */
91  //itg_conf.int_cfg = (0x01 | (0x1<<4) | (0x1<<5) | 0x01<<7);
92 
93  /* initialize mag and set default options */
94  hmc58xx_init(&imu_aspirin.mag_hmc, &(ASPIRIN_I2C_DEV), HMC58XX_ADDR);
95 #ifdef IMU_ASPIRIN_VERSION_1_0
96  imu_aspirin.mag_hmc.type = HMC_TYPE_5843;

98 
99 #if ASPIRIN_ARCH_INDEP

107 void imu_aspirin_periodic(void)
108 {
109  adxl345_spi_periodic(&imu_aspirin.acc_adxl);
110 
111  // Start reading the latest gyroscope data
112  itg3200_periodic(&imu_aspirin.gyro_itg);

118 void imu_aspirin_event(void)
119 {
120  uint32_t now_ts = get_sys_time_usec();
121 
122  adxl345_spi_event(&imu_aspirin.acc_adxl);
123  if (imu_aspirin.acc_adxl.data_available) {
124  VECT3_COPY(imu.accel_unscaled, imu_aspirin.acc_adxl.data.vect);
125  imu_aspirin.acc_adxl.data_available = false;
126  imu_aspirin.accel_valid = true;
127  }
128 
129  /* If the itg3200 I2C transaction has succeeded: convert the data */
130  itg3200_event(&imu_aspirin.gyro_itg);
131  if (imu_aspirin.gyro_itg.data_available) {
132  RATES_COPY(imu.gyro_unscaled, imu_aspirin.gyro_itg.data.rates);
133  imu_aspirin.gyro_itg.data_available = false;
134  imu_aspirin.gyro_valid = true;
135  }
136 
137  /* HMC58XX event task */
138  hmc58xx_event(&imu_aspirin.mag_hmc);
139  if (imu_aspirin.mag_hmc.data_available) {
140 #ifdef IMU_ASPIRIN_VERSION_1_0
141  VECT3_COPY(imu.mag_unscaled, imu_aspirin.mag_hmc.data.vect);
142 #else // aspirin 1.5 with hmc5883
143  imu.mag_unscaled.x = imu_aspirin.mag_hmc.data.vect.y;
144  imu.mag_unscaled.y = -imu_aspirin.mag_hmc.data.vect.x;
145  imu.mag_unscaled.z = imu_aspirin.mag_hmc.data.vect.z;
146 #endif
147  imu_aspirin.mag_hmc.data_available = false;
148  imu_aspirin.mag_valid = true;
149  }
150 
151  if (imu_aspirin.gyro_valid) {
152  imu_aspirin.gyro_valid = false;
153  imu_scale_gyro(&imu);
154  AbiSendMsgIMU_GYRO_INT32(IMU_ASPIRIN_ID, now_ts, &imu.gyro);
155  }
156  if (imu_aspirin.accel_valid) {
157  imu_aspirin.accel_valid = false;
158  imu_scale_accel(&imu);
159  AbiSendMsgIMU_ACCEL_INT32(IMU_ASPIRIN_ID, now_ts, &imu.accel);
160  }
161  if (imu_aspirin.mag_valid) {
162  imu_aspirin.mag_valid = false;
163  imu_scale_mag(&imu);
164  AbiSendMsgIMU_MAG_INT32(IMU_ASPIRIN_ID, now_ts, &imu.mag);
165  }
166 }
