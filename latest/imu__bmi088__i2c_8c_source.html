113void imu_bmi088_event(void)
114{
115 uint32_t now_ts = get_sys_time_usec();
116
117 // If the BMI088 I2C transaction has succeeded: convert the data
118 bmi088_i2c_event(&imu_bmi088.bmi);
119
120 if (imu_bmi088.bmi.gyro_available) {
121 struct Int32Rates rates = {
122 IMU_BMI088_X_SIGN *(int32_t)(imu_bmi088.bmi.data_rates.value[IMU_BMI088_CHAN_X]),
123 IMU_BMI088_Y_SIGN *(int32_t)(imu_bmi088.bmi.data_rates.value[IMU_BMI088_CHAN_Y]),
124 IMU_BMI088_Z_SIGN *(int32_t)(imu_bmi088.bmi.data_rates.value[IMU_BMI088_CHAN_Z])
125 };
126
127 imu_bmi088.bmi.gyro_available = false;
128 AbiSendMsgIMU_GYRO_RAW(IMU_BMI088_ID, now_ts, &rates, 1, imu_bmi088.bmi.config.gyro_samplerate, NAN);
129 }
130 if (imu_bmi088.bmi.accel_available) {
131 // set channel order
132 struct Int32Vect3 accel = {
133 IMU_BMI088_X_SIGN *(int32_t)(imu_bmi088.bmi.data_accel.value[IMU_BMI088_CHAN_X]),
134 IMU_BMI088_Y_SIGN *(int32_t)(imu_bmi088.bmi.data_accel.value[IMU_BMI088_CHAN_Y]),
135 IMU_BMI088_Z_SIGN *(int32_t)(imu_bmi088.bmi.data_accel.value[IMU_BMI088_CHAN_Z])
136 };
137
138 imu_bmi088.bmi.accel_available = false;
139 AbiSendMsgIMU_ACCEL_RAW(IMU_BMI088_ID, now_ts, &accel, 1, imu_bmi088.bmi.config.accel_samplerate, NAN);
140 }
141}
