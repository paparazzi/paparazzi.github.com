71 void baro_event(void)
72 {

74 
75  if (baro_bmp085.data_available) {
76  uint32_t now_ts = get_sys_time_usec();
77  float pressure = (float)baro_bmp085.pressure;
78  AbiSendMsgBARO_ABS(BARO_BOARD_SENDER_ID, now_ts, pressure);
79  float temp = baro_bmp085.temperature / 10.0f;
80  AbiSendMsgTEMPERATURE(BARO_BOARD_SENDER_ID, temp);
81  baro_bmp085.data_available = false;
