177void OA_update()
178{
179 float v_x = 0;
180 float v_y = 0;
181
182 if (opti_speed_flag == 1) {
183 //rotation_vector.psi = stateGetNedToBodyEulers_f()->psi;
184 //opti_speed = stateGetSpeedNed_f();
185
186 //Transform to body frame.
187 //float_rmat_of_eulers_312(&T, &rotation_vector)Attractforce_goal_send;
188 //MAT33_VECT3_MUL(*opti_speed, T, *opti_speed);
189
190 // Calculate the speed in body frame
191 struct FloatVect2 speed_cur;
192 float psi = stateGetNedToBodyEulers_f()->psi;
193 float s_psi = sin(psi);
194 float c_psi = cos(psi);
195 speed_cur.x = c_psi * stateGetSpeedNed_f()->x + s_psi * stateGetSpeedNed_f()->y;
196 speed_cur.y = -s_psi * stateGetSpeedNed_f()->x + c_psi * stateGetSpeedNed_f()->y;
197
198 opti_speed_read.x = speed_cur.x * 100;
199 opti_speed_read.y = speed_cur.y * 100;
200
201 //set result_vel
202 v_x = speed_cur.y * 100;
203 v_y = speed_cur.x * 100;
204 } else {
205 }
206
207 if (OA_method_flag == NO_OBSTACLE_AVOIDANCE) {
208 /* Calculate the error if we have enough flow */
209 opticflow_stab.desired_vx = 0;
210 opticflow_stab.desired_vy = 0;
211
212 err_vx = opticflow_stab.desired_vx - v_x;
213 err_vy = opticflow_stab.desired_vy - v_y;
214
215 /* Calculate the integrated errors (TODO: bound??) */
216 opticflow_stab.err_vx_int += err_vx / 100;
217 opticflow_stab.err_vy_int += err_vy / 100;
218
219 /* Calculate the commands */
220 opticflow_stab.cmd.phi = opticflow_stab.phi_pgain * err_vx / 100
221 + opticflow_stab.phi_igain * opticflow_stab.err_vx_int;
222 opticflow_stab.cmd.theta = -(opticflow_stab.theta_pgain * err_vy / 100
223 + opticflow_stab.theta_igain * opticflow_stab.err_vy_int);
224
225 /* Bound the roll and pitch commands */
226 BoundAbs(opticflow_stab.cmd.phi, CMD_OF_SAT);
227 BoundAbs(opticflow_stab.cmd.theta, CMD_OF_SAT);
228 }
229
230 if (OA_method_flag == PINGPONG) {
231 opticflow_stab.cmd.phi = ANGLE_BFP_OF_REAL(ref_roll);
232 opticflow_stab.cmd.theta = ANGLE_BFP_OF_REAL(ref_pitch);
233 }
234
235 if (OA_method_flag == 2) {
236 Total_Kan_x = r_dot_new;
237 Total_Kan_y = speed_pot;
238
239 alpha_fil = 0.1;
240
241 yaw_rate = (int32_t)(alpha_fil * ANGLE_BFP_OF_REAL(r_dot_new));
242 opticflow_stab.cmd.psi = stateGetNedToBodyEulers_i()->psi + yaw_rate;
243
244 INT32_ANGLE_NORMALIZE(opticflow_stab.cmd.psi);
245
246 opticflow_stab.desired_vx = 0;
247 opticflow_stab.desired_vy = speed_pot;
248
249 /* Calculate the error if we have enough flow */
250
251 err_vx = opticflow_stab.desired_vx - v_x;
252 err_vy = opticflow_stab.desired_vy - v_y;
253
254 /* Calculate the integrated errors (TODO: bound??) */
255 opticflow_stab.err_vx_int += err_vx / 100;
256 opticflow_stab.err_vy_int += err_vy / 100;
257
258 /* Calculate the commands */
259 opticflow_stab.cmd.phi = opticflow_stab.phi_pgain * err_vx / 100
260 + opticflow_stab.phi_igain * opticflow_stab.err_vx_int;
261
262 opticflow_stab.cmd.theta = -(opticflow_stab.theta_pgain * err_vy / 100
263 + opticflow_stab.theta_igain * opticflow_stab.err_vy_int);
264
265 /* Bound the roll and pitch commands */
266 BoundAbs(opticflow_stab.cmd.phi, CMD_OF_SAT);
267 BoundAbs(opticflow_stab.cmd.theta, CMD_OF_SAT);
268
269 }
270 if (OA_method_flag == POT_HEADING) {
271 new_heading = ref_pitch;
272
273 opticflow_stab.desired_vx = sin(new_heading) * speed_pot * 100;
274 opticflow_stab.desired_vy = cos(new_heading) * speed_pot * 100;
275
276 /* Calculate the error if we have enough flow */
277 err_vx = opticflow_stab.desired_vx - v_x;
278 err_vy = opticflow_stab.desired_vy - v_y;
279
280
281 /* Calculate the integrated errors (TODO: bound??) */
282 opticflow_stab.err_vx_int += err_vx / 100;
283 opticflow_stab.err_vy_int += err_vy / 100;
284
285 /* Calculate the commands */
286 opticflow_stab.cmd.phi = opticflow_stab.phi_pgain * err_vx / 100
287 + opticflow_stab.phi_igain * opticflow_stab.err_vx_int;
288 opticflow_stab.cmd.theta = -(opticflow_stab.theta_pgain * err_vy / 100
289 + opticflow_stab.theta_igain * opticflow_stab.err_vy_int);
290
291 /* Bound the roll and pitch commands */
292 BoundAbs(opticflow_stab.cmd.phi, CMD_OF_SAT);
293 BoundAbs(opticflow_stab.cmd.theta, CMD_OF_SAT)
294
295 }
296
297 if (OA_method_flag == VECTOR || OA_method_flag == SAFETYZONE || OA_method_flag == LOGICBASED) {
298 //vector field method
299 float v_desired_total;
300
301 Total_Kan_x = ref_pitch;
302 Total_Kan_y = ref_roll;
303
304 opticflow_stab.desired_vx = alpha_fil *
305 Total_Kan_y; //alpha_fil*(Repulsionforce_Kan.y+Attractforce_goal.y) + result->vel_x;
306 opticflow_stab.desired_vy = alpha_fil *
307 Total_Kan_x; //alpha_fil*(Repulsionforce_Kan.x+Attractforce_goal.x) + result->vel_y;
308
309 v_desired_total = sqrt(opticflow_stab.desired_vx * opticflow_stab.desired_vx + opticflow_stab.desired_vy *
310 opticflow_stab.desired_vy);
311
312 if (v_desired_total >= vref_max) {
313 opticflow_stab.desired_vx = (vref_max / v_desired_total) * opticflow_stab.desired_vx;
314 opticflow_stab.desired_vy = (vref_max / v_desired_total) * opticflow_stab.desired_vy;
315 }
316
317 /* Calculate the error if we have enough flow */
318
319 //alpha_fil needs to be tuned!
320 err_vx = opticflow_stab.desired_vx - v_x;
321 err_vy = opticflow_stab.desired_vy - v_y;
322
323 /* Calculate the integrated errors (TODO: bound??) */
324 opticflow_stab.err_vx_int += err_vx / 100;
325 opticflow_stab.err_vy_int += err_vy / 100;
326
327 /* Calculate the commands */
328 opticflow_stab.cmd.phi = opticflow_stab.phi_pgain * err_vx / 100
329 + opticflow_stab.phi_igain * opticflow_stab.err_vx_int;
330 opticflow_stab.cmd.theta = -(opticflow_stab.theta_pgain * err_vy / 100
331 + opticflow_stab.theta_igain * opticflow_stab.err_vy_int);
332
333 /* Bound the roll and pitch commands */
334 BoundAbs(opticflow_stab.cmd.phi, CMD_OF_SAT);
335 BoundAbs(opticflow_stab.cmd.theta, CMD_OF_SAT);
336
337 }
