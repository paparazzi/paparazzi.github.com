140static void force_sensor_autoset_offset(void)
141{
142 if (!force_sensor_valid) {
143 return;
144 }
145
146 for (uint8_t i = 0; i < force_sensor.count; i++) {
147 force_sensor.offsets[i] += force_sensor.values_filt[i];
148 }
149}
150
151static void force_sensor_cb(uint8_t sender_id UNUSED, uint32_t stamp UNUSED, uint8_t count, int32_t *values)
152{
153 uint8_t n = count;
154 if (n > FORCE_SENSOR_MAX_NB) {
155 n = FORCE_SENSOR_MAX_NB;
156 }
157
158 force_sensor.count = n;
159 force_sensor_valid = true;
160 for (uint8_t i = 0; i < n; i++) {
161 force_sensor.values_filt[i] = update_median_filter_i(&force_sensor_filt[i], values[i] - force_sensor.offsets[i]);
162 }
163}
164
165static bool force_sensor_detect_ground(void)
166{
167 if (!force_sensor_valid) {
168 return false;
169 }
170
171 for (uint8_t i = 0; i < force_sensor.count; i++) {
172 if (fabsf((float)force_sensor.values_filt[i]) > force_sensor_ground_threshold) {
173 return true;
174 }
175 }
176 return false;
177}
178#endif

199void ground_detect_init()
200{
201 // Initialize the ground detection status
202 ground_detect_status.value = 0;
203 ground_detect_values.speed_down = 0.0;
204 ground_detect_values.spec_thrust_down = 0.0;
205 ground_detect_values.agl_dist_value_filtered = 0.0;
206 force_sensor.count = 1; // Needed for force sensor field in ground detect message
207 for (uint8_t i = 0; i < FORCE_SENSOR_MAX_NB; i++) {
208 force_sensor.offsets[i] = 0;
209 force_sensor.values_filt[i] = 0;
210 }
211
212 float tau = 1.0 / (2.0 * M_PI * GROUND_DETECT_FILT_FREQ);
213 float sample_time = 1.0 / PERIODIC_FREQUENCY;
214 init_butterworth_2_low_pass(&ground_detect_values.accel_down_filt, tau, sample_time, 0.0);
215
216#if GROUND_DETECT_USE_FORCE_SENSOR
217 for (uint8_t i = 0; i < FORCE_SENSOR_MAX_NB; i++) {
218 init_median_filter_i(&force_sensor_filt[i], GROUND_DETECT_FORCE_SENSOR_MEDIAN_FILT_SIZE);
219 }
220 force_sensor.count = 0;
221 force_sensor_valid = false;
222 AbiBindMsgFORCE_SENSOR(GROUND_DETECT_FORCE_SENSOR_ID, &force_sensor_ev, force_sensor_cb);
223#endif
224
225#if PERIODIC_TELEMETRY
226 register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_GROUND_DETECT, send_ground_detect);
227#endif
228}

241void ground_detect_periodic()
242{
243
244 // Evaluate thrust given (less than hover thrust)
245 // Use the control effectiveness in thrust in order to estimate the thrust delivered (only works for multicopters)
246 float specific_thrust = 0.0; // m/s2
247
248#if GROUND_DETECT_USE_INDI_THRUST
249 uint8_t i;
250 for (i = 0; i < INDI_NUM_ACT; i++) {
251 specific_thrust += actuator_state_filt_vect[i] * g1g2[3][i] * -((int32_t) act_is_servo[i] - 1);
252 }
253#endif
254
255 // vertical component
256 struct FloatRMat *ned_to_body_rmat = stateGetNedToBodyRMat_f();
257 ground_detect_values.spec_thrust_down = ned_to_body_rmat->m[8] * specific_thrust;
258
259 // Evaluate vertical speed (close to zero, not at terminal velocity)
260 ground_detect_values.speed_down = stateGetSpeedNed_f()->z;
261
262 // Detect free fall (to be done, rearm?)
263
264 // Detect noise level (to be done)
265
266 // Detect ground based on AND of some triggers
267#if GROUND_DETECT_USE_FORCE_SENSOR
268 ground_detect_status.force_sensor_trigger = force_sensor_detect_ground();
269#else
270 ground_detect_status.force_sensor_trigger = false;
271#endif
272
273 ground_detect_status.vspeed_trigger = (fabsf(ground_detect_values.speed_down) < GROUND_DETECT_VERTICAL_SPEED_THRESHOLD)? 1:0;
274
275#if GROUND_DETECT_USE_INDI_THRUST
276 ground_detect_status.spec_thrust_trigger = (ground_detect_values.spec_thrust_down > GROUND_DETECT_SPECIFIC_THRUST_THRESHOLD)? 1:0;
277#else
278 ground_detect_status.spec_thrust_trigger = false;
279#endif
280
281 ground_detect_status.accel_filt_trigger = (fabsf(ground_detect_values.accel_down_filt.o[0]) < GROUND_DETECT_VERTICAL_ACCEL_THRESHOLD)? 1:0;
282
283#if GROUND_DETECT_USE_AGL_DIST
284 ground_detect_status.agl_trigger = (agl_dist_valid && (agl_dist_value_filtered < GROUND_DETECT_AGL_MIN_VALUE))? 1:0;
285 ground_detect_values.agl_dist_value_filtered = agl_dist_value_filtered;
286#else
287 ground_detect_status.agl_trigger = false;
288#endif
289
290 int trigger_sum = 0;
291 for(uint8_t i = 0; i < 16; i++) trigger_sum += (ground_detect_status.value >> i) & 0x1;
292
293 if (trigger_sum >= GROUND_DETECT_NUM_TRIGGERS) {
294 counter += 1;
295 if (counter > GROUND_DETECT_COUNTER_TRIGGER) {
296 ground_detected = true;
297
298 if (disarm_on_not_in_flight) {
299 autopilot_set_motors_on(false);
300 disarm_on_not_in_flight = false;
301 }
302 }
303 } else {
304 ground_detected = false;
305 counter = 0;
306 }
307}
