97void nps_autopilot_run_step(double time)
98{
99
100 nps_electrical_run_step(time);
101
102#if RADIO_CONTROL && !RADIO_CONTROL_TYPE_DATALINK
103 if (nps_radio_control_available(time)) {
104 radio_control_feed();
105 main_ap_event();
106 }
107#endif
108
109 if (nps_sensors_gyro_available()) {
110 imu_feed_gyro_accel();
111 main_ap_event();
112 }
113
114 if (nps_sensors_mag_available()) {
115 imu_feed_mag();
116 main_ap_event();
117 }
118
119 if (nps_sensors_baro_available()) {
120 uint32_t now_ts = get_sys_time_usec();
121 float pressure = (float) sensors.baro.value;
122 AbiSendMsgBARO_ABS(BARO_SIM_SENDER_ID, now_ts, pressure);
123 main_ap_event();
124 }
125
126 if (nps_sensors_temperature_available()) {
127 AbiSendMsgTEMPERATURE(BARO_SIM_SENDER_ID, (float)sensors.temp.value);
128 main_ap_event();
129 }
130
131#if USE_AIRSPEED
132 if (nps_sensors_airspeed_available()) {
133 AbiSendMsgAIRSPEED(AIRSPEED_NPS_ID, (float)sensors.airspeed.value);
134 main_ap_event();
135 }
136#endif
137
138#if USE_SONAR
139 if (nps_sensors_sonar_available()) {
140 uint32_t now_ts = get_sys_time_usec();
141 float dist = (float) sensors.sonar.value;
142 if (dist >= 0.0) {
143 AbiSendMsgAGL(AGL_SONAR_NPS_ID, now_ts, dist);
144 }
145
146#ifdef SENSOR_SYNC_SEND_SONAR
147 uint16_t foo = 0;
148 DOWNLINK_SEND_SONAR(DefaultChannel, DefaultDevice, &foo, &dist);
149#endif
150
151 main_ap_event();
152 }
153#endif
154
155#if USE_GPS
156 if (nps_sensors_gps_available()) {
157 gps_feed_value();
158 main_ap_event();
159 }
160#endif
161
162 if (nps_bypass_ahrs) {
163 sim_overwrite_ahrs();
164 }
165
166 if (nps_bypass_ins) {
167 sim_overwrite_ins();
168 }
169
170 main_ap_periodic();
171
172 /* scale final motor commands to 0-1 for feeding the fdm */
173 for (uint8_t i = 0; i < NPS_COMMANDS_NB; i++) {
174#if NPS_NO_MOTOR_MIXING
175 #if NPS_USE_COMMANDS
176 commands[i] = autopilot_get_motors_on() ? commands[i] : 0;
177 nps_autopilot.commands[i] = (double)commands[i] / MAX_PPRZ;
178 #else
179 actuators_pprz[i] = autopilot_get_motors_on() ? actuators_pprz[i] : 0;
180 nps_autopilot.commands[i] = (double)actuators_pprz[i] / MAX_PPRZ;
181 #endif
182#else
183 nps_autopilot.commands[i] = (double)motor_mixing.commands[i] / MAX_PPRZ;
184#endif
185 }
186}
