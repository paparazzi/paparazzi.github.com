112static void rc_cb(uint8_t sender_id UNUSED, struct RadioControl *rc)
113{
114 /* used in RC_DIRECT directly and as saturation in CLIMB and HOVER */
115 guidance_v.rc_delta_t = (int32_t)rc->values[RADIO_THROTTLE];
116
117 /* used in RC_CLIMB */
118 guidance_v.rc_zd_sp = (MAX_PPRZ / 2) - (int32_t)rc->values[RADIO_THROTTLE];
119 DeadBand(guidance_v.rc_zd_sp, GUIDANCE_V_CLIMB_RC_DEADBAND);
120
121 static const int32_t climb_scale = ABS(SPEED_BFP_OF_REAL(GUIDANCE_V_MAX_RC_CLIMB_SPEED) /
122 (MAX_PPRZ / 2 - GUIDANCE_V_CLIMB_RC_DEADBAND));
123 static const int32_t descent_scale = ABS(SPEED_BFP_OF_REAL(GUIDANCE_V_MAX_RC_DESCENT_SPEED) /
124 (MAX_PPRZ / 2 - GUIDANCE_V_CLIMB_RC_DEADBAND));
125
126 if (guidance_v.rc_zd_sp > 0) {
127 guidance_v.rc_zd_sp *= descent_scale;
128 } else {
129 guidance_v.rc_zd_sp *= climb_scale;
130 }
131}

133void guidance_v_mode_changed(uint8_t new_mode)
134{
135
136 if (new_mode == guidance_v.mode) {
137 return;
138 }
139
140 switch (new_mode) {
141 case GUIDANCE_V_MODE_GUIDED:
142 case GUIDANCE_V_MODE_HOVER:
143 guidance_v_guided_enter();
144 break;
145
146 case GUIDANCE_V_MODE_RC_CLIMB:
147 case GUIDANCE_V_MODE_CLIMB:
148 guidance_v.zd_sp = 0;
149 guidance_v.zdd_sp = 0;
150 /* Falls through. */
151 case GUIDANCE_V_MODE_NAV:
152 guidance_v_run_enter();
153 GuidanceVSetRef(stateGetPositionNed_i()->z, stateGetSpeedNed_i()->z, 0);
154 break;
155
156 default:
157 break;
158
159 }
160
161 guidance_v.mode = new_mode;
162

173static int32_t get_vertical_thrust_coeff(void)
174{
175 // cos(30°) = 0.8660254
176 static const int32_t max_bank_coef = BFP_OF_REAL(0.8660254f, INT32_TRIG_FRAC);
177
178 struct Int32RMat *att = stateGetNedToBodyRMat_i();
179 /* thrust vector:
180 * int32_rmat_vmult(&thrust_vect, &att, &zaxis)
181 * same as last colum of rmat with INT32_TRIG_FRAC
182 * struct Int32Vect thrust_vect = {att.m[2], att.m[5], att.m[8]};
183 *
184 * Angle between two vectors v1 and v2:
185 * angle = acos(dot(v1, v2) / (norm(v1) * norm(v2)))
186 * since here both are already of unit length:
187 * angle = acos(dot(v1, v2))
188 * since we we want the cosine of the angle we simply need
189 * thrust_coeff = dot(v1, v2)
190 * also can be simplified considering: v1 is zaxis with (0,0,1)
191 * dot(v1, v2) = v1.z * v2.z = v2.z
192 */
193 int32_t coef = att->m[8];
194 if (coef < max_bank_coef) {
195 coef = max_bank_coef;
196 }
197 return coef;
198}

201void guidance_v_thrust_adapt(bool in_flight)
202{
203 guidance_v.thrust_coeff = get_vertical_thrust_coeff();
204
205 if (in_flight) {
206 /* Only run adaptive throttle estimation if we are in flight and
207 * the desired vertical velocity (zd) was updated (i.e. we ran hover_loop before).
208 * This means that the estimation is not updated when using direct throttle commands.
209 *
210 * FIXME... SATURATIONS NOT TAKEN INTO ACCOUNT, AKA SUPERVISION and co
211 * FIXME get this out of here !!!! and don't get stab cmd directly
212 */
213 if (desired_zd_updated) {
214 int32_t vertical_thrust = (stabilization.cmd[COMMAND_THRUST] * guidance_v.thrust_coeff) >> INT32_TRIG_FRAC;
215 gv_adapt_run(stateGetAccelNed_i()->z, vertical_thrust, guidance_v.zd_ref);
216 }
217 } else {
218 /* reset estimate while not in_flight */
219 gv_adapt_init();
220 }
221}

223struct ThrustSetpoint guidance_v_run(bool in_flight)
224{
225 guidance_v_thrust_adapt(in_flight);
226
227 /* reset flag indicating if desired zd was updated */
228 desired_zd_updated = false;
229 /* reset setpoint */
230 THRUST_SP_SET_ZERO(guidance_v.thrust);
231
232 switch (guidance_v.mode) {
233
234 case GUIDANCE_V_MODE_RC_DIRECT:
235 guidance_v.z_sp = stateGetPositionNed_i()->z; // for display only
236 guidance_v.thrust = th_sp_from_thrust_i(guidance_v.rc_delta_t, THRUST_AXIS_Z);
237 break;
238
239 case GUIDANCE_V_MODE_RC_CLIMB:
240 guidance_v.zd_sp = guidance_v.rc_zd_sp;
241 gv_update_ref_from_zd_sp(guidance_v.zd_sp, stateGetPositionNed_i()->z);
242 guidance_v.thrust = guidance_v_run_speed(in_flight, &guidance_v);
243 break;
244
245 case GUIDANCE_V_MODE_CLIMB:
246 gv_update_ref_from_zd_sp(guidance_v.zd_sp, stateGetPositionNed_i()->z);
247 guidance_v.thrust = guidance_v_run_speed(in_flight, &guidance_v);
248 break;
249
250 case GUIDANCE_V_MODE_HOVER:
251 guidance_v_guided_mode = GUIDANCE_V_GUIDED_MODE_ZHOLD;
252 /* Falls through. */
253 case GUIDANCE_V_MODE_GUIDED:
254 guidance_v.thrust = guidance_v_guided_run(in_flight);
255 break;
256
257 case GUIDANCE_V_MODE_NAV: {
258 guidance_v.thrust = guidance_v_from_nav(in_flight);
259 break;
260 }
261
262 default:
263 break;
264 }
265 return guidance_v.thrust;
266}

292void guidance_v_update_ref(void)
293{
294 /* convert our reference to generic representation */
295 int64_t tmp = gv_z_ref >> (GV_Z_REF_FRAC - INT32_POS_FRAC);
296 guidance_v.z_ref = (int32_t)tmp;
297 guidance_v.zd_ref = gv_zd_ref << (INT32_SPEED_FRAC - GV_ZD_REF_FRAC);
298 guidance_v.zdd_ref = gv_zdd_ref << (INT32_ACCEL_FRAC - GV_ZDD_REF_FRAC);

303struct ThrustSetpoint guidance_v_from_nav(bool in_flight)
304{
305 struct ThrustSetpoint sp;
306 THRUST_SP_SET_ZERO(sp);
307 if (nav.vertical_mode == NAV_VERTICAL_MODE_ALT) {
308 guidance_v.z_sp = -POS_BFP_OF_REAL(nav.nav_altitude);
309 guidance_v.zd_sp = 0;
310 guidance_v.zdd_sp = 0;
311 gv_update_ref_from_z_sp(guidance_v.z_sp);
312 guidance_v_update_ref();
313 sp = guidance_v_run_pos(in_flight, &guidance_v);
314 } else if (nav.vertical_mode == NAV_VERTICAL_MODE_CLIMB) {
315 guidance_v.z_sp = stateGetPositionNed_i()->z;
316 guidance_v.zd_sp = -SPEED_BFP_OF_REAL(nav.climb);
317 guidance_v.zdd_sp = 0;
318 gv_update_ref_from_zd_sp(guidance_v.zd_sp, stateGetPositionNed_i()->z);
319 guidance_v_update_ref();
320 sp = guidance_v_run_speed(in_flight, &guidance_v);
321 } else if (nav.vertical_mode == NAV_VERTICAL_MODE_MANUAL) {
322 guidance_v.z_sp = stateGetPositionNed_i()->z;
323 guidance_v.zd_sp = stateGetSpeedNed_i()->z;
324 guidance_v.zdd_sp = 0;
325 GuidanceVSetRef(guidance_v.z_sp, guidance_v.zd_sp, 0);
326 guidance_v_run_enter();
327 sp = th_sp_from_thrust_i((int32_t)nav.throttle, THRUST_AXIS_Z);
328 } else if (nav.vertical_mode == NAV_VERTICAL_MODE_GUIDED) {
329 sp = guidance_v_guided_run(in_flight);
330 }
331 return sp;
332}

344struct ThrustSetpoint guidance_v_guided_run(bool in_flight)
345{
346 struct ThrustSetpoint sp;
347 THRUST_SP_SET_ZERO(sp);
348 switch(guidance_v_guided_mode)
349 {
350 case GUIDANCE_V_GUIDED_MODE_ZHOLD:
351 // Altitude Hold
352 guidance_v.zd_sp = 0;
353 gv_update_ref_from_z_sp(guidance_v.z_sp);
354 guidance_v_update_ref();
355 sp = guidance_v_run_pos(in_flight, &guidance_v);
356 break;
357 case GUIDANCE_V_GUIDED_MODE_CLIMB:
358 // Climb
359 gv_update_ref_from_zd_sp(guidance_v.zd_sp, stateGetPositionNed_i()->z);
360 guidance_v_update_ref();
361 sp = guidance_v_run_speed(in_flight, &guidance_v);
362 break;
363 case GUIDANCE_V_GUIDED_MODE_THROTTLE:
364 // Throttle
365 guidance_v.z_sp = stateGetPositionNed_i()->z; // for display only
366 sp = th_sp_from_thrust_i(guidance_v.th_sp, THRUST_AXIS_Z);
367 break;
368 case GUIDANCE_V_GUIDED_MODE_ALL:
369 // update full reference
370 guidance_v_set_ref(guidance_v.z_sp, guidance_v.zd_sp, guidance_v.zdd_sp);
371 sp = guidance_v_run_pos(in_flight, &guidance_v);
372 break;
373 default:
374 break;
375 }
376 return sp;
377}
