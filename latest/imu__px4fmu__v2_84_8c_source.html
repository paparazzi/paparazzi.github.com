67 void imu_px4_event(void) {
68 
69  uint32_t now_ts = get_sys_time_usec();
70 
71  /* L3GD20 event task */
72  l3gd20_spi_event(&imu_px4.l3g);
73  if (imu_px4.l3g.data_available) {
74  //the p and q seem to be swapped on the Pixhawk board compared to the acc
75  imu.gyro_unscaled.p = imu_px4.l3g.data_rates.rates.q;
76  imu.gyro_unscaled.q = -imu_px4.l3g.data_rates.rates.p;
77  imu.gyro_unscaled.r = imu_px4.l3g.data_rates.rates.r;
78  imu_px4.l3g.data_available = FALSE;
79  imu_scale_gyro(&imu);
80  AbiSendMsgIMU_GYRO_INT32(IMU_PX4_ID, now_ts, &imu.gyro);
81  }
82 
83  /* LSM303d event task */
84  lsm303d_spi_event(&imu_px4.lsm_acc);
85  if (imu_px4.lsm_acc.data_available_acc) {
86  VECT3_COPY(imu.accel_unscaled, imu_px4.lsm_acc.data_accel.vect);
87  imu_px4.lsm_acc.data_available_acc = FALSE;
88  imu_scale_accel(&imu);
89  AbiSendMsgIMU_ACCEL_INT32(IMU_PX4_ID, now_ts, &imu.accel);
90  }
91 #if !IMU_PX4_DISABLE_MAG
92  lsm303d_spi_event(&imu_px4.lsm_mag);
93  if (imu_px4.lsm_mag.data_available_mag) {
94  VECT3_COPY(imu.mag_unscaled, imu_px4.lsm_mag.data_mag.vect);
95  imu_px4.lsm_mag.data_available_mag = FALSE;
96  imu_scale_mag(&imu);
97  AbiSendMsgIMU_MAG_INT32(IMU_PX4_ID, now_ts, &imu.mag);
98  }
99 #endif
100 
101 }
