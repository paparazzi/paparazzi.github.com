41void imu_px4_init(void) {
42 /* L3GD20 gyro init */
43 /* initialize gyro and set default options */
44 l3gd20_spi_init(&imu_px4.l3g, &IMU_PX4FMU_SPI_DEV, IMU_L3G_SPI_SLAVE_IDX);
45
46 /* LSM303d acc init */
47 lsm303d_spi_init(&imu_px4.lsm_acc, &IMU_PX4FMU_SPI_DEV, IMU_LSM_SPI_SLAVE_IDX, LSM303D_TARGET_ACC);
48
49 // Set the default scaling
50 const struct FloatRates gyro_scale_f = {L3GD20_SENS_2000, L3GD20_SENS_2000, L3GD20_SENS_2000};
51 const struct FloatVect3 accel_scale_f = {LSM303D_ACCEL_SENS_16G, LSM303D_ACCEL_SENS_16G, LSM303D_ACCEL_SENS_16G};
52 imu_set_defaults_gyro(IMU_PX4_ID, NULL, NULL, &gyro_scale_f);
53 imu_set_defaults_accel(IMU_PX4_ID, NULL, NULL, &accel_scale_f);
54
55#if !IMU_PX4_DISABLE_MAG
56 /* LSM303d mag init */
57 lsm303d_spi_init(&imu_px4.lsm_mag, &IMU_PX4FMU_SPI_DEV, IMU_LSM_SPI_SLAVE_IDX, LSM303D_TARGET_MAG);
58#endif

60}

62void imu_px4_periodic(void) {
63 l3gd20_spi_periodic(&imu_px4.l3g);
64 lsm303d_spi_periodic(&imu_px4.lsm_acc);
65
66#if !IMU_PX4_DISABLE_MAG
67 /* Read magneto's every 10 times of main freq
68 * at ~50Hz (main loop for rotorcraft: 512Hz)
69 */
70 RunOnceEvery(10, lsm303d_spi_periodic(&imu_px4.lsm_mag));
71#endif
72}

74void imu_px4_event(void) {
75
76 uint32_t now_ts = get_sys_time_usec();
77
78 /* L3GD20 event task */
79 l3gd20_spi_event(&imu_px4.l3g);
80 if (imu_px4.l3g.data_available) {
81 //the p and q seem to be swapped on the Pixhawk board compared to the acc
82 struct Int32Rates gyro = {
83 imu_px4.l3g.data_rates.rates.q,
84 -imu_px4.l3g.data_rates.rates.p,
85 imu_px4.l3g.data_rates.rates.r
86 };
87 imu_px4.l3g.data_available = FALSE;
88 AbiSendMsgIMU_GYRO_RAW(IMU_PX4_ID, now_ts, &gyro, 1, IMU_PX4_PERIODIC_FREQ, NAN);
89 }
90
91 /* LSM303d event task */
92 lsm303d_spi_event(&imu_px4.lsm_acc);
93 if (imu_px4.lsm_acc.data_available_acc) {
94 struct Int32Vect3 accel;
95 VECT3_COPY(accel, imu_px4.lsm_acc.data_accel.vect);
96 AbiSendMsgIMU_ACCEL_RAW(IMU_PX4_ID, now_ts, &accel, 1, IMU_PX4_PERIODIC_FREQ, NAN);
97 imu_px4.lsm_acc.data_available_acc = FALSE;
98 }
99#if !IMU_PX4_DISABLE_MAG
100 lsm303d_spi_event(&imu_px4.lsm_mag);
101 if (imu_px4.lsm_mag.data_available_mag) {
102 struct Int32Vect3 mag;
103 VECT3_ASSIGN(mag,
104 imu_px4.lsm_mag.data_mag.vect.x,
105 imu_px4.lsm_mag.data_mag.vect.y,
106 imu_px4.lsm_mag.data_mag.vect.z);
107 AbiSendMsgIMU_MAG_RAW(IMU_PX4_ID, now_ts, &mag);
108 imu_px4.lsm_mag.data_available_mag = FALSE;
109 }
110#endif
111
112}
