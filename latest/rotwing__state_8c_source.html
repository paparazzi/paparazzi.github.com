167void rotwing_state_init(void)
168{
169 // Initialize rotwing state
170 rotwing_state.state = ROTWING_STATE_FORCE_HOVER; // For takeoff
171 rotwing_state.nav_state = ROTWING_STATE_FORCE_HOVER;
172 rotwing_state.hover_motors_enabled = true;
173 rotwing_state.fw_min_airspeed = ROTWING_FW_MIN_AIRSPEED;
174 rotwing_state.cruise_airspeed = ROTWING_FW_CRUISE_AIRSPEED;
175 rotwing_state.sp_skew_angle_deg = 0;
176 rotwing_state.meas_skew_angle_deg = 0;
177 rotwing_state.meas_skew_angle_time = 0;
178 rotwing_state.skew_cmd = 0;
179 rotwing_state.force_skew = false;
180 for (int i = 0; i < 5; i++) {
181 rotwing_state.meas_rpm[i] = 0;
182 rotwing_state.meas_rpm_time[i] = 0;
183 }
184 rotwing_state.fail_skew_angle = false;
185 rotwing_state.fail_hover_motor = false;
186 rotwing_state.fail_pusher_motor = false;
187 rotwing_state.ref_model_skew_angle_deg = 0;
188 // Bind ABI messages
189 AbiBindMsgACT_FEEDBACK(ROTWING_STATE_ACT_FEEDBACK_ID, &rotwing_state_feedback_ev, rotwing_state_feedback_cb);
190
191#if PERIODIC_TELEMETRY
192 register_periodic_telemetry(DefaultPeriodic, PPRZ_MSG_ID_ROTATING_WING_STATE, send_rotating_wing_state);
193#endif
194
195 /* Register preflight checks */
196#if PREFLIGHT_CHECKS
197 preflight_check_register(&rotwing_state_skew_pfc, rotwing_state_skew_preflight);
198#endif
199}

205bool rotwing_state_hover_motors_idling(void) {
206 static float last_idle_time = 0;
207 // Check if hover motors are idling and reset timer
208 if(stabilization.cmd[COMMAND_THRUST] > ROTWING_QUAD_IDLE_MIN_THRUST) {
209 last_idle_time = get_sys_time_float();
210 }
211
212 // Check if we exceeded the idle timeout
213 if(get_sys_time_float() - last_idle_time > ROTWING_QUAD_IDLE_TIMEOUT) {
214 return true;
215 }
216 return false;
217}

219void rotwing_state_periodic(void)
220{
221 /* Get some genericly used variables */
222 static float last_stall_time = 0;
223 float current_time = get_sys_time_float();
224 float meas_airspeed = stateGetAirspeed_f();
225 float meas_skew_angle = rotwing_state.meas_skew_angle_deg;
226 Bound(meas_skew_angle, 0, 90); // Bound to prevent errors
227
228 if(meas_airspeed > rotwing_state.fw_min_airspeed) {
229 last_stall_time = current_time;
230 }
231
232 /* Override modes if flying with RC */
233 rotwing_state.state = rotwing_state.nav_state;
234 if (autopilot.mode == AP_MODE_FAILSAFE) {
235 rotwing_state.state = ROTWING_STATE_FORCE_HOVER;
236 }
237 else if (guidance_h.mode == GUIDANCE_H_MODE_NONE) {
238 // Kill mode
239 if(stabilization.mode == STABILIZATION_MODE_NONE) {
240 rotwing_state.state = ROTWING_STATE_FORCE_HOVER;
241 }
242 // ATT and ATT_FWD
243 else if(stabilization.mode == STABILIZATION_MODE_ATTITUDE) {
244 if (stabilization.att_submode == STABILIZATION_ATT_SUBMODE_FORWARD) {
245 rotwing_state.state = ROTWING_STATE_FORCE_FW;
246 } else {
247 rotwing_state.state = ROTWING_STATE_FORCE_HOVER;
248 }
249 }
250 }
251
252 /* Handle the quad motors on/off state */
253 if(rotwing_state.state == ROTWING_STATE_FORCE_HOVER || rotwing_state.state == ROTWING_STATE_REQUEST_HOVER) {
254 rotwing_state.hover_motors_enabled = true;
255 }
256 else if((current_time - last_stall_time) < ROTWING_FW_STALL_TIMEOUT && rotwing_state_hover_motors_idling() && rotwing_state_pusher_motor_running() && meas_skew_angle >= ROTWING_FW_SKEW_ANGLE
257 && (gi_unbounded_airspeed_sp >= rotwing_state.fw_min_airspeed || rotwing_state.state != ROTWING_STATE_FREE)) {
258 rotwing_state.hover_motors_enabled = false;
259 }
260 else {
261 rotwing_state.hover_motors_enabled = true;
262 }
263
264
265 /* Calculate min/max airspeed bounds based on skew angle */
266 float skew_min_airspeed = ROTWING_FW_QUAD_MIN_AIRSPEED * (meas_skew_angle - ROTWING_MIN_AIRSPEED_SLOPE_START_ANGLE) / (90.f - ROTWING_MIN_AIRSPEED_SLOPE_START_ANGLE);
267 float skew_max_airspeed = ROTWING_QUAD_MAX_AIRSPEED + (ROTWING_FW_MAX_AIRSPEED - ROTWING_QUAD_MAX_AIRSPEED) * meas_skew_angle / ROTWING_FW_SKEW_ANGLE;
268 Bound(skew_min_airspeed, 0, rotwing_state.fw_min_airspeed);
269 Bound(skew_max_airspeed, ROTWING_QUAD_MAX_AIRSPEED, ROTWING_FW_MAX_AIRSPEED);
270
271 if(!rotwing_state_hover_motors_running() || !rotwing_state.hover_motors_enabled) {
272 skew_min_airspeed = rotwing_state.fw_min_airspeed;
273 skew_max_airspeed = ROTWING_FW_MAX_AIRSPEED;
274 }
275
276
277 /* Handle the skew angle setpoint */
278 if (rotwing_state.force_skew) {
279 // Do nothing
280 }
281 else if(rotwing_state.state == ROTWING_STATE_FORCE_HOVER) {
282 rotwing_state.sp_skew_angle_deg = 0.f;
283 }
284 else if(!rotwing_state_hover_motors_running() || !rotwing_state.hover_motors_enabled || rotwing_state.state == ROTWING_STATE_FORCE_FW) {
285 rotwing_state.sp_skew_angle_deg = 90.f;
286 }
287 else if(!rotwing_state_pusher_motor_running()) {
288 rotwing_state.sp_skew_angle_deg = 0.f;
289 }
290 else if(rotwing_state.state == ROTWING_STATE_REQUEST_HOVER && meas_skew_angle <= (ROTWING_SKEW_ANGLE_STEP + ROTWING_SKEW_BACK_MARGIN)) {
291 rotwing_state.sp_skew_angle_deg = 0.f;
292 }
293 else {
294 // SKEWING function based on Vair
295 if (meas_airspeed < ROTWING_SKEW_DOWN_AIRSPEED) {
296 rotwing_state.sp_skew_angle_deg = 0.f;
297 } else if (meas_airspeed < ROTWING_SKEW_UP_AIRSPEED) {
298 // Hysteresis do nothing
299 } else if (meas_airspeed < ROTWING_QUAD_MAX_AIRSPEED) {
300 rotwing_state.sp_skew_angle_deg = ROTWING_SKEW_ANGLE_STEP;
301 } else {
302 rotwing_state.sp_skew_angle_deg = ((meas_airspeed - ROTWING_QUAD_MAX_AIRSPEED)) / (rotwing_state.fw_min_airspeed - ROTWING_QUAD_MAX_AIRSPEED) * (90.f - ROTWING_SKEW_ANGLE_STEP) + ROTWING_SKEW_ANGLE_STEP;
303 }
304 }
305 Bound(rotwing_state.sp_skew_angle_deg, 0.f, 90.f);
306
307
308 /* Handle the airspeed bounding */
309 Bound(rotwing_state.cruise_airspeed, rotwing_state.fw_min_airspeed, ROTWING_FW_MAX_AIRSPEED);
310 if((!rotwing_state_hover_motors_running() && rotwing_state.state != ROTWING_STATE_FORCE_HOVER) || rotwing_state.state == ROTWING_STATE_FORCE_FW) {
311 rotwing_state.min_airspeed = rotwing_state.cruise_airspeed;
312 rotwing_state.max_airspeed = rotwing_state.cruise_airspeed;
313 }
314 else if(!rotwing_state_pusher_motor_running()) {
315 rotwing_state.min_airspeed = 0;
316 rotwing_state.max_airspeed = ROTWING_QUAD_NOPUSH_AIRSPEED;
317 }
318 else if(rotwing_state.state == ROTWING_STATE_FORCE_HOVER || rotwing_state.state == ROTWING_STATE_REQUEST_HOVER) {
319 rotwing_state.min_airspeed = skew_min_airspeed;
320 rotwing_state.max_airspeed = fmax(ROTWING_QUAD_MAX_AIRSPEED, skew_min_airspeed);
321 }
322 else if(rotwing_state.state == ROTWING_STATE_REQUEST_FW) {
323 rotwing_state.min_airspeed = fmin(rotwing_state.cruise_airspeed, skew_max_airspeed);
324 rotwing_state.max_airspeed = fmin(rotwing_state.cruise_airspeed, skew_max_airspeed);
325 }
326 else{
327 rotwing_state.min_airspeed = skew_min_airspeed;
328 rotwing_state.max_airspeed = skew_max_airspeed;
329 }
330
331 /* Bound max bank angle, climb speed and descend speed if the rotwing drone is transitioning and not in FREE mode */
332 if (meas_skew_angle < ROTWING_FW_SKEW_ANGLE && meas_skew_angle > ROTWING_QUAD_SKEW_ANGLE
333 && rotwing_state.state != ROTWING_STATE_FREE) {
334 guidance_set_max_bank_angle(ROTWING_TRANSITION_MAX_BANK);
335 guidance_set_max_climb_speed(ROTWING_TRANSITION_MAX_CLIMB_SPEED, ROTWING_TRANSITION_MAX_CLIMB_SPEED);
336 guidance_set_max_descend_speed(ROTWING_TRANSITION_MAX_DESCEND_SPEED, ROTWING_TRANSITION_MAX_DESCEND_SPEED);
337 } else {
338 guidance_set_max_bank_angle(GUIDANCE_H_MAX_BANK);
339 guidance_set_max_climb_speed(GUIDANCE_INDI_QUAD_CLIMB_SPEED, GUIDANCE_INDI_FWD_CLIMB_SPEED);
340 guidance_set_max_descend_speed(GUIDANCE_INDI_QUAD_DESCEND_SPEED, GUIDANCE_INDI_FWD_DESCEND_SPEED);
341 }
342
343 // Override failing skewing while fwd
344 /*if(meas_skew_angle > 70 && rotwing_state_.skewing_failing) {
345 rotwing_state.min_airspeed = 0; // Vstall + margin
346 rotwing_state.max_airspeed = 0; // Max airspeed FW
347 }*/
348
349 guidance_set_min_max_airspeed(rotwing_state.min_airspeed, rotwing_state.max_airspeed);
350
351 /* Set navigation/guidance settings */
352 nav_max_deceleration_sp = ROTWING_FW_MAX_DECELERATION * meas_skew_angle / 90.f + ROTWING_QUAD_MAX_DECELERATION * (90.f - meas_skew_angle) / 90.f; //TODO: Do we really want to based this on the skew?
353
354
355 /* Calculate the skew command */
356 float servo_pprz_cmd = MAX_PPRZ * (rotwing_state.sp_skew_angle_deg - 45.f) / 45.f;
357 BoundAbs(servo_pprz_cmd, MAX_PPRZ);
358
359 // Rotate with second order filter
360 static float rotwing_state_skew_p_cmd = -MAX_PPRZ;
361 static float rotwing_state_skew_d_cmd = 0;
362
363 float speed_sp = ROTWING_SKEW_REF_MODEL_P_GAIN * (servo_pprz_cmd - rotwing_state_skew_p_cmd);
364 BoundAbs(speed_sp, ROTWING_SKEW_REF_MODEL_MAX_SPEED);
365 rotwing_state_skew_d_cmd += ROTWING_SKEW_REF_MODEL_D_GAIN * (speed_sp - rotwing_state_skew_d_cmd);
366 rotwing_state_skew_p_cmd += rotwing_state_skew_d_cmd;
367 BoundAbs(rotwing_state_skew_p_cmd, MAX_PPRZ);
368 rotwing_state.ref_model_skew_angle_deg = 45.0 / MAX_PPRZ * rotwing_state_skew_p_cmd + 45.0;
369
370#if (ROTWING_SKEW_REF_MODEL || USE_NPS)
371 rotwing_state.skew_cmd = rotwing_state_skew_p_cmd;
372#else
373 rotwing_state.skew_cmd = servo_pprz_cmd;
374#endif
375#ifdef COMMAND_ROT_MECH
376 commands[COMMAND_ROT_MECH] = rotwing_state.skew_cmd;
377#endif
378
379 /* Add simulation feedback for the skewing and RPM */
380#if USE_NPS
381 // Export to the index of the SKEW in the NPS_ACTUATOR_NAMES array
382#ifdef COMMAND_ROT_MECH
383 commands[COMMAND_ROT_MECH] = (rotwing_state.skew_cmd + MAX_PPRZ) / 2.f; // Scale to simulation command
384#else
385 actuators_pprz[INDI_NUM_ACT] = (rotwing_state.skew_cmd + MAX_PPRZ) / 2.f; // Scale to simulation command
386#endif
387 // SEND ABI Message to ctr_eff_sched, ourself and other modules that want Actuator position feedback
388 struct act_feedback_t feedback;
389 feedback.idx = SERVO_ROTATION_MECH_IDX;
390 feedback.position = 0.5f * M_PI - RadOfDeg((float) rotwing_state.skew_cmd / MAX_PPRZ * 45.f + 45.f);
391 feedback.set.position = true;
392
393 // Send ABI message (or simulate failure)
394 if(!rotwing_state.fail_skew_angle) {
395 AbiSendMsgACT_FEEDBACK(ACT_FEEDBACK_BOARD_ID, &feedback, 1);
396 }
397
398 // Simulate to always have RPM if on and active feedback
399 rotwing_state.meas_rpm[0] = (actuators[SERVO_MOTOR_FRONT_IDX].pprz_val >= 0)? (ROTWING_QUAD_MIN_RPM + 100) : 0;
400 rotwing_state.meas_rpm[1] = (actuators[SERVO_MOTOR_RIGHT_IDX].pprz_val >= 0)? (ROTWING_QUAD_MIN_RPM + 100) : 0;
401 rotwing_state.meas_rpm[2] = (actuators[SERVO_MOTOR_BACK_IDX].pprz_val >= 0)? (ROTWING_QUAD_MIN_RPM + 100) : 0;
402 rotwing_state.meas_rpm[3] = (actuators[SERVO_MOTOR_LEFT_IDX].pprz_val >= 0)? (ROTWING_QUAD_MIN_RPM + 100) : 0;
403 rotwing_state.meas_rpm[4] = (actuators[SERVO_MOTOR_PUSH_IDX].pprz_val >= 0)? (ROTWING_PUSH_MIN_RPM + 100) : 0;
404 for(uint8_t i = 0; i < 5; i++) {
405 rotwing_state.meas_rpm_time[i] = current_time;
406 }
407
408#ifdef SITL
409 if(rotwing_state.fail_hover_motor) {
410 rotwing_state.meas_rpm[0] = 0;
411 rotwing_state.meas_rpm[1] = 0;
412 rotwing_state.meas_rpm[2] = 0;
413 rotwing_state.meas_rpm[3] = 0;
414 }
415
416 if(rotwing_state.fail_pusher_motor) {
417 rotwing_state.meas_rpm[4] = 0;
418 }
419#endif
420
421#endif
422}

424static void rotwing_state_feedback_cb(uint8_t __attribute__((unused)) sender_id,
425 struct act_feedback_t UNUSED *feedback_msg, uint8_t UNUSED num_act_message)
426{
427 float current_time = get_sys_time_float();
428 for (int i = 0; i < num_act_message; i++) {
429 int idx = feedback_msg[i].idx;
430
431 // Check for wing rotation feedback
432 if ((feedback_msg[i].set.position) && (idx == SERVO_ROTATION_MECH_IDX || idx == SERVO_BROTATION_MECH_IDX)) {
433 // Get wing rotation angle from sensor
434 float skew_angle_rad = 0.5 * M_PI - feedback_msg[i].position;
435 rotwing_state.meas_skew_angle_deg = DegOfRad(skew_angle_rad);
436 rotwing_state.meas_skew_angle_time = current_time;
437 }
438
439 // Get the RPM feedbacks of the motors
440 if (feedback_msg[i].set.rpm) {
441 if ((idx == SERVO_MOTOR_FRONT_IDX) || (idx == SERVO_BMOTOR_FRONT_IDX)) {
442 rotwing_state.meas_rpm[0] = feedback_msg->rpm;
443 rotwing_state.meas_rpm_time[0] = current_time;
444 } else if ((idx == SERVO_MOTOR_RIGHT_IDX) || (idx == SERVO_BMOTOR_RIGHT_IDX)) {
445 rotwing_state.meas_rpm[1] = feedback_msg->rpm;
446 rotwing_state.meas_rpm_time[1] = current_time;
447 } else if ((idx == SERVO_MOTOR_BACK_IDX) || (idx == SERVO_BMOTOR_BACK_IDX)) {
448 rotwing_state.meas_rpm[2] = feedback_msg->rpm;
449 rotwing_state.meas_rpm_time[2] = current_time;
450 } else if ((idx == SERVO_MOTOR_LEFT_IDX) || (idx == SERVO_BMOTOR_LEFT_IDX)) {
451 rotwing_state.meas_rpm[3] = feedback_msg->rpm;
452 rotwing_state.meas_rpm_time[3] = current_time;
453 } else if ((idx == SERVO_MOTOR_PUSH_IDX) || (idx == SERVO_BMOTOR_PUSH_IDX)) {
454 rotwing_state.meas_rpm[4] = feedback_msg->rpm;
455 rotwing_state.meas_rpm_time[4] = current_time;
456 }
457 }
458 }

461bool rotwing_state_hover_motors_running(void) {
462 float current_time = get_sys_time_float();
463 // Check if hover motors are running
464 if (rotwing_state.meas_rpm[0] > ROTWING_QUAD_MIN_RPM
465 && rotwing_state.meas_rpm[1] > ROTWING_QUAD_MIN_RPM
466 && rotwing_state.meas_rpm[2] > ROTWING_QUAD_MIN_RPM
467 && rotwing_state.meas_rpm[3] > ROTWING_QUAD_MIN_RPM
468 && (current_time - rotwing_state.meas_rpm_time[0]) < ROTWING_RPM_TIMEOUT
469 && (current_time - rotwing_state.meas_rpm_time[1]) < ROTWING_RPM_TIMEOUT

501bool rotwing_state_choose_circle_direction(uint8_t wp_id) {
502 // Get circle waypoint coordinates in NED
503 struct FloatVect3 circ_ned = {.x = waypoints[wp_id].enu_f.y,
504 .y = waypoints[wp_id].enu_f.x,
505 .z = -waypoints[wp_id].enu_f.z};
506
507 // Get drone position coordinates in NED
508 struct FloatVect3 pos_ned = {.x = stateGetPositionNed_f()->x,
509 .y = stateGetPositionNed_f()->y,
510 .z = stateGetPositionNed_f()->z};
511
512 // Get vector pointing from drone to waypoint
513 struct FloatVect3 vect_pos_circ;
514 VECT3_DIFF(vect_pos_circ, circ_ned, pos_ned);
515
516 static struct FloatVect3 x_axis = {.x = 1, .y = 0, .z = 0};
517 static struct FloatVect3 z_axis = {.x = 0, .y = 0, .z = 1};
518 struct FloatVect3 att_NED;
519 struct FloatVect3 cross_att_circ;
520
521 float_rmat_transp_vmult(&att_NED, stateGetNedToBodyRMat_f(), &x_axis);
522
523 VECT3_CROSS_PRODUCT(cross_att_circ, vect_pos_circ, att_NED);
524 float y = VECT3_DOT_PRODUCT(cross_att_circ, z_axis);
525 float x = VECT3_DOT_PRODUCT(vect_pos_circ, att_NED);
526
527 float body_to_wp_angle_rad = atan2f(y, x);
528
529 if (body_to_wp_angle_rad >= 0.f) {
530 return true; // CCW circle
531 } else {
532 return false; // CW circle
533 }
534}
