183 void periodic_rotwing_state(void)
184 {
185  // Check and set the current state
186  rotwing_check_set_current_state();
187 
188  // Check and update desired state
189  if (guidance_h.mode == GUIDANCE_H_MODE_NAV) {
190  rotwing_switch_state();
191  } else if (guidance_h.mode == GUIDANCE_H_MODE_ATTITUDE) {
192  rotwing_state_set_hover_settings();
193  } else if (guidance_h.mode == GUIDANCE_H_MODE_FORWARD) {
194  rotwing_state_set_fw_settings();
195  }
196 
197  // Run the wing skewer
198  rotwing_state_skewer();
199 
200  //TODO: incorparate motor active / disbaling depending on called flight state
201  // Switch on motors if flight mode is attitude
202  if (guidance_h.mode == GUIDANCE_H_MODE_ATTITUDE) {
203  bool_disable_hover_motors = false;
204  } else if (guidance_h.mode == GUIDANCE_H_MODE_FORWARD) {
205  bool_disable_hover_motors = false;
206  }
207 }

233 void rotwing_check_set_current_state(void)
234 {
235  // if !in_flight, set state to hover
236  if (!autopilot.in_flight) {
237  rotwing_state_hover_counter = 0;
238  rotwing_state_skewing_counter = 0;
239  rotwing_state_fw_counter = 0;
240  rotwing_state_fw_idle_counter = 0;
241  rotwing_state_fw_m_off_counter = 0;
242  rotwing_state.current_state = ROTWING_STATE_HOVER;
243  return;
244  }
245 
246  // States can be checked according to wing angle sensor, setpoints .....
247  uint8_t prev_state = rotwing_state.current_state;
248  switch (prev_state) {
249  case ROTWING_STATE_HOVER:
250  // Check if state needs to be set to skewing
251  if (rotwing_state_skewing.wing_angle_deg > ROTWING_MIN_SKEW_ANGLE_DEG_QUAD) {
252  rotwing_state_skewing_counter++;
253  } else {
254  rotwing_state_skewing_counter = 0;
255  }
256 
257  // Switch state if necessary
258  if (rotwing_state_skewing_counter > ROTWING_MIN_SKEW_ANGLE_COUNTER) {
259  rotwing_state.current_state = ROTWING_STATE_SKEWING;
260  rotwing_state_skewing_counter = 0;
261  }
262  break;
263 
264  case ROTWING_STATE_SKEWING:
265  // Check if state needs to be set to hover
266  if (rotwing_state_skewing.wing_angle_deg < ROTWING_MIN_SKEW_ANGLE_DEG_QUAD) {
267  rotwing_state_hover_counter++;
268  } else {
269  rotwing_state_hover_counter = 0;
270  }
271 
272  // Check if state needs to be set to fixed wing
273  if (rotwing_state_skewing.wing_angle_deg > ROTWING_MIN_FW_SKEW_ANGLE_DEG) {
274  rotwing_state_fw_counter++;
275  } else {
276  rotwing_state_fw_counter = 0;
277  }
278 
279  // Switch state if necessary
280  if (rotwing_state_hover_counter > ROTWING_MIN_SKEW_ANGLE_COUNTER) {
281  rotwing_state.current_state = ROTWING_STATE_HOVER;
282  rotwing_state_hover_counter = 0;
283  }
284 
285  if (rotwing_state_fw_counter > ROTWING_MIN_FW_COUNTER) {
286  rotwing_state.current_state = ROTWING_STATE_FW;
287  rotwing_state_fw_counter = 0;
288  }
289  break;
290 
291  case ROTWING_STATE_FW:
292  // Check if state needs to be set to skewing
293  if (rotwing_state_skewing.wing_angle_deg < ROTWING_MIN_FW_SKEW_ANGLE_DEG) {
294  rotwing_state_skewing_counter++;
295  } else {
296  rotwing_state_skewing_counter = 0;
297  }
298 
299  // Check if state needs to be set to fixed wing with hover motors idle (If hover thrust below threshold)
300  if (stabilization_cmd[COMMAND_THRUST] < ROTWING_MIN_THRUST_IDLE && rotwing_state.desired_state > ROTWING_STATE_FW) {
301  rotwing_state_fw_idle_counter++;
302  } else {
303  rotwing_state_fw_idle_counter = 0;
304  }
305 
306  // Switch state if necessary
307  if (rotwing_state_skewing_counter > ROTWING_MIN_FW_COUNTER) {
308  rotwing_state.current_state = ROTWING_STATE_SKEWING;
309  rotwing_state_skewing_counter = 0;
310  rotwing_state_fw_idle_counter = 0;
311  } else if (rotwing_state_fw_idle_counter > ROTWING_MIN_THRUST_IDLE_COUNTER
312  && rotwing_state_skewing_counter == 0) {
313  rotwing_state.current_state = ROTWING_STATE_FW_HOV_MOT_IDLE;
314  rotwing_state_skewing_counter = 0;
315  rotwing_state_fw_idle_counter = 0;
316  }
317  break;
318 
319  case ROTWING_STATE_FW_HOV_MOT_IDLE:
320  // Check if state needs to be set to fixed wing with hover motors activated
321  if (stabilization_cmd[COMMAND_THRUST] > ROTWING_MIN_THRUST_IDLE
322  || rotwing_state.desired_state < ROTWING_STATE_FW_HOV_MOT_IDLE) {
323  rotwing_state_fw_counter++;
324  } else {
325  rotwing_state_fw_counter = 0;
326  }
327 
328  // Check if state needs to be set to fixed wing with hover motors off (if zero rpm on hover motors)
329  if (rotwing_state_hover_rpm[0] == 0
330  && rotwing_state_hover_rpm[1] == 0
331  && rotwing_state_hover_rpm[2] == 0
332  && rotwing_state_hover_rpm[3] == 0) {
333 #if !USE_NPS
334  rotwing_state_fw_m_off_counter++;
335 #endif
336  } else {
337  rotwing_state_fw_m_off_counter = 0;
338  }
339 
340  // Switch state if necessary
341  if (rotwing_state_fw_counter > ROTWING_MIN_THRUST_IDLE_COUNTER) {
342  rotwing_state.current_state = ROTWING_STATE_FW;
343  rotwing_state_fw_counter = 0;
344  rotwing_state_fw_m_off_counter = 0;
345  } else if (rotwing_state_fw_m_off_counter > ROTWING_HOV_MOT_OFF_COUNTER
346  && rotwing_state_fw_counter == 0) {
347  rotwing_state.current_state = ROTWING_STATE_FW_HOV_MOT_OFF;
348  rotwing_state_fw_counter = 0;
349  rotwing_state_fw_m_off_counter = 0;
350  }
351  break;
352 
353  case ROTWING_STATE_FW_HOV_MOT_OFF:
354  // Check if state needs to be set to fixed wing with hover motors idle (if rpm on hover motors)
355  if (rotwing_state_hover_rpm[0] > ROTWING_HOV_MOT_OFF_RPM_TH
356  && rotwing_state_hover_rpm[1] > ROTWING_HOV_MOT_OFF_RPM_TH
357  && rotwing_state_hover_rpm[2] > ROTWING_HOV_MOT_OFF_RPM_TH
358  && rotwing_state_hover_rpm[3] > ROTWING_HOV_MOT_OFF_RPM_TH) {
359  rotwing_state_fw_idle_counter++;
360  } else {
361  rotwing_state_fw_idle_counter = 0;
362  }
363 
364  // Switch state if necessary
365  if (rotwing_state_fw_idle_counter > ROTWING_MIN_THRUST_IDLE_COUNTER) {
366  rotwing_state.current_state = ROTWING_STATE_FW_HOV_MOT_IDLE;
367  rotwing_state_fw_idle_counter = 0;
368  }
369  break;
370 
371  default:
372  break;
373  }
374 }

376 // Function that handles settings for switching state(s)
377 void rotwing_switch_state(void)
378 {
379  switch (rotwing_state.current_state) {
380  case ROTWING_STATE_HOVER:
381  if (rotwing_state.desired_state > ROTWING_STATE_HOVER) {
382  rotwing_state_set_skewing_settings();
383  } else {
384  rotwing_state_set_hover_settings();
385  }
386  break;
387 
388  case ROTWING_STATE_SKEWING:
389  if (rotwing_state.desired_state < ROTWING_STATE_SKEWING && stateGetAirspeed_f() < ROTWING_MAX_QUAD_AIRSPEED) {
390  rotwing_state_set_hover_settings();
391  } else {
392  rotwing_state_set_skewing_settings();
393  }
394  break;
395 
396  case ROTWING_STATE_FW:
397  if (rotwing_state.desired_state > ROTWING_STATE_FW) {
398  rotwing_state_set_fw_hov_mot_idle_settings();
399  } else if (rotwing_state.desired_state < ROTWING_STATE_FW) {
400  rotwing_state_set_skewing_settings();
401  } else {
402  rotwing_state_set_fw_settings();
403  }
404  break;
405 
406  case ROTWING_STATE_FW_HOV_MOT_IDLE:
407  if (rotwing_state.desired_state > ROTWING_STATE_FW_HOV_MOT_IDLE) {
408  rotwing_state_set_fw_hov_mot_off_settings();
409  } else if (rotwing_state.desired_state < ROTWING_STATE_FW_HOV_MOT_IDLE) {
410  rotwing_state_set_fw_settings();
411  } else {
412  rotwing_state_set_fw_hov_mot_idle_settings();
413  }
414  break;
415 
416  case ROTWING_STATE_FW_HOV_MOT_OFF:
417  if (rotwing_state.desired_state < ROTWING_STATE_FW_HOV_MOT_OFF) {
418  rotwing_state_set_fw_hov_mot_idle_settings();
419  } else {
420  rotwing_state_set_fw_hov_mot_off_settings();
421  }
422  break;
423  }
424 }

501 void rotwing_state_set_state_settings(void)
502 {
503 
504  if (!rotwing_state_skewing.force_rotation_angle) {
505  switch (rotwing_state_settings.wing_scheduler) {
506  case ROTWING_STATE_WING_QUAD_SETTING:
507  rotwing_state_skewing.airspeed_scheduling = false;
508  rotwing_state_skewing.wing_angle_deg_sp = 0;
509  break;
510  case ROTWING_STATE_WING_SCHEDULING_SETTING:
511  rotwing_state_skewing.airspeed_scheduling = true;
512  break;
513  case ROTWING_STATE_WING_FW_SETTING:
514  rotwing_state_skewing.airspeed_scheduling = true;
515  break;
516  }
517  } else {
518  rotwing_state_skewing.airspeed_scheduling = false;
519  }
520 
521  hover_motors_active = rotwing_state_settings.hover_motors_active;
522 
523  bool_disable_hover_motors = rotwing_state_settings.hover_motors_disable;
524 
525  force_forward = rotwing_state_settings.force_forward;
526

632 void guidance_indi_hybrid_set_wls_settings(float body_v[3], float roll_angle, float pitch_angle)
633 {
634  float pitch_priority_factor = 11.;
635  float roll_priority_factor = 10.;
636  float thrust_priority_factor = 7.;
637  float pusher_priority_factor = 30.;
638 
639  float horizontal_accel_weight = 10.;
640  float vertical_accel_weight = 10.;
641 
642  // Set weights
643  Wu_gih[0] = roll_priority_factor * 10.414;
644  Wu_gih[1] = pitch_priority_factor * 27.53;
645  Wu_gih[2] = thrust_priority_factor * 0.626;
646  Wu_gih[3] = pusher_priority_factor * 1.0;
647 
648  // adjust weights
649  float thrust_command = (actuator_state_filt_vect[0] + actuator_state_filt_vect[1] + actuator_state_filt_vect[2] +
650  actuator_state_filt_vect[3]) / 4;
651  Bound(thrust_command, 0, MAX_PPRZ);
652  float fixed_wing_percentage = !hover_motors_active; // TODO: when hover props go below 40%, ...
653  Bound(fixed_wing_percentage, 0, 1);
654 #define AIRSPEED_IMPORTANCE_IN_FORWARD_WEIGHT 16
655 
656  Wv_gih[0] = horizontal_accel_weight * (1.0f + fixed_wing_percentage *
657  AIRSPEED_IMPORTANCE_IN_FORWARD_WEIGHT); // stall n low hover motor_off (weight 16x more important than vertical weight)
658  Wv_gih[1] = horizontal_accel_weight;
659  Wv_gih[2] = vertical_accel_weight;
660 
661  struct FloatEulers eulers_zxy;
662  float_eulers_of_quat_zxy(&eulers_zxy, stateGetNedToBodyQuat_f());
663 
664  // Evaluate motors_on boolean
665  if (!hover_motors_active) {
666  if (stateGetAirspeed_f() < 15.) {
667  hover_motors_active = true;
668  bool_disable_hover_motors = false;
669  } else if (eulers_zxy.theta > RadOfDeg(15.0)) {
670  hover_motors_active = true;
671  bool_disable_hover_motors = false;
672  }
673  } else {
674  bool_disable_hover_motors = false;
675  }
676 
677  float du_min_thrust_z = ((MAX_PPRZ - actuator_state_filt_vect[0]) * g1g2[3][0] + (MAX_PPRZ -
678  actuator_state_filt_vect[1]) * g1g2[3][1] + (MAX_PPRZ - actuator_state_filt_vect[2]) * g1g2[3][2] +
679  (MAX_PPRZ - actuator_state_filt_vect[3]) * g1g2[3][3]) * hover_motors_active;
680  Bound(du_min_thrust_z, -50., 0.);
681  float du_max_thrust_z = -(actuator_state_filt_vect[0] * g1g2[3][0] + actuator_state_filt_vect[1] * g1g2[3][1] +
682  actuator_state_filt_vect[2] * g1g2[3][2] + actuator_state_filt_vect[3] * g1g2[3][3]);
683  Bound(du_max_thrust_z, 0., 50.);
684 
685  float roll_limit_rad = 2.0; // big roll limit hacked in to overcome wls problems at roll limit
686  float max_pitch_limit_rad = RadOfDeg(GUIDANCE_INDI_MAX_PITCH);
687  float min_pitch_limit_rad = RadOfDeg(GUIDANCE_INDI_MIN_PITCH);
688 
689  float scheduled_pitch_angle = 0;
690  float pitch_angle_range = 3.;
691  if (rotwing_state_skewing.wing_angle_deg < 55) {
692  scheduled_pitch_angle = 0;
693  } else {
694  float pitch_progression = (rotwing_state_skewing.wing_angle_deg - 55) / 35.;
695  scheduled_pitch_angle = pitch_angle_range * pitch_progression;
696  }
697  if (!hover_motors_active) {
698  scheduled_pitch_angle = 8.;
699  }
700  Bound(scheduled_pitch_angle, -5., 8.);
701  guidance_indi_pitch_pref_deg = scheduled_pitch_angle;
702 
703  float pitch_pref_rad = RadOfDeg(guidance_indi_pitch_pref_deg);
704 
705  // Set lower limits
706  du_min_gih[0] = -roll_limit_rad - roll_angle; //roll
707  du_min_gih[1] = min_pitch_limit_rad - pitch_angle; // pitch
708  du_min_gih[2] = du_min_thrust_z;
709  du_min_gih[3] = (-actuator_state_filt_vect[8] * g1g2[4][8]);
710 
711  // Set upper limits limits
712  du_max_gih[0] = roll_limit_rad - roll_angle; //roll
713  du_max_gih[1] = max_pitch_limit_rad - pitch_angle; // pitch
714  du_max_gih[2] = du_max_thrust_z;
715  du_max_gih[3] = 9.0; // Hacky value to prevent drone from pitching down in transition
716 
717  // Set prefered states
718  du_pref_gih[0] = 0; // prefered delta roll angle
719  du_pref_gih[1] = -pitch_angle + pitch_pref_rad;// prefered delta pitch angle
720  du_pref_gih[2] = du_max_gih[2]; // Low thrust better for efficiency
721  du_pref_gih[3] = body_v[0]; // solve the body acceleration
