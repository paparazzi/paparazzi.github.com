116 static void gyro_cb(uint8_t sender_id __attribute__((unused)),
117  uint32_t stamp, struct Int32Rates *gyro)
118 {
119  struct FloatRates gyro_f;
120  RATES_FLOAT_OF_BFP(gyro_f, *gyro);
121 
122 #if USE_AUTO_AHRS_FREQ || !defined(AHRS_PROPAGATE_FREQUENCY)
123  PRINT_CONFIG_MSG("Calculating dt for AHRS float_invariant propagation.")
124  /* timestamp in usec when last callback was received */
125  static uint32_t last_stamp = 0;
126 
127  if (last_stamp > 0 && ahrs_float_inv.is_aligned) {
128  float dt = (float)(stamp - last_stamp) * 1e-6;
129  ahrs_float_invariant_propagate(&gyro_f, dt);
130  compute_body_orientation_and_rates();
131  }
132  last_stamp = stamp;
133 #else
134  PRINT_CONFIG_MSG("Using fixed AHRS_PROPAGATE_FREQUENCY for AHRS float_invariant propagation.")
135  PRINT_CONFIG_VAR(AHRS_PROPAGATE_FREQUENCY)
136  const float dt = 1. / (AHRS_PROPAGATE_FREQUENCY);
137  if (ahrs_float_inv.is_aligned) {
138  ahrs_float_invariant_propagate(&gyro_f, dt);
139  compute_body_orientation_and_rates();
140  }
141 #endif
142 
143  ahrs_finv_last_stamp = stamp;
144 }
