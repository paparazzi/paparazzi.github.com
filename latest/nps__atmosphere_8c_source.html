53void nps_atmosphere_init(void)
54{
55 nps_atmosphere.qnh = NPS_QNH;
56 FLOAT_VECT3_ZERO(nps_atmosphere.wind);
57 nps_atmosphere_set_wind_speed(NPS_WIND_SPEED);
58 nps_atmosphere_set_wind_dir(NPS_WIND_DIR);
59 nps_atmosphere.turbulence_severity = NPS_TURBULENCE_SEVERITY;
60 nps_atmosphere.last_world_env_req = 0.;

63void nps_atmosphere_set_wind_speed(double speed)
64{
65 nps_atmosphere.wind_speed = speed;
66 /* recalc wind in north and east */
67 nps_atmosphere.wind.x = -speed * sin(M_PI_2 - nps_atmosphere.wind_dir);
68 nps_atmosphere.wind.y = -speed * cos(M_PI_2 - nps_atmosphere.wind_dir);
69}

71void nps_atmosphere_set_wind_dir(double dir)
72{
73 /* normalize dir to 0-2Pi */
74 while (dir < 0.0) { dir += 2 * M_PI; }
75 while (dir >= 2 * M_PI) { dir -= 2 * M_PI; }
76
77 nps_atmosphere.wind_dir = dir;
78 /* recalc wind in north and east */
79 nps_atmosphere.wind.x = -nps_atmosphere.wind_speed * sin(M_PI_2 - dir);
80 nps_atmosphere.wind.y = -nps_atmosphere.wind_speed * cos(M_PI_2 - dir);
81}

83void nps_atmosphere_set_wind_ned(double wind_north, double wind_east, double wind_down)
84{
85 nps_atmosphere.wind.x = wind_north;
86 nps_atmosphere.wind.y = wind_east;
87 nps_atmosphere.wind.z = wind_down;
88 /* recalc horizontal wind speed and dir */
89 nps_atmosphere.wind_speed = FLOAT_VECT2_NORM(nps_atmosphere.wind);
90
91 double dir = atan2(-wind_east, -wind_north);
92 /* normalize dir to 0-2Pi */
93 while (dir < 0.0) { dir += 2 * M_PI; }
94 while (dir >= 2 * M_PI) { dir -= 2 * M_PI; }
95 nps_atmosphere.wind_dir = dir;
96}

98void nps_atmosphere_update(double dt)
99{
100 static double req_time = 0.;
101 req_time += dt;
102 if (req_time - nps_atmosphere.last_world_env_req >= NPS_WORLD_ENV_UPDATE) {
103 nps_atmosphere.last_world_env_req = req_time;
104 nps_ivy_send_world_env = true;
105 }
106
107 nps_fdm_set_wind_ned(nps_atmosphere.wind.x, nps_atmosphere.wind.y, nps_atmosphere.wind.z);
108 nps_fdm_set_turbulence(nps_atmosphere.wind_speed, nps_atmosphere.turbulence_severity);
109}
