44static void superbitrf_rc_normalize(int16_t *in, int16_t *out, uint8_t count)
45{
46 uint8_t i;
47 for (i = 0; i < count; i++) {
48 if (i == RADIO_THROTTLE) {
49 out[i] = (in[i] + MAX_PPRZ) / 2;
50 Bound(out[i], 0, MAX_PPRZ);
51 } else {
52 out[i] = in[i];
53 Bound(out[i], MIN_PPRZ, MAX_PPRZ);
54 }
55 }
56}

58void superbitrf_rc_event(void)
59{
60 cyrf6936_event(&superbitrf.cyrf6936);
61 superbitrf_event();
62 if (superbitrf.rc_frame_available) {
63 radio_control.frame_cpt++;
64 radio_control.time_since_last_frame = 0;
65 radio_control.radio_ok_cpt = 0;
66 radio_control.status = RC_OK;
67 superbitrf_rc_normalize(superbitrf.rc_values, radio_control.values,
68 superbitrf.num_channels);
69 AbiSendMsgRADIO_CONTROL(RADIO_CONTROL_SUPERBITRF_RC_ID, &radio_control);
70 superbitrf.rc_frame_available = false;
71 }
