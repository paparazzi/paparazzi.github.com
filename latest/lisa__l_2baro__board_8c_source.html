84void baro_periodic(void)
85{
86 // check i2c_done
87 if (!i2c_idle(&i2c2)) { return; }
88
89 switch (baro_board.status) {
90 case LBS_UNINITIALIZED:
91 baro_board_send_reset();
92 baro_board.status = LBS_RESETED;
93 break;
94 case LBS_RESETED:
95 baro_board_send_config_abs();
96 baro_board.status = LBS_INITIALIZING_ABS;
97 break;
98 case LBS_INITIALIZING_ABS:
99 baro_board_set_current_register(BARO_ABS_ADDR, 0x00);
100 baro_board.status = LBS_INITIALIZING_ABS_1;
101 break;
102 case LBS_INITIALIZING_ABS_1:
103 baro_board_send_config_diff();
104 baro_board.status = LBS_INITIALIZING_DIFF;
105 break;
106 case LBS_INITIALIZING_DIFF:
107 baro_board_set_current_register(BARO_DIFF_ADDR, 0x00);
108 baro_board.status = LBS_INITIALIZING_DIFF_1;
109 // baro_board.status = LBS_UNINITIALIZED;
110 break;
111 case LBS_INITIALIZING_DIFF_1:
112 baro_board.running = true;
113 /* Falls through. */
114 case LBS_READ_DIFF:
115 baro_board_read_from_current_register(BARO_ABS_ADDR);
116 baro_board.status = LBS_READING_ABS;
117 break;
118 case LBS_READ_ABS:
119 baro_board_read_from_current_register(BARO_DIFF_ADDR);
120 baro_board.status = LBS_READING_DIFF;
121 break;
122 default:
123 break;
124 }
125
126#ifdef BARO_LED
127 if (baro_board.running == TRUE) {
128 LED_ON(BARO_LED);
129 } else {
130 LED_TOGGLE(BARO_LED);
131 }
132#endif
133}

135void lisa_l_baro_event(void)
136{
137 if (baro_board.status == LBS_READING_ABS &&
138 baro_trans.status != I2CTransPending) {
139 baro_board.status = LBS_READ_ABS;
140 if (baro_trans.status == I2CTransSuccess) {
141 uint32_t now_ts = get_sys_time_usec();
142 int16_t tmp = baro_trans.buf[0] << 8 | baro_trans.buf[1];
143 float pressure = LISA_L_BARO_SENS * (float)tmp;
144 AbiSendMsgBARO_ABS(BARO_BOARD_SENDER_ID, now_ts, pressure);
145 }
146 } else if (baro_board.status == LBS_READING_DIFF &&
147 baro_trans.status != I2CTransPending) {
148 baro_board.status = LBS_READ_DIFF;
149 if (baro_trans.status == I2CTransSuccess) {
150 int16_t tmp = baro_trans.buf[0] << 8 | baro_trans.buf[1];
151 float diff = LISA_L_DIFF_SENS * (float)tmp;
152 AbiSendMsgBARO_DIFF(BARO_BOARD_SENDER_ID, diff);
153 }
154 }
155}

157static inline void baro_board_send_config_abs(void)
158{
159#ifndef BARO_LOW_GAIN
160 INFO("Using High LisaL Baro Gain: Do not use below 1000hPa")
161 baro_board_write_to_register(BARO_ABS_ADDR, 0x01, 0x86, 0x83);
162#else
163 INFO("Using Low LisaL Baro Gain, capable of measuring below 1000hPa or more")
164 //config register should be 0x84 in low countries, or 0x86 in normal countries
165 baro_board_write_to_register(BARO_ABS_ADDR, 0x01, 0x84, 0x83);
166#endif
167}

174static inline void baro_board_send_reset(void)
175{
176 baro_trans.type = I2CTransTx;
177 baro_trans.slave_addr = 0x00;
178 baro_trans.len_w = 1;
179 baro_trans.buf[0] = 0x06;
180 i2c_submit(&i2c2, &baro_trans);

184{
185 baro_trans.type = I2CTransTx;
