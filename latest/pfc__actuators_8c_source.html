155 void pfc_actuators_run(void) {
156  // Only actuate when running
157  if(pfc_actuators.state != PFC_ACTUATORS_STATE_RUNNING || pfc_actuators.act_idx >= pfc_actuators.act_nb) {
158  return;
159  }
160 
161  // Get the actuators
162  struct pfc_actuator_t *act = &pfc_acts[pfc_actuators.act_idx];
163 
164  // Verify the result and continue
165  if((get_sys_time_float() - pfc_actuators.act_start_time) > act->timeout) {
166  switch(pfc_actuators.act_state) {
167  case PFC_ACTUATOR_STATE_WAIT:
168  pfc_actuators_debug("Actuator %d starting LOW", pfc_actuators.act_idx);
169  pfc_actuators.act_state = PFC_ACTUATOR_STATE_LOW;
170  pfc_actuators.act_start_time = get_sys_time_float();
171  pfc_actuators.last_feedback = NAN;
172  pfc_actuators.last_feedback2 = NAN;
173  break;
174  case PFC_ACTUATOR_STATE_LOW:
175  // Check if the feedback is correct
176  if(act->feedback_id != 255 &&
177  (pfc_actuators.last_feedback < (act->low_feedback - pfc_actuators.last_feedback_err) || pfc_actuators.last_feedback > (act->low_feedback + pfc_actuators.last_feedback_err))) {
178  pfc_actuators_error("Actuator %d not responding correctly LOW %.2f < %.2f < %.2f", pfc_actuators.act_idx, act->low_feedback - pfc_actuators.last_feedback_err, pfc_actuators.last_feedback, act->low_feedback + pfc_actuators.last_feedback_err);
179  pfc_actuators.state = PFC_ACTUATORS_STATE_ERROR;
180  } else if(act->feedback_id2 != 255 &&
181  (pfc_actuators.last_feedback2 < (act->low_feedback - pfc_actuators.last_feedback_err2) || pfc_actuators.last_feedback2 > (act->low_feedback + pfc_actuators.last_feedback_err2))) {
182  pfc_actuators_error("Actuator %d not responding correctly LOW2 %.2f < %.2f < %.2f", pfc_actuators.act_idx, act->low_feedback - pfc_actuators.last_feedback_err2, pfc_actuators.last_feedback2, act->low_feedback + pfc_actuators.last_feedback_err2);
183  pfc_actuators.state = PFC_ACTUATORS_STATE_ERROR;
184  }
185  if(pfc_actuators.state != PFC_ACTUATORS_STATE_ERROR || PFC_ACTUATORS_DEBUG) {
186  pfc_actuators_debug("Actuator %d LOW ok %.2f, %.2f starting HIGH", pfc_actuators.act_idx, pfc_actuators.last_feedback, pfc_actuators.last_feedback2);
187  pfc_actuators.state = PFC_ACTUATORS_STATE_RUNNING;
188  pfc_actuators.act_state = PFC_ACTUATOR_STATE_HIGH;
189  pfc_actuators.act_start_time = get_sys_time_float();
190  pfc_actuators.last_feedback = NAN;
191  pfc_actuators.last_feedback2 = NAN;
192  }
193  break;
194  case PFC_ACTUATOR_STATE_HIGH:
195  // Check if the feedback is correct
196  if(act->feedback_id != 255 &&
197  (pfc_actuators.last_feedback < (act->high_feedback - pfc_actuators.last_feedback_err) || pfc_actuators.last_feedback > (act->high_feedback + pfc_actuators.last_feedback_err))) {
198  pfc_actuators_error("Actuator %d not responding correctly HIGH %.2f < %.2f < %.2f", pfc_actuators.act_idx, act->high_feedback - pfc_actuators.last_feedback_err, pfc_actuators.last_feedback, act->high_feedback + pfc_actuators.last_feedback_err);
199  pfc_actuators.state = PFC_ACTUATORS_STATE_ERROR;
200  } else if(act->feedback_id2 != 255 &&
201  (pfc_actuators.last_feedback2 < (act->high_feedback - pfc_actuators.last_feedback_err2) || pfc_actuators.last_feedback2 > (act->high_feedback + pfc_actuators.last_feedback_err2))) {
202  pfc_actuators_error("Actuator %d not responding correctly HIGH2 %.2f < %.2f < %.2f", pfc_actuators.act_idx, act->high_feedback - pfc_actuators.last_feedback_err2, pfc_actuators.last_feedback2, act->high_feedback + pfc_actuators.last_feedback_err2);
203  pfc_actuators.state = PFC_ACTUATORS_STATE_ERROR;
204  }
205  if(pfc_actuators.state != PFC_ACTUATORS_STATE_ERROR || PFC_ACTUATORS_DEBUG) {
206  pfc_actuators_debug("Actuator %d HIGH ok %.2f, %.2f", pfc_actuators.act_idx, pfc_actuators.last_feedback, pfc_actuators.last_feedback2);
207  pfc_actuators.state = PFC_ACTUATORS_STATE_RUNNING;
208  pfc_actuators.act_state = PFC_ACTUATOR_STATE_WAIT;
209  pfc_actuators.act_start_time = 0;
210  pfc_actuators.act_idx++;
211  }
212  break;
213  }
214  }
215 
216  // Finished testing
217  if(pfc_actuators.act_idx >= pfc_actuators.act_nb) {
218  pfc_actuators.state = PFC_ACTUATORS_STATE_SUCCESS;
219  pfc_actuators_error("Actuators checks done %d", pfc_actuators.act_idx);
220  }
221 }

226 void pfc_actuators_start(bool start) {
227  if(start && pfc_actuators.state != PFC_ACTUATORS_STATE_RUNNING) {
228  pfc_actuators.act_idx = 0;
229  pfc_actuators.act_start_time = 0;
230  pfc_actuators.state = PFC_ACTUATORS_STATE_RUNNING;
231  pfc_actuators.act_state = PFC_ACTUATOR_STATE_WAIT;
232  pfc_actuators.last_feedback = NAN;
233  pfc_actuators.last_feedback2 = NAN;
234  }
235  else if(!start && pfc_actuators.state == PFC_ACTUATORS_STATE_RUNNING) {
236  pfc_actuators.act_idx = 0;
237  pfc_actuators.act_start_time = 0;
238  pfc_actuators.state = PFC_ACTUATORS_STATE_INIT;
239  pfc_actuators.act_state = PFC_ACTUATOR_STATE_WAIT;
240  }
241 }

248 int16_t pfc_actuators_value(uint8_t idx, int16_t value) {
249  if(idx != pfc_actuators.act_idx || pfc_actuators.state != PFC_ACTUATORS_STATE_RUNNING) {
250  return value;
251  }
252

254  return pfc_acts[pfc_actuators.act_idx].low;

258  }
259

268 static void pfc_actuators_cb(struct preflight_result_t *result)
269 {
270  switch(pfc_actuators.state) {
271  case PFC_ACTUATORS_STATE_INIT:
272  preflight_error(result, "Actuators not checked perform checks first[%d]", pfc_actuators.act_nb);
273  break;
274  case PFC_ACTUATORS_STATE_RUNNING:
275  preflight_error(result, "Actuators are currently being checked[%d/%d]",pfc_actuators.act_idx, pfc_actuators.act_nb);
276  break;
277  case PFC_ACTUATORS_STATE_SUCCESS:
278  preflight_success(result, "Actuators checked and moved succesfully[%d]", pfc_actuators.act_nb);
279  break;
280  case PFC_ACTUATORS_STATE_ERROR:
281  preflight_error(result, "Actuators not responding correctly[%d/%d]",pfc_actuators.act_idx, pfc_actuators.act_nb);
282  break;
283  }
284 }

289 static void pfc_act_feedback_cb(uint8_t sender_id __attribute__((unused)), struct act_feedback_t *feedback, uint8_t num_act) {
290  if(pfc_actuators.state != PFC_ACTUATORS_STATE_RUNNING || pfc_actuators.act_idx >= pfc_actuators.act_nb) {
291  return;
292  }
293 
294  // Save the last feedback values
295  for(uint8_t i = 0; i < num_act; i++) {
296  if(feedback[i].idx == pfc_acts[pfc_actuators.act_idx].feedback_id && feedback[i].set.rpm) {
297  pfc_actuators.last_feedback = feedback[i].rpm;
298  pfc_actuators.last_feedback_err = PFC_ACTUATORS_MAX_RPM_ERROR;
299  } else if(feedback[i].idx == pfc_acts[pfc_actuators.act_idx].feedback_id && feedback[i].set.position) {
300  pfc_actuators.last_feedback = feedback[i].position;
301  pfc_actuators.last_feedback_err = PFC_ACTUATORS_MAX_ANGLE_ERROR;
302  }
303 
304  if(feedback[i].idx == pfc_acts[pfc_actuators.act_idx].feedback_id2 && feedback[i].set.rpm) {
305  pfc_actuators.last_feedback2 = feedback[i].rpm;
306  pfc_actuators.last_feedback_err2 = PFC_ACTUATORS_MAX_RPM_ERROR;
307  } else if(feedback[i].idx == pfc_acts[pfc_actuators.act_idx].feedback_id2 && feedback[i].set.position) {
308  pfc_actuators.last_feedback2 = feedback[i].position;
309  pfc_actuators.last_feedback_err2 = PFC_ACTUATORS_MAX_ANGLE_ERROR;
310  }
311  }
