141void pfc_actuators_init(void) {
142 pfc_actuators.state = PFC_ACTUATORS_STATE_INIT;
143 pfc_actuators.act_state = PFC_ACTUATOR_STATE_WAIT;
144 pfc_actuators.act_idx = 0;
145 pfc_actuators.act_nb = sizeof(pfc_acts) / sizeof(struct pfc_actuator_t);
146 pfc_actuators.last_feedback = NAN;
147 pfc_actuators.last_feedback2 = NAN;

151}

156void pfc_actuators_run(void) {
157 // Only actuate when running
158 if(pfc_actuators.state != PFC_ACTUATORS_STATE_RUNNING || pfc_actuators.act_idx >= pfc_actuators.act_nb) {
159 return;
160 }
161
162 // Get the actuators
163 struct pfc_actuator_t *act = &pfc_acts[pfc_actuators.act_idx];
164
165 // Verify the result and continue
166 if((get_sys_time_float() - pfc_actuators.act_start_time) > act->timeout) {
167 switch(pfc_actuators.act_state) {
168 case PFC_ACTUATOR_STATE_WAIT:
169 pfc_actuators_debug("Actuator %d starting LOW", pfc_actuators.act_idx);
170 pfc_actuators.act_state = PFC_ACTUATOR_STATE_LOW;
171 pfc_actuators.act_start_time = get_sys_time_float();
172 pfc_actuators.last_feedback = NAN;
173 pfc_actuators.last_feedback2 = NAN;
174 break;
175 case PFC_ACTUATOR_STATE_LOW:
176 // Check if the feedback is correct
177 if(act->feedback_id != 255 &&
178 (!isfinite(pfc_actuators.last_feedback) || (pfc_actuators.last_feedback < (act->low_feedback - pfc_actuators.last_feedback_err) || pfc_actuators.last_feedback > (act->low_feedback + pfc_actuators.last_feedback_err)))) {
179 pfc_actuators_error("Actuator %d not responding correctly LOW %.2f < %.2f < %.2f", pfc_actuators.act_idx, act->low_feedback - pfc_actuators.last_feedback_err, pfc_actuators.last_feedback, act->low_feedback + pfc_actuators.last_feedback_err);
180 pfc_actuators.state = PFC_ACTUATORS_STATE_ERROR;
181 } else if(act->feedback_id2 != 255 &&
182 (!isfinite(pfc_actuators.last_feedback2) || (pfc_actuators.last_feedback2 < (act->low_feedback - pfc_actuators.last_feedback_err2) || pfc_actuators.last_feedback2 > (act->low_feedback + pfc_actuators.last_feedback_err2)))) {
183 pfc_actuators_error("Actuator %d not responding correctly LOW2 %.2f < %.2f < %.2f", pfc_actuators.act_idx, act->low_feedback - pfc_actuators.last_feedback_err2, pfc_actuators.last_feedback2, act->low_feedback + pfc_actuators.last_feedback_err2);
184 pfc_actuators.state = PFC_ACTUATORS_STATE_ERROR;
185 }
186 if(pfc_actuators.state != PFC_ACTUATORS_STATE_ERROR || PFC_ACTUATORS_DEBUG) {
187 pfc_actuators_debug("Actuator %d LOW ok %.2f, %.2f starting HIGH", pfc_actuators.act_idx, pfc_actuators.last_feedback, pfc_actuators.last_feedback2);
188 pfc_actuators.state = PFC_ACTUATORS_STATE_RUNNING;
189 pfc_actuators.act_state = PFC_ACTUATOR_STATE_HIGH;
190 pfc_actuators.act_start_time = get_sys_time_float();
191 pfc_actuators.last_feedback = NAN;
192 pfc_actuators.last_feedback2 = NAN;
193 }
194 break;
195 case PFC_ACTUATOR_STATE_HIGH:
196 // Check if the feedback is correct
197 if(act->feedback_id != 255 &&
198 (!isfinite(pfc_actuators.last_feedback) || (pfc_actuators.last_feedback < (act->high_feedback - pfc_actuators.last_feedback_err) || pfc_actuators.last_feedback > (act->high_feedback + pfc_actuators.last_feedback_err)))) {
199 pfc_actuators_error("Actuator %d not responding correctly HIGH %.2f < %.2f < %.2f", pfc_actuators.act_idx, act->high_feedback - pfc_actuators.last_feedback_err, pfc_actuators.last_feedback, act->high_feedback + pfc_actuators.last_feedback_err);
200 pfc_actuators.state = PFC_ACTUATORS_STATE_ERROR;
201 } else if(act->feedback_id2 != 255 &&
202 (!isfinite(pfc_actuators.last_feedback2) || (pfc_actuators.last_feedback2 < (act->high_feedback - pfc_actuators.last_feedback_err2) || pfc_actuators.last_feedback2 > (act->high_feedback + pfc_actuators.last_feedback_err2)))) {
203 pfc_actuators_error("Actuator %d not responding correctly HIGH2 %.2f < %.2f < %.2f", pfc_actuators.act_idx, act->high_feedback - pfc_actuators.last_feedback_err2, pfc_actuators.last_feedback2, act->high_feedback + pfc_actuators.last_feedback_err2);
204 pfc_actuators.state = PFC_ACTUATORS_STATE_ERROR;
205 }
206 if(pfc_actuators.state != PFC_ACTUATORS_STATE_ERROR || PFC_ACTUATORS_DEBUG) {
207 pfc_actuators_debug("Actuator %d HIGH ok %.2f, %.2f", pfc_actuators.act_idx, pfc_actuators.last_feedback, pfc_actuators.last_feedback2);
208 pfc_actuators.state = PFC_ACTUATORS_STATE_RUNNING;
209 pfc_actuators.act_state = PFC_ACTUATOR_STATE_WAIT;
210 pfc_actuators.act_start_time = 0;
211 pfc_actuators.act_idx++;
212 }
213 break;
214 }
215 }
216
217 // Finished testing
218 if(pfc_actuators.act_idx >= pfc_actuators.act_nb) {
219 pfc_actuators.state = PFC_ACTUATORS_STATE_SUCCESS;
220 pfc_actuators_error("Actuators checks done %d", pfc_actuators.act_idx);
221 }
222}

227void pfc_actuators_start(bool start) {
228 if(start && pfc_actuators.state != PFC_ACTUATORS_STATE_RUNNING) {
229 pfc_actuators.act_idx = 0;
230 pfc_actuators.act_start_time = 0;
231 pfc_actuators.state = PFC_ACTUATORS_STATE_RUNNING;
232 pfc_actuators.act_state = PFC_ACTUATOR_STATE_WAIT;
233 pfc_actuators.last_feedback = NAN;
234 pfc_actuators.last_feedback2 = NAN;
235 }
236 else if(!start && pfc_actuators.state == PFC_ACTUATORS_STATE_RUNNING) {
237 pfc_actuators.act_idx = 0;
238 pfc_actuators.act_start_time = 0;
239 pfc_actuators.state = PFC_ACTUATORS_STATE_INIT;
240 pfc_actuators.act_state = PFC_ACTUATOR_STATE_WAIT;
241 }
242}

249int16_t pfc_actuators_value(uint8_t idx, int16_t value) {
250 if(idx != pfc_actuators.act_idx || pfc_actuators.state != PFC_ACTUATORS_STATE_RUNNING) {
251 return value;
252 }
253
254 if(pfc_actuators.act_state == PFC_ACTUATOR_STATE_LOW) {
255 return pfc_acts[pfc_actuators.act_idx].low;
256 }
257 else if(pfc_actuators.act_state == PFC_ACTUATOR_STATE_HIGH) {
258 return pfc_acts[pfc_actuators.act_idx].high;
259 }
260
261 return value;
262}

269static void pfc_actuators_cb(struct preflight_result_t *result)
270{
271 switch(pfc_actuators.state) {
272 case PFC_ACTUATORS_STATE_INIT:
273 preflight_error(result, "Actuators not checked [act: %d]", pfc_actuators.act_nb);
274 break;
275 case PFC_ACTUATORS_STATE_RUNNING:
276 preflight_error(result, "Actuators are currently being checked [act: %d/%d]",pfc_actuators.act_idx, pfc_actuators.act_nb);
277 break;
278 case PFC_ACTUATORS_STATE_SUCCESS:
279 preflight_success(result, "Actuators checked and succesfull [act: %d]", pfc_actuators.act_nb);
280 break;
281 case PFC_ACTUATORS_STATE_ERROR:
282 preflight_error(result, "Actuators not responding correctly [act: %d/%d]",pfc_actuators.act_idx, pfc_actuators.act_nb);
283 break;
284 }
285}

290static void pfc_act_feedback_cb(uint8_t sender_id __attribute__((unused)), struct act_feedback_t *feedback, uint8_t num_act) {
291 if(pfc_actuators.state != PFC_ACTUATORS_STATE_RUNNING || pfc_actuators.act_idx >= pfc_actuators.act_nb) {
292 return;
293 }
294
295 // Save the last feedback values
296 for(uint8_t i = 0; i < num_act; i++) {
297 if(feedback[i].idx == pfc_acts[pfc_actuators.act_idx].feedback_id && feedback[i].set.rpm) {
298 pfc_actuators.last_feedback = feedback[i].rpm;
299 pfc_actuators.last_feedback_err = PFC_ACTUATORS_MAX_RPM_ERROR;
300 } else if(feedback[i].idx == pfc_acts[pfc_actuators.act_idx].feedback_id && feedback[i].set.position) {
301 pfc_actuators.last_feedback = feedback[i].position;
302 pfc_actuators.last_feedback_err = PFC_ACTUATORS_MAX_ANGLE_ERROR;
303 }
304
305 if(feedback[i].idx == pfc_acts[pfc_actuators.act_idx].feedback_id2 && feedback[i].set.rpm) {
306 pfc_actuators.last_feedback2 = feedback[i].rpm;
307 pfc_actuators.last_feedback_err2 = PFC_ACTUATORS_MAX_RPM_ERROR;
308 } else if(feedback[i].idx == pfc_acts[pfc_actuators.act_idx].feedback_id2 && feedback[i].set.position) {
309 pfc_actuators.last_feedback2 = feedback[i].position;
310 pfc_actuators.last_feedback_err2 = PFC_ACTUATORS_MAX_ANGLE_ERROR;
311 }
312 }
