154inline float bound_or_zero(float value, float low_lim, float up_lim) {
155 float output = value;
156 if (low_lim > 0.f) {
157 if (value < low_lim) {
158 output = 0.f;
159 } else if (value > up_lim) {
160 output = up_lim;
161 }
162 } else {
163 if (value > up_lim) {
164 output = 0.f;
165 } else if (value < low_lim) {
166 output = low_lim;
167 }
168 }
169 return output;
170}

205{

218void eff_scheduling_rotwing_init(void)
219{
220 // Initialize variables to quad values
221 eff_sched_var.Ixx = eff_sched_p.Ixx_body + eff_sched_p.Ixx_wing;
222 eff_sched_var.Iyy = eff_sched_p.Iyy_body + eff_sched_p.Iyy_wing;
223 eff_sched_var.wing_rotation_rad = 0; // ABI input
224 eff_sched_var.wing_rotation_deg = 0;
225 eff_sched_var.cosr = 1;
226 eff_sched_var.sinr = 0;
227 eff_sched_var.cosr2 = 1;
228 eff_sched_var.sinr2 = 0;
229 eff_sched_var.sinr3 = 0;
230
231 // Set moment derivative variables
232 float hover_thrust = 6000;
233 eff_sched_var.pitch_motor_dMdpprz = (eff_sched_p.DMdpprz_hover_pitch[0] + 2*hover_thrust * eff_sched_p.DMdpprz_hover_pitch[1]) / 10000.; // Dmdpprz hover pitch for hover thrust
234 eff_sched_var.roll_motor_dMdpprz = (eff_sched_p.DMdpprz_hover_roll[0] + 2*hover_thrust * eff_sched_p.DMdpprz_hover_roll[1]) / 10000.; // Dmdpprz hover roll for hover thrust
235
236 eff_sched_var.cmd_elevator = 0;
237 eff_sched_var.cmd_pusher = 0;
238 eff_sched_var.cmd_pusher_scaled = 0;
239 eff_sched_var.cmd_T_mean_scaled = 0;
240
241 eff_sched_var.airspeed = 0;
242 eff_sched_var.airspeed2 = 0;
243
244 // Get wing angle
245 AbiBindMsgACT_FEEDBACK(WING_ROTATION_CAN_ROTWING_ID, &wing_position_ev, wing_position_cb);
246}

267void eff_scheduling_rotwing_update_wing_angle(void)
268{
269 // Calculate sin and cosines of rotation
270 eff_sched_var.wing_rotation_deg = eff_sched_var.wing_rotation_rad / M_PI * 180.;
271
272 eff_sched_var.cosr = cosf(eff_sched_var.wing_rotation_rad);
273 eff_sched_var.sinr = sinf(eff_sched_var.wing_rotation_rad);
274
275 eff_sched_var.cosr2 = eff_sched_var.cosr * eff_sched_var.cosr;
276 eff_sched_var.sinr2 = eff_sched_var.sinr * eff_sched_var.sinr;
277
278 eff_sched_var.sinr3 = eff_sched_var.sinr2 * eff_sched_var.sinr;
279
280}

282void eff_scheduling_rotwing_update_MMOI(void)
283{
284 eff_sched_var.Ixx = eff_sched_p.Ixx_body + eff_sched_var.cosr2 * eff_sched_p.Ixx_wing + eff_sched_var.sinr2 * eff_sched_p.Iyy_wing;
285 eff_sched_var.Iyy = eff_sched_p.Iyy_body + eff_sched_var.sinr2 * eff_sched_p.Ixx_wing + eff_sched_var.cosr2 * eff_sched_p.Iyy_wing;
286
287 // Bound inertia
288 Bound(eff_sched_var.Ixx, 0.01, 100.);
289 Bound(eff_sched_var.Iyy, 0.01, 100.);
290}

309void eff_scheduling_rotwing_update_hover_motor_effectiveness(void)
310{
311 float cmd_quat[4];
312 float dM_dpprz[4];
313 // Quadratic thrust (and therefore moment) model of the hover propellers
314 for (uint8_t i = 0; i < 4; i++) {
315 cmd_quat[i] = actuator_state_filt_vect[i];
316 Bound(cmd_quat[i], 2500, MAX_PPRZ);
317
318 if(i==0 || i==2) { // pitch motors
319 dM_dpprz[i] = (eff_sched_p.DMdpprz_hover_pitch[0] + 2*cmd_quat[i] * eff_sched_p.DMdpprz_hover_pitch[1]) / 10000.;
320 // Bound dM_dpprz to half and 2 times the hover effectiveness
321 Bound(dM_dpprz[i], eff_sched_var.pitch_motor_dMdpprz * 0.5, eff_sched_var.pitch_motor_dMdpprz * 2.0);
322 } else { // roll motors
323 dM_dpprz[i] = (eff_sched_p.DMdpprz_hover_roll[0] + 2*cmd_quat[i] * eff_sched_p.DMdpprz_hover_roll[1]) / 10000.;
324 Bound(dM_dpprz[i], eff_sched_var.roll_motor_dMdpprz * 0.5, eff_sched_var.roll_motor_dMdpprz * 2.0);
325 }
326 }
327
328 // Roll motor effectiveness
329 float dM_dpprz_right = dM_dpprz[1];
330 float dM_dpprz_left = dM_dpprz[3];;
331
332 float roll_motor_p_eff_right = -(dM_dpprz_right * eff_sched_var.cosr + eff_sched_p.hover_roll_roll_coef[0] * eff_sched_var.wing_rotation_rad * eff_sched_var.wing_rotation_rad * eff_sched_var.airspeed * eff_sched_var.cosr) / eff_sched_var.Ixx;
333 roll_motor_p_eff_right = bound_or_zero(roll_motor_p_eff_right, -1.f, -0.00001f);
334
335 float roll_motor_p_eff_left = (dM_dpprz_left * eff_sched_var.cosr + eff_sched_p.hover_roll_roll_coef[0] * eff_sched_var.wing_rotation_rad * eff_sched_var.wing_rotation_rad * eff_sched_var.airspeed * eff_sched_var.cosr) / eff_sched_var.Ixx;
336 if(autopilot.in_flight) {
337 float roll_motor_airspeed_compensation = eff_sched_p.hover_roll_roll_coef[1] * eff_sched_var.airspeed * eff_sched_var.cosr / eff_sched_var.Ixx;
338 roll_motor_p_eff_left += roll_motor_airspeed_compensation;
339 }
340 roll_motor_p_eff_left = bound_or_zero(roll_motor_p_eff_left, 0.00001f, 1.f);
341
342 float roll_motor_q_eff = (eff_sched_p.hover_roll_pitch_coef[0] * eff_sched_var.wing_rotation_rad + eff_sched_p.hover_roll_pitch_coef[1] * eff_sched_var.wing_rotation_rad * eff_sched_var.wing_rotation_rad * eff_sched_var.sinr) / eff_sched_var.Iyy;
343 Bound(roll_motor_q_eff, 0, 1);
344
345 // Update front pitch motor q effectiveness
346 g1g2[1][0] = dM_dpprz[0] / eff_sched_var.Iyy; // pitch effectiveness front motor
347
348 // Update back motor q effectiveness
349 g1g2[1][2] = -dM_dpprz[2] / eff_sched_var.Iyy; // pitch effectiveness back motor
350
351 g1g2[0][1] = roll_motor_p_eff_right; // roll effectiveness right motor (no airspeed compensation)
352 g1g2[1][1] = roll_motor_q_eff; // pitch effectiveness right motor
353
354 // Update left motor p and q effectiveness
355 g1g2[0][3] = roll_motor_p_eff_left; // roll effectiveness left motor
356 g1g2[1][3] = -roll_motor_q_eff; // pitch effectiveness left motor
357}

359void eff_scheduling_rotwing_update_elevator_effectiveness(void)
360{
361 float de = eff_sched_p.k_elevator_deflection[0] + eff_sched_p.k_elevator_deflection[1] * eff_sched_var.cmd_elevator;
362
363 float dMyde = (eff_sched_p.k_elevator[0] * de * eff_sched_var.airspeed2 +
364 eff_sched_p.k_elevator[1] * eff_sched_var.cmd_pusher_scaled * eff_sched_var.cmd_pusher_scaled * eff_sched_var.airspeed +
365 eff_sched_p.k_elevator[2] * eff_sched_var.airspeed2) / 10000.;
366
367 // scale the effectiveness of the elevator down if it has a large deflection to encourage it to become flat quickly (pragmatic, not physically inpsired)
368 float elevator_ineffectiveness_scaling = (50-de)/40;
369 Bound(elevator_ineffectiveness_scaling, 0.5, 1.0);
370
371 float dMydpprz = dMyde * eff_sched_p.k_elevator_deflection[1];
372
373 // Convert moment to effectiveness
374 float eff_y_elev = dMydpprz / eff_sched_var.Iyy;
375
376 Bound(eff_y_elev, 0.00001, 0.1);
377
378 g1g2[1][5] = eff_y_elev;
379}

381void eff_scheduling_rotwing_update_rudder_effectiveness(void)
382{
383 float dMzdr = (eff_sched_p.k_rudder[0] * eff_sched_var.cmd_pusher_scaled * eff_sched_var.cmd_T_mean_scaled +
384 eff_sched_p.k_rudder[1] * eff_sched_var.cmd_T_mean_scaled * eff_sched_var.airspeed2 * eff_sched_var.cosr +
385 eff_sched_p.k_rudder[2] * eff_sched_var.airspeed2) / 10000.;
386
387 // Convert moment to effectiveness
388
389 float dMzdpprz = dMzdr * eff_sched_p.d_rudder_d_pprz;
390
391 // Convert moment to effectiveness
392 float eff_z_rudder = dMzdpprz / eff_sched_p.Izz;
393
394 Bound(eff_z_rudder, 0.000001, 0.1);
395
396 g1g2[2][4] = eff_z_rudder;
397}

399void eff_scheduling_rotwing_update_aileron_effectiveness(void)
400{
401 float dMxdpprz = (eff_sched_p.k_aileron * eff_sched_var.airspeed2 * eff_sched_var.sinr3) / 1000000.;
402 float eff_x_aileron = dMxdpprz / eff_sched_var.Ixx;
403 Bound(eff_x_aileron, 0, 0.005)
404 g1g2[0][6] = eff_x_aileron;
405
406 float dMydpprz = 4.0*(eff_sched_p.k_aileron * eff_sched_var.airspeed2 * eff_sched_var.sinr2 * eff_sched_var.cosr) / 1000000.;
407 float eff_y_aileron = dMydpprz / eff_sched_var.Iyy;
408 eff_y_aileron = bound_or_zero(eff_y_aileron, 0.00003f, 0.005f);
409 g1g2[1][6] = eff_y_aileron;
410}

420void eff_scheduling_rotwing_update_pusher_effectiveness(void)
421{
422 float rpmP = eff_sched_p.k_rpm_pprz_pusher[0] + eff_sched_p.k_rpm_pprz_pusher[1] * eff_sched_var.cmd_pusher + eff_sched_p.k_rpm_pprz_pusher[2] * eff_sched_var.cmd_pusher * eff_sched_var.cmd_pusher;
423
424 float dFxdrpmP = eff_sched_p.k_pusher[0]*rpmP + eff_sched_p.k_pusher[1] * eff_sched_var.airspeed;
425 float drpmPdpprz = eff_sched_p.k_rpm_pprz_pusher[1] + 2. * eff_sched_p.k_rpm_pprz_pusher[2] * eff_sched_var.cmd_pusher;
426
427 float eff_pusher = (dFxdrpmP * drpmPdpprz / eff_sched_p.m) / 10000.;
428
429 Bound(eff_pusher, 0.00030, 0.0015);
430 g1g2[4][8] = eff_pusher;
431}

435void eff_scheduling_rotwing_schedule_liftd(void)
436{
437 float lift_d_wing = (eff_sched_p.k_lift_wing[0] + eff_sched_p.k_lift_wing[1] * eff_sched_var.sinr2) * eff_sched_var.airspeed2 / eff_sched_p.m;
438 float lift_d_fuselage = eff_sched_p.k_lift_fuselage * eff_sched_var.airspeed2 / eff_sched_p.m;
439 float lift_d_tail = eff_sched_p.k_lift_tail * eff_sched_var.airspeed2 / eff_sched_p.m;
440
441 float lift_d = lift_d_wing + lift_d_fuselage + lift_d_tail;
442 if (eff_sched_var.wing_rotation_deg < 60.) {
443 lift_d = 0.0;
444 }
445 Bound(lift_d, -130., 0.);
446 eff_scheduling_rotwing_lift_d = lift_d;
447}

454void stabilization_indi_set_wls_settings(void)
455{
456 // Calculate the min and max increments
457 for (uint8_t i = 0; i < INDI_NUM_ACT; i++) {
458 wls_stab_p.u_min[i] = -MAX_PPRZ * act_is_servo[i];
459 wls_stab_p.u_max[i] = MAX_PPRZ;
460 wls_stab_p.u_pref[i] = act_pref[i];
461 if (i == 5) { // elevator
462 wls_stab_p.u_pref[i] = actuator_state_filt_vect[i]; // Set change in prefered state to 0 for elevator
463 wls_stab_p.u_min[i] = 0; // cmd 0 is lowest position for elevator
464 }
465 if (i == 7) { // flaperons
466 // If an offset is used, limit the max differential command to prevent unilateral saturation.
467 int32_t flap_saturation_limit = MAX_PPRZ - abs(rw_flap_offset);
468 BoundAbs(flap_saturation_limit, MAX_PPRZ);
469 wls_stab_p.u_min[i] = -flap_saturation_limit;
470 wls_stab_p.u_max[i] = flap_saturation_limit;
471 }
472 if (i==8) { // pusher
473 // dt (min to max) MAX_PPRZ / (dt * f) dt_min == 0.002
474 Bound(eff_sched_pusher_time, 0.002, 5.);
475 float max_increment = MAX_PPRZ / (eff_sched_pusher_time * 500);
476 wls_stab_p.u_min[i] = actuators_pprz[i] - max_increment;
477 wls_stab_p.u_max[i] = actuators_pprz[i] + max_increment;
478
479 Bound(wls_stab_p.u_min[i], 0, MAX_PPRZ);
480 Bound(wls_stab_p.u_max[i], 0, MAX_PPRZ);
481 }
482 }

485void guidance_indi_hybrid_set_wls_settings(float body_v[3], float roll_angle, float pitch_angle)
486{
487 // adjust weights
488 float fixed_wing_percentile = (rotwing_state_hover_motors_idling())? 1:0; // TODO: when hover props go below 40%, ...
489 Bound(fixed_wing_percentile, 0, 1);
490#define AIRSPEED_IMPORTANCE_IN_FORWARD_WEIGHT 16
491
492 float Wv_original[GUIDANCE_INDI_HYBRID_V] = GUIDANCE_INDI_WLS_PRIORITIES;
493
494 // Increase importance of forward acceleration in forward flight
495 wls_guid_p.Wv[0] = Wv_original[0] * (1.0f + fixed_wing_percentile *
496 AIRSPEED_IMPORTANCE_IN_FORWARD_WEIGHT); // stall n low hover motor_off (weight 16x more important than vertical weight)
497
498 struct FloatEulers eulers_zxy;
499 float_eulers_of_quat_zxy(&eulers_zxy, stateGetNedToBodyQuat_f());
500
501 float du_min_thrust_z = ((MAX_PPRZ - actuator_state_filt_vect[0]) * g1g2[3][0] + (MAX_PPRZ -
502 actuator_state_filt_vect[1]) * g1g2[3][1] + (MAX_PPRZ - actuator_state_filt_vect[2]) * g1g2[3][2] +
503 (MAX_PPRZ - actuator_state_filt_vect[3]) * g1g2[3][3]) * rotwing_state_hover_motors_running();
504 Bound(du_min_thrust_z, -50., 0.);
505 float du_max_thrust_z = -(actuator_state_filt_vect[0] * g1g2[3][0] + actuator_state_filt_vect[1] * g1g2[3][1] +
506 actuator_state_filt_vect[2] * g1g2[3][2] + actuator_state_filt_vect[3] * g1g2[3][3]);
507 Bound(du_max_thrust_z, 0., 50.);
508
509 float roll_limit_rad = guidance_indi_max_bank;
510 float max_pitch_limit_rad = RadOfDeg(GUIDANCE_INDI_MAX_PITCH);
511 float min_pitch_limit_rad = RadOfDeg(GUIDANCE_INDI_MIN_PITCH);
512
513 float fwd_pitch_limit_rad = RadOfDeg(GUIDANCE_INDI_MAX_PITCH);
514 float quad_pitch_limit_rad = RadOfDeg(5.0);
515
516 float airspeed = stateGetAirspeed_f();
517
518 float scheduled_pitch_angle = 0.f;
519 float pitch_angle_range = 3.;
520 float meas_skew_angle = rotwing_state.meas_skew_angle_deg;
521 Bound(meas_skew_angle, 0, 90); // Bound to prevent errors
522 if (meas_skew_angle < ROTWING_SKEW_ANGLE_STEP) {
523 scheduled_pitch_angle = ROTWING_QUAD_PREF_PITCH;
524 wls_guid_p.Wu[1] = Wu_gih_original[1];
525 max_pitch_limit_rad = quad_pitch_limit_rad;
526 } else {
527 float pitch_progression = (airspeed - rotwing_state.fw_min_airspeed) / 2.f;
528 Bound(pitch_progression, 0.f, 1.f);
529 scheduled_pitch_angle = pitch_angle_range * pitch_progression + ROTWING_QUAD_PREF_PITCH*(1.f-pitch_progression);
530 wls_guid_p.Wu[1] = Wu_gih_original[1] * (1.f - pitch_progression*0.99);
531 max_pitch_limit_rad = quad_pitch_limit_rad + (fwd_pitch_limit_rad - quad_pitch_limit_rad) * pitch_progression;
532 }
533 if (!rotwing_state_hover_motors_running()) {
534 scheduled_pitch_angle = 8.;
535 max_pitch_limit_rad = fwd_pitch_limit_rad;
536 }
537 Bound(scheduled_pitch_angle, -5., 8.);
538 guidance_indi_pitch_pref_deg = scheduled_pitch_angle;
539
540 float pitch_pref_rad = RadOfDeg(guidance_indi_pitch_pref_deg);
541
542 // Set lower limits
543 wls_guid_p.u_min[0] = -roll_limit_rad - roll_angle; //roll
544 wls_guid_p.u_min[1] = min_pitch_limit_rad - pitch_angle; // pitch
545
546 // Set upper limits limits
547 wls_guid_p.u_max[0] = roll_limit_rad - roll_angle; //roll
548 wls_guid_p.u_max[1] = max_pitch_limit_rad - pitch_angle; // pitch
549
550 if(rotwing_state_hover_motors_running()) {
551 wls_guid_p.u_min[2] = du_min_thrust_z;
552 wls_guid_p.u_max[2] = du_max_thrust_z;
553 } else {
554 wls_guid_p.u_min[2] = 0.;
555 wls_guid_p.u_max[2] = 0.;
556 }
557
558 if(rotwing_state_pusher_motor_running()) {
559 wls_guid_p.u_min[3] = (-actuator_state_filt_vect[8] * g1g2[4][8]);
560 wls_guid_p.u_max[3] = 9.0; // Hacky value to prevent drone from pitching down in transition
561 } else {
562 wls_guid_p.u_min[3] = 0.;
563 wls_guid_p.u_max[3] = 0.;
564 }
565
566 // Set prefered states
567 wls_guid_p.u_pref[0] = 0; // prefered delta roll angle
568 wls_guid_p.u_pref[1] = -pitch_angle + pitch_pref_rad;// prefered delta pitch angle
569 wls_guid_p.u_pref[2] = wls_guid_p.u_max[2]; // Low thrust better for efficiency
570 wls_guid_p.u_pref[3] = body_v[0]; // solve the body acceleration
571}
