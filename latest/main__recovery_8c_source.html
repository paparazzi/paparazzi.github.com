59  struct RadioControl *rc)
60 {
61  recovery_mode = RECOVERY_MODE_OF_PPRZ(rc->values[RADIO_MODE]);

106

136 void main_recovery_periodic(void)
137 {
138  if (sys_time_check_and_ack_timer(periodic_tid)) {
139  modules_mcu_periodic_task();
140  modules_radio_control_periodic_task();
141  // if RC is lost or not in manual, set failsafe commands
142  if (radio_control.status == RC_REALLY_LOST ||
143  recovery_mode == RECOVERY_MODE_FAILSAFE) {
144  recovery_mode = RECOVERY_MODE_FAILSAFE;
145  SetCommands(commands_failsafe);
146  }
147  modules_actuators_periodic_task();
148  }
149 
150 #if !(RECOVERY_DISABLE_ELECTRICAL)
151  if (sys_time_check_and_ack_timer(electrical_tid)) {
152  electrical_periodic();
153  }
154 #endif
155 
156 #if PERIODIC_TELEMETRY
157  if (sys_time_check_and_ack_timer(telemetry_tid)) {
158  // only fbw part during recovery
159  periodic_telemetry_send_Fbw(DefaultPeriodic, &(DefaultChannel).trans_tx, &(DefaultDevice).device);
160  }
161 #endif
162 
163 }
