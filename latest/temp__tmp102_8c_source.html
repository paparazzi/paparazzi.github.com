63void tmp102_init(void)
64{
65 tmp_meas_started = false;
66 /* configure 8Hz and enhanced mode */
67 tmp_trans.buf[0] = TMP102_CONF_REG;
68 tmp_trans.buf[1] = TMP102_CONF1;
69 tmp_trans.buf[2] = TMP102_CONF2;
70 i2c_transmit(&TMP_I2C_DEV, &tmp_trans, TMP102_SLAVE_ADDR, 3);
71}

73void tmp102_periodic(void)
74{
75 tmp_trans.buf[0] = TMP102_TEMP_REG;
76 i2c_transceive(&TMP_I2C_DEV, &tmp_trans, TMP102_SLAVE_ADDR, 1, 2);
77 tmp_meas_started = true;
78}

80void tmp102_event(void)
81{
82
83 if ((tmp_trans.status == I2CTransSuccess) && (tmp_meas_started == TRUE)) {
84
85 uint16_t tmp_temperature;
86
87 /* read two byte temperature */
88 tmp_temperature = tmp_trans.buf[0] << 8;
89 tmp_temperature |= tmp_trans.buf[1];
90 tmp_temperature >>= 3;
91 if (tmp_temperature & 0x1000) {
92 tmp_temperature |= 0xE000;
93 }
94
95 ftmp_temperature = ((int16_t) tmp_temperature) / 16.;
96
97 DOWNLINK_SEND_TMP_STATUS(DefaultChannel, DefaultDevice, &tmp_temperature, &ftmp_temperature);
98 tmp_trans.status = I2CTransDone;
99 }
