117void nav_parse_MOVE_WP(uint8_t *buf)
118{
119 uint8_t ac_id = DL_MOVE_WP_ac_id(buf);
120 if (ac_id != AC_ID) { return; }
121 if (stateIsLocalCoordinateValid()) {
122 uint8_t wp_id = DL_MOVE_WP_wp_id(buf);
123 struct LlaCoor_i lla;
124 lla.lat = DL_MOVE_WP_lat(buf);
125 lla.lon = DL_MOVE_WP_lon(buf);
126 /* WP_alt from message is alt above MSL in mm
127 * lla.alt is above ellipsoid in mm
128 */
129 lla.alt = DL_MOVE_WP_alt(buf) - stateGetHmslOrigin_i() +
130 stateGetLlaOrigin_i().alt;
131 waypoint_move_lla(wp_id, &lla);
132 }
133}

227float get_dist2_to_point(struct EnuCoor_f *p)
228{
229 struct EnuCoor_f *pos = stateGetPositionEnu_f();
230 struct FloatVect2 pos_diff = {
231 .x = p->x - pos->x,
232 .y = p->y - pos->y
233 };
234 return pos_diff.x * pos_diff.x + pos_diff.y * pos_diff.y;
235}

238float get_dist2_to_waypoint(uint8_t wp_id)
239{
240 return get_dist2_to_point(&waypoints[wp_id].enu_f);
241}

246void compute_dist2_to_home(void)
247{
248 nav.dist2_to_home = get_dist2_to_waypoint(WP_HOME);
249 nav.too_far_from_home = nav.dist2_to_home > max_dist2_from_home;
250#ifdef InGeofenceSector
251 struct EnuCoor_f *pos = stateGetPositionEnu_f();
252 nav.too_far_from_home = nav.too_far_from_home || !(InGeofenceSector(pos->x, pos->y));
253#endif
254}

270void nav_set_heading_towards(float x, float y)
271{
272 struct FloatVect2 target = {x, y};
273 struct FloatVect2 pos_diff;
274 VECT2_DIFF(pos_diff, target, *stateGetPositionEnu_f());
275 // don't change heading if closer than 0.5m to target
276 if (VECT2_NORM2(pos_diff) > 0.25f) {
277 nav.heading = atan2f(pos_diff.x, pos_diff.y);
278 }
279}
