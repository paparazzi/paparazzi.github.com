101void mcu_init(void)
102{
103 /* If we have a board specific init function, call it.
104 * Otherwise it will simply call the empty weak function.
105 *
106 * For example the ARDrone2 has this implemented to prevent stray data of IMU
107 * from OEM program still running and also accessing AC sensors
108 */
109 board_init();
110
111 mcu_arch_init();
112
113 /* First enable the power of the MCU if needed */
114#if defined MCU_PWR
115 gpio_setup_output(MCU_PWR, MCU_PWR_PIN);
116 MCU_PWR_ON(MCU_PWR, MCU_PWR_PIN);
117
118#if defined BTN_ON
119 gpio_setup_input(BTN_ON, BTN_ON_PIN);
120 if (gpio_get(BTN_ON, BTN_ON_PIN)) {
121 MCU_PWR_ON(MCU_PWR, MCU_PWR_PIN);
122 } else {
123 // Turn off and stop: wait until all power is off
124 while (true) {
125 MCU_PWR_OFF(MCU_PWR, MCU_PWR_PIN);
126 }
127 }
128#endif //BTN_ON
129
130#endif //MCU_PWR
131
132#ifdef PERIPHERALS_AUTO_INIT
133 sys_time_init();
134#ifdef USE_LED
135 led_init();
136#endif
137 /* First enable power of RC */
138#if defined RADIO_CONTROL_POWER_PORT
139 gpio_setup_output(RADIO_CONTROL_POWER_PORT, RADIO_CONTROL_POWER_PIN);
140 RADIO_CONTROL_POWER_ON(RADIO_CONTROL_POWER_PORT, RADIO_CONTROL_POWER_PIN);
141#endif
142#ifdef PERIPHERAL3V3_ENABLE_PORT
143 gpio_setup_output(PERIPHERAL3V3_ENABLE_PORT, PERIPHERAL3V3_ENABLE_PIN);
144 PERIPHERAL3V3_ENABLE_ON(PERIPHERAL3V3_ENABLE_PORT, PERIPHERAL3V3_ENABLE_PIN);
145#endif
146 /* for now this means using spektrum */
147#if defined RADIO_CONTROL && defined RADIO_CONTROL_BIND_IMPL_FUNC && defined SPEKTRUM_BIND_PIN_PORT
148 RADIO_CONTROL_BIND_IMPL_FUNC();
149#endif

152#endif
