132 static void compute_points_from_bungee(void)
133 {
134  // Store init point (current position, where the plane will be released)
135  VECT2_ASSIGN(init_point, stateGetPositionEnu_f()->x, stateGetPositionEnu_f()->y);
136  // Compute unitary 2D vector (bungee_point - init_point) = takeoff direction
137  VECT2_DIFF(takeoff_dir, bungee_point, init_point);
138  float_vect2_normalize(&takeoff_dir);
139  // Find throttle point (the point where the throttle line and launch line intersect)
140  // If TakeOff_Distance is positive, throttle point is after bungee point, before otherwise
141  VECT2_SMUL(throttle_point, takeoff_dir, BUNGEE_TAKEOFF_DISTANCE);
142  VECT2_SUM(throttle_point, bungee_point, throttle_point);
143 }

145 void nav_bungee_takeoff_setup(uint8_t bungee_wp)
146 {

148  VECT3_ASSIGN(bungee_point, WaypointX(bungee_wp), WaypointY(bungee_wp), stateGetPositionUtm_f()->alt);

158 bool nav_bungee_takeoff_run(void)
159 {
160  float cross = 0.;
161 
162  // Get current position
163  struct FloatVect2 pos;
164  VECT2_ASSIGN(pos, stateGetPositionEnu_f()->x, stateGetPositionEnu_f()->y);
165 
166  switch (CTakeoffStatus) {
167  case Launch:
168  // Recalculate lines if below min speed
169  if (stateGetHorizontalSpeedNorm_f() < BUNGEE_TAKEOFF_MIN_SPEED) {
170  compute_points_from_bungee();
171  }
172 
173  // Follow Launch Line with takeoff pitch and no throttle
174  NavVerticalAutoThrottleMode(BUNGEE_TAKEOFF_PITCH);
175  NavVerticalThrottleMode(0);
176  // FIXME previously using altitude mode, maybe not wise without motors
177  //NavVerticalAltitudeMode(bungee_point.z + BUNGEE_TAKEOFF_HEIGHT, 0.);
178  nav_route_xy(init_point.x, init_point.y, throttle_point.x, throttle_point.y);
179 
180  autopilot_set_kill_throttle(true);
181 
182  // Find out if UAV has crossed the line
183  VECT2_DIFF(pos, pos, throttle_point); // position local to throttle_point
184  cross = VECT2_DOT_PRODUCT(pos, takeoff_dir);
185 
186  if (cross > 0. && stateGetHorizontalSpeedNorm_f() > BUNGEE_TAKEOFF_MIN_SPEED) {
187  CTakeoffStatus = Throttle;
188  autopilot_set_kill_throttle(false);
189  nav_init_stage();
190  } else {
191  // If not crossed stay in this status
192  break;
193  }
194  // Start throttle imidiatelly
195  case Throttle:
196  //Follow Launch Line
197  NavVerticalAutoThrottleMode(BUNGEE_TAKEOFF_PITCH);
198  NavVerticalThrottleMode(MAX_PPRZ * (BUNGEE_TAKEOFF_THROTTLE));
199  autopilot.launch = true; // turn on motor
200  nav_route_xy(init_point.x, init_point.y, throttle_point.x, throttle_point.y);
201  autopilot_set_kill_throttle(false);
202 
203  if ((stateGetPositionUtm_f()->alt > bungee_point.z + BUNGEE_TAKEOFF_HEIGHT)
204 #if USE_AIRSPEED
205  && (stateGetAirspeed_f() > BUNGEE_TAKEOFF_AIRSPEED)
206 #endif
207  ) {
208  CTakeoffStatus = Finished;
209  return false;
210  } else {
211  return true;
212  }
213  break;
214  default:
215  // Invalid status or Finished, end function
216  return false;
217  }
218  return true;
219 }
