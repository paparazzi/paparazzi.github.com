100 void periodic_sysmon(void)
101 {
103  uint32_t periodic_usec = SysTimeTimer(periodic_timer);
104  SysTimeTimerStart(periodic_timer);
105  sum_time_periodic += periodic_usec;
106 
107  /* only periodic cycle : periodic_cycle = periodic_usec - sum_time_event; */
108  periodic_cycle = periodic_usec - n_event * min_time_event;
109  sum_cycle_periodic += periodic_cycle;
110 
111  /* remember min and max periodic times */
112  if (periodic_usec < sys_mon.periodic_time_min) {
113  sys_mon.periodic_time_min = periodic_usec;
114  }
115  if (periodic_usec > sys_mon.periodic_time_max) {
116  sys_mon.periodic_time_max = periodic_usec;
117  }
118 
119  /* remember min and max periodic cycle times */
120  if (periodic_cycle < sys_mon.periodic_cycle_min) {
121  sys_mon.periodic_cycle_min = periodic_cycle;
122  }
123  if (periodic_cycle > sys_mon.periodic_cycle_max) {
124  sys_mon.periodic_cycle_max = periodic_cycle;
125  }
126 
127  n_periodic++;
128  sum_n_event += n_event;
129  n_event = 0;
130  sum_time_event = 0;
131 }

133 void event_sysmon(void)
134 {
136  if (n_event > 0) {
137  uint32_t t = SysTimeTimer(event_timer);
138  if (t < min_time_event) {
139  min_time_event = t;
140  }
141  sum_time_event += t;
142  }
143  SysTimeTimerStart(event_timer);
144  n_event++;
145 }
