81 bool nps_radio_control_available(double time)
82 {
83  if (time >= nps_radio_control.next_update && nps_radio_control.type != NORC) {
84  nps_radio_control.next_update += RADIO_CONTROL_DT;
85 
86  if (nps_radio_control.type == JOYSTICK) {
87  nps_radio_control_joystick_update();
88  nps_radio_control.throttle = nps_joystick.throttle;
89  nps_radio_control.roll = nps_joystick.roll;
90  nps_radio_control.pitch = nps_joystick.pitch;
91  nps_radio_control.yaw = nps_joystick.yaw;
92  nps_radio_control.mode = nps_joystick.mode;
93  //printf("throttle: %f, roll: %f, pitch: %f, yaw: %f\n", nps_joystick.throttle, nps_joystick.roll, nps_joystick.pitch, nps_joystick.yaw);
94  } else if (nps_radio_control.type == SCRIPT) {
95  if (time < RADIO_CONTROL_TAKEOFF_TIME) {
96  radio_control_script_takeoff(time);
97  } else {
98  scripts[nps_radio_control.num_script](time);
99  }
100  }
101  return TRUE;
102  }
103  return FALSE;
104 }
