166static void data_actuators_t4_out(uint8_t sender_id __attribute__((unused)), struct ActuatorsT4Out * actuators_t4_out_ptr, float * actuators_t4_extra_data_out_ptr){
167
168 /* Copying the struct to be transmitted and the extra data in a local variable: */
169 memcpy(&actuators_t4_out,actuators_t4_out_ptr,sizeof(struct ActuatorsT4Out));
170 memcpy(&actuators_t4_extra_data_out,actuators_t4_extra_data_out_ptr, sizeof(actuators_t4_extra_data_out) );
171
172 /* Increase the counter to track the sending messages: */
173 actuators_t4_out.rolling_msg_out = actuators_t4_extra_data_out[actuators_t4_out_msg_id];
174 actuators_t4_out.rolling_msg_out_id = actuators_t4_out_msg_id;
175 actuators_t4_out_msg_id++;
176 if(actuators_t4_out_msg_id == 255){
177 actuators_t4_out_msg_id = 0;
178 }
179
180 /* Send the message over UART to the T4 Actuators Board: */
181 uint8_t *buf_send = (uint8_t *)&actuators_t4_out;
182 //Calculating the checksum
183 uint8_t checksum_out_local = 0;
184 for(uint16_t i = 0; i < sizeof(struct ActuatorsT4Out) - 1; i++){
185 checksum_out_local += buf_send[i];
186 }
187 actuators_t4_out.checksum_out = checksum_out_local;
188
189#ifdef ACTUATORS_T4_SIM
190 /* don't send the data if it is SIM */
191 //TODO: why not, could be useful for HITL
192#else
193 /* Do send the bytes */
194 uart_put_byte(&(ACTUATORS_T4_PORT), 0, START_BYTE_ACTUATORS_T4);
195 for(uint8_t i = 0; i < sizeof(struct ActuatorsT4Out) ; i++){
196 uart_put_byte(&(ACTUATORS_T4_PORT), 0, buf_send[i]);
197 }
198#endif
199}

215/* Send the received message over ABI so then another module can use this actuator state info in e.g. control modules */
216void actuators_t4_uart_parse_msg_in(void)
217{
218 memcpy(&actuators_t4_in, &actuators_t4_msg_buf_in[1], sizeof(struct ActuatorsT4In)); //Starting from 1 to avoid reading the starting byte
219 /* Assign the rolling message: */
220 actuators_t4_extra_data_in[actuators_t4_in.rolling_msg_in_id] = actuators_t4_in.rolling_msg_in;
221 /* Send msg through ABI: */
222 AbiSendMsgACTUATORS_T4_IN(ABI_ACTUATORS_T4_IN_ID, &actuators_t4_in, &actuators_t4_extra_data_in[0]);
223}

225/* Event checking if serial packet are available on the bus */
226void actuators_t4_uart_event()
227{
228 if(fabs(get_sys_time_float() - actuators_t4_last_ts) > 5){ //Reset received packets to zero every 5 second to update the statistics
229 actuators_t4_received_packets = 0;
230 actuators_t4_last_ts = get_sys_time_float();
231 }
232#ifdef ACTUATORS_T4_SIM //TODO: use the SIM, NPS and HITL flags ,but if HIL is used it should be able to send the data to the T4 board
233 /* Don't do anything if it is SIM */
234#else
235 while(uart_char_available(&(ACTUATORS_T4_PORT)) > 0) {
236 uint8_t actuators_t4_byte_in;
237 actuators_t4_byte_in = uart_getch(&(ACTUATORS_T4_PORT));
238 if ((actuators_t4_byte_in == START_BYTE_ACTUATORS_T4) || (actuators_t4_buf_in_cnt > 0)) {
239 actuators_t4_msg_buf_in[actuators_t4_buf_in_cnt] = actuators_t4_byte_in;
240 actuators_t4_buf_in_cnt++;
241 }
242 if (actuators_t4_buf_in_cnt > sizeof(struct ActuatorsT4In) ) {
243 actuators_t4_buf_in_cnt = 0;
244 uint8_t checksum_in_local = 0;
245 for(uint16_t i = 1; i < sizeof(struct ActuatorsT4In) ; i++){
246 checksum_in_local += actuators_t4_msg_buf_in[i];
247 }
248 if(checksum_in_local == actuators_t4_msg_buf_in[sizeof(struct ActuatorsT4In)]){
249 actuators_t4_uart_parse_msg_in();
250 actuators_t4_received_packets++;
251 }
252 else {
253 actuators_t4_missed_packets_in++;
254 }
255 }
256 }
257#endif
258 actuators_t4_message_frequency_in = (uint16_t) actuators_t4_received_packets/(get_sys_time_float() - actuators_t4_last_ts);
259}
