68void sonar_adc_read(void)
69{
70#ifndef SITL
71 sonar_adc.meas = sonar_adc_buf.sum / sonar_adc_buf.av_nb_sample;
72 sonar_adc.distance = (float)(sonar_adc.meas - sonar_adc.offset) * SONAR_SCALE;
73#else // SITL
74 sonar_adc.distance = stateGetPositionEnu_f()->z;
75 Bound(sonar_adc.distance, 0.1f, 7.0f);
76#endif // SITL
77
78 // Send ABI message
79 uint32_t now_ts = get_sys_time_usec();
80 AbiSendMsgAGL(AGL_SONAR_ADC_ID, now_ts, sonar_adc.distance);
81
82#ifdef SENSOR_SYNC_SEND_SONAR
83 // Send Telemetry report
84 DOWNLINK_SEND_SONAR(DefaultChannel, DefaultDevice, &sonar_adc.meas, &sonar_adc.distance);
85#endif
86}
