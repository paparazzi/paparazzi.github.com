113static void send_svinfo_id(struct transport_tx *trans, struct link_device *dev,
114 uint8_t svid)
115{
116 if (svid < GPS_NB_CHANNELS) {
117 pprz_msg_send_SVINFO(trans, dev, AC_ID, &svid,
118 &gps.svinfos[svid].svid, &gps.svinfos[svid].flags,
119 &gps.svinfos[svid].qi, &gps.svinfos[svid].cno,
120 &gps.svinfos[svid].elev, &gps.svinfos[svid].azim);
121 }
122}

125static void send_svinfo(struct transport_tx *trans, struct link_device *dev)
126{
127 static uint8_t i = 0;
128 if (i >= gps.nb_channels) { i = 0; }
129 send_svinfo_id(trans, dev, i);
130 i++;
131}
132
137static inline void send_svinfo_available(struct transport_tx *trans, struct link_device *dev)
138{
139 static uint8_t i = 0;
140 static uint8_t last_cnos[GPS_NB_CHANNELS];
141 if (i >= gps.nb_channels) { i = 0; }
142 // send SVINFO for all satellites while no GPS fix,
143 // after 3D fix, send avialable sats if they were updated
144 if (gps.fix < GPS_FIX_3D) {
145 send_svinfo_id(trans, dev, i);
146 } else if (gps.svinfos[i].cno != last_cnos[i]) {
147 send_svinfo_id(trans, dev, i);
148 last_cnos[i] = gps.svinfos[i].cno;
149 }
150 i++;
151}

153static void send_gps(struct transport_tx *trans, struct link_device *dev)
154{
155 uint8_t zero = (gps_disable_fix <= 0)? 0 : gps_disable_fix;
156 int16_t climb = -gps.ned_vel.z;
157 int16_t course = (DegOfRad(gps.course) / ((int32_t)1e6));
158 struct UtmCoor_i utm = utm_int_from_gps(&gps, 0);
159#if PPRZLINK_DEFAULT_VER == 2 && GPS_POS_BROADCAST
160 // broadcast GPS message
161 struct pprzlink_msg msg;
162 msg.trans = trans;
163 msg.dev = dev;
164 msg.sender_id = AC_ID;
165 msg.receiver_id = PPRZLINK_MSG_BROADCAST;
166 msg.component_id = 0;
167 pprzlink_msg_send_GPS(&msg,
168#else
169 pprz_msg_send_GPS(trans, dev, AC_ID,
170#endif
171 &gps.fix,
172 &utm.east, &utm.north,
173 &course, &gps.hmsl, &gps.gspeed, &climb,
174 &gps.week, &gps.tow, &utm.zone, &zero);
175 // send SVINFO for available satellites that have new data
176 send_svinfo_available(trans, dev);
177}

179static void send_gps_relpos(struct transport_tx *trans, struct link_device *dev)
180{
181 static uint8_t idx = 0;
182 if(gps_relposned[idx].tow == 0)
183 return;
184
185 pprz_msg_send_GPS_RELPOS(trans, dev, AC_ID,
186 &gps_relposned[idx].reference_id,
187 &gps_relposned[idx].tow,
188 &gps_relposned[idx].pos.x, &gps_relposned[idx].pos.y, &gps_relposned[idx].pos.z,
189 &gps_relposned[idx].distance,
190 &gps_relposned[idx].heading,
191 &gps_relposned[idx].pos_acc.x, &gps_relposned[idx].pos_acc.y, &gps_relposned[idx].pos_acc.z,
192 &gps_relposned[idx].distance_acc,
193 &gps_relposned[idx].heading_acc);
194
195 // Send the next index that is set
196 idx++;
197 if(idx >= GPS_RELPOS_MAX || gps_relposned[idx].tow == 0)
198 idx = 0;
199}

260static uint8_t gps_multi_switch(struct GpsState *gps_s)
261{
262 static uint32_t time_since_last_gps_switch = 0;
263
264 if (multi_gps_mode == GPS_MODE_PRIMARY) {
265 return GpsId(PRIMARY_GPS);
266 } else if (multi_gps_mode == GPS_MODE_SECONDARY) {
267 return GpsId(SECONDARY_GPS);
268 } else {
269 if (gps_s->fix > gps.fix) {
270 return gps_s->comp_id;
271 } else if (gps.fix > gps_s->fix) {
272 return gps.comp_id;
273 } else {
274 if (get_sys_time_msec() - time_since_last_gps_switch > TIME_TO_SWITCH) {
275 if (gps_s->num_sv > gps.num_sv) {
276 current_gps_id = gps_s->comp_id;
277 time_since_last_gps_switch = get_sys_time_msec();
278 } else if (gps.num_sv > gps_s->num_sv) {
279 current_gps_id = gps.comp_id;
280 time_since_last_gps_switch = get_sys_time_msec();
281 }
282 }
283 }
284 }
285 return current_gps_id;
286}
287#endif /*SECONDARY_GPS*/

290void gps_periodic_check(struct GpsState *gps_s)
291{
292 if (sys_time.nb_sec - gps_s->last_msg_time > GPS_TIMEOUT) {
293 gps_s->fix = GPS_FIX_NONE;
294 }
295
296#ifdef SECONDARY_GPS
297 current_gps_id = gps_multi_switch(gps_s);
298 if (gps_s->comp_id == current_gps_id) {
299 gps = *gps_s;
300 }
301#else
302 gps = *gps_s;
303#endif
304}

306bool gps_fix_valid(void)
307{
308 bool gps_3d_timeout_valid = false;
309#ifdef GPS_FIX_TIMEOUT
310 if (get_sys_time_float() - gps_time_since_last_3dfix() < GPS_FIX_TIMEOUT) {
311 gps_3d_timeout_valid = true;
312 }
313#endif
314 return (gps.fix >= GPS_FIX_3D || gps_3d_timeout_valid);
315}

318static abi_event gps_ev;
319static void gps_cb(uint8_t sender_id,
320 uint32_t stamp __attribute__((unused)),
321 struct GpsState *gps_s)
322{
323 /* ignore callback from own AbiSendMsgGPS */
324 if (sender_id == GPS_MULTI_ID) {
325 return;
326 }
327
328 uint8_t old_fix = gps_s->fix;
329 uint32_t now_ts = get_sys_time_usec();
330#ifdef SECONDARY_GPS
331 current_gps_id = gps_multi_switch(gps_s);
332 if (gps_s->comp_id == current_gps_id) {
333 gps = *gps_s;
334#if GPS_LOSS_TEST_TIMER
335 if(gps_disable_fix > 0) {
336 gps_s->fix = GPS_FIX_NONE;
337 }
338#endif
339 AbiSendMsgGPS(GPS_MULTI_ID, now_ts, gps_s);
340 gps_s->fix = old_fix;
341 }
342#else
343 gps = *gps_s;
344#if GPS_LOSS_TEST_TIMER
345 if(gps_disable_fix > 0) {
346 gps_s->fix = GPS_FIX_NONE;
347 }
348#endif
349 AbiSendMsgGPS(GPS_MULTI_ID, now_ts, gps_s);
350 gps_s->fix = old_fix;
351#endif
352 if (gps.tow != gps_time_sync.t0_tow) {
353 gps_time_sync.t0_ticks = sys_time.nb_tick;
354 gps_time_sync.t0_tow = gps.tow;
355 }
356}

359static void gps_relpos_cb(uint8_t sender_id __attribute__((unused)),
360 uint32_t stamp __attribute__((unused)),
361 struct RelPosNED *relpos)

363 for(uint8_t i = 0; i < GPS_RELPOS_MAX; i++) {

366 {
367 // Copy and save result
368 gps_relposned[i] = *relpos;
369 break;
370 }
371 }

419uint32_t gps_tow_from_sys_ticks(uint32_t sys_ticks)
420{
421 uint32_t clock_delta;
422 uint32_t time_delta;
423 uint32_t itow_now;
424
425 if (sys_ticks < gps_time_sync.t0_ticks) {
426 clock_delta = (0xFFFFFFFF - sys_ticks) + gps_time_sync.t0_ticks + 1;
427 } else {
428 clock_delta = sys_ticks - gps_time_sync.t0_ticks;
429 }
430
431 time_delta = msec_of_sys_time_ticks(clock_delta);
432
433 itow_now = gps_time_sync.t0_tow + time_delta;
434 if (itow_now > MSEC_PER_WEEK) {
435 itow_now %= MSEC_PER_WEEK;
436 }
437
438 return itow_now;
439}

603struct UtmCoor_f utm_float_from_gps(struct GpsState *gps_s, uint8_t zone)
604{
605 struct UtmCoor_f utm = {.east = 0., .north = 0., .alt = 0., .zone = zone};
606
607 if (bit_is_set(gps_s->valid_fields, GPS_VALID_POS_UTM_BIT)) {
608 /* A real UTM position is available, use the correct zone */
609 UTM_FLOAT_OF_BFP(utm, gps_s->utm_pos);
610 } else if (bit_is_set(gps_s->valid_fields, GPS_VALID_POS_LLA_BIT)) {
611 /* Recompute UTM coordinates in this zone */
612 struct UtmCoor_i utm_i;
613 utm_i.zone = zone;
614 utm_of_lla_i(&utm_i, &gps_s->lla_pos);
615 UTM_FLOAT_OF_BFP(utm, utm_i);
616
617 /* set utm.alt in hsml */
618 if (bit_is_set(gps_s->valid_fields, GPS_VALID_HMSL_BIT)) {
619 utm.alt = gps_s->hmsl / 1000.;
620 } else {
621 utm.alt = wgs84_ellipsoid_to_geoid_i(gps_s->lla_pos.lat, gps_s->lla_pos.lon) / 1000.;
622 }
623 }
624
625 return utm;
626}

628struct UtmCoor_i utm_int_from_gps(struct GpsState *gps_s, uint8_t zone)
629{
630 struct UtmCoor_i utm = {.east = 0, .north = 0, .alt = 0, .zone = zone};
631
632 if (bit_is_set(gps_s->valid_fields, GPS_VALID_POS_UTM_BIT)) {
633 // A real UTM position is available, use the correct zone
634 UTM_COPY(utm, gps_s->utm_pos);
635 } else if (bit_is_set(gps_s->valid_fields, GPS_VALID_POS_LLA_BIT)) {
636 /* Recompute UTM coordinates in zone */
637 utm_of_lla_i(&utm, &gps_s->lla_pos);
638
639 /* set utm.alt in hsml */
640 if (bit_is_set(gps_s->valid_fields, GPS_VALID_HMSL_BIT)) {
641 utm.alt = gps_s->hmsl;
642 } else {
643 utm.alt = wgs84_ellipsoid_to_geoid_i(gps_s->lla_pos.lat, gps_s->lla_pos.lon);
644 }
645 }
646
647 return utm;
648}

655// Major index 0 is for non-leap years, and 1 is for leap years
656// Minor index is for month number 1 .. 12, 0 at index 0 is number of days before January
657static const uint16_t month_days[2][13] = {
658 { 0, 31, 59, 90, 120, 151, 181, 212, 243, 273, 304, 334, 365 },
659 { 0, 31, 60, 91, 121, 152, 182, 213, 244, 274, 305, 335, 366 }
660};
661
662// Count the days since start of 1980, 6th of January
663// Counts year * 356 days + leap days + month lengths + days in month
664// The leap days counting needs the "+ 1" because GPS year 0 (i.e. 1980) was a leap year
665uint16_t gps_day_number(uint16_t year, uint8_t month, uint8_t day)
666{
667 uint16_t gps_years = year - 1980;
668 uint16_t leap_year = (gps_years % 4 == 0) ? 1 : 0;
669 uint16_t day_of_year = month_days[leap_year][month - 1] + day;
670 if (gps_years == 0)
671 return day_of_year;
672 return gps_years * 365 + ((gps_years - 1) / 4) + 1 + day_of_year - 6;
673}
674
675uint16_t gps_week_number(uint16_t year, uint8_t month, uint8_t day)
676{
677 return gps_day_number(year, month, day) / 7;
678}
