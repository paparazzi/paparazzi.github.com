168void airspeed_ets_read_periodic(void)
169{
170#ifndef SITL
171 if (!airspeed_ets_delay_done) {
172 if (SysTimeTimer(airspeed_ets_delay_time) < USEC_OF_SEC(AIRSPEED_ETS_START_DELAY)) { return; }
173 else { airspeed_ets_delay_done = true; }
174 }
175 if (airspeed_ets_i2c_trans.status == I2CTransDone) {
176 i2c_receive(&AIRSPEED_ETS_I2C_DEV, &airspeed_ets_i2c_trans, AIRSPEED_ETS_ADDR, 2);
177 }
178#elif !defined USE_NPS
179 extern float sim_air_speed;
180 stateSetAirspeed_f(MODULE_AIRSPEED_ETS_ID, sim_air_speed);
181#endif //SITL
182}

184void airspeed_ets_read_event(void)
185{
186 int n;
187 float airspeed_tmp = 0.0;
188
189 // Get raw airspeed from buffer
190 airspeed_ets_raw = ((uint16_t)(airspeed_ets_i2c_trans.buf[1]) << 8) | (uint16_t)(airspeed_ets_i2c_trans.buf[0]);
191 // Check if this is valid airspeed
192 if (airspeed_ets_raw == 0) {
193 airspeed_ets_valid = false;
194 } else {
195 airspeed_ets_valid = true;
196 }
197
198 // Continue only if a new airspeed value was received
199 if (airspeed_ets_valid) {
200#if !AIRSPEED_ETS_3RD_PARTY_MODE
201 // Calculate offset average if not done already
202 if (!airspeed_ets_offset_init) {
203 --airspeed_ets_cnt;
204 // Check if averaging completed
205 if (airspeed_ets_cnt == 0) {
206 // Calculate average
207 airspeed_ets_offset = (uint16_t)(airspeed_ets_offset_tmp / AIRSPEED_ETS_OFFSET_NBSAMPLES_AVRG);
208 // Limit offset
209 if (airspeed_ets_offset < AIRSPEED_ETS_OFFSET_MIN) {
210 airspeed_ets_offset = AIRSPEED_ETS_OFFSET_MIN;
211 }
212 if (airspeed_ets_offset > AIRSPEED_ETS_OFFSET_MAX) {
213 airspeed_ets_offset = AIRSPEED_ETS_OFFSET_MAX;
214 }
215 airspeed_ets_offset_init = true;
216 }
217 // Check if averaging needs to continue
218 else if (airspeed_ets_cnt <= AIRSPEED_ETS_OFFSET_NBSAMPLES_AVRG) {
219 airspeed_ets_offset_tmp += airspeed_ets_raw;
220 }
221 }
222 // Convert raw to m/s
223#ifdef AIRSPEED_ETS_REVERSE
224 if (airspeed_ets_offset_init && airspeed_ets_raw < airspeed_ets_offset) {
225 airspeed_tmp = AIRSPEED_ETS_SCALE * sqrtf((float)(airspeed_ets_offset - airspeed_ets_raw)) - AIRSPEED_ETS_OFFSET;
226 }
227#else
228 if (airspeed_ets_offset_init && airspeed_ets_raw > airspeed_ets_offset) {
229 airspeed_tmp = AIRSPEED_ETS_SCALE * sqrtf((float)(airspeed_ets_raw - airspeed_ets_offset)) - AIRSPEED_ETS_OFFSET;
230 }
231#endif
232 else {
233 airspeed_tmp = 0.0;
234 }
235//use raw value for sensor set to third-party mode
236#else
237 airspeed_tmp = airspeed_ets_raw;
238#endif //AIRSPEED_ETS_3RD_PARTY_MODE
239
240 // Airspeed should always be positive
241 if (airspeed_tmp < 0.0) {
242 airspeed_tmp = 0.0;
243 }
244 // Moving average
245 airspeed_ets_buffer[airspeed_ets_buffer_idx++] = airspeed_tmp;
246 if (airspeed_ets_buffer_idx >= AIRSPEED_ETS_NBSAMPLES_AVRG) {
247 airspeed_ets_buffer_idx = 0;
248 }
249 airspeed_ets = 0.0;
250 for (n = 0; n < AIRSPEED_ETS_NBSAMPLES_AVRG; ++n) {
251 airspeed_ets += airspeed_ets_buffer[n];
252 }
253 airspeed_ets = airspeed_ets / (float)AIRSPEED_ETS_NBSAMPLES_AVRG;
254
255 // Publish airspeed sensor
256 AbiSendMsgAIRSPEED(AIRSPEED_ETS_ID, airspeed_ets);
257
258#if USE_AIRSPEED_ETS
259 stateSetAirspeed_f(MODULE_AIRSPEED_ETS_ID, airspeed_ets);
260#endif
261 } else {
262 airspeed_ets = 0.0;
263 }
264
265
266#if AIRSPEED_ETS_SDLOG
267#ifndef SITL
268 if (pprzLogFile != -1) {
269 if (!log_airspeed_ets_started) {
270 sdLogWriteLog(pprzLogFile, "AIRSPEED_ETS: raw offset airspeed(m/s) GPS_fix TOW(ms) Week Lat(1e7deg) Lon(1e7deg) HMSL(mm) gpseed(cm/s) course(1e7rad) climb(cm/s)\n");
271 log_airspeed_ets_started = true;
272 }
273 sdLogWriteLog(pprzLogFile, "airspeed_ets: %d %d %8.4f %d %d %d %d %d %d %d %d %d\n",
274 airspeed_ets_raw, airspeed_ets_offset, airspeed_ets,
275 gps.fix, gps.tow, gps.week,
276 gps.lla_pos.lat, gps.lla_pos.lon, gps.hmsl,
277 gps.gspeed, gps.course, -gps.ned_vel.z);
278 }
279#endif
280#endif
