242void dma1_c4_irq_handler(void)
243{
244
245 DMA_ITConfig(DMA1_Channel4, DMA_IT_TC, DISABLE);
246 /* Disable SPI_2 Rx and TX request */
247 SPI_I2S_DMACmd(SPI2, SPI_I2S_DMAReq_Rx, DISABLE);
248 SPI_I2S_DMACmd(SPI2, SPI_I2S_DMAReq_Tx, DISABLE);
249 /* Disable DMA1 Channel4 and 5 */
250 DMA_Cmd(DMA1_Channel4, DISABLE);
251 DMA_Cmd(DMA1_Channel5, DISABLE);
252
253 switch (sc18is600.transaction) {
254 case Sc18Is600ReadRegister:
255 case Sc18Is600WriteRegister:
256 sc18is600.status = Sc18Is600TransactionComplete;
257 Sc18Is600Unselect();
258 break;
259 case Sc18Is600Transmit:
260 if (sc18is600.status == Sc18Is600SendingRequest) {
261 sc18is600.status = Sc18Is600WaitingForI2C;
262 Sc18Is600Unselect();
263 } else if (sc18is600.status == Sc18Is600ReadingI2CStat) {
264 sc18is600.i2c_status = sc18is600.priv_rx_buf[2];
265 sc18is600.status = Sc18Is600TransactionComplete;
266 Sc18Is600Unselect();
267 }
268 break;
269 case Sc18Is600Receive:
270 case Sc18Is600Transcieve:
271 if (sc18is600.status == Sc18Is600SendingRequest) {
272 sc18is600.status = Sc18Is600WaitingForI2C;
273 Sc18Is600Unselect();
274 } else if (sc18is600.status == Sc18Is600ReadingI2CStat) {
275 sc18is600.status = Sc18Is600ReadingBuffer;
276 sc18is600.priv_tx_buf[0] = Sc18Is600_Cmd_Read_Buffer;
277 // debug
278 for (int i = 1; i < sc18is600.rx_len + 1; i++) { sc18is600.priv_tx_buf[i] = 0; }
279 Sc18Is600Select();
280 sc18is600_setup_SPI_DMA(sc18is600.rx_len + 1);
281 } else if (sc18is600.status == Sc18Is600ReadingBuffer) {
282 sc18is600.status = Sc18Is600TransactionComplete;
283 Sc18Is600Unselect();
284 }
285 break;
286 default:
287 break;
288 }
