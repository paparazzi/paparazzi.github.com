44 {
45  struct Int32Eulers sp_euler = stab_sp_to_eulers_i(sp);
46

48  const int32_t angle2cmd = (MAX_PPRZ / TRAJ_MAX_BANK);
49  cmd[COMMAND_ROLL] = sp_euler.phi * angle2cmd;
