88 gvf_parametric_trajectory.p_len = 7 + gvf_parametric_p_len_wps;
89 gvf_parametric_p_len_wps = 0;
90
91 float f1, f2, f1d, f2d, f1dd, f2dd;
92 float wb = gvf_parametric_control.w * gvf_parametric_control.beta * gvf_parametric_control.s;
93
94 gvf_parametric_2d_trefoil_info(&f1, &f2, &f1d, &f2d, &f1dd, &f2dd, wb);
95 gvf_parametric_control_2D(gvf_parametric_2d_trefoil_par.kx, gvf_parametric_2d_trefoil_par.ky, f1, f2, f1d, f2d, f1dd,
96 f2dd);
97
98 return true;
99}
