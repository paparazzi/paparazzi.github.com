97// Steering control (GVF)
98void rover_guidance_steering_heading_ctrl(float omega) //GVF give us this omega
99{
100 // Speed is bounded to avoid GPS noise while driving at small velocity
101 float speed = BoundSpeed(stateGetHorizontalSpeedNorm_f());
102 // Compute steering angle
103 float delta = DegOfRad(atanf(omega*DRIVE_SHAFT_DISTANCE/speed));
104
105 guidance_control.cmd.delta = BoundDelta(delta);
106}

108// Speed control (feed feed forward + propotional + integral controler) (PID)
109void rover_guidance_steering_speed_ctrl(void)
110{
111 // Updating PID
112 guidance_control.speed_error = (guidance_control.cmd.speed - stateGetHorizontalSpeedNorm_f());
113 update_pid_f(&rover_pid, guidance_control.speed_error, time_step);
114
115 guidance_control.throttle = BoundThrottle(guidance_control.kf*guidance_control.cmd.speed + get_pid_f(&rover_pid));
116}

118void rover_guidance_steering_setpoints(struct EnuCoor_f pos_sp, float *heading_sp)
119{
120 // compute position error
121 struct FloatVect2 pos_2d;
122 VECT2_COPY(pos_2d, pos_sp);
123 struct FloatVect2 pos_err;
124 VECT2_DIFF(pos_err, pos_2d, *stateGetPositionEnu_f());
125 float pos_err_norm = float_vect2_norm(&pos_err);
126 BoundAbs(pos_err_norm, ROVER_GUIDANCE_MAX_POS_ERR);
127
128 // speed and heading update when far enough
129 if (pos_err_norm > ROVER_GUIDANCE_PROXIMITY_DISTANCE) {
130 // compute speed error
131 guidance_control.cmd.speed = guidance_control.pos_kp * pos_err_norm;
132 Bound(guidance_control.cmd.speed, 0.f, ROVER_GUIDANCE_MAX_SPEED);
133 // if not close to WP, compute desired heading
134 guidance_control.heading_sp = atan2f(pos_err.x, pos_err.y);
135 // update nav sp
136 *heading_sp = guidance_control.heading_sp;
137 // angular error
138 float heading_err = guidance_control.heading_sp - stateGetNedToBodyEulers_f()->psi;
139 NormRadAngle(heading_err);
140 // compute omega setpoint
141 guidance_control.omega_sp = guidance_control.heading_kp * heading_err;
142 }
143 else {
144 guidance_control.cmd.speed = 0.f;
145 guidance_control.heading_sp = *heading_sp;
146 guidance_control.omega_sp = 0.f;
147 }
148}
