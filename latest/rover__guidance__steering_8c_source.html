49void rover_guidance_steering_init(void)
50{
51 guidance_control.cmd.delta = 0.0;
52 guidance_control.cmd.speed = 0.0;
53 guidance_control.throttle = 0.0;
54
55 last_speed_cmd = 0.0;
56 last_ap_mode = AP_MODE_KILL;
57
58 guidance_control.speed_error = 0.0;
59 guidance_control.kf = SR_MEASURED_KF;
60 guidance_control.kp = 10;
61 guidance_control.ki = 100;
62
63 init_pid_f(&rover_pid, guidance_control.kp, 0.f, guidance_control.ki, MAX_PPRZ*0.2);
64
65 // Based on autopilot state machine frequency
66 time_step = 1.f/PERIODIC_FREQUENCY;
67}

70// Steering control (GVF)
71void rover_guidance_steering_heading_ctrl(float omega) //GVF give us this omega
72{
73 float delta = 0.0;
74
75 // Speed is bounded to avoid GPS noise while driving at small velocity
76 float speed = BoundSpeed(stateGetHorizontalSpeedNorm_f());
77
78 if (fabs(omega)>0.0) {
79 delta = DegOfRad(-atanf(omega*DRIVE_SHAFT_DISTANCE/speed));
80 }
81
82 guidance_control.cmd.delta = BoundDelta(delta);
83}

85// Speed control (feed feed forward + propotional + integral controler) (PID)
86void rover_guidance_steering_speed_ctrl(void)
87{
88 // - Looking for setting update
89 if (guidance_control.kp != rover_pid.g[0] || guidance_control.ki != rover_pid.g[2]) {
90 set_gains_pid_f(&rover_pid, guidance_control.kp, 0.f, guidance_control.ki);
91 }
92 if (guidance_control.cmd.speed != last_speed_cmd) {
93 last_speed_cmd = guidance_control.cmd.speed;
94 //reset_pid_f(&rover_pid);
95 }
96
97 // - Updating PID
98 guidance_control.speed_error = (guidance_control.cmd.speed - stateGetHorizontalSpeedNorm_f());
99 update_pid_f(&rover_pid, guidance_control.speed_error, time_step);
100
101 guidance_control.throttle = BoundThrottle(guidance_control.kf*guidance_control.cmd.speed + get_pid_f(&rover_pid));
102}
