108 static void gyro_cb(uint8_t __attribute__((unused)) sender_id,
109  uint32_t stamp, struct Int32Rates *gyro)
110 {
111  ahrs_mlkf_last_stamp = stamp;
112  struct FloatRates gyro_f;
113  RATES_FLOAT_OF_BFP(gyro_f, *gyro);
114 
115 #if USE_AUTO_AHRS_FREQ || !defined(AHRS_PROPAGATE_FREQUENCY)
116  PRINT_CONFIG_MSG("Calculating dt for AHRS_MLKF propagation.")
117  /* timestamp in usec when last callback was received */
118  static uint32_t last_stamp = 0;
119 
120  if (last_stamp > 0 && ahrs_mlkf.is_aligned) {
121  float dt = (float)(stamp - last_stamp) * 1e-6;
122  ahrs_mlkf_propagate(&gyro_f, dt);
123  set_body_state_from_quat();
124  }
125  last_stamp = stamp;
126 #else
127  PRINT_CONFIG_MSG("Using fixed AHRS_PROPAGATE_FREQUENCY for AHRS_MLKF propagation.")
128  PRINT_CONFIG_VAR(AHRS_PROPAGATE_FREQUENCY)
129  if (ahrs_mlkf.status == AHRS_MLKF_RUNNING) {
130  const float dt = 1. / (AHRS_PROPAGATE_FREQUENCY);
131  ahrs_mlkf_propagate(&gyro_f, dt);
132  set_body_state_from_quat();
133  }
134 #endif
135 }

161 static void aligner_cb(uint8_t __attribute__((unused)) sender_id,
162  uint32_t stamp __attribute__((unused)),
163  struct Int32Rates *lp_gyro, struct Int32Vect3 *lp_accel,
164  struct Int32Vect3 *lp_mag)
165 {
166  if (!ahrs_mlkf.is_aligned) {
167  /* convert to float */
168  struct FloatRates gyro_f;
169  RATES_FLOAT_OF_BFP(gyro_f, *lp_gyro);
170  struct FloatVect3 accel_f;
171  ACCELS_FLOAT_OF_BFP(accel_f, *lp_accel);
172  struct FloatVect3 mag_f;
173  MAGS_FLOAT_OF_BFP(mag_f, *lp_mag);
174  /* set initial body orientation in state interface if alignment was successful */
175  if (ahrs_mlkf_align(&gyro_f, &accel_f, &mag_f)) {
176  set_body_state_from_quat();
177  }
178  }
179 }
