113void tfmini_parse(uint8_t byte)
114{
115 switch (tfmini.parse_status) {
116 case TFMINI_INITIALIZE:
117 break;
118 case TFMINI_PARSE_HEAD:
119 if (byte == 0x59) {
120 tfmini.parse_crc = byte;
121 tfmini.parse_status++;
122 }
123 break;
124 case TFMINI_PARSE_HEAD2:
125 if (byte == 0x59) {
126 tfmini.parse_crc += byte;
127 tfmini.parse_status++;
128 } else {
129 tfmini.parse_status = TFMINI_PARSE_HEAD;
130 }
131 break;
132
133 case TFMINI_PARSE_DIST_L:
134 tfmini.raw_dist = byte;
135 tfmini.parse_crc += byte;
136 tfmini.parse_status++;
137 break;
138 case TFMINI_PARSE_DIST_H:
139 tfmini.raw_dist |= (byte << 8);
140 tfmini.parse_crc += byte;
141 tfmini.parse_status++;
142 break;
143
144 case TFMINI_PARSE_STRENGTH_L:
145 tfmini.raw_strength = byte;
146 tfmini.parse_crc += byte;
147 tfmini.parse_status++;
148 break;
149 case TFMINI_PARSE_STRENGTH_H:
150 tfmini.raw_strength |= (byte << 8);
151 tfmini.parse_crc += byte;
152 tfmini.parse_status++;
153 break;
154
155 case TFMINI_PARSE_MODE:
156 tfmini.raw_mode = byte;
157 tfmini.parse_crc += byte;
158 tfmini.parse_status++;
159 break;
160 case TFMINI_PARSE_BYTE7:
161 tfmini.parse_crc += byte;
162 tfmini.parse_status++;
163 break;
164
165 case TFMINI_PARSE_CHECKSUM:
166 // When the CRC matches
167 if (tfmini.parse_crc == byte) {
168 tfmini.distance = tfmini.raw_dist / 100.f;
169 tfmini.strength = tfmini.raw_strength;
170 tfmini.mode = tfmini.raw_mode;
171
172 // When the distance is valid
173 if (tfmini.distance != 0xFFFF) {
174 // compensate AGL measurement for body rotation
175 if (tfmini.compensate_rotation) {
176 // If it is a rover, we need to compensate the distance
177 if (tfmini.is_rover) {
178 float theta = stateGetNedToBodyEulers_f()->theta;
179 float ground_distance;
180 if (fabs(theta) < 0.01) {
181 ground_distance = 100; // If it is 0 it is straight
182 } else {
183 ground_distance = lidar_height / sinf(-theta) - lidar_offset;
184 }
185
186 if ((tfmini.distance >= ground_distance) && (ground_distance > 0)) {
187 tfmini.distance = 0;
188 }
189 }
190 // If it is not a rover (like a drone), we need to compensate the distance differently
191 else {
192 float phi = stateGetNedToBodyEulers_f()->phi;
193 float theta = stateGetNedToBodyEulers_f()->theta;
194 float gain = (float)fabs((double)(cosf(phi) * cosf(theta)));
195 tfmini.distance = tfmini.distance * gain;
196 }
197 }
198
199 // Send the AGL message
200 tfmini_send_abi();
201 }
202 }
203
204 // Start reading again
205 tfmini.parse_status = TFMINI_PARSE_HEAD;
206 break;
207
208 default:
209 // Error, return to start
210 tfmini.parse_status = TFMINI_PARSE_HEAD;
211 break;
212 }
213}
