108 void actuators_disco_commit(void)
109 {
110  // Handle PWM outputs
111  // already done in set function (FIXME ?)
112 
113  // Handle motor speed controller
114 
115  // Receive the status
116  actuators_disco.i2c_trans.buf[0] = ACTUATORS_DISCO_GET_OBS_DATA;
117  i2c_blocking_transceive(&i2c1, &actuators_disco.i2c_trans, actuators_disco.i2c_trans.slave_addr, 1, sizeof(obs_data));
118  // copy data from buffer
119 #pragma GCC diagnostic push
120 #pragma GCC diagnostic ignored "-Wcast-qual"
121  memcpy(&obs_data, (uint8_t*)actuators_disco.i2c_trans.buf, sizeof(obs_data));
122 #pragma GCC diagnostic pop
123 
124  // Update status
125  electrical.vsupply = (float)(be16toh(obs_data.batt_mv)) / 1000.f;
126  // extract 'rpm saturation bit'
127  actuators_disco.rpm_saturated = (obs_data.rpm & (1 << 7)) ? true : false;
128  // clear 'rpm saturation bit' and fix endianess
129  obs_data.rpm &= (uint16_t)(~(1 << 7));
130  obs_data.rpm = be16toh(obs_data.rpm);
131  if (obs_data.rpm == 0) {
132  actuators_disco.rpm_saturated = false;
133  }
134 
135  // When detected a suicide
136  uint8_t bldc_status = obs_data.status & 0x07;
137  if (obs_data.error == 2 && bldc_status != DISCO_BLDC_STATUS_STOPPED) {
138  autopilot_set_kill_throttle(true); //FIXME: make behaviour definable low flying should not stop it
139  }
140 
141  // Start the motors
142  if (bldc_status != DISCO_BLDC_STATUS_RUNNING &&
143  bldc_status != DISCO_BLDC_STATUS_RAMPUP && //FIXME also on rampdown?
144  actuators_disco.motor_rpm > DISCO_BLDC_START_MOTOR_THRESHOLD) {
145  // Reset the error
146  actuators_disco.i2c_trans.buf[0] = ACTUATORS_DISCO_CLEAR_ERROR;
147  i2c_blocking_transmit(&i2c1, &actuators_disco.i2c_trans, actuators_disco.i2c_trans.slave_addr, 1);
148 
149  // Start the motors
150  actuators_disco.i2c_trans.buf[0] = ACTUATORS_DISCO_START_PROP;
151  i2c_blocking_transmit(&i2c1, &actuators_disco.i2c_trans, actuators_disco.i2c_trans.slave_addr, 1);
152  }
153  // Stop the motors
154  else if ((bldc_status == DISCO_BLDC_STATUS_RUNNING || bldc_status == DISCO_BLDC_STATUS_RAMPUP) &&
155  actuators_disco.motor_rpm < DISCO_BLDC_START_MOTOR_THRESHOLD) {
156  actuators_disco.i2c_trans.buf[0] = ACTUATORS_DISCO_STOP_PROP;
157  i2c_blocking_transmit(&i2c1, &actuators_disco.i2c_trans, actuators_disco.i2c_trans.slave_addr, 1);
158  } else if (bldc_status == DISCO_BLDC_STATUS_RUNNING) {
159  // Send the commands
160  actuators_disco.i2c_trans.buf[0] = ACTUATORS_DISCO_SET_REF_SPEED;
161  actuators_disco.i2c_trans.buf[1] = actuators_disco.motor_rpm >> 8;
162  actuators_disco.i2c_trans.buf[2] = actuators_disco.motor_rpm & 0xFF;
163  actuators_disco.i2c_trans.buf[3] = 0x00; // enable security
164 #pragma GCC diagnostic push
165 #pragma GCC diagnostic ignored "-Wcast-qual"
166  actuators_disco.i2c_trans.buf[4] = actuators_disco_checksum((uint8_t *)actuators_disco.i2c_trans.buf, 3);
167 #pragma GCC diagnostic pop
168  i2c_blocking_transmit(&i2c1, &actuators_disco.i2c_trans, actuators_disco.i2c_trans.slave_addr, 11);
169  }
170 
171  // Send ABI message
172  struct act_feedback_t feedback = {0};
173 #ifdef SERVOS_DISCO_OFFSET
174  feedback.idx = SERVOS_DISCO_OFFSET;
175 #else
176  feedback.idx = SERVOS_DEFAULT_OFFSET;
177 #endif
178  feedback.rpm = actuators_disco.rpm_obs;
179  feedback.set.rpm = true;
180  AbiSendMsgACT_FEEDBACK(ACT_FEEDBACK_BOARD_ID, &feedback, 1);
181 }
