89void actuators_disco_init(void)
90{
91 /* Initialize the I2C connection */
92 actuators_disco.i2c_trans.slave_addr = ACTUATORS_DISCO_ADDR;
93 actuators_disco.i2c_trans.status = I2CTransDone;
94 actuators_disco.motor_rpm = 0;
95 int i = 0;
96 for (i = 0; i < ACTUATORS_DISCO_PWM_NB; i++) {
97 pwm_sysfs_init(&actuators_disco.pwm[i], "/sys/class/pwm", "export", "run", "duty_ns", "period_ns", disco_channels[i]);
98 }
99}

101void actuators_disco_set(uint8_t idx, uint16_t val)
102{
103 if (idx == ACTUATORS_DISCO_MOTOR_IDX) {
104 actuators_disco.motor_rpm = val;
105 } else if (idx > ACTUATORS_DISCO_PWM_NB) {
106 // wrong index, do nothing
107 } else {
108 // val is a PWM value in ms, convert to ns
109 pwm_sysfs_set_duty(&actuators_disco.pwm[idx-1], val * 1000);
110 }
111}

113void actuators_disco_commit(void)
114{
115 // Handle PWM outputs
116 // already done in set function (FIXME ?)
117
118 // Handle motor speed controller
119
120 // Receive the status
121 actuators_disco.i2c_trans.buf[0] = ACTUATORS_DISCO_GET_OBS_DATA;
122 i2c_blocking_transceive(&i2c1, &actuators_disco.i2c_trans, actuators_disco.i2c_trans.slave_addr, 1, sizeof(obs_data));
123 // copy data from buffer
124#pragma GCC diagnostic push
125#pragma GCC diagnostic ignored "-Wcast-qual"
126 memcpy(&obs_data, (uint8_t*)actuators_disco.i2c_trans.buf, sizeof(obs_data));
127#pragma GCC diagnostic pop
128
129 // Update status
130 electrical.vsupply = (float)(be16toh(obs_data.batt_mv)) / 1000.f;
131 // extract 'rpm saturation bit'
132 actuators_disco.rpm_saturated = (obs_data.rpm & (1 << 7)) ? true : false;
133 // clear 'rpm saturation bit' and fix endianess
134 obs_data.rpm &= (uint16_t)(~(1 << 7));
135 obs_data.rpm = be16toh(obs_data.rpm);
136 if (obs_data.rpm == 0) {
137 actuators_disco.rpm_saturated = false;
138 }
139
140 // When detected a suicide
141 uint8_t bldc_status = obs_data.status & 0x07;
142 if (obs_data.error == 2 && bldc_status != DISCO_BLDC_STATUS_STOPPED) {
143 autopilot_set_kill_throttle(true); //FIXME: make behaviour definable low flying should not stop it
144 }
145
146 // Start the motors
147 if (bldc_status != DISCO_BLDC_STATUS_RUNNING &&
148 bldc_status != DISCO_BLDC_STATUS_RAMPUP && //FIXME also on rampdown?
149 actuators_disco.motor_rpm > DISCO_BLDC_START_MOTOR_THRESHOLD) {
150 // Reset the error
151 actuators_disco.i2c_trans.buf[0] = ACTUATORS_DISCO_CLEAR_ERROR;
152 i2c_blocking_transmit(&i2c1, &actuators_disco.i2c_trans, actuators_disco.i2c_trans.slave_addr, 1);
153
154 // Start the motors
155 actuators_disco.i2c_trans.buf[0] = ACTUATORS_DISCO_START_PROP;
156 i2c_blocking_transmit(&i2c1, &actuators_disco.i2c_trans, actuators_disco.i2c_trans.slave_addr, 1);
157 }
158 // Stop the motors
159 else if ((bldc_status == DISCO_BLDC_STATUS_RUNNING || bldc_status == DISCO_BLDC_STATUS_RAMPUP) &&
160 actuators_disco.motor_rpm < DISCO_BLDC_START_MOTOR_THRESHOLD) {
161 actuators_disco.i2c_trans.buf[0] = ACTUATORS_DISCO_STOP_PROP;
162 i2c_blocking_transmit(&i2c1, &actuators_disco.i2c_trans, actuators_disco.i2c_trans.slave_addr, 1);
163 } else if (bldc_status == DISCO_BLDC_STATUS_RUNNING) {
164 // Send the commands
165 actuators_disco.i2c_trans.buf[0] = ACTUATORS_DISCO_SET_REF_SPEED;
166 actuators_disco.i2c_trans.buf[1] = actuators_disco.motor_rpm >> 8;
167 actuators_disco.i2c_trans.buf[2] = actuators_disco.motor_rpm & 0xFF;
168 actuators_disco.i2c_trans.buf[3] = 0x00; // enable security
169#pragma GCC diagnostic push
170#pragma GCC diagnostic ignored "-Wcast-qual"
171 actuators_disco.i2c_trans.buf[4] = actuators_disco_checksum((uint8_t *)actuators_disco.i2c_trans.buf, 3);
172#pragma GCC diagnostic pop
173 i2c_blocking_transmit(&i2c1, &actuators_disco.i2c_trans, actuators_disco.i2c_trans.slave_addr, 11);
174 }
175
176 // Send ABI message
177 struct act_feedback_t feedback = {0};
178 feedback.idx = SERVO_MOTOR_IDX;
179 feedback.rpm = actuators_disco.rpm_obs;
180 feedback.set.rpm = true;
181 AbiSendMsgACT_FEEDBACK(ACT_FEEDBACK_BOARD_ID, &feedback, 1);
182}

184static uint8_t actuators_disco_checksum(uint8_t *bytes, uint8_t size)
185{
186 uint8_t checksum = 0;
187 for (int i = 0; i < size; i++) {
188 checksum = checksum ^ bytes[i];
189 }
190
191 return checksum;
192}
