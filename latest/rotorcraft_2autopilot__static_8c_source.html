94 void autopilot_static_init(void)
95 {
96  /* Mode is finally set by autopilot_static_set_mode if MODE_STARTUP is not KILL.
97  * For autopilot_static_set_mode to do anything, the requested mode needs to differ
98  * from previous mode, so we set it to a safe KILL first.
99  */
100  autopilot.mode = AP_MODE_KILL;
101 
102  /* set startup mode, propagates through to guidance h/v */
103  autopilot_static_set_mode(MODE_STARTUP);
104 
105  /* init arming */
106  autopilot_arming_init();
107 }

110 #define NAV_PRESCALER (PERIODIC_FREQUENCY / NAVIGATION_FREQUENCY)
111 void autopilot_static_periodic(void)
112 {
113 
114  RunOnceEvery(NAV_PRESCALER, compute_dist2_to_home());
115 
116  if (autopilot_in_flight() && autopilot.mode == AP_MODE_NAV) {
117  if (nav.too_far_from_home || datalink_lost() || higher_than_max_altitude()) {
118  if (nav.dist2_to_home > nav.failsafe_mode_dist2) {
119  autopilot_static_set_mode(FAILSAFE_MODE_TOO_FAR_FROM_HOME);
120  } else {
121  autopilot_static_set_mode(AP_MODE_HOME);
122  }
123  }
124  }
125 
126  if (autopilot.mode == AP_MODE_HOME) {
127  RunOnceEvery(NAV_PRESCALER, nav_home());
128  } else {
129  // otherwise always call nav_periodic_task so that carrot is always updated in GCS for other modes
130  RunOnceEvery(NAV_PRESCALER, nav_periodic_task());
131  }
132 
133 
134  /* If in FAILSAFE mode and either already not in_flight anymore
135  * or just "detected" ground, go to KILL mode.
136  */
137  if (autopilot.mode == AP_MODE_FAILSAFE) {
138  if (!autopilot_in_flight()) {
139  autopilot_static_set_mode(AP_MODE_KILL);
140  }
141 
142 #if FAILSAFE_GROUND_DETECT
143  INFO("Using FAILSAFE_GROUND_DETECT: KILL")
144  if (autopilot.ground_detected) {
145  autopilot_static_set_mode(AP_MODE_KILL);
146  }
147 #endif
148  }
149 
150  /* Reset ground detection _after_ running flight plan
151  */
152  if (!autopilot_in_flight()) {
153  autopilot.ground_detected = false;
154  autopilot.detect_ground_once = false;
155  }
156 
157  /* Set fixed "failsafe" commands from airframe file if in KILL mode.
158  * If in FAILSAFE mode, run normal loops with failsafe attitude and
159  * downwards velocity setpoints.
160  */
161  if (autopilot.mode == AP_MODE_KILL) {
162  SetCommands(commands_failsafe);
163  } else {
164  guidance_v_run(autopilot_in_flight());
165  guidance_h_run(autopilot_in_flight());
166  SetRotorcraftCommands(stabilization_cmd, autopilot.in_flight, autopilot.motors_on);
167  }
168  autopilot.throttle = commands[COMMAND_THRUST];
169 
170 }

192 void autopilot_static_set_mode(uint8_t new_autopilot_mode)
193 {
194  if (new_autopilot_mode != autopilot.mode) {
195  /* horizontal mode */
196  switch (new_autopilot_mode) {
197  case AP_MODE_FAILSAFE:
198 #ifndef KILL_AS_FAILSAFE
199  stabilization_attitude_set_failsafe_setpoint();
200  guidance_h_mode_changed(GUIDANCE_H_MODE_ATTITUDE);
201  break;
202 #endif
203  case AP_MODE_KILL:
204  autopilot_set_in_flight(false);
205  guidance_h_mode_changed(GUIDANCE_H_MODE_KILL);
206  break;
207  case AP_MODE_RC_DIRECT:
208  guidance_h_mode_changed(GUIDANCE_H_MODE_RC_DIRECT);
209  break;
210  case AP_MODE_RATE_RC_CLIMB:
211  case AP_MODE_RATE_DIRECT:
212  case AP_MODE_RATE_Z_HOLD:
213 #if USE_STABILIZATION_RATE
214  guidance_h_mode_changed(GUIDANCE_H_MODE_RATE);
215 #else
216  return;
217 #endif
218  break;
219  case AP_MODE_ATTITUDE_RC_CLIMB:
220  case AP_MODE_ATTITUDE_DIRECT:
221  case AP_MODE_ATTITUDE_CLIMB:
222  case AP_MODE_ATTITUDE_Z_HOLD:
223  guidance_h_mode_changed(GUIDANCE_H_MODE_ATTITUDE);
224  break;
225  case AP_MODE_FORWARD:
226  guidance_h_mode_changed(GUIDANCE_H_MODE_FORWARD);
227  break;
228  case AP_MODE_CARE_FREE_DIRECT:
229  guidance_h_mode_changed(GUIDANCE_H_MODE_CARE_FREE);
230  break;
231  case AP_MODE_HOVER_DIRECT:
232  case AP_MODE_HOVER_CLIMB:
233  case AP_MODE_HOVER_Z_HOLD:
234  guidance_h_mode_changed(GUIDANCE_H_MODE_HOVER);
235  break;
236  case AP_MODE_HOME:
237  case AP_MODE_NAV:
238  guidance_h_mode_changed(GUIDANCE_H_MODE_NAV);
239  break;
240  case AP_MODE_MODULE:
241 #ifdef GUIDANCE_H_MODE_MODULE_SETTING
242  guidance_h_mode_changed(GUIDANCE_H_MODE_MODULE_SETTING);
243 #endif
244  break;
245  case AP_MODE_FLIP:
246  guidance_h_mode_changed(GUIDANCE_H_MODE_FLIP);
247  break;
248  case AP_MODE_GUIDED:
249  guidance_h_mode_changed(GUIDANCE_H_MODE_GUIDED);
250  break;
251  default:
252  break;
253  }
254  /* vertical mode */
255  switch (new_autopilot_mode) {
256  case AP_MODE_FAILSAFE:
257 #ifndef KILL_AS_FAILSAFE
258  guidance_v_mode_changed(GUIDANCE_V_MODE_CLIMB);
259  guidance_v_set_vz(FAILSAFE_DESCENT_SPEED);
260  break;
261 #endif
262  case AP_MODE_KILL:
263  autopilot_set_motors_on(FALSE);
264  stabilization_cmd[COMMAND_THRUST] = 0;
265  guidance_v_mode_changed(GUIDANCE_V_MODE_KILL);
266  break;
267  case AP_MODE_RC_DIRECT:
268  case AP_MODE_RATE_DIRECT:
269  case AP_MODE_ATTITUDE_DIRECT:
270  case AP_MODE_HOVER_DIRECT:
271  case AP_MODE_CARE_FREE_DIRECT:
272  case AP_MODE_FORWARD:
273  guidance_v_mode_changed(GUIDANCE_V_MODE_RC_DIRECT);
274  break;
275  case AP_MODE_RATE_RC_CLIMB:
276  case AP_MODE_ATTITUDE_RC_CLIMB:
277  guidance_v_mode_changed(GUIDANCE_V_MODE_RC_CLIMB);
278  break;
279  case AP_MODE_ATTITUDE_CLIMB:
280  case AP_MODE_HOVER_CLIMB:
281  guidance_v_mode_changed(GUIDANCE_V_MODE_CLIMB);
282  break;
283  case AP_MODE_RATE_Z_HOLD:
284  case AP_MODE_ATTITUDE_Z_HOLD:
285  case AP_MODE_HOVER_Z_HOLD:
286  guidance_v_mode_changed(GUIDANCE_V_MODE_HOVER);
287  break;
288  case AP_MODE_HOME:
289  case AP_MODE_NAV:
290  guidance_v_mode_changed(GUIDANCE_V_MODE_NAV);
291  break;
292  case AP_MODE_MODULE:
293 #ifdef GUIDANCE_V_MODE_MODULE_SETTING
294  guidance_v_mode_changed(GUIDANCE_V_MODE_MODULE_SETTING);
295 #endif
296  break;
297  case AP_MODE_FLIP:
298  guidance_v_mode_changed(GUIDANCE_V_MODE_FLIP);
299  break;
300  case AP_MODE_GUIDED:
301  guidance_v_mode_changed(GUIDANCE_V_MODE_GUIDED);
302  break;
303  default:
304  break;
305  }
306  //if switching to rate mode but rate mode is not defined, the function returned
307  autopilot.mode = new_autopilot_mode;
308  }

322 void autopilot_static_on_rc_frame(void)
323 {
324 
325  if (kill_switch_is_on()) {
326  autopilot_static_set_mode(AP_MODE_KILL);
327  } else {
328 #ifdef RADIO_AUTO_MODE
329  INFO("Using RADIO_AUTO_MODE to switch between AUTO1 and AUTO2.")
330  uint8_t new_autopilot_mode = ap_mode_of_two_switches();
331 #else
332 #ifdef RADIO_MODE_2x3
333  uint8_t new_autopilot_mode = ap_mode_of_3x2way_switch();
334 #else
335  uint8_t new_autopilot_mode = ap_mode_of_3way_switch();
336 #endif
337 #endif
338 
339  /* don't enter NAV mode if GPS is lost (this also prevents mode oscillations) */
340  if (!(new_autopilot_mode == AP_MODE_NAV && GpsIsLost())) {
341  /* always allow to switch to manual */
342  if (new_autopilot_mode == MODE_MANUAL) {
343  autopilot_static_set_mode(new_autopilot_mode);
344  }
345  /* if in HOME or FAILSAFE mode, don't allow switching to non-manual modes */
346  else if (((autopilot.mode != AP_MODE_HOME) && (autopilot.mode != AP_MODE_FAILSAFE))
347 
348 #if UNLOCKED_HOME_MODE
349  /* Allowed to leave home mode when UNLOCKED_HOME_MODE */
350  || !nav.too_far_from_home
351 #endif
352  ) {
353  autopilot_static_set_mode(new_autopilot_mode);
354  }
355  }
356  }
357 
358  /* an arming sequence is used to start/stop motors.
359  */
360  autopilot_arming_check_motors_on();
361  autopilot.kill_throttle = ! autopilot.motors_on;
362 
363  /* if not in FAILSAFE or HOME mode, read RC and set commands accordingly */
364  if (autopilot.mode != AP_MODE_FAILSAFE && autopilot.mode != AP_MODE_HOME) {
365 
366  /* if there are some commands that should always be set from RC, do it */
367 #ifdef SetAutoCommandsFromRC
368  SetAutoCommandsFromRC(commands, radio_control.values);
369 #endif
370 
371  /* if not in NAV_MODE set commands from the rc */
372 #ifdef SetCommandsFromRC
373  if (autopilot.mode != AP_MODE_NAV) {
374  SetCommandsFromRC(commands, radio_control.values);
375  }
376 #endif
377 
378  guidance_v_read_rc();
379  guidance_h_read_rc(autopilot_in_flight());
380  }
381

389 void autopilot_failsafe_checks(void)
390 {
391  if (radio_control.status == RC_REALLY_LOST &&
392  autopilot_get_mode() != AP_MODE_KILL &&
393  autopilot_get_mode() != AP_MODE_HOME &&
394  autopilot_get_mode() != AP_MODE_FAILSAFE &&
395  autopilot_get_mode() != AP_MODE_NAV &&
396  autopilot_get_mode() != AP_MODE_MODULE &&
397  autopilot_get_mode() != AP_MODE_FLIP &&
398  autopilot_get_mode() != AP_MODE_GUIDED) {
399  autopilot_set_mode(RC_LOST_MODE);
400  }
401 
402 #if FAILSAFE_ON_BAT_CRITICAL
403  if (autopilot_get_mode() != AP_MODE_KILL &&
404  electrical.bat_critical) {
405  autopilot_set_mode(AP_MODE_FAILSAFE);
406  }
407 #endif
408 
409 #if USE_GPS
410  if (autopilot_get_mode() == AP_MODE_NAV &&
411  autopilot_get_motors_on() &&
412 #if NO_GPS_LOST_WITH_RC_VALID
413  radio_control.status != RC_OK &&
414 #endif
415 #ifdef NO_GPS_LOST_WITH_DATALINK_TIME
416  datalink_time > NO_GPS_LOST_WITH_DATALINK_TIME &&
417 #endif
418  GpsIsLost()) {
419  autopilot_set_mode(AP_MODE_FAILSAFE);
420  }
421 
422  if (autopilot_get_mode() == AP_MODE_HOME &&
423  autopilot_get_motors_on() && GpsIsLost()) {
424  autopilot_set_mode(AP_MODE_FAILSAFE);
425  }
426 #endif
427 
428 }
