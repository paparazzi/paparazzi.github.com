93void autopilot_static_init(void)
94{
95 /* Mode is finally set by autopilot_static_set_mode if MODE_STARTUP is not KILL.
96 * For autopilot_static_set_mode to do anything, the requested mode needs to differ
97 * from previous mode, so we set it to a safe KILL first.
98 */
99 autopilot.mode = AP_MODE_KILL;
100
101 /* set startup mode, propagates through to guidance h/v */
102 autopilot_static_set_mode(MODE_STARTUP);
103
104 /* init arming */
105 autopilot_arming_init();
106}

109#define NAV_PRESCALER (PERIODIC_FREQUENCY / NAVIGATION_FREQUENCY)
110void autopilot_static_periodic(void)
111{
112
113 RunOnceEvery(NAV_PRESCALER, compute_dist2_to_home());
114
115 if (autopilot_in_flight() && autopilot.mode == AP_MODE_NAV) {
116 if (nav.too_far_from_home || datalink_lost() || higher_than_max_altitude()) {
117 if (nav.dist2_to_home > nav.failsafe_mode_dist2) {
118 autopilot_static_set_mode(FAILSAFE_MODE_TOO_FAR_FROM_HOME);
119 } else {
120 autopilot_static_set_mode(AP_MODE_HOME);
121 }
122 }
123 }
124
125 if (autopilot.mode == AP_MODE_HOME) {
126 RunOnceEvery(NAV_PRESCALER, nav_home());
127 } else {
128 // otherwise always call nav_periodic_task so that carrot is always updated in GCS for other modes
129 RunOnceEvery(NAV_PRESCALER, nav_periodic_task());
130 }
131
132
133 /* If in FAILSAFE mode and either already not in_flight anymore
134 * or just "detected" ground, go to KILL mode.
135 */
136 if (autopilot.mode == AP_MODE_FAILSAFE) {
137 if (!autopilot_in_flight()) {
138 autopilot_static_set_mode(AP_MODE_KILL);
139 }
140
141#if FAILSAFE_GROUND_DETECT
142 INFO("Using FAILSAFE_GROUND_DETECT: KILL")
143 if (autopilot.ground_detected) {
144 autopilot_static_set_mode(AP_MODE_KILL);
145 }
146#endif
147 }
148
149 /* Reset ground detection _after_ running flight plan
150 */
151 if (!autopilot_in_flight()) {
152 autopilot.ground_detected = false;
153 autopilot.detect_ground_once = false;
154 }
155
156 /* Set fixed "failsafe" commands from airframe file if in KILL mode.
157 * If in FAILSAFE mode, run normal loops with failsafe attitude and
158 * downwards velocity setpoints.
159 */
160 struct StabilizationSetpoint stab_sp;
161 struct ThrustSetpoint thrust_sp;
162 switch (autopilot.mode) {
163 case AP_MODE_FAILSAFE:
164#ifndef KILL_AS_FAILSAFE
165#ifdef FAILSAFE_THROTTLE
166 thrust_sp = th_sp_from_thrust_i(FAILSAFE_THROTTLE, THRUST_AXIS_Z);
167#else
168 thrust_sp = guidance_v_run(autopilot_in_flight());
169#endif
170 stab_sp = stabilization_get_failsafe_sp();
171 stabilization_run(autopilot_in_flight(), &stab_sp, &thrust_sp, stabilization.cmd);
172 SetRotorcraftCommands(stabilization.cmd, autopilot.in_flight, autopilot.motors_on);
173 break;
174#endif
175 case AP_MODE_KILL:
176 SetCommands(commands_failsafe);
177 break;
178 default:
179 thrust_sp = guidance_v_run(autopilot_in_flight());
180 if (guidance_h.mode != GUIDANCE_H_MODE_NONE) {
181 stab_sp = guidance_h_run(autopilot_in_flight());
182 } else {
183 stab_sp = stabilization.rc_sp;
184 }
185 stabilization_run(autopilot_in_flight(), &stab_sp, &thrust_sp, stabilization.cmd);
186 // TODO maybe add RC limiter here as an option ?
187 SetRotorcraftCommands(stabilization.cmd, autopilot.in_flight, autopilot.motors_on);
188 break;
189 }
190#ifdef COMMAND_THRUST
191 autopilot.throttle = commands[COMMAND_THRUST];
192#endif
193
194}

211 }

216void autopilot_static_set_mode(uint8_t new_autopilot_mode)
217{
218 if (new_autopilot_mode != autopilot.mode) {
219 /* horizontal mode */
220 switch (new_autopilot_mode) {
221 case AP_MODE_FAILSAFE:
222#ifndef KILL_AS_FAILSAFE
223 guidance_h_mode_changed(GUIDANCE_H_MODE_NONE);
224 stabilization_mode_changed(STABILIZATION_MODE_ATTITUDE,
225 STABILIZATION_ATT_SUBMODE_HEADING);
226 break;
227#endif
228 case AP_MODE_KILL:
229 autopilot_set_in_flight(false);
230 guidance_h_mode_changed(GUIDANCE_H_MODE_NONE);
231 stabilization_mode_changed(STABILIZATION_MODE_NONE, 0);
232 break;
233 case AP_MODE_RC_DIRECT:
234 guidance_h_mode_changed(GUIDANCE_H_MODE_NONE);
235 stabilization_mode_changed(STABILIZATION_MODE_DIRECT, 0);
236 break;
237 case AP_MODE_RATE_RC_CLIMB:
238 case AP_MODE_RATE_DIRECT:
239 case AP_MODE_RATE_Z_HOLD:
240#if USE_STABILIZATION_RATE
241 guidance_h_mode_changed(GUIDANCE_H_MODE_NONE);
242 stabilization_mode_changed(STABILIZATION_MODE_RATE, 0);
243#else
244 return;
245#endif
246 break;
247 case AP_MODE_ATTITUDE_RC_CLIMB:
248 case AP_MODE_ATTITUDE_DIRECT:
249 case AP_MODE_ATTITUDE_CLIMB:
250 case AP_MODE_ATTITUDE_Z_HOLD:
251 guidance_h_mode_changed(GUIDANCE_H_MODE_NONE);
252 stabilization_mode_changed(STABILIZATION_MODE_ATTITUDE,
253 STABILIZATION_ATT_SUBMODE_HEADING);
254 break;
255 case AP_MODE_FORWARD:
256 guidance_h_mode_changed(GUIDANCE_H_MODE_NONE);
257 stabilization_mode_changed(STABILIZATION_MODE_ATTITUDE,
258 STABILIZATION_ATT_SUBMODE_FORWARD);
259 break;
260 case AP_MODE_CARE_FREE_DIRECT:
261 guidance_h_mode_changed(GUIDANCE_H_MODE_NONE);
262 stabilization_mode_changed(STABILIZATION_MODE_ATTITUDE,
263 STABILIZATION_ATT_SUBMODE_CARE_FREE);
264 break;
265 case AP_MODE_HOVER_DIRECT:
266 case AP_MODE_HOVER_CLIMB:
267 case AP_MODE_HOVER_Z_HOLD:
268 guidance_h_mode_changed(GUIDANCE_H_MODE_HOVER);
269 stabilization_mode_changed(STABILIZATION_MODE_ATTITUDE,
270 STABILIZATION_ATT_SUBMODE_HEADING);
271 break;
272 case AP_MODE_HOME:
273 case AP_MODE_NAV:
274 guidance_h_mode_changed(GUIDANCE_H_MODE_NAV);
275 stabilization_mode_changed(STABILIZATION_MODE_ATTITUDE,
276 STABILIZATION_ATT_SUBMODE_HEADING); // TODO check
277 break;
278 case AP_MODE_GUIDED:
279 guidance_h_mode_changed(GUIDANCE_H_MODE_GUIDED);
280 stabilization_mode_changed(STABILIZATION_MODE_ATTITUDE,
281 STABILIZATION_ATT_SUBMODE_HEADING);
282 break;
283 default:
284 break;
285 }
286 /* vertical mode */
287 switch (new_autopilot_mode) {
288 case AP_MODE_FAILSAFE:
289#ifndef KILL_AS_FAILSAFE
290 guidance_v_mode_changed(GUIDANCE_V_MODE_CLIMB);
291 guidance_v_set_vz(FAILSAFE_DESCENT_SPEED);
292 break;
293#endif
294 case AP_MODE_KILL:
295 autopilot_set_motors_on(false);
296#ifdef COMMAND_THRUST
297 stabilization.cmd[COMMAND_THRUST] = 0; // FIXME maybe not needed ?
298#endif
299 guidance_v_mode_changed(GUIDANCE_V_MODE_KILL);
300 break;
301 case AP_MODE_RC_DIRECT:
302 case AP_MODE_RATE_DIRECT:
303 case AP_MODE_ATTITUDE_DIRECT:
304 case AP_MODE_HOVER_DIRECT:
305 case AP_MODE_CARE_FREE_DIRECT:
306 case AP_MODE_FORWARD:
307 guidance_v_mode_changed(GUIDANCE_V_MODE_RC_DIRECT);
308 break;
309 case AP_MODE_RATE_RC_CLIMB:
310 case AP_MODE_ATTITUDE_RC_CLIMB:
311 guidance_v_mode_changed(GUIDANCE_V_MODE_RC_CLIMB);
312 break;
313 case AP_MODE_ATTITUDE_CLIMB:
314 case AP_MODE_HOVER_CLIMB:
315 guidance_v_mode_changed(GUIDANCE_V_MODE_CLIMB);
316 break;
317 case AP_MODE_RATE_Z_HOLD:
318 case AP_MODE_ATTITUDE_Z_HOLD:
319 case AP_MODE_HOVER_Z_HOLD:
320 guidance_v_mode_changed(GUIDANCE_V_MODE_HOVER);
321 break;
322 case AP_MODE_HOME:
323 case AP_MODE_NAV:
324 guidance_v_mode_changed(GUIDANCE_V_MODE_NAV);
325 break;
326 case AP_MODE_GUIDED:
327 guidance_v_mode_changed(GUIDANCE_V_MODE_GUIDED);
328 break;
329 default:
330 break;
331 }
332 //if switching to rate mode but rate mode is not defined, the function returned
333 autopilot.mode = new_autopilot_mode;
334 }
335}

348void autopilot_static_on_rc_frame(void)
349{
350
351 if (kill_switch_is_on()) {
352 autopilot_static_set_mode(AP_MODE_KILL);
353 } else {
354#ifdef RADIO_AUTO_MODE
355 INFO("Using RADIO_AUTO_MODE to switch between AUTO1 and AUTO2.")
356 uint8_t new_autopilot_mode = ap_mode_of_two_switches();
357#else
358#ifdef RADIO_MODE_2x3
359 uint8_t new_autopilot_mode = ap_mode_of_3x2way_switch();
360#else
361 uint8_t new_autopilot_mode = ap_mode_of_3way_switch();
362#endif
363#endif
364
365 /* don't enter NAV mode if GPS is lost (this also prevents mode oscillations) */
366 if (!(new_autopilot_mode == AP_MODE_NAV && GpsIsLost())) {
367 /* always allow to switch to manual */
368 if (new_autopilot_mode == MODE_MANUAL) {
369 autopilot_static_set_mode(new_autopilot_mode);
370 }
371 /* if in HOME or FAILSAFE mode, don't allow switching to non-manual modes */
372 else if (((autopilot.mode != AP_MODE_HOME) && (autopilot.mode != AP_MODE_FAILSAFE))
373
374#if UNLOCKED_HOME_MODE
375 /* Allowed to leave home mode when UNLOCKED_HOME_MODE */
376 || !nav.too_far_from_home
377#endif
378 ) {
379 autopilot_static_set_mode(new_autopilot_mode);
380 }
381 }
382 }
383
384 /* an arming sequence is used to start/stop motors.
385 */
386 autopilot_arming_check_motors_on();
387 autopilot.kill_throttle = ! autopilot.motors_on;
388
389 /* if not in FAILSAFE or HOME mode, read RC and set commands accordingly */
390 if (autopilot.mode != AP_MODE_FAILSAFE && autopilot.mode != AP_MODE_HOME) {
391
392 /* if there are some commands that should always be set from RC, do it */
393#ifdef SetAutoCommandsFromRC
394 SetAutoCommandsFromRC(commands, radio_control.values);
395#endif
396
397 /* if not in NAV_MODE set commands from the rc */
398#ifdef SetCommandsFromRC
399 if (autopilot.mode != AP_MODE_NAV) {
400 SetCommandsFromRC(commands, radio_control.values);
401 }
402#endif
403 }
404
405}

412void autopilot_failsafe_checks(void)
413{
414 if (radio_control.status == RC_REALLY_LOST &&
415 autopilot_get_mode() != AP_MODE_KILL &&
416 autopilot_get_mode() != AP_MODE_HOME &&
417 autopilot_get_mode() != AP_MODE_FAILSAFE &&
418 autopilot_get_mode() != AP_MODE_NAV &&
419 autopilot_get_mode() != AP_MODE_GUIDED) {
420 autopilot_set_mode(RC_LOST_MODE);
421 }
422
423#if FAILSAFE_ON_BAT_CRITICAL
424 if (autopilot_get_mode() != AP_MODE_KILL &&
425 electrical.bat_critical) {
426 autopilot_set_mode(AP_MODE_FAILSAFE);
427 }
428#endif
429
430#if USE_GPS
431 if (autopilot_get_mode() == AP_MODE_NAV &&
432 autopilot_get_motors_on() &&
433#if NO_GPS_LOST_WITH_RC_VALID
434 radio_control.status != RC_OK &&
435#endif
436#ifdef NO_GPS_LOST_WITH_DATALINK_TIME
437 datalink_time > NO_GPS_LOST_WITH_DATALINK_TIME &&
438#endif
439 GpsIsLost()) {
440 autopilot_set_mode(AP_MODE_FAILSAFE);
441 }
442
443 if (autopilot_get_mode() == AP_MODE_HOME &&
444 autopilot_get_motors_on() && GpsIsLost()) {
445 autopilot_set_mode(AP_MODE_FAILSAFE);
446 }
447#endif
448
449}
