79bool nav_gvf_parametric_3D_ellipse_XYZ(float xo, float yo, float r, float zl, float zh, float alpha)
80{
81 horizontal_mode = HORIZONTAL_MODE_CIRCLE; // Circle for the 2D GCS
82
83 // Safety first! If the asked altitude is low
84 if (zl > zh) {
85 zl = zh;
86 }
87 if (zl < 1 || zh < 1) {
88 zl = 10;
89 zh = 10;
90 }
91 if (r < 1) {
92 r = 60;
93 }
94
95 gvf_parametric_trajectory.type = ELLIPSE_3D;
96 gvf_parametric_trajectory.p_parametric[0] = xo;
97 gvf_parametric_trajectory.p_parametric[1] = yo;
98 gvf_parametric_trajectory.p_parametric[2] = r;
99 gvf_parametric_trajectory.p_parametric[3] = zl;
100 gvf_parametric_trajectory.p_parametric[4] = zh;
101 gvf_parametric_trajectory.p_parametric[5] = alpha;
102 gvf_parametric_trajectory.p_len = 6 + gvf_parametric_p_len_wps;
103 gvf_parametric_p_len_wps = 0;
104
105 float f1, f2, f3, f1d, f2d, f3d, f1dd, f2dd, f3dd;
106 float wb = gvf_parametric_control.w * gvf_parametric_control.beta * gvf_parametric_control.s;
107
108 gvf_parametric_3d_ellipse_info(&f1, &f2, &f3, &f1d, &f2d, &f3d, &f1dd, &f2dd, &f3dd, wb);
109 gvf_parametric_control_3D(gvf_parametric_3d_ellipse_par.kx, gvf_parametric_3d_ellipse_par.ky,
110 gvf_parametric_3d_ellipse_par.kz, f1, f2, f3, f1d, f2d, f3d, f1dd, f2dd, f3dd);
111
112 return true;
113}
