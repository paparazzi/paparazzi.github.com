32#define DELAYED_FIRST_ORDER_LOWPASS_FILTER_BUFFER_SIZE 20
33#define DELAYED_FIRST_ORDER_LOWPASS_FILTER_FILTER_ALPHA_SHIFT 14

35struct delayed_first_order_lowpass_filter_t {
36 uint16_t sample_frequency;
37 uint32_t omega;
38 uint8_t delay;
39 int32_t alpha;
40 uint16_t max_inc;
41 int32_t buffer[DELAYED_FIRST_ORDER_LOWPASS_FILTER_BUFFER_SIZE];
42 uint8_t idx;
43};

54static inline int32_t delayed_first_order_lowpass_propagate(struct delayed_first_order_lowpass_filter_t *f,
55 int32_t input)
56{
57 int32_t prev = f->buffer[f->idx];
58 uint8_t next_idx = ++f->idx % DELAYED_FIRST_ORDER_LOWPASS_FILTER_BUFFER_SIZE;
59 f->idx = next_idx;
60 f->buffer[next_idx] = (f->alpha * prev + ((1 << DELAYED_FIRST_ORDER_LOWPASS_FILTER_FILTER_ALPHA_SHIFT) - f->alpha) *
61 input) >> DELAYED_FIRST_ORDER_LOWPASS_FILTER_FILTER_ALPHA_SHIFT;
62
63 /* Check if new value exceeds maximum increase */
64 if ((f->buffer[next_idx] - prev) > f->max_inc) {
65 f->buffer[next_idx] = prev + f->max_inc;
66 }
67 /* Also negative case */
68 if ((f->buffer[next_idx] - prev) < -f->max_inc) {
69 f->buffer[next_idx] = prev - f->max_inc;
70 }
71
72 uint8_t req_idx = (f->idx - f->delay + DELAYED_FIRST_ORDER_LOWPASS_FILTER_BUFFER_SIZE) %
73 DELAYED_FIRST_ORDER_LOWPASS_FILTER_BUFFER_SIZE;
74 return f->buffer[req_idx];
75}

84static inline void delayed_first_order_lowpass_set_omega(struct delayed_first_order_lowpass_filter_t *f, uint32_t omega)
85{
86 /* alpha = 1 / ( 1 + omega_c * Ts) */
87 f->omega = omega;
88 f->alpha = (f->sample_frequency << DELAYED_FIRST_ORDER_LOWPASS_FILTER_FILTER_ALPHA_SHIFT) /
89 (f->sample_frequency + f->omega);
90}

101static inline void delayed_first_order_lowpass_set_delay(struct delayed_first_order_lowpass_filter_t *f, uint8_t delay)
102{
103 /* Delay cannot be more than buffer size minus one */
104 if (delay >= DELAYED_FIRST_ORDER_LOWPASS_FILTER_BUFFER_SIZE) {
105 f->delay = DELAYED_FIRST_ORDER_LOWPASS_FILTER_BUFFER_SIZE - 1;
106 } else {
107 f->delay = delay;
108 }
109}

120static inline void delayed_first_order_lowpass_initialize(struct delayed_first_order_lowpass_filter_t *f,
121 uint32_t omega, uint8_t delay, uint16_t max_inc, uint16_t sample_frequency)
122{
123 /* Set sample frequency */
124 f->sample_frequency = sample_frequency;
125 /* Set delay */
126 delayed_first_order_lowpass_set_delay(f, delay);
127
128 /* Set omega and calculate alpha */
129 delayed_first_order_lowpass_set_omega(f, omega);
130
131 /* Set maximum increase per cycle */
132 f->max_inc = max_inc;
133
134 /* Clear the buffer */
135 f->idx = 0;
136 for (uint8_t i = 0; i < DELAYED_FIRST_ORDER_LOWPASS_FILTER_BUFFER_SIZE; i++) {
137 f->buffer[i] = 0;
138 }
139}
