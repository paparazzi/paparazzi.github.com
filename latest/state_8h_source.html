457 static inline void stateSetLocalOrigin_i(struct LtpDef_i *ltp_def)
458 {
459  state.ned_origin_i = *ltp_def;
460  /* convert to float */
461  ECEF_FLOAT_OF_BFP(state.ned_origin_f.ecef, state.ned_origin_i.ecef);
462  LLA_FLOAT_OF_BFP(state.ned_origin_f.lla, state.ned_origin_i.lla);
463  HIGH_RES_RMAT_FLOAT_OF_BFP(state.ned_origin_f.ltp_of_ecef, state.ned_origin_i.ltp_of_ecef);
464  state.ned_origin_f.hmsl = M_OF_MM(state.ned_origin_i.hmsl);
465 
466  /* clear bits for all local frame representations */
467  state.pos_status &= ~(POS_LOCAL_COORD);
468  state.speed_status &= ~(SPEED_LOCAL_COORD);
469  ClearBit(state.accel_status, ACCEL_NED_I);
470  ClearBit(state.accel_status, ACCEL_NED_F);
471 
472  state.ned_initialized_i = true;
473  state.ned_initialized_f = true;
474 }
