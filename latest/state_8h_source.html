519static inline void stateSetLocalOrigin_i(uint16_t id, struct LtpDef_i *ltp_def)
520{
521 STATE_CHECK_INPUT_AND_RETURN(state.origin_input_filter, id);
522
523 state.ned_origin_i = *ltp_def;
524 /* convert to float */
525 ECEF_FLOAT_OF_BFP(state.ned_origin_f.ecef, state.ned_origin_i.ecef);
526 LLA_FLOAT_OF_BFP(state.ned_origin_f.lla, state.ned_origin_i.lla);
527 HIGH_RES_RMAT_FLOAT_OF_BFP(state.ned_origin_f.ltp_of_ecef, state.ned_origin_i.ltp_of_ecef);
528 state.ned_origin_f.hmsl = M_OF_MM(state.ned_origin_i.hmsl);
529
530 /* clear bits for all local frame representations */
531 state.pos_status &= ~(POS_LOCAL_COORD);
532 state.speed_status &= ~(SPEED_LOCAL_COORD);
533 ClearBit(state.accel_status, ACCEL_NED_I);
534 ClearBit(state.accel_status, ACCEL_NED_F);
535
536 state.ned_initialized_i = true;
537 state.ned_initialized_f = true;
538}
