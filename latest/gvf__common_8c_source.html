37void gvf_low_level_getState(void)
38{
39
40 gvf_c_state.px = stateGetPositionEnu_f()->x;
41 gvf_c_state.py = stateGetPositionEnu_f()->y;
42
43 #if defined(FIXEDWING_FIRMWARE)
44 float ground_speed = stateGetHorizontalSpeedNorm_f();
45 gvf_c_state.course = stateGetHorizontalSpeedDir_f();
46 gvf_c_state.px_dot = ground_speed * sinf(gvf_c_state.course);
47 gvf_c_state.py_dot = ground_speed * cosf(gvf_c_state.course);
48
49 #elif defined(ROVER_FIRMWARE) || defined(ROTORCRAFT_FIRMWARE)
50 // We assume that the course and psi
51 // of the rover (steering wheel) are the same
52 gvf_c_state.course = stateGetNedToBodyEulers_f()->psi;
53 gvf_c_state.px_dot = stateGetSpeedEnu_f()->x;
54 gvf_c_state.py_dot = stateGetSpeedEnu_f()->y;
55 #endif
56}

58void gvf_low_level_control_2D(float omega)
59{
60
61 #if defined(FIXEDWING_FIRMWARE)
62 if (autopilot_get_mode() == AP_MODE_AUTO2) {
63
64 // Coordinated turn
65 struct FloatEulers *att = stateGetNedToBodyEulers_f();
66 float ground_speed = stateGetHorizontalSpeedNorm_f();
67
68 lateral_mode = LATERAL_MODE_ROLL;
69
70 h_ctl_roll_setpoint =
71 - gvf_c_params.k_roll * atanf(omega * ground_speed / GVF_GRAVITY / cosf(att->theta));
72 BoundAbs(h_ctl_roll_setpoint, h_ctl_roll_max_setpoint);
73 }
74 #endif
75
76 gvf_c_ctrl.omega = omega;
77}

79void gvf_low_level_control_3D(float heading_rate, float climbing_rate)
80{
81 #if defined(FIXEDWING_FIRMWARE)
82 if (autopilot_get_mode() == AP_MODE_AUTO2) {
83 // Vertical Z coordinate
84 v_ctl_mode = V_CTL_MODE_AUTO_CLIMB;
85 v_ctl_speed_mode = V_CTL_SPEED_THROTTLE;
86
87 v_ctl_climb_setpoint = gvf_c_params.k_climb * climbing_rate; // Setting point for vertical speed
88
89 // Lateral XY coordinates
90 lateral_mode = LATERAL_MODE_ROLL;
91
92 struct FloatEulers *att = stateGetNedToBodyEulers_f();
93 float ground_speed = stateGetHorizontalSpeedNorm_f();
94
95 h_ctl_roll_setpoint =
96 -gvf_c_params.k_roll * atanf(heading_rate * ground_speed / GVF_GRAVITY / cosf(att->theta));
97 BoundAbs(h_ctl_roll_setpoint, h_ctl_roll_max_setpoint); // Setting point for roll angle
98 }
99 #endif
100
101 gvf_low_level_control_2D(heading_rate);
102
103 (void)(climbing_rate); // Avoid unused parameter warning
104}
