74 void link_mcu_event_task(void)
75 {
76  if (link_mcu_trans.status == SPITransSuccess) {
77  /* Got a message on SPI. */
78  link_mcu_trans.status = SPITransDone;
79 
80  /* A message has been received */
81  ComputeChecksum(link_mcu_from_ap_msg);
82  link_mcu_received = true;
83  if (link_mcu_from_ap_msg.checksum == crc) {
84  inter_mcu_received_ap = true;
85  } else {
86  fbw_state->nb_err++;
87  }
88  }
89  if (link_mcu_trans.status == SPITransFailed) {
90  link_mcu_trans.status = SPITransDone;
91  link_mcu_received = true;
92  fbw_state->nb_err++;
93  }
94 }
