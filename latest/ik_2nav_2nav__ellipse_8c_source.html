64bool nav_gvf_ik_ellipse_XY(float x, float y, float a, float b, float alpha)
65{
66 float e;
67 struct gvf_grad grad_ellipse;
68 struct gvf_Hess Hess_ellipse;
69
70 gvf_trajectory.type = 1;
71 gvf_trajectory.p[0] = x;
72 gvf_trajectory.p[1] = y;
73 gvf_trajectory.p[2] = a;
74 gvf_trajectory.p[3] = b;
75 gvf_trajectory.p[4] = alpha;
76 gvf_trajectory.p_len= 5 + gvf_p_len_wps;
77 gvf_p_len_wps = 0;
78
79 // SAFE MODE
80 if (a < 1 || b < 1) {
81 gvf_trajectory.p[2] = 60;
82 gvf_trajectory.p[3] = 60;
83 }
84
85 if ((int)gvf_trajectory.p[2] == (int)gvf_trajectory.p[3]) {
86 gvf_setNavMode(GVF_MODE_CIRCLE);
87
88 } else {
89 gvf_setNavMode(GVF_MODE_WAYPOINT);
90 }
91
92 gvf_ellipse_info(&e, &grad_ellipse, &Hess_ellipse);
93 gvf_ik_control.ke = gvf_ik_ellipse_par.ke;
94 gvf_ik_control_2D(gvf_ik_ellipse_par.ke, gvf_ik_ellipse_par.kn,
95 e, &grad_ellipse, &Hess_ellipse);
96
97 gvf_ik_control.error = e;
98
99 return true;
100}
