59 {
60  // Do not know whether is a good idea to do this check here or to include
61  // this plen in gvf_trajectory

63 
64  switch (gvf_trajectory.type) {
65  case LINE:
66  plen = 3;
67  break;
68  case ELLIPSE:
69  plen = 5;
70  break;
71  case SIN:
72  plen = 6;
73  break;
74  default:
75  plen = 1;
76  break;
77  }
78 
79  uint8_t traj_type = (uint8_t)gvf_trajectory.type;
80 
81  uint32_t now = get_sys_time_msec();
82  uint32_t delta_T = now - gvf_t0;
83 
84  if (delta_T < 200)
85  pprz_msg_send_GVF(trans, dev, AC_ID, &gvf_control.error, &traj_type,
86  &gvf_control.s, &gvf_control.ke, plen, gvf_trajectory.p);
87 }

118 static int out_of_segment_area(float x1, float y1, float x2, float y2, float d1, float d2)
119 {
120  struct EnuCoor_f *p = stateGetPositionEnu_f();
121  float px = p->x - x1;
122  float py = p->y - y1;
123 
124  float zx = x2 - x1;
125  float zy = y2 - y1;
126  float alpha = atan2f(zy, zx);
127 
128  float cosa = cosf(-alpha);
129  float sina = sinf(-alpha);
130 
131  float pxr = px * cosa - py * sina;
132  float zxr = zx * cosa - zy * sina;
133 
134  int s = 0;
135 
136  if (pxr < -d1) {
137  s = 1;
138  } else if (pxr > (zxr + d2)) {
139  s = -1;
140  }
141 
142  if (zy < 0) {
143  s *= -1;
144  }
145 
146  return s;
147 }

163 // GENERIC TRAJECTORY CONTROLLER
164 void gvf_control_2D(float ke, float kn, float e,
165  struct gvf_grad *grad, struct gvf_Hess *hess)
166 {
167  gvf_t0 = get_sys_time_msec();
168 
169  #if defined(FIXEDWING_FIRMWARE)
170  float ground_speed = stateGetHorizontalSpeedNorm_f();
171  float course = stateGetHorizontalSpeedDir_f();
172  float px_dot = ground_speed * sinf(course);
173  float py_dot = ground_speed * cosf(course);
174  #elif defined(ROVER_FIRMWARE)
175  // We assume that the course and psi
176  // of the rover (steering wheel) are the same
177  float course = stateGetNedToBodyEulers_f()->psi;
178  float px_dot = stateGetSpeedEnu_f()->x;
179  float py_dot = stateGetSpeedEnu_f()->y;
180  #endif
181 
182  int s = gvf_control.s;
183 
184  // gradient Phi
185  float nx = grad->nx;
186  float ny = grad->ny;
187 
188  // tangent to Phi
189  float tx = s * grad->ny;
190  float ty = -s * grad->nx;
191 
192  // Hessian
193  float H11 = hess->H11;
194  float H12 = hess->H12;
195  float H21 = hess->H21;
196  float H22 = hess->H22;
197 
198  // Calculation of the desired angular velocity in the vector field
199  float pdx_dot = tx - ke * e * nx;
200  float pdy_dot = ty - ke * e * ny;
201 
202  float norm_pd_dot = sqrtf(pdx_dot * pdx_dot + pdy_dot * pdy_dot);
203  float md_x = pdx_dot / norm_pd_dot;
204  float md_y = pdy_dot / norm_pd_dot;
205 
206  float Apd_dot_dot_x = -ke * (nx * px_dot + ny * py_dot) * nx;
207  float Apd_dot_dot_y = -ke * (nx * px_dot + ny * py_dot) * ny;
208 
209  float Bpd_dot_dot_x = ((-ke * e * H11) + s * H21) * px_dot
210  + ((-ke * e * H12) + s * H22) * py_dot;
211  float Bpd_dot_dot_y = -(s * H11 + (ke * e * H21)) * px_dot
212  - (s * H12 + (ke * e * H22)) * py_dot;
213 
214  float pd_dot_dot_x = Apd_dot_dot_x + Bpd_dot_dot_x;
215  float pd_dot_dot_y = Apd_dot_dot_y + Bpd_dot_dot_y;
216 
217  float md_dot_const = -(md_x * pd_dot_dot_y - md_y * pd_dot_dot_x)
218  / norm_pd_dot;
219 
220  float md_dot_x = md_y * md_dot_const;
221  float md_dot_y = -md_x * md_dot_const;
222 
223  float omega_d = -(md_dot_x * md_y - md_dot_y * md_x);
224 
225  float mr_x = sinf(course);
226  float mr_y = cosf(course);
227 
228  float omega = omega_d + kn * (mr_x * md_y - mr_y * md_x);
229 
230  #if defined(FIXEDWING_FIRMWARE)
231  if (autopilot_get_mode() == AP_MODE_AUTO2) {
232 
233  // Coordinated turn
234  struct FloatEulers *att = stateGetNedToBodyEulers_f();
235  float ground_speed = stateGetHorizontalSpeedNorm_f();
236 
237  lateral_mode = LATERAL_MODE_ROLL;
238 
239  h_ctl_roll_setpoint =
240  -atanf(omega * ground_speed / GVF_GRAVITY / cosf(att->theta));
241  BoundAbs(h_ctl_roll_setpoint, h_ctl_roll_max_setpoint);
242  }
243 
244  #elif defined(ROVER_FIRMWARE)
245  if (autopilot_get_mode() != AP_MODE_DIRECT) {
246  guidance_control.gvf_omega = omega; //TODO: set omega into external GVF variable
247  }
248 
249  #else
250  #error GVF does not support your firmware yet
251  #endif
252 }

607 bool gvf_segment_XY1_XY2(float x1, float y1, float x2, float y2)
608 {
609  struct EnuCoor_f *p = stateGetPositionEnu_f();
610  float px = p->x - x1;
611  float py = p->y - y1;
612 
613  float zx = x2 - x1;

616  float beta = atan2f(zy, zx);
617  float cosb = cosf(-beta);
618  float sinb = sinf(-beta);
619  float zxr = zx * cosb - zy * sinb;
620  float pxr = px * cosb - py * sinb;
