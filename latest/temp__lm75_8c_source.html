63void lm75_event(void)
64{
65 if (lm75_trans.status == I2CTransSuccess) {
66 uint16_t lm75_temperature;
67 float flm75_temperature;
68
69 /* read two byte temperature */
70 lm75_temperature = lm75_trans.buf[0] << 8;
71 lm75_temperature |= lm75_trans.buf[1];
72 lm75_temperature >>= 7;
73 if (lm75_temperature & 0x0100) {
74 lm75_temperature |= 0xFE00;
75 }
76
77 flm75_temperature = ((int16_t) lm75_temperature) / 2.;
78
79 DOWNLINK_SEND_TMP_STATUS(DefaultChannel, DefaultDevice, &lm75_temperature, &flm75_temperature);
80 lm75_trans.status = I2CTransDone;
81 }
