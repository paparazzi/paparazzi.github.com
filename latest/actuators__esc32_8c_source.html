83void actuators_esc32_config_cmd(uint8_t i)
84{
85 if (actuators_esc32.status != ESC32_STATUS_UNARMED ||
86 actuators_esc32.config_cmd != ESC32_CONFIG_CMD_IDLE) {
87 return;
88 }
89
90 // Switch the command
91 switch (i) {
92 case ESC32_CONFIG_CMD_BEEP: {
93 actuators_esc32_beep(ESC32_CAN_TT_NODE, actuators_esc32.escs_sorted[actuators_esc32.config_idx] + 1,
94 600, 500);
95 break;
96 }
97
98 case ESC32_CONFIG_CMD_TURN: {
99 uint16_t cmds[4] = {10000, 10000, 10000, 10000};
100 actuators_esc32_arm(ESC32_CAN_TT_NODE, actuators_esc32.escs_sorted[actuators_esc32.config_idx] + 1);
101 actuators_esc32_duty(ESC32_CAN_TT_NODE, actuators_esc32.escs_sorted[actuators_esc32.config_idx] + 1,
102 cmds);
103 actuators_esc32_start(ESC32_CAN_TT_NODE,
104 actuators_esc32.escs_sorted[actuators_esc32.config_idx] + 1);
105 actuators_esc32_disarm(ESC32_CAN_TT_NODE,
106 actuators_esc32.escs_sorted[actuators_esc32.config_idx] + 1);
107 break;
108 }
109
110 case ESC32_CONFIG_CMD_DIR: {
111 actuators_esc32_dir(ESC32_CAN_TT_NODE, actuators_esc32.escs_sorted[actuators_esc32.config_idx] + 1);
112 //actuators_esc32_beep(ESC32_CAN_TT_NODE, actuators_esc32.escs_sorted[actuators_esc32.config_idx]+1, 300, 200);
113 break;
114 }
115
116 default:
117 break;
118 }
119}

122void actuators_esc32_init(void)
123{
124 int i;
125 // Set default values
126 actuators_esc32.status = ESC32_STATUS_INIT;
127 actuators_esc32.melody_status = 0;
128
129 // Convert the melody
130 for (i = 1; i < actuators_esc32_melody_size; i++) {
131 actuators_esc32_melody[i][1] += actuators_esc32_melody[i - 1][1];
132 }
133
134 // Set everyone to free
135 for (i = 0; i < SERVOS_ESC32_NB; i++) {
136 actuators_esc32.escs[i].status = ESC32_STATUS_ESC_FREE;
137 }
138
139 // Initialize the the can bus
140 ppz_can_init(actuators_esc32_can_rx_cb);
141
142 // Reset all devices on the bus
143 actuators_esc32_send(ESC32_CAN_LCC_EXCEPTION | ESC32_CAN_TT_GROUP | ESC32_CAN_FID_RESET_BUS, 0, 0, 0);
144}

147void actuators_esc32_commit(void)
148{
149 uint8_t i;
150
151 // Go trough the groups of ESCs
152 for (i = 1; i <= (SERVOS_ESC32_NB - 1) / 4 + 1; i++) {
153
154 switch (actuators_esc32.status) {
155 // When the ESCs are running
156 case ESC32_STATUS_RUNNING:
157 actuators_esc32_duty(ESC32_CAN_TT_GROUP, i, &actuators_esc32.cmds[(i - 1) * 4]);
158
159 if (!autopilot_get_motors_on()) {
160 actuators_esc32_disarm(ESC32_CAN_TT_GROUP, i);
161 actuators_esc32.status = ESC32_STATUS_UNARMED;
162 }
163 break;
164
165 // When the ESCs are unarmed
166 case ESC32_STATUS_UNARMED:
167 actuators_esc32_duty(ESC32_CAN_TT_GROUP, i, &actuators_esc32.cmds[(i - 1) * 4]);
168
169 if (autopilot_get_motors_on()) {
170 actuators_esc32_arm(ESC32_CAN_TT_GROUP, i);
171 actuators_esc32_start(ESC32_CAN_TT_GROUP, i);
172 actuators_esc32.status = ESC32_STATUS_RUNNING;
173 }
174 break;
175
176 // When the ESC's are playing a melody
177 case ESC32_STATUS_MELODY:
178 if (actuators_esc32_play_melody(ESC32_CAN_TT_GROUP, i, &actuators_esc32.melody_status,
179 actuators_esc32_melody, actuators_esc32_melody_size)) {
180 actuators_esc32.status = ESC32_STATUS_UNARMED;
181 }
182 break;
183
184 default:
185 break;
186 }
187 }
188}

192static uint8_t actuators_esc32_send(uint32_t id, uint8_t tid, uint8_t length, uint8_t *data)
193{
194 uint8_t ret_idx = actuators_esc32.can_seq_idx;
195 actuators_esc32.responses[ret_idx].fid = 0;
196
197 ppz_can_transmit((id >> 3) | ((tid & 0x1f) << 6) | actuators_esc32.can_seq_idx, data, length);
198 actuators_esc32.can_seq_idx = (actuators_esc32.can_seq_idx + 1) % ESC32_RESPONSE_CNT;
199 return ret_idx;
200}

241static inline void actuators_esc32_dir(uint32_t tt, uint8_t tid)
242{
243 char *name = "DIRECTION";
244
245 // Request the param id
246 actuators_esc32_send(ESC32_CAN_LCC_NORMAL | tt | ESC32_CAN_FID_GET |
247 (ESC32_CAN_DATA_PARAM_NAME1 << 19), tid, 0, 0);
248 uint8_t retId = actuators_esc32_send(ESC32_CAN_LCC_NORMAL | tt | ESC32_CAN_FID_GET |
249 (ESC32_CAN_DATA_PARAM_NAME2 << 19), tid, 8, (uint8_t *)&name[8]);
250
251 // Busy waiting really bad (FIXME) TIMEOUT
252 while (actuators_esc32.responses[retId].fid == 0);
253
254 // Request the parameter value
255 uint16_t *paramId = (uint16_t *) actuators_esc32.responses[retId].data;
256 retId = actuators_esc32_send(ESC32_CAN_LCC_NORMAL | tt | ESC32_CAN_FID_GET |
257 (ESC32_CAN_DATA_PARAM_ID << 19), tid, 2, actuators_esc32.responses[retId].data);
258
259 // Busy waiting really bad (FIXME) TIMEOUT
260 while (actuators_esc32.responses[retId].fid == 0);
261
262 // Invert the direction
263 float *value = ((float *) actuators_esc32.responses[retId].data);
264 *value = *value * -1;
265
266 // Send the new direction
267 uint32_t data[2];
268 data[0] = *paramId;
269 data[1] = (uint32_t) * value;
270 actuators_esc32_send(ESC32_CAN_LCC_NORMAL | tt | ESC32_CAN_FID_SET |
271 (ESC32_CAN_DATA_PARAM_ID << 19), tid, 8, (uint8_t *)&data);
272}

275static bool actuators_esc32_play_melody(uint32_t tt, uint8_t tid, uint32_t *status_sub,
276 uint16_t melody[][2], uint8_t length)
277{
278 uint32_t timer = (*status_sub & 0x00FFFFFF) << 8;
279 uint8_t counter = (*status_sub & 0xFF000000) >> 24;
280 uint32_t timeout = counter == 0 ? ACTUATORS_ESC32_START_DELAY * 1000 :
281 (melody[counter - 1][1] + ACTUATORS_ESC32_START_DELAY) * 1000;
282
283 if (*status_sub == 0) {
284 SysTimeTimerStart(*status_sub);
285 *status_sub = *status_sub >> 8;
286 } else if (counter < length && SysTimeTimer(timer) > timeout) { // Start delay
287 if (melody[counter][0] != 0) {
288 actuators_esc32_beep(tt, tid, melody[counter][0],
289 melody[counter][1] - SysTimeTimer(timer) / 1000 + ACTUATORS_ESC32_START_DELAY);
290 }
291 *status_sub = *status_sub + (1 << 24);
292 } else if (counter == length && SysTimeTimer(timer) > timeout) {
293 return true;
294 }
295
296 return false;
297}

300void actuators_esc32_can_rx_cb(uint32_t id, uint8_t *data, int len __attribute__((unused)))
301{
302 id = id << 3;
303 //uint8_t doc = (id & ESC32_CAN_DOC_MASK)>>19;
304 //uint8_t sid = (id & ESC32_CAN_SID_MASK)>>14;
305 uint8_t seqId = (id & ESC32_CAN_SEQ_MASK) >> 3;
306
307 // Switch the different messages
308 switch (id & ESC32_CAN_FID_MASK) {
309 case ESC32_CAN_FID_REQ_ADDR:
310 actuators_esc32_grant_idx(data);
311 break;
312
313 case ESC32_CAN_FID_TELEM:
314 actuators_esc32_proc_telem(data);
315 break;
316
317 case ESC32_CAN_FID_ACK:
318 case ESC32_CAN_FID_NACK:
319 case ESC32_CAN_FID_REPLY:
320 actuators_esc32.responses[seqId].fid = (id & ESC32_CAN_FID_MASK) >> 25;
321 memcpy(actuators_esc32.responses[seqId].data, data, 8);
322 break;
323
324 default:
325 break;
326 }
327}

330static inline void actuators_esc32_grant_idx(uint8_t *data)
331{
332 uint8_t i, j;
333 uint32_t uuid = *((uint32_t *)data);
334
335 // Look if UUID is already registered
336 for (i = 0; actuators_esc32.escs[i].status != ESC32_STATUS_ESC_FREE; i++)
337 if (actuators_esc32.escs[i].uuid == uuid) {
338 break;
339 }
340
341 // Store the esc information
342 if (i < (ESC32_CAN_TID_MASK >> 9)) {
343 actuators_esc32.escs[i].status = ESC32_STATUS_ESC_INIT;
344 actuators_esc32.escs[i].network_id = i + 1;
345 actuators_esc32.escs[i].uuid = uuid;
346 actuators_esc32.escs[i].type = data[4];
347 actuators_esc32.escs[i].can_id = data[5];
348
349 data[4] = (i / 4) + 1; //GroupId
350 data[5] = (i % 4) + 1; //SubGroupId
351 actuators_esc32_send(ESC32_CAN_LCC_HIGH | ESC32_CAN_TT_NODE | ESC32_CAN_FID_GRANT_ADDR, i + 1, 6,
352 data);
353 }
354
355 // When this is the last motor assign groups
356 if (i >= SERVOS_ESC32_NB - 1) {
357
358 // Go trough all servos to sort
359 for (i = 0; i < SERVOS_ESC32_NB; i++) {
360 // Search for next lowest uuid
361 uint8_t lowest = UCHAR_MAX;
362 for (j = 0; j < SERVOS_ESC32_NB; j++) {
363 if ((i == 0 ||
364 actuators_esc32.escs[j].uuid > actuators_esc32.escs[actuators_esc32.escs_sorted[i - 1]].uuid)
365 && (lowest == UCHAR_MAX || actuators_esc32.escs[j].uuid < actuators_esc32.escs[lowest].uuid)) {
366 lowest = j;
367 }
368 }
369
370 actuators_esc32.escs[lowest].status = ESC32_STATUS_ESC_RDY;
371 actuators_esc32.escs_sorted[i] = lowest;
372 }
373
374 actuators_esc32.status = ESC32_STATUS_MELODY;
375 }
376}
