88static void update_barycenter(struct EnuCoor_f *bary, struct EnuCoor_f *border, struct EnuCoor_f *first, float alt_sp, float dt, float tau)
89{
90 if (first != NULL) {
91 bary->x = (border->x + first->x) / 2.f;
92 bary->y = (border->y + first->y) / 2.f;
93 bary->z = alt_sp;
94 }
95 else {
96 if (tau > 1e-4) {
97 float alpha = dt / (dt + tau);
98 bary->x = (border->x * alpha) + (bary->x * (1.f - alpha));
99 bary->y = (border->y * alpha) + (bary->y * (1.f - alpha));
100 bary->z = alt_sp;
101 }
102 }
103}

105static struct EnuCoor_f process_new_point_rosette(struct EnuCoor_f *position, float uav_direction)
106{
107 struct EnuCoor_f new_point;
108 float rot_angle;
109
110 if (nav_rosette.rotation == RSTT_RIGHT) {
111 rot_angle = -M_PI_2;
112 } else {
113 rot_angle = M_PI_2;
114 }
115
116 if (nav_rosette.inside_cloud == true) {
117 new_point.x = nav_rosette.target.x;
118 new_point.y = nav_rosette.target.y;
119 new_point.z = nav_rosette.target.z;
120 }
121 else if (nav_rosette.inside_cloud == false) {
122 new_point.x = position->x + (cosf(rot_angle + uav_direction) * nav_rosette.radius);
123 new_point.y = position->y + (sinf(rot_angle + uav_direction) * nav_rosette.radius);
124 new_point.z = nav_rosette.target.z;
125 }
126
127 return new_point;
128}

133static bool nav_rosette_mission(uint8_t nb, float *params, enum MissionRunFlag flag)
134{
135 if (flag == MissionInit && nb == 8) {
136 float start_x = params[0];
137 float start_y = params[1];
138 float start_z = params[2];
139 int first_turn = params[3];
140 float circle_radius = params[4];
141 float vx = params[5];
142 float vy = params[6];
143 float vz = params[7];
144 nav_rosette_setup(start_x, start_y, start_z, first_turn, circle_radius, vx, vy, vz);
145 return true;
146 }
147 else if (flag == MissionUpdate && nb == 3) {
148 // update target 3D position (ENU frame, above ground alt)
149 float x = params[0];
150 float y = params[1];
151 float z = params[2];
152 VECT3_ASSIGN(nav_rosette.target, x, y, z);
153 return true;
154 }
155 else if (flag == MissionUpdate && nb == 2) {
156 // update horizontal speed
157 float vx = params[0];
158 float vy = params[1];
159 nav_rosette.pos_incr.x = vx * nav_dt;
160 nav_rosette.pos_incr.y = vy * nav_dt;
161 return true;
162 }
163 else if (flag == MissionUpdate && nb == 1) {
164 // update vertical speed
165 float vz = params[0];
166 nav_rosette.pos_incr.z = vz * nav_dt;
167 return true;
168 }
169 else if (flag == MissionRun) {
170 return nav_rosette_run();
171 }
172 return false; // not a valid case
173}
174#endif

207void nav_rosette_setup(float init_x, float init_y, float init_z,
208 int turn, float desired_radius,
209 float vx, float vy, float vz)
210{
211 struct EnuCoor_f start = {init_x, init_y, init_z};
212 // increment based on speed
213 VECT3_ASSIGN(nav_rosette.pos_incr, vx*nav_dt, vy*nav_dt, vz*nav_dt);
214
215 nav_rosette.target = start;
216 nav_rosette.status = RSTT_ENTER;
217 nav_rosette.inside_cloud = false;
218 nav_rosette.radius = desired_radius;
219 nav_rosette.first = NULL;
220
221 if (turn == 1) {
222 nav_rosette.rotation = RSTT_RIGHT;
223 nav_rosette.radius_sign = 1.0f;
224 } else {
225 nav_rosette.rotation = RSTT_LEFT;
226 nav_rosette.radius_sign = -1.0f;
227 }
228
229 nav_rosette.actual = *stateGetPositionEnu_f();
230 nav_rosette.last_border_time = get_sys_time_float();
231}

233bool nav_rosette_run(void)
234{
235 float pre_climb = 0.f;
236 float time = get_sys_time_float();
237
238 NavVerticalAutoThrottleMode(0.f); /* No Pitch */
239
240 switch (nav_rosette.status) {
241 case RSTT_ENTER:
242 // init stage
243 nav_init_stage();
244 // reach target point
245 nav_route_xy(nav_rosette.actual.x, nav_rosette.actual.y, nav_rosette.target.x, nav_rosette.target.y);
246 NavVerticalAltitudeMode(nav_rosette.target.z + ground_alt, pre_climb);
247
248 if (nav_rosette.inside_cloud) {
249 // found border or already inside
250 nav_rosette.status = RSTT_CROSSING_FIRST;
251 }
252 break;
253 case RSTT_CROSSING_FIRST:
254 // prepare first crossing
255 nav_rosette.actual = *stateGetPositionEnu_f();
256 nav_rosette.last_border_time = time;
257 nav_rosette.estim_border = nav_rosette.actual;
258 nav_rosette.first = &nav_rosette.estim_border;
259 // init stage
260 nav_init_stage();
261 // cross
262 nav_rosette.status = RSTT_CROSSING;
263 break;
264 case RSTT_CROSSING_START:
265 // prepare crossing
266 nav_rosette.actual = *stateGetPositionEnu_f();
267 update_barycenter(&nav_rosette.target, &nav_rosette.actual, nav_rosette.first,
268 nav_rosette.target.z, time - nav_rosette.last_border_time, NAV_ROSETTE_BORDER_FILTER);
269 nav_rosette.last_border_time = time;
270 nav_rosette.estim_border = nav_rosette.actual;
271 // init stage
272 nav_init_stage();
273 // cross
274 nav_rosette.status = RSTT_CROSSING;
275 break;
276 case RSTT_CROSSING:
277 // cross towards target
278 pre_climb = nav_rosette.pos_incr.z / nav_dt;
279 nav_route_xy(nav_rosette.actual.x, nav_rosette.actual.y, nav_rosette.target.x, nav_rosette.target.y);
280 NavVerticalAltitudeMode(nav_rosette.target.z + ground_alt, pre_climb);
281
282 if (!nav_rosette.inside_cloud) {
283 // found a border, starting turning back inside
284 // note that you don't need a recover as you always expect to go out after some time
285 nav_rosette.status = RSTT_TURNING_START;
286 }
287 break;
288 case RSTT_TURNING_START:
289 // update target
290 nav_rosette.actual = *stateGetPositionEnu_f();
291 update_barycenter(&nav_rosette.target, &nav_rosette.actual, nav_rosette.first,
292 nav_rosette.target.z, time - nav_rosette.last_border_time, NAV_ROSETTE_BORDER_FILTER);
293 nav_rosette.last_border_time = time;
294 nav_rosette.estim_border = nav_rosette.actual;
295 nav_rosette.first = NULL;
296 // prepare next circle
297 nav_rosette.direction = change_rep(stateGetHorizontalSpeedDir_f());
298 nav_rosette.circle = process_new_point_rosette(&nav_rosette.actual, nav_rosette.direction);
299 // init stage
300 nav_init_stage();
301 // turn
302 nav_rosette.status = RSTT_TURNING;
303 break;
304 case RSTT_TURNING:
305 // update circle center
306 VECT3_ADD(nav_rosette.circle, nav_rosette.pos_incr);
307 pre_climb = nav_rosette.pos_incr.z / nav_dt;
308 nav_circle_XY(nav_rosette.circle.x, nav_rosette.circle.y , nav_rosette.radius_sign * nav_rosette.radius);
309 NavVerticalAltitudeMode(nav_rosette.circle.z + ground_alt, pre_climb);
310
311 if (nav_rosette.inside_cloud) {
312 // found a border, cross again
313 nav_rosette.status = RSTT_CROSSING_START;
314 }
315 else if (NavCircleCountNoRewind() > NAV_ROSETTE_RECOVER_MAX_TURN) {
316 // most likely lost outside
317 nav_rosette.status = RSTT_RECOVER_START;
318 }
319 break;
320 case RSTT_RECOVER_START:
321 // prepare recovery circle
322 nav_rosette.recover_circle = nav_rosette.target;
323 nav_rosette.actual = *stateGetPositionEnu_f();
324 // initial recovery radius
325 nav_rosette.recover_radius = nav_rosette.radius;
326 nav_rosette.max_recover_radius = 2.0f * nav_rosette.recover_radius; // FIXME ?
327 // init stage
328 nav_init_stage();
329 // recover
330 nav_rosette.status = RSTT_RECOVER_OUTSIDE;
331 break;
332 case RSTT_RECOVER_OUTSIDE:
333 // increment center position
334 VECT3_ADD(nav_rosette.recover_circle, nav_rosette.pos_incr);
335 nav_circle_XY(nav_rosette.recover_circle.x, nav_rosette.recover_circle.y , nav_rosette.radius_sign * nav_rosette.recover_radius);
336 // increment recover circle radius
337 if (nav_rosette.recover_radius < nav_rosette.max_recover_radius) {
338 nav_rosette.recover_radius += 0.5;
339 }
340 // found a new border
341 if (nav_rosette.inside_cloud) {
342 nav_rosette.status = RSTT_CROSSING_START;
343 }
344 break;
345 default:
346 // error, leaving
347 return false;
348 }
349 // update target and border positions
350 VECT3_ADD(nav_rosette.estim_border, nav_rosette.pos_incr);
351 VECT3_ADD(nav_rosette.target, nav_rosette.pos_incr);
352#ifdef WP_TARGET
353 nav_move_waypoint_enu(WP_TARGET, nav_rosette.target.x, nav_rosette.target.y, nav_rosette.target.z + ground_alt);
354 RunOnceEvery(10, nav_send_waypoint(WP_TARGET));
355#endif
356
357 return true;
358}
