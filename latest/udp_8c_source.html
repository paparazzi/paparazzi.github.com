94 void WEAK udp_put_byte(struct udp_periph *p, long fd __attribute__((unused)), uint8_t data)
95 {
96  if (p->tx_insert_idx >= UDP_TX_BUFFER_SIZE) {
97  return; // no room
98  }
99 
100  p->tx_buf[p->tx_insert_idx] = data;
101  p->tx_insert_idx++;
102 }

104 void WEAK udp_put_buffer(struct udp_periph *p, long fd __attribute__((unused)), const uint8_t *data, uint16_t len)
105 {
106  if (p->tx_insert_idx + len >= UDP_TX_BUFFER_SIZE) {
107  return; // no room
108  }
109 
110  memcpy(&(p->tx_buf[p->tx_insert_idx]), data, len);
111  p->tx_insert_idx += len;
