64void eff_scheduling_cyfoam_init(void)
65{
66 //sum of G1 and G2
67 int8_t i;
68 int8_t j;
69 for (i = 0; i < INDI_OUTPUTS; i++) {
70 for (j = 0; j < INDI_NUM_ACT; j++) {
71 if (i != 2) {
72 g1g2_hover[i][j] = g1g2_hover[i][j] / INDI_G_SCALING;
73 g1g2_forward[i][j] = g1g2_forward[i][j] / INDI_G_SCALING;
74 } else {
75 g1g2_forward[i][j] = (g1g2_forward[i][j] + g2_both[j]) / INDI_G_SCALING;
76 g1g2_hover[i][j] = (g1g2_hover[i][j] + g2_both[j]) / INDI_G_SCALING;
77 }
78 }
79 }
80}

82static void eff_scheduling_periodic_a(void)
83{
84 int8_t i;
85 int8_t j;
86 const float ratio = stabilization.transition_ratio;
87 for (i = 0; i < INDI_OUTPUTS; i++) {
88 for (j = 0; j < INDI_NUM_ACT; j++) {
89 g1g2[i][j] = g1g2_hover[i][j] * (1.0 - ratio) + g1g2_forward[i][j] * ratio;
90 }
91 }
92}

94static void eff_scheduling_periodic_b(void)
95{
96 float airspeed = stateGetAirspeed_f();
97 struct FloatEulers eulers_zxy;
98 if(airspeed < 6.0) {
99 float_eulers_of_quat_zxy(&eulers_zxy, stateGetNedToBodyQuat_f());
100 float pitch_interp = DegOfRad(eulers_zxy.theta);
101 Bound(pitch_interp, -60.0, -30.0);
102 float ratio = (pitch_interp + 30.0)/(-30.);
103
104 /*pitch*/
105 g1g2[1][0] = g1g2_hover[1][0]*(1-ratio) + -PITCH_EFF_AT_60/1000*ratio;
106 g1g2[1][1] = g1g2_hover[1][1]*(1-ratio) + PITCH_EFF_AT_60/1000*ratio;
107 /*yaw*/
108 g1g2[2][0] = g1g2_hover[2][0]*(1-ratio) + -YAW_EFF_AT_60/1000*ratio;
109 g1g2[2][1] = g1g2_hover[2][1]*(1-ratio) + -YAW_EFF_AT_60/1000*ratio;
110 } else {
111 // calculate squared airspeed
112 Bound(airspeed, 0.0, 30.0);
113 float airspeed2 = airspeed*airspeed;
114
115 float pitch_eff = CE_PITCH_A + CE_PITCH_B*airspeed2;
116 g1g2[1][0] = -pitch_eff/1000;
117 g1g2[1][1] = pitch_eff/1000;
118
119 float yaw_eff = CE_YAW_A + CE_YAW_B*airspeed2;
120 g1g2[2][0] = -yaw_eff/1000;
121 g1g2[2][1] = -yaw_eff/1000;
122 }
123
124 g1g2[0][2] = -actuator_state_filt_vect[2]/1000*SQUARED_ROLL_EFF;
125 g1g2[0][3] = actuator_state_filt_vect[3]/1000*SQUARED_ROLL_EFF;
126 Bound(g1g2[0][2], -30.0/1000, -2.0/1000);
127 Bound(g1g2[0][3], 2.0/1000, 30.0/1000);
128
129 /*Make pitch gain equal to roll gain for turns forward flight*/
130 if(airspeed > 12.0) {
131 indi_gains.att.q = 107.0;
132 } else {
133 indi_gains.att.q = 200.0;
134 }
135
136}

138void eff_scheduling_cyfoam_periodic(void)
139{
140 if (radio_control.values[INDI_FUNCTIONS_RC_CHANNEL] > 0) {
141 eff_scheduling_periodic_b();
142 } else {
143 eff_scheduling_periodic_a();
144 }
145
146#ifdef INDI_THRUST_ON_PITCH_EFF
147 //State prioritization {W Roll, W pitch, W yaw, TOTAL THRUST}
148 if(radio_control.values[INDI_FUNCTIONS_RC_CHANNEL] > 0 && (actuator_state_filt_vect[0] < -7000) && (actuator_state_filt_vect[1] > 7000)) {
149 Bwls[1][2] = INDI_THRUST_ON_PITCH_EFF/INDI_G_SCALING;
150 Bwls[1][3] = INDI_THRUST_ON_PITCH_EFF/INDI_G_SCALING;
151 } else if(radio_control.values[INDI_FUNCTIONS_RC_CHANNEL] > 0 && (actuator_state_filt_vect[0] > 7000) && (actuator_state_filt_vect[1] < -7000)) {
152 Bwls[1][2] = -INDI_THRUST_ON_PITCH_EFF/INDI_G_SCALING;
153 Bwls[1][3] = -INDI_THRUST_ON_PITCH_EFF/INDI_G_SCALING;
154 } else {
155 Bwls[1][2] = 0.0;
156 Bwls[1][3] = 0.0;
157 }
158#endif
159}
