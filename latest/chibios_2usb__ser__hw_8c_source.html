297 /*
298  * Handles the USB driver global events.
299  */
300 static void usb_event(USBDriver *usbp, usbevent_t event) {
301 
302  switch (event) {
303  case USB_EVENT_RESET:
304  return;
305  case USB_EVENT_ADDRESS:
306  return;
307  case USB_EVENT_CONFIGURED:
308  chSysLockFromISR();
309 
310  /* Enables the endpoints specified into the configuration.
311  Note, this callback is invoked from an ISR so I-Class functions
312  must be used.*/
313  usbInitEndpointI(usbp, USBD1_DATA_REQUEST_EP, &ep1config);
314  usbInitEndpointI(usbp, USBD1_INTERRUPT_REQUEST_EP, &ep2config);
315 
316  /* Resetting the state of the CDC subsystem.*/
317  sduConfigureHookI(&SDU);
318 
319  chSysUnlockFromISR();
320  return;
321  case USB_EVENT_UNCONFIGURED:
322  return;
323  case USB_EVENT_SUSPEND:
324  chSysLockFromISR();
325 
326  /* Disconnection event on suspend.*/
327  sduSuspendHookI(&SDU);
328 
329  chSysUnlockFromISR();
330  return;
331  case USB_EVENT_WAKEUP:
332  return;
333  case USB_EVENT_STALLED:
334  return;
335  }
336  return;
337 }

438 static int usb_serial_char_available(struct usb_serial_periph *p __attribute__((unused)))
439 {
440  // some bytes in the buffer
441  if(p->nb_bytes > 0) {
442  return p->nb_bytes;
443  }
444 
445  uint16_t write_idx = (p->rx_read_idx + p->nb_bytes) % USB_RX_BUFFER_SIZE;
446 
447  // no bytes in the buffer, try to get one
448  msg_t res = ibqGetTimeout(&SDU.ibqueue, TIME_IMMEDIATE);
449  if(res >= 0) {
450  // got a byte. put it in the buffer.
451  p->rx_buf[write_idx] = (uint8_t) res;
452  p->nb_bytes++;
453  return p->nb_bytes;
454  } else {
455  // timeout or queue suspended. No data available.
456  return 0;
457  }
458 }

460 static uint8_t usb_serial_getch(struct usb_serial_periph *p __attribute__((unused)))
461 {
462  if(p->nb_bytes > 0) {
463  uint8_t ret = p->rx_buf[p->rx_read_idx];
464  p->rx_read_idx = (p->rx_read_idx + 1) % USB_RX_BUFFER_SIZE;
465  p->nb_bytes--;
466  return ret;
467  } else {
468  // blocking get
469  return streamGet(&SDU);
470  }
471 }
