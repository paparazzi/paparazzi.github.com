50 #define Polynom 0xD5
51 uint8_t faulhaber_crc8(uint8_t u8Byte, uint8_t u8CRC)
52 {
53  uint8_t i;
54  u8CRC = u8CRC ^ u8Byte;
55  for (i = 0; i < 8; i++) {
56  if (u8CRC & 0x01) {
57  u8CRC = (u8CRC >> 1) ^ Polynom;
58  } else {
59  u8CRC >>= 1;
60  }
61  }
62  return u8CRC;
63 }

65 static void faulhaber_send_command(struct uart_periph *dev, uint8_t cmd_code, uint8_t *data, uint8_t data_length)
66 {
67  uart_put_byte(dev, 0, 'S'); // Character (S) as Start of Frame
68  uart_put_byte(dev, 0, data_length + 4); // Telegram length without SOF/EOF (packet length)
69  uint8_t crc8 = faulhaber_crc8(data_length + 4, 0xFF); // Start CRC with 0xFF
70  uart_put_byte(dev, 0, 0x01); // Node number of the slave (0 = Broadcast)
71  crc8 = faulhaber_crc8(0x01, crc8);
72  uart_put_byte(dev, 0, cmd_code); // See Tab. 2
73  crc8 = faulhaber_crc8(cmd_code, crc8);
74  for (uint8_t i = 0; i < data_length; i++) { // Data area (length = packet length – 4)
75  uart_put_byte(dev, 0, data[i]);
76  crc8 = faulhaber_crc8(data[i], crc8);
77  }
78  uart_put_byte(dev, 0, crc8); // CRC8 with polynomial 0xD5 over byte 2–N
79  uart_put_byte(dev, 0, 'E'); // Character (E) as End of Frame
80  uart_send_message(dev, 0);
81 }

92 static void faulhaber_parser(struct faulhaber_parser_t *p, uint8_t c)
93 {
94  switch (p->state) {
95  case UINIT:
96  if (c == 'S') {
97  p->state = GOT_SOF;
98  p->calc_crc8 = 0xFF;
99  }
100  break;
101  case GOT_SOF:
102  if (c - 4 < 0) {
103  p->state = UINIT;
104  } else {
105  p->data_length = c - 4;
106  p->calc_crc8 = faulhaber_crc8(c, p->calc_crc8);
107  p->state = GOT_LENGTH;
108  }
109  break;
110  case GOT_LENGTH:
111  p->node_nb = c;
112  p->calc_crc8 = faulhaber_crc8(c, p->calc_crc8);
113  p->state = GOT_NODE_NB;
114  break;
115  case GOT_NODE_NB:
116  p->cmd_code = c;
117  p->data_idx = 0;
118  p->calc_crc8 = faulhaber_crc8(c, p->calc_crc8);
119  p->state = GET_DATA;
120  break;
121  case GET_DATA:
122  p->data[p->data_idx++] = c;
123  p->calc_crc8 = faulhaber_crc8(c, p->calc_crc8);
124  if (p->data_idx >= p->data_length) {
125  p->state = GET_CRC;
126  }
127  break;
128  case GET_CRC:
129  if (p->calc_crc8 == c) {
130  p->state = GET_EOF;
131  } else {
132  p->state = UINIT;
133  }
134  break;
135  case GET_EOF:
136  if (c == 'E') {
137  p->state = GOT_FULL_PACKET;
138  } else {
139  p->state = UINIT;
140  }
141  break;
142  default:
143  p->state = UINIT;
144  break;
145  }
146 }

167 void actuators_faulhaber_periodic(void)
168 {
169  switch (faulhaber.mode) {
170 
171  /* HOME MODE */
172  case FH_MODE_HOME:
173  switch (faulhaber.state) {
174  case 0: {
175  // Set the homing mode
176  uint8_t data[] = { 0x60, 0x60, 0x00, 0x06 }; // Set 0x6060.00 to 0x06: Homing mode
177  faulhaber_send_command(faulhaber_dev, 0x02, data, 4);
178  faulhaber.state++;
179  break;
180  }
181  case 1: {
182  // Set the homing method
183  uint8_t data[] = { 0x98, 0x60, 0x00, 0x11 }; // Set 0x6098.00 to 0x11: Homing method to 17
184  faulhaber_send_command(faulhaber_dev, 0x02, data, 4);
185  faulhaber.state++;
186  break;
187  }
188  case 2: {
189  // Enable operation
190  uint8_t data[] = { 0x40, 0x60, 0x00, 0x0F, 0x00}; // Set 0x6040.00 to 0x000F: Enable operation
191  faulhaber_send_command(faulhaber_dev, 0x02, data, 5);
192  faulhaber.state++;
193  break;
194  }
195  case 3: {
196  // Start moving
197  uint8_t data[] = { 0x40, 0x60, 0x00, 0x1F, 0x00}; // Set 0x6040.00 to 0x001F: Start moving
198  faulhaber_send_command(faulhaber_dev, 0x02, data, 5);
199  faulhaber.state++;
200  break;
201  }
202  default:
203  // Goto position mode
204  faulhaber.mode = FH_MODE_INIT;
205  faulhaber.state = 0;
206  break;
207  }
208  break;
209 
210  /* POSITION MODE */
211  case FH_MODE_POSITION:
212  switch (faulhaber.state) {
213  case 0: {
214  uint8_t data[] = { 0x60, 0x60, 0x00, 0x01 }; // Set 0x6060.00 to 0x01: Position mode
215  faulhaber_send_command(faulhaber_dev, 0x02, data, 4);
216  faulhaber.state++;
217  break;
218  }
219  case 1: {
220  // Set the target position
221  faulhaber.target_position = faulhaber.setpoint_position;
222  uint8_t data[] = { 0x7A, 0x60, 0x00, (faulhaber.target_position & 0xFF), ((faulhaber.target_position >> 8) & 0xFF), ((faulhaber.target_position >> 16) & 0xFF), ((faulhaber.target_position >> 24) & 0xFF) }; // Set 0x607A.00 to 0x00000000: Target position 0
223  faulhaber_send_command(faulhaber_dev, 0x02, data, 7);
224  faulhaber.state++;
225  break;
226  }
227  case 2: {
228  // Enable operation
229  uint8_t data[] = { 0x40, 0x60, 0x00, 0x0F, 0x00}; // Set 0x6040.00 to 0x000F: Enable operation
230  faulhaber_send_command(faulhaber_dev, 0x02, data, 5);
231  faulhaber.state++;
232  break;
233  }
234  case 3: {
235  // Start moving
236  uint8_t data[] = { 0x40, 0x60, 0x00, 0x3F, 0x00}; // Set 0x6040.00 to 0x003F: Start moving
237  faulhaber_send_command(faulhaber_dev, 0x02, data, 5);
238  faulhaber.state++;
239  break;
240  }
241  default:
242  faulhaber.mode = FH_MODE_IDLE;
243  faulhaber.state = 0;
244  break;
245 
246  }
247  break;
248 
249  /* FH_MODE_IDLE */
250  case FH_MODE_IDLE: {
251  // Move to the new position
252  if (faulhaber.target_position != faulhaber.setpoint_position) {
253  faulhaber.mode = FH_MODE_POSITION;
254  faulhaber.state = 0;
255  break;
256  }
257 
258  // Request the position
259  uint8_t data[] = { 0x64, 0x60, 0x00}; // Get 0x6064.00: Get the actual position
260  faulhaber_send_command(faulhaber_dev, 0x01, data, 3);
261  faulhaber.state = 0;
262  break;
263  }
264 
265  case FH_MODE_ENABLE:
266  switch (faulhaber.state) {
267  case 0: {
268  // Disable drive
269  uint8_t data[] = { 0x40, 0x60, 0x00, 0x80, 0x00}; // Set 0x6040.00 to 0x0007: Disable drive
270  faulhaber_send_command(faulhaber_dev, 0x02, data, 5);
271  faulhaber.state++;
272  //break;
273  }
274  case 1: {
275  // Disable drive
276  uint8_t data[] = { 0x40, 0x60, 0x00, 0x06, 0x00}; // Set 0x6040.00 to 0x0006: Disable drive
277  faulhaber_send_command(faulhaber_dev, 0x02, data, 5);
278  faulhaber.state++;
279  //break;
280  }
281  case 2: {
282  // Enable drive
283  uint8_t data[] = { 0x40, 0x60, 0x00, 0x07, 0x00}; // Set 0x6040.00 to 0x0007: Enable drive
284  faulhaber_send_command(faulhaber_dev, 0x02, data, 5);
285  faulhaber.state++;
286  //break;
287  }
288  case 3: {
289  // Execute
290  uint8_t data[] = { 0x40, 0x60, 0x00, 0x0F, 0x00}; // Set 0x6040.00 to 0x000F: Execute
291  faulhaber_send_command(faulhaber_dev, 0x02, data, 5);
292  faulhaber.state++;
293  //break;
294  }
295  default:
296  faulhaber.mode = FH_MODE_INIT;
297  faulhaber.state = 0;
298  break;
299  }
300  break;
301 
302  /* Do nothing */
303  default:
304  break;
305  }

309 static void faulhaber_parse_msg(struct faulhaber_parser_t *p)
310 {
311  // Process position message
312  if (p->cmd_code == 0x01 && p->data[0] == 0x64 && p->data[1] == 0x60 && p->data[2] == 0x00) {
313  faulhaber.real_position = p->data[3] | (p->data[4] << 8) | (p->data[5] << 16) | (p->data[6] << 24);
314 
315  struct act_feedback_t feedback = {0};
316  feedback.position = ((faulhaber.real_position/ACTUATORS_FAULHABER_COMMAND_SCALE) - get_servo_min_FAULHABER(0)) / (double)(get_servo_max_FAULHABER(
317  0) - get_servo_min_FAULHABER(0)) * M_PI_2; // In radians
318  feedback.set.position = true;
319  feedback.idx = get_servo_idx_FAULHABER(0);
320 
321  // Send ABI message
322  AbiSendMsgACT_FEEDBACK(ACT_FEEDBACK_FAULHABER_ID, &feedback, 1);
