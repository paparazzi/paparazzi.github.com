116void dc_shoot_pwm_periodic(void)
117{
118 // Manage the shutter opening time each DC_CAM_PWM_SHUTTER_DELAY seconds
119 if (shutter_timer == 0) {
120 CamActuatorSet(DC_CAM_PWM_SERVO, DC_CAM_PWM_OFF_VALUE);
121 } else {
122 shutter_timer--;
123 }
124
125 // Common DC Periodic task
126 dc_periodic();
127}
