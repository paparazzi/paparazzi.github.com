39void imu_pixhawk6x_init(void)
40{
41 struct Int32RMat rmat;
42 struct Int32Eulers eulers;
43
44 /* IMU 1 */
45 imu1.abi_id = IMU_PIXHAWK1_ID;
46 imu1.parser = INVENSENSE3_PARSER_FIFO;
47 imu1.bus = INVENSENSE3_SPI;
48 imu1.spi.p = &PIXHAWK6X_IMU1_SPI_DEV;
49 imu1.spi.slave_idx = PIXHAWK6X_IMU1_SPI_SLAVE_IDX;
50
51 imu1.gyro_odr = INVENSENSE3_GYRO_ODR_4KHZ;
52 imu1.gyro_range = INVENSENSE3_GYRO_RANGE_2000DPS;
53 imu1.accel_odr = INVENSENSE3_ACCEL_ODR_4KHZ;
54 imu1.accel_range = INVENSENSE3_ACCEL_RANGE_32G;
55 imu1.gyro_aaf = 977; // ~ODR/4
56 imu1.accel_aaf = 213; // Fixed
57
58 invensense3_init(&imu1);
59

61 eulers.phi = ANGLE_BFP_OF_REAL(0);

63 eulers.psi = ANGLE_BFP_OF_REAL(RadOfDeg(90));
64 int32_rmat_of_eulers(&rmat, &eulers);
65 imu_set_defaults_gyro(IMU_PIXHAWK1_ID, &rmat, NULL, NULL);
66 imu_set_defaults_accel(IMU_PIXHAWK1_ID, &rmat, NULL, NULL);
67
68 /* IMU 2 */
69 imu2.abi_id = IMU_PIXHAWK2_ID;
70 imu2.parser = INVENSENSE3_PARSER_FIFO;
71 imu2.bus = INVENSENSE3_SPI;
72 imu2.spi.p = &PIXHAWK6X_IMU2_SPI_DEV;
73 imu2.spi.slave_idx = PIXHAWK6X_IMU2_SPI_SLAVE_IDX;
74
75 imu2.gyro_odr = INVENSENSE3_GYRO_ODR_4KHZ;
76 imu2.gyro_range = INVENSENSE3_GYRO_RANGE_2000DPS;
77 imu2.accel_odr = INVENSENSE3_ACCEL_ODR_4KHZ;
78 imu2.accel_range = INVENSENSE3_ACCEL_RANGE_32G;
