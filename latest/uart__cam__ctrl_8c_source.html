65void digital_cam_uart_event(void)
66{
67 while (CameraLinkChAvailable()) {
68 parse_mora(&mora_protocol, CameraLinkGetch());
69 if (mora_protocol.msg_received) {
70 switch (mora_protocol.msg_id) {
71 case MORA_STATUS:
72 for (int i = 0; i < MORA_STATUS_MSG_SIZE; i++) {
73 mora_status_msg.bin[i] = mora_protocol.payload[i];
74 }
75 digital_cam_uart_status = mora_status_msg.data.shots;
76 break;
77 case MORA_PAYLOAD:
78 for (int i = 0; i < MORA_PAYLOAD_MSG_SIZE; i++) {
79 thumbs[thumb_pointer][i] = mora_protocol.payload[i];
80 }
81 break;
82 default:
83 break;
84 }
85 mora_protocol.msg_received = 0;
86 }
87 }
88}

91static void send_thumbnails(struct transport_tx *trans, struct link_device *dev)
92{
93 static int cnt = 0;
94 if (digital_cam_uart_thumbnails > 0) {
95 if (digital_cam_uart_thumbnails == 1) {
96 cnt++;
97 if (cnt > 1) {
98 cnt = 0;
99 return;
100 }
101 }
102 pprz_msg_send_PAYLOAD(trans, dev, AC_ID, THUMB_MSG_SIZE, thumbs[thumb_pointer]);
103
104 // Update the write/read pointer: if we receive a new thumb part, that will be sent, otherwise the oldest infor is repeated
105 thumb_pointer++;
106 if (thumb_pointer >= THUMB_COUNT) {
107 thumb_pointer = 0;
108 }
109
110 MoraHeader(MORA_BUFFER_EMPTY, 0);
111 MoraTrailer();
112 }
113}

116void digital_cam_uart_init(void)
117{

140/* Command The Camera */
141void dc_send_command(uint8_t cmd)
142{
143 switch (cmd) {
144 case DC_SHOOT:
145 // Send Photo Position To Camera
146 dc_shot_msg.data.nr = dc_photo_nr + 1;
147 dc_shot_msg.data.lat = stateGetPositionLla_i()->lat;
148 dc_shot_msg.data.lon = stateGetPositionLla_i()->lon;
149 dc_shot_msg.data.alt = stateGetPositionLla_i()->alt;
150 dc_shot_msg.data.phi = stateGetNedToBodyEulers_i()->phi;
151 dc_shot_msg.data.theta = stateGetNedToBodyEulers_i()->theta;
152 dc_shot_msg.data.psi = stateGetNedToBodyEulers_i()->psi;
153 dc_shot_msg.data.vground = stateGetHorizontalSpeedNorm_i();
154 dc_shot_msg.data.course = stateGetHorizontalSpeedDir_i();
155 dc_shot_msg.data.groundalt = POS_BFP_OF_REAL(state.alt_agl_f);
156
157 MoraHeader(MORA_SHOOT, MORA_SHOOT_MSG_SIZE);
158 for (int i = 0; i < (MORA_SHOOT_MSG_SIZE); i++) {
159 MoraPutUint8(dc_shot_msg.bin[i]);
160 }
161 MoraTrailer();
162 dc_send_shot_position();
163 break;
164 case DC_TALLER:
165 break;
166 case DC_WIDER:
167 break;
168 case DC_ON:
169 break;
170 case DC_OFF:
171 break;
172 default:
173 break;
174 }
175
176 // call command send_command function
177 dc_send_command_common(cmd);
178}
