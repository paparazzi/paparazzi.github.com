111bool nav_gvf_parametric_3D_lissajous_XYZ(float xo, float yo, float zo, float cx, float cy, float cz, float wx, float wy,
112 float wz, float dx, float dy, float dz, float alpha)
113{
114 // Safety first! If the asked altitude is low
115 if ((zo - cz) < 1) {
116 zo = 10;
117 cz = 0;
118 }
119
120 gvf_parametric_trajectory.type = LISSAJOUS_3D;
121 gvf_parametric_trajectory.p_parametric[0] = xo;
122 gvf_parametric_trajectory.p_parametric[1] = yo;
123 gvf_parametric_trajectory.p_parametric[2] = zo;
124 gvf_parametric_trajectory.p_parametric[3] = cx;
125 gvf_parametric_trajectory.p_parametric[4] = cy;
126 gvf_parametric_trajectory.p_parametric[5] = cz;
127 gvf_parametric_trajectory.p_parametric[6] = wx;
128 gvf_parametric_trajectory.p_parametric[7] = wy;
129 gvf_parametric_trajectory.p_parametric[8] = wz;
130 gvf_parametric_trajectory.p_parametric[9] = dx;
131 gvf_parametric_trajectory.p_parametric[10] = dy;
132 gvf_parametric_trajectory.p_parametric[11] = dz;
133 gvf_parametric_trajectory.p_parametric[12] = alpha;
134 gvf_parametric_trajectory.p_len = 13 + gvf_parametric_p_len_wps;
135 gvf_parametric_p_len_wps = 0;
136
137 float f1, f2, f3, f1d, f2d, f3d, f1dd, f2dd, f3dd;
138 float wb = gvf_parametric_control.w * gvf_parametric_control.beta * gvf_parametric_control.s;
139
140 gvf_parametric_3d_lissajous_info(&f1, &f2, &f3, &f1d, &f2d, &f3d, &f1dd, &f2dd, &f3dd, wb);
141 gvf_parametric_control_3D(gvf_parametric_3d_lissajous_par.kx, gvf_parametric_3d_lissajous_par.ky,
142 gvf_parametric_3d_lissajous_par.kz, f1, f2, f3, f1d, f2d, f3d, f1dd, f2dd, f3dd);
143
144 return true;
145}
