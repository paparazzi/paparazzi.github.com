43struct FloatRates stabilization_indi_rate_controller(struct FloatRates rates, struct FloatRates sp)
44{
45 struct FloatRates rate_error;
46 RATES_DIFF(rate_error, sp, rates);
47
48 struct FloatRates accel_ref;
49 if (autopilot_in_flight()) {
50 accel_ref.p = Cd * rate_state.p + Dd * rate_error.p;
51 rate_state.p = Ad * rate_state.p + Bd * rate_error.p;
52
53 accel_ref.q = Cd * rate_state.q + Dd * rate_error.q;
54 rate_state.q = Ad * rate_state.q + Bd * rate_error.q;
55
56 accel_ref.r = rate_error.r * indi_gains.rate.r;
57 } else {
58 FLOAT_RATES_ZERO(rate_state);
59 FLOAT_RATES_ZERO(accel_ref);
60 }
61
62 return accel_ref;
63}

70struct FloatRates WEAK stabilization_indi_attitude_controller(struct FloatQuat att, struct FloatQuat att_sp, struct FloatRates rates_ff)
71{
72 /* attitude error */
73 struct FloatQuat att_err;
74 float_quat_inv_comp_norm_shortest(&att_err, &att, &att_sp);
75
76 struct FloatVect3 att_fb;
77#if TILT_TWIST_CTRL
78 struct FloatQuat tilt;
79 struct FloatQuat twist;
80 float_quat_tilt_twist(&tilt, &twist, &att_err);
81 att_fb.x = tilt.qx;
82 att_fb.y = tilt.qy;
83 att_fb.z = twist.qz;
84#else
85 att_fb.x = att_err.qx;
86 att_fb.y = att_err.qy;
87 att_fb.z = att_err.qz;
88#endif
89
90 struct FloatRates rate_sp;
91 if (autopilot_in_flight()) {
92 rate_sp.p = Cp * att_state.p + Dp * att_fb.x;
93 att_state.p = Ap * att_state.p + Bp * att_fb.x;
94
95 rate_sp.q = Cp * att_state.q + Dp * att_fb.y;
96 att_state.q = Ap * att_state.q + Bp * att_fb.y;
97
98 rate_sp.r = indi_gains.att.r * att_fb.z / indi_gains.rate.r;
99
100 // add feed-forward term
101 RATES_ADD(rate_sp, rates_ff);
102 } else {
103 FLOAT_RATES_ZERO(att_state);
104 FLOAT_RATES_ZERO(rate_sp);
105 }
106
107 return rate_sp;
108}
