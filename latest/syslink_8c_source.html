109 A := A + D[i]

112 void syslink_compute_cksum(syslink_message_t *msg)
113 {
114  uint8_t a = 0, b = 0;
115  uint8_t *Di = (uint8_t *)msg, *end = Di + (2 + msg->length) * sizeof(uint8_t);
116 
117  while (Di < end) {
118  a = a + *Di;
119  b = b + a;

121  }
