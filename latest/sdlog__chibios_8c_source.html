196 void sdlog_chibios_finish(const bool flush)
197 {
198  if (pprzLogFile != -1) {
199  // if a FF_FS_REENTRANT is true, we can umount fs without closing
200  // file, fatfs lock will assure that umount is done after a write,
201  // and umount will close all open files cleanly. Thats the fatest
202  // way to umount cleanly filesystem.
203  //
204  // if FF_FS_REENTRANT is false,
205  // we have to flush and close files before unmounting filesystem
206 #if FF_FS_REENTRANT == 0
207  sdLogCloseAllLogs(flush);
208 #else
209  (void) flush;
210 #endif
211 
212  sdLogFinish();
213  pprzLogFile = -1;
214 #if FLIGHTRECORDER_SDLOG
215  flightRecorderLogFile = -1;
216 #endif
217  }
218  chibios_sdlog_status = SDLOG_STOPPED;
219 }

221 static void thd_startlog(void *arg)
222 {
223  (void) arg;
224  chRegSetThreadName("start log");
225  char tmpFilename[32];
226 
227  // Wait before starting the log if needed
228  chThdSleepSeconds(SDLOG_START_DELAY);
229  // Check if we are already in USB Storage mode
230  if (usbStorageIsItRunning()) {
231  chThdSleepSeconds(20000); // stuck here for hours FIXME stop the thread ?
232  }
233 
234  // Init sdlog struct
235  chibios_sdlog_init(&chibios_sdlog, &pprzLogFile);
236 
237  // Check for init errors
238  sdOk = true;
239 
240  if (sdLogInit(NULL) != SDLOG_OK) {
241  sdOk = false;
242  } else {
243  removeEmptyLogs(PPRZ_LOG_DIR, PPRZ_LOG_NAME, 50);
244  if (sdLogOpenLog(&pprzLogFile, PPRZ_LOG_DIR,
245  PPRZ_LOG_NAME, SDLOG_AUTO_FLUSH_PERIOD, LOG_APPEND_TAG_AT_CLOSE_DISABLED,
246  SDLOG_CONTIGUOUS_STORAGE_MEM, LOG_PREALLOCATION_DISABLED, tmpFilename, sizeof(tmpFilename)) != SDLOG_OK) {
247  sdOk = false;
248  }
249  chsnprintf(chibios_sdlog_filenames, sizeof(chibios_sdlog_filenames), "%s", tmpFilename);
250 #if FLIGHTRECORDER_SDLOG
251  removeEmptyLogs(FR_LOG_DIR, FLIGHTRECORDER_LOG_NAME, 50);
252  if (sdLogOpenLog(&flightRecorderLogFile, FR_LOG_DIR, FLIGHTRECORDER_LOG_NAME,
253  SDLOG_AUTO_FLUSH_PERIOD, LOG_APPEND_TAG_AT_CLOSE_DISABLED,
254  SDLOG_CONTIGUOUS_STORAGE_MEM, LOG_PREALLOCATION_DISABLED, tmpFilename, sizeof(tmpFilename)) != SDLOG_OK) {
255  sdOk = false;
256  }
257  chsnprintf(chibios_sdlog_filenames, sizeof(chibios_sdlog_filenames), "%s,%s", chibios_sdlog_filenames, tmpFilename);
258 #endif
259  }
260 
261  if (sdOk) {
262  // Create Battery Survey Thread with event
263  chEvtObjectInit(&powerOutageSource);
264  chThdCreateStatic(wa_thd_bat_survey, sizeof(wa_thd_bat_survey),
265  NORMALPRIO + 2, thd_bat_survey, NULL);
266 
267  chibios_sdlog_status = SDLOG_RUNNING;
268  } else {
269  chibios_sdlog_status = SDLOG_ERROR;
270  }
271 
272  while (true) {
273 #ifdef SDLOG_LED
274  LED_TOGGLE(SDLOG_LED);
275 #endif
276  // Blink faster if init has errors
277  chThdSleepMilliseconds(sdOk == true ? 1000 : 200);
278  if (sdLogGetStorageStatus() != SDLOG_OK) {
279  chibios_sdlog_status = SDLOG_ERROR;
280  sdOk = false;
281  } else {
282  chibios_sdlog_status = SDLOG_RUNNING;
283  sdOk = true;
284  }
285 
286 #if HAL_USE_RTC && USE_GPS
287  static uint32_t timestamp = 0;
288  // FIXME this could be done somewhere else, like in sys_time
289  // we sync gps time to rtc every 5 seconds
290  if (chVTGetSystemTime() - timestamp > TIME_S2I(5)) {
291  timestamp = chVTGetSystemTime();
292  if (gps.tow != 0) {
293  // Unix timestamp of the GPS epoch 1980-01-06 00:00:00 UTC
294  const uint32_t unixToGpsEpoch = 315964800;
295  struct tm time_tm;
296  time_t univTime = ((gps.week * 7 * 24 * 3600) + (gps.tow / 1000)) + unixToGpsEpoch;
297  gmtime_r(&univTime, &time_tm);
298  // Chibios date struct
299  RTCDateTime date;
300  rtcConvertStructTmToDateTime(&time_tm, 0, &date);
301  rtcSetTime(&RTCD1, &date);
302  }
303  }
304 #endif
305 
306  }
307 }

310 static void thd_bat_survey(void *arg)
311 {
312  (void)arg;
313  chRegSetThreadName("battery survey");
314  chEvtRegister(&powerOutageSource, &powerOutageListener, 1);
315  chThdSleepMilliseconds(2000);
316 
317  register_adc_watchdog(&SDLOG_BAT_ADC, SDLOG_BAT_CHAN, V_ALERT, &powerOutageIsr);
318 
319  chEvtWaitOne(EVENT_MASK(1));
320  // Only try to energy save is it is really a problem and not powered through USB
321  if (palReadPad(SDLOG_USB_VBUS_PORT, SDLOG_USB_VBUS_PIN) == PAL_LOW) {
322  // disable all required periph to save energy and maximize chance to flush files
323  // to mass storage and avoid infamous dirty bit on filesystem
324  mcu_energy_save();
325  }
326 
327  // in case of powerloss, we should go fast and avoid to flush ram buffer
328  sdlog_chibios_finish(false);
329 
330  // Only put to deep sleep in case there is no power on the USB
331  if (palReadPad(SDLOG_USB_VBUS_PORT, SDLOG_USB_VBUS_PIN) == PAL_LOW) {
332  mcu_reboot(MCU_REBOOT_POWEROFF);
333  }
334  chThdExit(0);
335  while (true); // never goes here, only to avoid compiler warning: 'noreturn' function does return
336 }

347 void logger_log_msg_up(uint8_t* buf) {
348  uint8_t ac_id = pprzlink_get_DL_INFO_MSG_UP_ac_id(buf);
349  if(ac_id != AC_ID && ac_id != 0xFF) {
350  return;
351  }
352  uint8_t fd = pprzlink_get_DL_INFO_MSG_UP_fd(buf);
353  if(pprzLogFile != -1 && (fd == DEST_INFO_MSG_ALL || fd == DEST_INFO_MSG_PPRZLOG)) {
354  uint8_t len = pprzlink_get_INFO_MSG_UP_msg_length(buf);
355  uint8_t* msg = (uint8_t*) pprzlink_get_DL_INFO_MSG_UP_msg(buf);
356  sdLogWriteRaw(pprzLogFile, msg, len);
357  sdLogWriteByte(pprzLogFile, '\n');
358  }
