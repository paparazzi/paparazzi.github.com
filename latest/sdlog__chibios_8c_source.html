142static int sdlog_check_free_space(struct chibios_sdlog *p __attribute__((unused)), long *fd, uint16_t len)
143{
144 SdLogBuffer *sdb;
145 SdioError status = sdLogAllocSDB(&sdb, len);
146 if (status != SDLOG_OK) {
147 return 0;
148 } else {
149 *fd = (long) sdb;
150 return 1;
151 }
152}

154static void sdlog_transmit(struct chibios_sdlog *p __attribute__((unused)), long fd, uint8_t byte)
155{
156 SdLogBuffer *sdb = (SdLogBuffer *) fd;
157 uint8_t *data = (uint8_t *) sdLogGetBufferFromSDB(sdb);
158 *data = byte;
159 sdLogSeekBufferFromSDB(sdb, 1);
160}
161
162static void sdlog_transmit_buffer(struct chibios_sdlog *p __attribute__((unused)), long fd, uint8_t *data, uint16_t len)
163{
164 SdLogBuffer *sdb = (SdLogBuffer *) fd;
165 memcpy(sdLogGetBufferFromSDB(sdb), data, len);
166 sdLogSeekBufferFromSDB(sdb, len);
167}

169static void sdlog_send(struct chibios_sdlog *p, long fd)
170{
171 SdLogBuffer *sdb = (SdLogBuffer *) fd;
172 sdLogWriteSDB(*(p->file), sdb);
173}

179{

185 sdlog->device.put_byte = (put_byte_t) sdlog_transmit;

213void sdlog_chibios_finish(const bool flush)
214{
215 if (pprzLogFile != -1) {
216 // if a FF_FS_REENTRANT is true, we can umount fs without closing
217 // file, fatfs lock will assure that umount is done after a write,
218 // and umount will close all open files cleanly. Thats the fatest
219 // way to umount cleanly filesystem.
220 //
221 // if FF_FS_REENTRANT is false,
222 // we have to flush and close files before unmounting filesystem
223#if FF_FS_REENTRANT == 0
224 sdLogCloseAllLogs(flush);
225#else
226 (void) flush;
227#endif
228
229 sdLogFinish();
230 pprzLogFile = -1;
231#if FLIGHTRECORDER_SDLOG
232 flightRecorderLogFile = -1;
233#endif
234 }
235 chibios_sdlog_status = SDLOG_STOPPED;
236}

245static void powerOutageIsr(void);

249#define DefaultAdcOfVoltage(voltage) ((uint32_t) (voltage/(DefaultVoltageOfAdc(1))))
250static const uint16_t V_ALERT = DefaultAdcOfVoltage(5.5f);

253/*
254 powerOutageIsr is called within a lock zone from an isr, so no lock/unlock is needed
255*/
256static void powerOutageIsr(void)
257{
258 chEvtBroadcastI(&powerOutageSource);
259}

261static void thd_bat_survey(void *arg)
262{
263 (void)arg;
264 chRegSetThreadName("battery survey");
265 chEvtRegister(&powerOutageSource, &powerOutageListener, 1);
266 chThdSleepMilliseconds(2000);
267
268 register_adc_watchdog(&SDLOG_BAT_ADC, SDLOG_BAT_CHAN, V_ALERT, &powerOutageIsr);
269
270 chEvtWaitOne(EVENT_MASK(1));
271 // Only try to energy save is it is really a problem and not powered through USB
272 if (palReadPad(SDLOG_USB_VBUS_PORT, SDLOG_USB_VBUS_PIN) == PAL_LOW) {
273 // disable all required periph to save energy and maximize chance to flush files
274 // to mass storage and avoid infamous dirty bit on filesystem
275 mcu_energy_save();
276 }
277
278 // in case of powerloss, we should go fast and avoid to flush ram buffer
279 sdlog_chibios_finish(false);
280
281 // Only put to deep sleep in case there is no power on the USB
282 if (palReadPad(SDLOG_USB_VBUS_PORT, SDLOG_USB_VBUS_PIN) == PAL_LOW) {
283 mcu_reboot(MCU_REBOOT_POWEROFF);
284 }
285 chThdExit(0);
286 while (true); // never goes here, only to avoid compiler warning: 'noreturn' function does return
287}
288#endif // SDLOG_ENABLE_LOWBAT_FLUSH
289#endif

291static void thd_startlog(void *arg)
292{
293 (void) arg;
294 chRegSetThreadName("start log");
295 char tmpFilename[32];
296
297 // Wait before starting the log if needed
298 chThdSleepSeconds(SDLOG_START_DELAY);
299 // Check if we are already in USB Storage mode
300 if (usbStorageIsItRunning()) {
301 chThdSleepSeconds(20000); // stuck here for hours FIXME stop the thread ?
302 }
303
304 // Init sdlog struct
305 chibios_sdlog_init(&chibios_sdlog, &pprzLogFile);
306
307 // Check for init errors
308 sdOk = true;
309
310 if (sdLogInit(NULL) != SDLOG_OK) {
311 sdOk = false;
312 } else {
313 removeEmptyLogs(PPRZ_LOG_DIR, PPRZ_LOG_NAME, 50);
314 if (sdLogOpenLog(&pprzLogFile, PPRZ_LOG_DIR,
315 PPRZ_LOG_NAME, SDLOG_AUTO_FLUSH_PERIOD, LOG_APPEND_TAG_AT_CLOSE_DISABLED,
316 SDLOG_CONTIGUOUS_STORAGE_MEM, LOG_PREALLOCATION_DISABLED, tmpFilename, sizeof(tmpFilename)) != SDLOG_OK) {
317 sdOk = false;
318 }
319 chsnprintf(chibios_sdlog_filenames, sizeof(chibios_sdlog_filenames), "%s", tmpFilename);
320#if FLIGHTRECORDER_SDLOG
321 removeEmptyLogs(FR_LOG_DIR, FLIGHTRECORDER_LOG_NAME, 50);
322 if (sdLogOpenLog(&flightRecorderLogFile, FR_LOG_DIR, FLIGHTRECORDER_LOG_NAME,
323 SDLOG_AUTO_FLUSH_PERIOD, LOG_APPEND_TAG_AT_CLOSE_DISABLED,
324 SDLOG_CONTIGUOUS_STORAGE_MEM, LOG_PREALLOCATION_DISABLED, tmpFilename, sizeof(tmpFilename)) != SDLOG_OK) {
325 sdOk = false;
326 }
327 chsnprintf(chibios_sdlog_filenames, sizeof(chibios_sdlog_filenames), "%s,%s", chibios_sdlog_filenames, tmpFilename);
328#endif
329 }
330
331 if (sdOk) {
332#if defined(SDLOG_BAT_ADC) && defined(SDLOG_BAT_CHAN) && SDLOG_ENABLE_LOWBAT_FLUSH
333 // Create Battery Survey Thread with event
334 chEvtObjectInit(&powerOutageSource);
335 chThdCreateStatic(wa_thd_bat_survey, sizeof(wa_thd_bat_survey),
336 NORMALPRIO + 2, thd_bat_survey, NULL);
337#endif
338
339 chibios_sdlog_status = SDLOG_RUNNING;
340 } else {
341 chibios_sdlog_status = SDLOG_ERROR;
342 }
343
344 while (true) {
345#ifdef SDLOG_LED
346 LED_TOGGLE(SDLOG_LED);
347#endif
348 // Blink faster if init has errors
349 chThdSleepMilliseconds(sdOk == true ? 1000 : 200);
350 if (sdLogGetStorageStatus() != SDLOG_OK) {
351 chibios_sdlog_status = SDLOG_ERROR;
352 sdOk = false;
353 } else {
354 chibios_sdlog_status = SDLOG_RUNNING;
355 sdOk = true;
356 }
357
358#if HAL_USE_RTC && USE_GPS
359 static uint32_t timestamp = 0;
360 // FIXME this could be done somewhere else, like in sys_time
361 // we sync gps time to rtc every 5 seconds
362 if (chVTGetSystemTime() - timestamp > TIME_S2I(5)) {
363 timestamp = chVTGetSystemTime();
364 if (gps.tow != 0) {
365 // Unix timestamp of the GPS epoch 1980-01-06 00:00:00 UTC
366 const uint32_t unixToGpsEpoch = 315964800;
367 struct tm time_tm;
368 time_t univTime = ((gps.week * 7 * 24 * 3600) + (gps.tow / 1000)) + unixToGpsEpoch;
369 gmtime_r(&univTime, &time_tm);
370 // Chibios date struct
371 RTCDateTime date;
372 rtcConvertStructTmToDateTime(&time_tm, 0, &date);
373 rtcSetTime(&RTCD1, &date);
374 }
375 }
376#endif
377
378 }
379}

382void logger_log_msg_up(uint8_t* buf) {
383 uint8_t ac_id = pprzlink_get_DL_INFO_MSG_UP_ac_id(buf);
384 if(ac_id != AC_ID && ac_id != 0xFF) {
385 return;
386 }
387 uint8_t fd = pprzlink_get_DL_INFO_MSG_UP_fd(buf);
388 if(pprzLogFile != -1 && (fd == DEST_INFO_MSG_ALL || fd == DEST_INFO_MSG_PPRZLOG)) {
389 uint8_t len = pprzlink_get_INFO_MSG_UP_msg_length(buf);
390 uint8_t* msg = (uint8_t*) pprzlink_get_DL_INFO_MSG_UP_msg(buf);
391 sdLogWriteRaw(pprzLogFile, msg, len);
392 sdLogWriteByte(pprzLogFile, '\n');
393 }
394}
