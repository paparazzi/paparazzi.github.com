43#define SATURATE_SPEED_TRIM_ACCEL(_rate, _accel, _max_rate) { \
44 if ((_rate) >= (_max_rate)) { \
45 (_rate) = (_max_rate); \
46 if ((_accel) > 0) { \
47 (_accel) = 0; \
