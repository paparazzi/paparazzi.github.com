52void gps_udp_parse(void)
53{
54 if (gps_network == NULL) { return; }
55
56 //Read from the network
57 int size = network_read(gps_network, &gps_udp_read_buffer[0], 256);
58
59 if (size > 0) {
60 // Correct MSG
61 if ((size == GPS_UDP_MSG_LEN) && (gps_udp_read_buffer[0] == STX)) {
62 gps_udp.lla_pos.lat = UDP_GPS_INT(gps_udp_read_buffer + 4);
63 gps_udp.lla_pos.lon = UDP_GPS_INT(gps_udp_read_buffer + 8);
64 gps_udp.lla_pos.alt = UDP_GPS_INT(gps_udp_read_buffer + 12);
65 SetBit(gps_udp.valid_fields, GPS_VALID_POS_LLA_BIT);
66
67 gps_udp.hmsl = UDP_GPS_INT(gps_udp_read_buffer + 16);
68 SetBit(gps_udp.valid_fields, GPS_VALID_HMSL_BIT);
69
70 gps_udp.ecef_pos.x = UDP_GPS_INT(gps_udp_read_buffer + 20);
71 gps_udp.ecef_pos.y = UDP_GPS_INT(gps_udp_read_buffer + 24);
72 gps_udp.ecef_pos.z = UDP_GPS_INT(gps_udp_read_buffer + 28);
73 SetBit(gps_udp.valid_fields, GPS_VALID_POS_ECEF_BIT);
74
75 gps_udp.ecef_vel.x = UDP_GPS_INT(gps_udp_read_buffer + 32);
76 gps_udp.ecef_vel.y = UDP_GPS_INT(gps_udp_read_buffer + 36);
77 gps_udp.ecef_vel.z = UDP_GPS_INT(gps_udp_read_buffer + 40);
78 SetBit(gps_udp.valid_fields, GPS_VALID_VEL_ECEF_BIT);
79
80 gps_udp.fix = GPS_FIX_3D;
81
82 // publish new GPS data
83 uint32_t now_ts = get_sys_time_usec();
84 gps_udp.last_msg_ticks = sys_time.nb_sec_rem;
85 gps_udp.last_msg_time = sys_time.nb_sec;
86 if (gps_udp.fix == GPS_FIX_3D) {
87 gps_udp.last_3dfix_ticks = sys_time.nb_sec_rem;
88 gps_udp.last_3dfix_time = sys_time.nb_sec;
89 }
90 AbiSendMsgGPS(GPS_UDP_ID, now_ts, &gps_udp);
91
92 } else {
93 printf("gps_udp error: msg len invalid %d bytes\n", size);
94 }
95 memset(&gps_udp_read_buffer[0], 0, sizeof(gps_udp_read_buffer));
96 }
97}
