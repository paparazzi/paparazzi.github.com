50void srf08_init(void)
51{
52 srf08_received = false;
53 srf08_got = false;
54
55 srf_trans.buf[0] = 0x00;
56 srf_trans.buf[1] = 0x51;
57 i2c_transmit(&SRF08_I2C_DEV, &srf_trans, SRF08_UNIT_0, 2);
58
60 srf_trans.buf[0] = SRF08_SET_GAIN;
61 srf_trans.buf[1] = SRF08_MIN_GAIN;
62 i2c_transmit(&SRF08_I2C_DEV, &srf_trans, SRF08_UNIT_0, 2);
63
64 return;
65}

77void srf08_receive(void)
78{

82 i2c_transmit(&SRF08_I2C_DEV, &srf_trans, SRF08_UNIT_0, 1);

107uint32_t srf08_read_register(uint8_t srf08_register)
108{
109 uint8_t cnt;
110
111 union i2c_union {
112 uint32_t rx_word;
113 uint8_t rx_byte[2];
114 } i2c;
115
116
117 srf_trans.buf[0] = srf08_register;
118
119 /* get high byte msb first */
120 if (srf08_register >= 2) {
121 cnt = 2;
122 } else {
123 cnt = 1;
124 }
125
126 i2c_transceive(&SRF08_I2C_DEV, &srf_trans, SRF08_UNIT_0, 1, cnt);
127
128 /* get high byte msb first */
129 if (srf08_register >= 2) {
130 i2c.rx_byte[1] = srf_trans.buf[1];
131 }
132
133 /* get low byte msb first */
134 i2c.rx_byte[0] = srf_trans.buf[0];
135
136 return (i2c.rx_word);
137}

139void srf08_event(void)
140{
141 float f = 0;
142 uint8_t i = 0;
143
145 if (srf_trans.status == I2CTransSuccess) {
146 if (srf08_received) {
147 srf08_received = false;
148 srf08_read();
149 } else if (srf08_got) {
150 srf08_got = false;
151 srf08_copy();
152 DOWNLINK_SEND_RANGEFINDER(DefaultChannel, DefaultDevice, &srf08_range, &f, &f, &f, &f, &f, &i);
153 }
154 }
