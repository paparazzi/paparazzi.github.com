80static void MPPT_ask(void)
81{
82 data_index++;
83 if (data_index >= NB_I2C_DATA) {
84 /* Setting the current value */
85 fbw_current_milliamp = MPPT_data[MPPT_IBAT_INDEX];
86
87 MPPT_data[MPPT_ITOTAL_INDEX] = MPPT_data[MPPT_IBAT_INDEX] + MPPT_data[MPPT_ICONV_INDEX];
88 DOWNLINK_SEND_MPPT(DefaultChannel, DefaultDevice, NB_DATA, MPPT_data);
89 data_index = 0;
90 }
91
92 mppt_trans.buf[0] = data_index;
93 i2c_transmit(&i2c0, &mppt_trans, MPPT_SLAVE_ADDR, 1);
94 MPPT_status = MPPT_STATUS_ASKING;
95}

97void MPPT_periodic(void)
98{
99
100 if (mppt_trans.status == I2CTransSuccess) {
101 switch (MPPT_status) {
102 case MPPT_STATUS_IDLE:
103 /* If free, change mode if needed */
104 if (MPPT_mode) {
105 mppt_trans.buf[0] = MPPT_MODE_ADDR;
106 mppt_trans.buf[1] = 0;
107 mppt_trans.buf[2] = MPPT_mode;
108 i2c_transmit(&i2c0, &mppt_trans, MPPT_SLAVE_ADDR, 3);
109 MPPT_mode = 0;
110 MPPT_status = MPPT_STATUS_WRITING;
111 } else {
112 MPPT_ask();
113 }
114 break;
115
116 case MPPT_STATUS_WRITING:
117 MPPT_status = MPPT_STATUS_IDLE;
118 break;
119
120 case MPPT_STATUS_ASKING:
121 /* The slave should send 2 bytes */
122 i2c_receive(&i2c0, &mppt_trans, MPPT_SLAVE_ADDR, 2);
123 MPPT_status = MPPT_STATUS_READING;
124 break;
125
126 case MPPT_STATUS_READING:
127 /* We got 2 bytes */
128 if (data_index < NB_I2C_DATA) {
129 MPPT_data[data_index] = (mppt_trans.buf[0] << 8) | mppt_trans.buf[1];
130 }
131 MPPT_status = MPPT_STATUS_IDLE;
132 break;
133 }
134 }
135}
