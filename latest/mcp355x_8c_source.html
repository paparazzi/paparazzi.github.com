60 void mcp355x_event(void)
61 {
62  if (mcp355x_spi_trans.status == SPITransSuccess) {
63  if ((mcp355x_spi_trans.input_buf[0] >> 4) == 0) {
64  mcp355x_data = (int32_t)(
65  ((uint32_t)mcp355x_spi_trans.input_buf[0] << 17) |
66  ((uint32_t)mcp355x_spi_trans.input_buf[1] << 9) |
67  ((uint32_t)mcp355x_spi_trans.input_buf[2] << 1) |
68  (mcp355x_spi_trans.input_buf[3] >> 7));
69  mcp355x_data_available = true;
70  }
71  mcp355x_spi_trans.status = SPITransDone;
72  }
73  if (mcp355x_spi_trans.status == SPITransFailed) {
74  mcp355x_spi_trans.status = SPITransDone;
