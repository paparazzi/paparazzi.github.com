74static void dw1000_range_arduino_parse(struct DW1000RangeArduino *dw, uint8_t c)
75{
76 switch (dw->state) {
77
78 case DW_WAIT_STX:
79 /* Waiting Synchro */
80 if (c == DW_STX) {
81 dw->idx = 0;
82 dw->ck = 0;
83 dw->state = DW_GET_DATA;
84 }
85 break;
86
87 case DW_GET_DATA:
88 /* Read Bytes */
89 dw->buf[dw->idx++] = c;
90 dw->ck += c;
91 if (dw->idx == DW_NB_DATA) {
92 dw->state = DW_GET_CK;
93 }
94 break;
95
96 case DW_GET_CK:
97 /* Checksum */
98 if (dw->ck == c) {
99 send_anchor_data(dw);
100 }
101 dw->state = DW_WAIT_STX;
102 break;
103
104 default:
105 dw->state = DW_WAIT_STX;
106 }
107}
