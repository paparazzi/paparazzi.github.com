128struct StabilizationSetpoint guidance_plane_attitude_from_nav(bool in_flight UNUSED)
129{
130 struct FloatEulers att_sp;
131
132 // course control loop
133 if (nav.horizontal_mode == NAV_HORIZONTAL_MODE_NONE || nav.horizontal_mode == NAV_HORIZONTAL_MODE_GUIDED) {
134 guidance_plane.roll_cmd = 0.f;
135 guidance_plane.pitch_cmd = 0.f;
136 } else if (nav.horizontal_mode == NAV_HORIZONTAL_MODE_ATTITUDE) {
137 if (nav.setpoint_mode == NAV_SETPOINT_MODE_QUAT) {
138 struct FloatEulers e;
139 float_eulers_of_quat(&e, &nav.quat);
140 guidance_plane.roll_cmd = e.phi;
141 guidance_plane.pitch_cmd = e.theta;
142 }
143 else {
144 guidance_plane.roll_cmd = nav.roll;
145 guidance_plane.pitch_cmd = nav.pitch;
146 }
147 } else {
148 // update carrot for GCS display and convert ENU float -> NED int
149 // even if sp is changed later
150 // FIXME really needed here ?
151 guidance_h.sp.pos.x = POS_BFP_OF_REAL(nav.carrot.y);
152 guidance_h.sp.pos.y = POS_BFP_OF_REAL(nav.carrot.x);
153
154 switch (nav.setpoint_mode) {
155 case NAV_SETPOINT_MODE_POS:
156 guidance_plane.course_setpoint = atan2f(nav.carrot.x - stateGetPositionEnu_f()->x, nav.carrot.y - stateGetPositionNed_f()->y);
157 break;
158 case NAV_SETPOINT_MODE_SPEED:
159 guidance_plane.course_setpoint = atan2f(nav.speed.x, nav.speed.y);
160 break;
161 case NAV_SETPOINT_MODE_ACCEL:
162 // not supported, flying towards HOME waypoint
163 guidance_plane.course_setpoint = atan2f(waypoint_get_x(WP_HOME) - stateGetPositionEnu_f()->x,
164 waypoint_get_y(WP_HOME) - stateGetPositionEnu_f()->y);
165 break;
166 default:
167 // nothing to do for other cases at the moment
168 break;
169 }
170 /* final attitude setpoint */
171
172 // Ground path error
173 static float last_err = 0.f;
174 float err = guidance_plane.course_setpoint - stateGetHorizontalSpeedDir_f();
175 NormRadAngle(err);
176 float d_err = err - last_err;
177 last_err = err;
178 NormRadAngle(d_err);
179 //float course_pre_bank = 0.f; // TODO: compute if flying a circle -> pb is that it is coming from nav
180 // guidance_plane.course_pre_bank_correction * course_pre_bank
181
182 guidance_plane.roll_cmd = guidance_plane.course_kp * err + guidance_plane.course_kd * d_err;
183 BoundAbs(guidance_plane.roll_cmd, guidance_plane.roll_max_setpoint);
184 }
185
186 att_sp.phi = guidance_plane.roll_cmd;
187 att_sp.theta = guidance_plane.pitch_cmd;
188 att_sp.psi = stateGetNedToBodyEulers_f()->psi;
189 return stab_sp_from_eulers_f(&att_sp);
190}

193static void guidance_plane_set_pitch(bool in_flight)
194{
195 static float last_err = 0.f;
196
197 if (!in_flight) {
198 pitch_sum_err = 0.f;
199 }
200
201 // Compute errors
202 float err = guidance_plane.climb_setpoint - stateGetSpeedEnu_f()->z;
203 float d_err = err - last_err;
204 last_err = err;
205
206 if (guidance_plane.p_ki > 0.) {
207 pitch_sum_err += err * (1. / PERIODIC_FREQUENCY);
208 BoundAbs(pitch_sum_err, GUIDANCE_PLANE_PITCH_MAX_SUM_ERR / guidance_plane.p_ki);
209 }
210
211 // PID loop + feedforward ctl
212 guidance_plane.pitch_cmd = nav.pitch
213 + guidance_plane.pitch_trim
214 + guidance_plane.pitch_of_vz * guidance_plane.climb_setpoint
215 + guidance_plane.p_kp * err
216 + guidance_plane.p_kd * d_err
217 + guidance_plane.p_ki * pitch_sum_err;
218
219}

221static void guidance_plane_set_throttle(bool in_flight)
222{
223 static float last_err = 0.;
224
225 if (!in_flight) {
226 throttle_sum_err = 0;
227 }
228
229 // Compute errors
230 float err = guidance_plane.climb_setpoint - stateGetSpeedEnu_f()->z;
231 float d_err = err - last_err;
232 last_err = err;
233
234 if (guidance_plane.t_ki > 0.) {
235 throttle_sum_err += err * (1. / PERIODIC_FREQUENCY);
236 BoundAbs(throttle_sum_err, GUIDANCE_PLANE_THROTTLE_MAX_SUM_ERR / guidance_plane.t_ki);
237 }
238
239 // PID loop + feedforward ctl
240 float th_cmd = guidance_plane.cruise_throttle
241 + guidance_plane.climb_throttle_increment * guidance_plane.climb_setpoint
242 + guidance_plane.t_kp * err
243 + guidance_plane.t_kd * d_err
244 + guidance_plane.t_ki * throttle_sum_err;
245
246 guidance_plane.throttle_cmd = TRIM_UPPRZ(MAX_PPRZ * th_cmd);
247
248}

253struct ThrustSetpoint guidance_plane_thrust_from_nav(bool in_flight)
254{
255 if (nav.vertical_mode == NAV_VERTICAL_MODE_MANUAL) {
256 guidance_plane.throttle_cmd = (int32_t) nav.throttle;
257 } else {
258 if (nav.vertical_mode == NAV_VERTICAL_MODE_ALT) {
259 guidance_plane.altitude_setpoint = nav.nav_altitude;
260 float alt_err = guidance_plane.altitude_setpoint - stateGetPositionEnu_f()->z;
261 guidance_plane.climb_setpoint = guidance_plane.climb_kp * alt_err;
262 } else if (nav.vertical_mode == NAV_VERTICAL_MODE_CLIMB) {
263 guidance_plane.climb_setpoint = nav.climb;
264 } else {
265 guidance_plane.climb_setpoint = 0.f;
266 }
267 BoundAbs(guidance_plane.climb_setpoint, guidance_plane.climb_max_setpoint);
268
269 guidance_plane_set_pitch(in_flight);
270 guidance_plane_set_throttle(in_flight);
271 }
272
273 return th_sp_from_thrust_i(guidance_plane.throttle_cmd, THRUST_AXIS_X);
274}
