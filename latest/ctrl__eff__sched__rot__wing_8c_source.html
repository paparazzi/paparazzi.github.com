172 static void wing_position_cb(uint8_t sender_id UNUSED, struct act_feedback_t *pos_msg, uint8_t num_act)
173 {
174  for (int i=0; i<num_act; i++){
175  if (pos_msg[i].set.position && (pos_msg[i].idx == SERVO_ROTATION_MECH))
176  {
177  // Get wing rotation angle from sensor
178  eff_sched_var.wing_rotation_rad = 0.5 * M_PI - pos_msg[i].position;
179 
180  // Bound wing rotation angle
181  Bound(eff_sched_var.wing_rotation_rad, 0, 0.5 * M_PI);
182  }
183  }

233 void ctrl_eff_sched_rot_wing_update_wing_angle(void)
234 {
235  // Calculate sin and cosines of rotation
236  eff_sched_var.wing_rotation_deg = eff_sched_var.wing_rotation_rad / M_PI * 180.;
237 
238  eff_sched_var.cosr = cosf(eff_sched_var.wing_rotation_rad);
239  eff_sched_var.sinr = sinf(eff_sched_var.wing_rotation_rad);
240 
241  eff_sched_var.cosr2 = eff_sched_var.cosr * eff_sched_var.cosr;
242  eff_sched_var.sinr2 = eff_sched_var.sinr * eff_sched_var.sinr;
243 
244  eff_sched_var.sinr3 = eff_sched_var.sinr2 * eff_sched_var.sinr;
245 
246 }

248 void ctrl_eff_sched_rot_wing_update_MMOI(void)
249 {
250  eff_sched_var.Ixx = eff_sched_p.Ixx_body + eff_sched_var.cosr2 * eff_sched_p.Ixx_wing + eff_sched_var.sinr2 * eff_sched_p.Iyy_wing;
251  eff_sched_var.Iyy = eff_sched_p.Iyy_body + eff_sched_var.sinr2 * eff_sched_p.Ixx_wing + eff_sched_var.cosr2 * eff_sched_p.Iyy_wing;
252 
253  // Bound inertia
254  Bound(eff_sched_var.Ixx, 0.01, 100.);
255  Bound(eff_sched_var.Iyy, 0.01, 100.);
256 }

273 void ctrl_eff_sched_rot_wing_update_hover_motor_effectiveness(void)
274 {
275  // Pitch motor effectiveness
276 
277  float pitch_motor_q_eff = eff_sched_var.pitch_motor_dMdpprz / eff_sched_var.Iyy;
278 
279  // Roll motor effectiveness
280 
281  float roll_motor_p_eff = (eff_sched_var.roll_motor_dMdpprz * eff_sched_var.cosr + eff_sched_p.hover_roll_roll_coef[0] * eff_sched_var.wing_rotation_rad * eff_sched_var.wing_rotation_rad * eff_sched_var.airspeed * eff_sched_var.cosr) / eff_sched_var.Ixx;
282  Bound(roll_motor_p_eff, 0.00001, 1);
283  float roll_motor_airspeed_compensation = eff_sched_p.hover_roll_roll_coef[1] * eff_sched_var.airspeed * eff_sched_var.cosr / eff_sched_var.Ixx;
284  float roll_motor_p_eff_compensated = roll_motor_p_eff + roll_motor_airspeed_compensation;
285  Bound(roll_motor_p_eff_compensated, 0.00001, 1);
286 
287  float roll_motor_q_eff = (eff_sched_p.hover_roll_pitch_coef[0] * eff_sched_var.wing_rotation_rad + eff_sched_p.hover_roll_pitch_coef[1] * eff_sched_var.wing_rotation_rad * eff_sched_var.wing_rotation_rad * eff_sched_var.sinr) / eff_sched_var.Iyy;
288  Bound(roll_motor_q_eff, 0, 1);
289 
290  // Update front pitch motor q effectiveness
291  g1g2[1][0] = pitch_motor_q_eff; // pitch effectiveness front motor
292 
293  // Update back motor q effectiveness
294  g1g2[1][2] = -pitch_motor_q_eff; // pitch effectiveness back motor
295 
296  // Update right motor p and q effectiveness
297  g1g2[0][1] = -roll_motor_p_eff; // roll effectiveness right motor (no airspeed compensation)
298  g1g2[1][1] = roll_motor_q_eff; // pitch effectiveness right motor
299 
300  // Update left motor p and q effectiveness
301  g1g2[0][3] = roll_motor_p_eff_compensated; // roll effectiveness left motor
302  g1g2[1][3] = -roll_motor_q_eff; // pitch effectiveness left motor
303 }

305 void ctrl_eff_sched_rot_wing_update_elevator_effectiveness(void)
306 {
307  float de = eff_sched_p.k_elevator_deflection[0] + eff_sched_p.k_elevator_deflection[1] * eff_sched_var.cmd_elevator;
308 
309  float dMyde = (eff_sched_p.k_elevator[0] * de * eff_sched_var.airspeed2 +
310  eff_sched_p.k_elevator[1] * eff_sched_var.cmd_pusher_scaled * eff_sched_var.cmd_pusher_scaled * eff_sched_var.airspeed +
311  eff_sched_p.k_elevator[2] * eff_sched_var.airspeed2) / 10000.;
312 
313  float dMydpprz = dMyde * eff_sched_p.k_elevator_deflection[1];
314 
315  // Convert moment to effectiveness
316  float eff_y_elev = dMydpprz / eff_sched_var.Iyy;
317 
318  Bound(eff_y_elev, 0.00001, 0.1);
319 
320  g1g2[1][5] = eff_y_elev;
321 }

323 void ctrl_eff_sched_rot_wing_update_rudder_effectiveness(void)
324 {

334  float eff_z_rudder = dMzdpprz / eff_sched_p.Izz;

358 {

365

372 void ctrl_eff_sched_rot_wing_schedule_liftd(void)
373 {
374  float lift_d_wing = (eff_sched_p.k_lift_wing[0] + eff_sched_p.k_lift_wing[1] * eff_sched_var.sinr2) * eff_sched_var.airspeed2 / eff_sched_p.m;

379  if (eff_sched_var.wing_rotation_deg < 60.) {
380  lift_d = 0.0;
381  }
382  Bound(lift_d, -130., 0.);
383  ctrl_eff_sched_rot_wing_lift_d = lift_d;
