123 static void wing_position_cb(uint8_t sender_id UNUSED, struct act_feedback_t *pos_msg, uint8_t num_act)
124 {
125  for (int i=0; i<num_act; i++){
126  if (pos_msg[i].set.position && (pos_msg[i].idx == SERVO_ROTATION_MECH))
127  {
128  // Get wing rotation angle from sensor

144 {

147  eff_sched_var.Iyy = eff_sched_p.Iyy_body + eff_sched_p.Iyy_wing;
148  eff_sched_var.wing_rotation_rad = 0;
149  eff_sched_var.cosr = 1;

187 void ctrl_eff_sched_rot_wing_update_wing_angle(void)
188 {
189  // Calculate sin and cosines of rotation
190  eff_sched_var.cosr = cosf(eff_sched_var.wing_rotation_rad);
191  eff_sched_var.sinr = sinf(eff_sched_var.wing_rotation_rad);
192 
193  eff_sched_var.cosr2 = eff_sched_var.cosr * eff_sched_var.cosr;
194  eff_sched_var.sinr2 = eff_sched_var.sinr * eff_sched_var.sinr;
195 
196  eff_sched_var.cosr3 = eff_sched_var.cosr2 * eff_sched_var.cosr;
197 }

199 void ctrl_eff_sched_rot_wing_update_MMOI(void)
200 {
201  eff_sched_var.Ixx = eff_sched_p.Ixx_body + eff_sched_var.cosr2 * eff_sched_p.Ixx_wing + eff_sched_var.sinr2 * eff_sched_p.Iyy_wing;
202  eff_sched_var.Iyy = eff_sched_p.Iyy_body + eff_sched_var.sinr2 * eff_sched_p.Ixx_wing + eff_sched_var.cosr2 * eff_sched_p.Iyy_wing;
203 
204  // Bound inertia
205  Bound(eff_sched_var.Ixx, 0.01, 100.);
206  Bound(eff_sched_var.Iyy, 0.01, 100.);

209 void ctrl_eff_sched_rot_wing_update_hover_motor_effectiveness(void)
210 {
211  // Pitch motor effectiveness
212 
213  float pitch_motor_q_eff = eff_sched_var.pitch_motor_dMdpprz / eff_sched_var.Iyy;
214 
215  // Roll motor effectiveness
216 
217  float roll_motor_p_eff = eff_sched_var.roll_motor_dMdpprz * eff_sched_var.cosr / eff_sched_var.Ixx;
218  // float roll_motor_q_eff = eff_sched_var.roll_motor_dMdpprz * eff_sched_var.sinr *
219  // (eff_sched_p.hover_roll_pitch_coef[0] + eff_sched_p.hover_roll_pitch_coef[1] * eff_sched_var.cosr2) / eff_sched_var.Iyy;
220 
221  float roll_motor_q_eff = (eff_sched_var.roll_motor_dMdpprz * eff_sched_var.sinr / eff_sched_var.Iyy) * (eff_sched_p.hover_roll_pitch_coef[0] + eff_sched_p.hover_roll_pitch_coef[1] * eff_sched_var.cosr3);
222 
223  // Update front pitch motor q effectiveness
224  g1g2[1][0] = pitch_motor_q_eff; // pitch effectiveness front motor
225 
226  // Update back motor q effectiveness
227  g1g2[1][2] = -pitch_motor_q_eff; // pitch effectiveness back motor
228 
229  // Update right motor p and q effectiveness
230  g1g2[0][1] = -roll_motor_p_eff; // roll effectiveness right motor
231  g1g2[1][1] = roll_motor_q_eff; // pitch effectiveness right motor
232 
233  // Update left motor p and q effectiveness
234  g1g2[0][3] = roll_motor_p_eff; // roll effectiveness left motor
235  g1g2[1][3] = -roll_motor_q_eff; // pitch effectiveness left motor
236 }
