76static inline void autopilot_arming_set(bool motors_on)
77{
78 if (motors_on) {
79 autopilot_check_motor_status = STATUS_MOTORS_ON;
80 } else if (kill_switch_is_on() && (autopilot_check_motor_status == STATUS_MOTORS_ON)) {
81 autopilot_check_motor_status = STATUS_MOTORS_RC_KILLED;
82 motor_kill_time = get_sys_time_float();
83 } else {
84 autopilot_check_motor_status = STATUS_MOTORS_AUTOMATICALLY_OFF;
85 }
86}

96static inline bool autopilot_arming_check_valid(bool yaw_must_be_centered)
97{
98 if (!THROTTLE_STICK_DOWN()) {
99 autopilot.arming_status = AP_ARMING_STATUS_THROTTLE_NOT_DOWN;
100 } else if (!PITCH_STICK_CENTERED()) {
101 autopilot.arming_status = AP_ARMING_STATUS_PITCH_NOT_CENTERED;
102 } else if (!ROLL_STICK_CENTERED()) {
103 autopilot.arming_status = AP_ARMING_STATUS_ROLL_NOT_CENTERED;
104 } else {
105 if (yaw_must_be_centered && !YAW_STICK_CENTERED()) {
106 autopilot.arming_status = AP_ARMING_STATUS_YAW_NOT_CENTERED;
107 } else if (!yaw_must_be_centered && YAW_STICK_CENTERED()) {
108 autopilot.arming_status = AP_ARMING_STATUS_YAW_CENTERED;
109 } else {
110 return true; // all checks valid
111 }
112 }
113 return false; // one of the checks failed
114}

122static inline void autopilot_arming_check_motors_on(void)
123{
124 /* only allow switching motor if not in KILL mode */
125 if (autopilot_get_mode() != AP_MODE_KILL) {
126
127 switch (autopilot_check_motor_status) {
128 case STATUS_INITIALISE_RC: // Wait until RC is initialised (it being centered is a good pointer to this)
129 if (autopilot_arming_check_valid(YAW_MUST_BE_CENTERED)) {
130 autopilot_check_motor_status = STATUS_MOTORS_OFF;
131 }
132 break;
133 case STATUS_MOTORS_RC_KILLED: // Motors were killed by kill mode
134 // If the vehicle was killed accidentally, allow rapid re-arm
135 if ( (get_sys_time_float() - motor_kill_time) < MOTOR_RE_ARM_TIME) {
136 autopilot_check_motor_status = STATUS_MOTORS_ON;
137 autopilot.motors_on = true; // Bypass preflight-checks and force the motors on
138 } else {
139 autopilot_check_motor_status = STATUS_MOTORS_AUTOMATICALLY_OFF;
140 autopilot_arming_motors_on(false);
141 }
142 break;
143 case STATUS_MOTORS_AUTOMATICALLY_OFF: // Motors were disarmed externally
144 //(possibly due to crash)
145 //wait extra delay before enabling the normal arming state machine
146 autopilot_arming_motors_on(false);
147 autopilot_motors_on_counter = 0;
148 if (autopilot_arming_check_valid(YAW_MUST_BE_CENTERED)) {
149 autopilot_check_motor_status = STATUS_MOTORS_AUTOMATICALLY_OFF_SAFETY_WAIT;
150 }
151 break;
152 case STATUS_MOTORS_AUTOMATICALLY_OFF_SAFETY_WAIT:
153 autopilot_motors_on_counter++;
154 if (autopilot_motors_on_counter >= MOTOR_ARMING_DELAY) {
155 autopilot_check_motor_status = STATUS_MOTORS_OFF;
156 } else {
157 autopilot.arming_status = AP_ARMING_STATUS_WAITING;
158 }
159 break;
160 case STATUS_MOTORS_OFF:
161 autopilot_arming_motors_on(false);
162 autopilot_motors_on_counter = 0;
163 autopilot.arming_status = AP_ARMING_STATUS_WAITING;
164 if (autopilot_arming_check_valid(YAW_MUST_BE_PUSHED)) { // stick pushed
165 autopilot_check_motor_status = STATUS_M_OFF_STICK_PUSHED;
166 }
167 break;
168 case STATUS_M_OFF_STICK_PUSHED:
169 autopilot_arming_motors_on(false);
170 autopilot_motors_on_counter++;
171 if (autopilot_motors_on_counter >= MOTOR_ARMING_DELAY) {
172 autopilot_check_motor_status = STATUS_START_MOTORS;
173 } else if (!autopilot_arming_check_valid(YAW_MUST_BE_PUSHED)) { // stick released too soon
174 autopilot_check_motor_status = STATUS_MOTORS_OFF;
175 } else {
176 autopilot.arming_status = AP_ARMING_STATUS_WAITING;
177 }
178 break;
179 case STATUS_START_MOTORS:
180 autopilot_arming_motors_on(true);
181 autopilot_motors_on_counter = MOTOR_ARMING_DELAY;
182 autopilot_set_in_flight(false); // stop fc from starting control (integration and yaw) till arm process is complete
183 if (YAW_STICK_CENTERED()) { // wait until stick released
184 // Verify if motors are really on else go back to STATUS_MOTORS_OFF
185 if(autopilot_get_motors_on())
186 autopilot_check_motor_status = STATUS_MOTORS_ON;
187 else
188 autopilot_check_motor_status = STATUS_MOTORS_OFF;
189 }
190 break;
191 case STATUS_MOTORS_ON:
192 autopilot.arming_status = AP_ARMING_STATUS_ARMED;
193 autopilot_arming_motors_on(true);
194 autopilot_motors_on_counter = MOTOR_ARMING_DELAY;
195 if (THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED()) { // stick pushed
196 autopilot_check_motor_status = STATUS_M_ON_STICK_PUSHED;
197 }
198 break;
199 case STATUS_M_ON_STICK_PUSHED:
200 autopilot_arming_motors_on(true);
201 autopilot_motors_on_counter--;
202 if (autopilot_motors_on_counter == 0) {
203 autopilot_check_motor_status = STATUS_STOP_MOTORS;
204 } else if (!(THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED())) { // stick released too soon
205 autopilot_check_motor_status = STATUS_MOTORS_ON;
206 } else {
207 autopilot.arming_status = AP_ARMING_STATUS_DISARMING;
208 }
209 break;
210 case STATUS_STOP_MOTORS:
211 autopilot_arming_motors_on(false);
212 autopilot_motors_on_counter = 0;
213 if (autopilot_arming_check_valid(YAW_MUST_BE_CENTERED)) { // wait till release disarm stick before allowing to re-arm
214 autopilot_check_motor_status = STATUS_MOTORS_OFF;
215 }
216 break;
217 default:
218 break;
219 }
220 } else {
221 autopilot.arming_status = AP_ARMING_STATUS_KILLED;
222 if (kill_switch_is_on()) {
223 autopilot_arming_motors_on(false);
224 }
225 }
226}
