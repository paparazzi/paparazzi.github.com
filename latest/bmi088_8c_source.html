70void bmi088_set_default_config(struct Bmi088Config *c)
71{
72 c->gyro_range = BMI088_DEFAULT_GYRO_RANGE;
73 c->gyro_odr = BMI088_DEFAULT_GYRO_ODR;
74 c->accel_range = BMI088_DEFAULT_ACCEL_RANGE;

79void bmi088_send_config(Bmi088ConfigSet bmi_set, void *bmi, struct Bmi088Config *config)
80{
81 switch (config->init_status) {
82 case BMI088_CONF_ACCEL_RANGE:
83 /* configure accelerometer range */
84 bmi_set(bmi, BMI088_ACCEL_RANGE, config->accel_range, BMI088_CONFIG_ACCEL);
85 config->init_status++;
86 break;
87 case BMI088_CONF_ACCEL_ODR:
88 /* configure accelerometer odr and bw */
89 bmi_set(bmi, BMI088_ACCEL_CONF, ((config->accel_bw << 4) | config->accel_odr), BMI088_CONFIG_ACCEL);
90 config->init_status++;
91 break;
92 case BMI088_CONF_ACCEL_PWR_CONF:
93 /* switch on accel */
94 bmi_set(bmi, BMI088_ACCEL_PWR_CONF, BMI088_ACCEL_ACTIVE, BMI088_CONFIG_ACCEL);
95 config->init_status++;
96 break;
97 case BMI088_CONF_ACCEL_PWR_CTRL:
98 /* switch on accel */
99 bmi_set(bmi, BMI088_ACCEL_PWR_CTRl, BMI088_ACCEL_POWER_ON, BMI088_CONFIG_ACCEL);
100 config->init_status++;
101 break;
102 case BMI088_CONF_GYRO_RANGE:
103 /* configure gyro range */
104 bmi_set(bmi, BMI088_GYRO_RANGE, config->gyro_range, BMI088_CONFIG_GYRO);
105 config->init_status++;
106 break;
107 case BMI088_CONF_GYRO_ODR:
108 /* configure gyro odr */
109 bmi_set(bmi, BMI088_GYRO_BAND_WIDTH, config->gyro_odr, BMI088_CONFIG_GYRO);
110 config->init_status++;
111 break;
112 case BMI088_CONF_GYRO_PWR:
113 /* switch on gyro */
114 bmi_set(bmi, BMI088_GYRO_LPM_1, BMI088_GYRO_NORMAL, BMI088_CONFIG_GYRO);
115 config->init_status++;
116 break;
117 case BMI088_CONF_DONE:
118 /* Set the samplerates from the ODR */
119 switch(config->gyro_odr) {
120 case BMI088_GYRO_ODR_2000_BW_532:
121 case BMI088_GYRO_ODR_2000_BW_230:
122 config->gyro_samplerate = 2000;
123 break;
124 case BMI088_GYRO_ODR_1000_BW_116:
125 config->gyro_samplerate = 1000;
126 break;
127 case BMI088_GYRO_ODR_400_BW_47:
128 config->gyro_samplerate = 400;
129 break;
130 case BMI088_GYRO_ODR_200_BW_23:
131 case BMI088_GYRO_ODR_200_BW_64:
132 config->gyro_samplerate = 200;
133 break;
134 case BMI088_GYRO_ODR_100_BW_12:
135 case BMI088_GYRO_ODR_100_BW_32:
136 config->gyro_samplerate = 100;
137 break;
138 }
139 switch(config->accel_odr) {
140 case BMI088_ACCEL_ODR_1600:
141 config->accel_samplerate = 1600;
142 break;
143 case BMI088_ACCEL_ODR_800:
144 config->accel_samplerate = 800;
145 break;
146 case BMI088_ACCEL_ODR_400:
147 config->accel_samplerate = 400;
148 break;
149 case BMI088_ACCEL_ODR_200:
150 config->accel_samplerate = 200;
151 break;
152 case BMI088_ACCEL_ODR_100:
153 config->accel_samplerate = 100;
154 break;
155 case BMI088_ACCEL_ODR_50:
156 config->accel_samplerate = 50;
157 break;
158 case BMI088_ACCEL_ODR_25:
159 config->accel_samplerate = 25;
160 break;
161 case BMI088_ACCEL_ODR_12:
162 config->accel_samplerate = 12;
163 break;
164 }
165
166 config->initialized = true;
167 break;
168 default:
169 break;
170 }
