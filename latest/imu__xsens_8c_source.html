53 static void handle_ins_msg(void)
54 {
55  uint32_t now_ts = get_sys_time_usec();
56 #ifdef XSENS_BACKWARDS
57  if (xsens.gyro_available) {
58  RATES_ASSIGN(imu.gyro_unscaled, -RATE_BFP_OF_REAL(xsens.gyro.p), -RATE_BFP_OF_REAL(xsens.gyro.q), RATE_BFP_OF_REAL(xsens.gyro.r));
59  xsens.gyro_available = FALSE;
60  imu_scale_gyro(&imu);
61  AbiSendMsgIMU_GYRO_INT32(IMU_XSENS_ID, now_ts, &imu.gyro);
62  }
63  if (xsens.accel_available) {
64  VECT3_ASSIGN(imu.accel_unscaled, -ACCEL_BFP_OF_REAL(xsens.accel.ax), -ACCEL_BFP_OF_REAL(xsens.accel.ay), ACCEL_BFP_OF_REAL(xsens.accel.az));
65  xsens.accel_available = FALSE;
66  imu_scale_accel(&imu);
67  AbiSendMsgIMU_ACCEL_INT32(IMU_XSENS_ID, now_ts, &imu.accel);
68  }
69  if (xsens.mag_available) {
70  VECT3_ASSIGN(imu.mag_unscaled, -MAG_BFP_OF_REAL(xsens.mag.mx), -MAG_BFP_OF_REAL(xsens.mag.my), MAG_BFP_OF_REAL(xsens.mag.mz));
71  xsens.mag_available = FALSE;
72  imu_scale_mag(&imu);
73  AbiSendMsgIMU_MAG_INT32(IMU_XSENS_ID, now_ts, &imu.mag);
74  }
75 #else
76  if (xsens.gyro_available) {
77  RATES_BFP_OF_REAL(imu.gyro_unscaled, xsens.gyro);
78  xsens.gyro_available = FALSE;
79  imu_scale_gyro(&imu);
80  AbiSendMsgIMU_GYRO_INT32(IMU_XSENS_ID, now_ts, &imu.gyro);
81  }
82  if (xsens.accel_available) {
83  ACCELS_BFP_OF_REAL(imu.accel_unscaled, xsens.accel);
84  xsens.accel_available = FALSE;
85  imu_scale_accel(&imu);
86  AbiSendMsgIMU_ACCEL_INT32(IMU_XSENS_ID, now_ts, &imu.accel);
87  }
88  if (xsens.mag_available) {
89  MAGS_BFP_OF_REAL(imu.mag_unscaled, xsens.mag);
90  xsens.mag_available = FALSE;
91  imu_scale_mag(&imu);
92  AbiSendMsgIMU_MAG_INT32(IMU_XSENS_ID, now_ts, &imu.mag);
93  }
94 #endif /* XSENS_BACKWARDS */
95 }
