53static void handle_ins_msg(void)
54{
55 uint32_t now_ts = get_sys_time_usec();
56#ifdef XSENS_BACKWARDS
57 if (xsens.gyro_available) {
58 struct Int32Rates gyro = {
59 -RATE_BFP_OF_REAL(xsens.gyro.p),
60 -RATE_BFP_OF_REAL(xsens.gyro.q),
61 RATE_BFP_OF_REAL(xsens.gyro.r)
62 };
63 xsens.gyro_available = FALSE
64 AbiSendMsgIMU_GYRO_RAW(IMU_XSENS_ID, now_ts, &gyro, 1, NAN, NAN); //FIXME: samplerate?
65 }
66 if (xsens.accel_available) {
67 struct Int32Vect3 accel = {
68 -ACCEL_BFP_OF_REAL(xsens.accel.x),
69 -ACCEL_BFP_OF_REAL(xsens.accel.y),
70 ACCEL_BFP_OF_REAL(xsens.accel.z)
71 };
72 xsens.accel_available = FALSE;
73 AbiSendMsgIMU_ACCEL_RAW(IMU_XSENS_ID, now_ts, &accel, 1, NAN, NAN); //FIXME: samplerate?
74 }
75 if (xsens.mag_available) {
76 struct Int32Vect3 mag = {
77 -MAG_BFP_OF_REAL(xsens.mag.x),
78 -MAG_BFP_OF_REAL(xsens.mag.y),
79 MAG_BFP_OF_REAL(xsens.mag.z)

81 xsens.mag_available = FALSE;

83 }
84#else
85 if (xsens.gyro_available) {
86 struct Int32Rates gyro = {
87 RATE_BFP_OF_REAL(xsens.gyro.p),

92 xsens.gyro_available = FALSE;
93 }

95 struct Int32Vect3 accel = {
