146static inline float pprz_isa_height_of_pressure_full(float pressure, float ref_p)
147{
148 if (ref_p > 0.) {
149 const float prel = pressure / ref_p;
150 const float inv_expo = PPRZ_ISA_GAS_CONSTANT * PPRZ_ISA_TEMP_LAPS_RATE /
151 PPRZ_ISA_GRAVITY / PPRZ_ISA_MOLAR_MASS;
152 return (1 - powf(prel, inv_expo)) * PPRZ_ISA_SEA_LEVEL_TEMP / PPRZ_ISA_TEMP_LAPS_RATE;
153 } else {
154 return 0.;
155 }
156}

166static inline float pprz_isa_ref_pressure_of_height_full(float pressure, float height)
167{
168 // Trel = 1 - L*h/T0;
169 const float Trel = 1.0 - PPRZ_ISA_TEMP_LAPS_RATE * height / PPRZ_ISA_SEA_LEVEL_TEMP;
170 const float expo = PPRZ_ISA_GRAVITY * PPRZ_ISA_MOLAR_MASS / PPRZ_ISA_GAS_CONSTANT /
171 PPRZ_ISA_TEMP_LAPS_RATE;
172 return pressure / pow(Trel, expo);
173}
