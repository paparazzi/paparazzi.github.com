91uint8_t battery_monitor_get_address(uint8_t chan) {
92 if ((chan>8) || (chan<1)){ // sanity check
93 return 0x80;
94 }
95 static uint8_t adr[] = {0x80, 0x90, 0xA0, 0xB0, 0xC0, 0xD0, 0xE0, 0xF0};
96 return adr[chan-1];
97}

102void battery_monitor_init_bus(void){
103 // can be tuned via datalink
104 batmon_current_offset = BATTERY_MONITOR_CURRENT_OFFSET;
105 batmon_current_sensitivity = BATTERY_MONITOR_CURRENT_SENSITIVITY;
106 batmon_temp_offset = BATTERY_MONITOR_TEMP_OFFSET;
107 batmon_temp_sensitivity = BATTERY_MONITOR_TEMP_SENSITIVITY;
108
109 // init Bus ADC
110 // transactions
111 batmonbus.bus_trans.status = I2CTransDone;
112 batmonbus.addr = BATTERY_MONITOR_BUS_ADC_I2C_ADDR;
113 batmonbus.bus_status = BATTERY_MONITOR_BUS_CURRENT_REQ; // device status
114 // Current readings
115 batmonbus.bus_current_mvolts = 0; // mV
116 batmonbus.bus_current = 0; // A
117 // Bus voltage readings
118 batmonbus.bus_voltage_mvolts = 0; //mV
119 // Temperature readings
120 batmonbus.bus_brd_tmp = 0; // C
121 memset(batmonbus.bus_tempsensors_mvolts, 0, sizeof(batmonbus.bus_tempsensors_mvolts));
122 memset(batmonbus.bus_temp, 0, sizeof(batmonbus.bus_temp));
123 batmonbus.t_idx = 0; // temp sensor index
124 batmonbus.data = 0;
125}

170void battery_monitor_check_i2c_transaction(struct i2c_transaction* t){
171 switch (t->status) {
172 case I2CTransPending:
173 // wait and do nothing
174 break;
175 case I2CTransRunning:
176 // wait and do nothing
177 break;
178 case I2CTransSuccess:
179 // set to done
180 t->status = I2CTransDone;
181 break;
182 case I2CTransFailed:
183 // set to done
184 t->status = I2CTransDone;
185 break;
186 case I2CTransDone:
187 // do nothing
188 break;
189 default:
190 break;
191 }
192}

197void battery_monitor_read_bus(void){
198 batmonbus.data = 0; // erase at each iteration
199
200 switch (batmonbus.bus_status) {
201 // reqtest BUS_CURRENT measurement
202 case BATTERY_MONITOR_BUS_CURRENT_REQ:
203 // set ADC to current channel
204 batmonbus.bus_trans.buf[0] = battery_monitor_get_address(
205 (uint8_t)BATTERY_MONITOR_BUS_CURRENT_CHANNEL);
206 //set to zero so we can detect an error
207 batmonbus.bus_current = 0;
208
209 // non-blocking transaction
210 if (i2c_transceive(&BATTERY_MONITOR_I2C_DEV, &batmonbus.bus_trans, batmonbus.addr, 1, 2)){
211 batmonbus.bus_status = BATTERY_MONITOR_BUS_CURRENT_READ;
212 }
213 break;
214
215 // read BUS_CURRENT data
216 case BATTERY_MONITOR_BUS_CURRENT_READ:
217 if (batmonbus.bus_trans.status == I2CTransDone) {
218 // read data
219 batmonbus.data = (uint16_t) (batmonbus.bus_trans.buf[0] << 8 | batmonbus.bus_trans.buf[1]);
220 // NOTE: we are not using the ALERT_FLAG at the moment,
221 // get counts
222 batmonbus.bus_current_mvolts = batmonbus.data & 0xFFF;
223 // shift right by 2 bits if 10-bit reads only
224 if (BATTERY_MONITOR_BIT_RES == 10){
225 batmonbus.bus_current_mvolts = (batmonbus.bus_current_mvolts) >> 2;
226 }
227 // convert to mV
228 batmonbus.bus_current_mvolts = (uint16_t)(
229 (float)batmonbus.bus_current_mvolts * BATTERY_MONITOR_VREF_MULT);
230 // convert to [A]
231 batmonbus.bus_current = ((float)batmonbus.bus_current_mvolts +
232 (float)batmon_current_offset) / batmon_current_sensitivity;
233 //update electrical subsystem
234 electrical.current = batmonbus.bus_current;
235
236 // increment status
237 batmonbus.bus_status = BATTERY_MONITOR_BUS_VOLTAGE_REQ;
238 }
239 break;
240
241 // request voltage data
242 case BATTERY_MONITOR_BUS_VOLTAGE_REQ:
243 // set ADC to voltage channel
244 batmonbus.bus_trans.buf[0] = battery_monitor_get_address(
245 (uint8_t)BATTERY_MONITOR_BUS_VOLTAGE_CHANNEL);
246 //set to zero so we can detect an error
247 batmonbus.bus_voltage_mvolts = 0;
248
249 // non-blocking transaction
250 if (i2c_transceive(&BATTERY_MONITOR_I2C_DEV, &batmonbus.bus_trans, batmonbus.addr, 1, 2)){
251 batmonbus.bus_status = BATTERY_MONITOR_BUS_VOLTAGE_READ;
252 }
253 break;
254
255 // read voltage data
256 case BATTERY_MONITOR_BUS_VOLTAGE_READ:
257 if (batmonbus.bus_trans.status == I2CTransDone) {
258 // read data
259 batmonbus.data = (uint16_t) (batmonbus.bus_trans.buf[0] << 8 | batmonbus.bus_trans.buf[1]);
260 // NOTE: we are not using the ALERT_FLAG at the moment,
261 // get counts
262 batmonbus.bus_voltage_mvolts = batmonbus.data & 0xFFF;
263 // shift right by 2 bits if 10-bit reads only
264 if (BATTERY_MONITOR_BIT_RES == 10){
265 batmonbus.bus_voltage_mvolts = (batmonbus.bus_voltage_mvolts) >> 2;
266 }
267 // convert to mV
268 batmonbus.bus_voltage_mvolts = (uint16_t)(
269 (float)batmonbus.bus_voltage_mvolts * BATTERY_MONITOR_VREF_MULT);
270 // convert to actual voltage
271 batmonbus.bus_voltage_mvolts = (uint16_t)(
272 (float)batmonbus.bus_voltage_mvolts * BatmonVbusGain);
273
274 //update electrical subsystem
275 if (batmonbus.bus_voltage_mvolts != 0) {
276 electrical.vsupply = (float)(batmonbus.bus_voltage_mvolts) / 1000.f;
277 }
278 else {
279 electrical.vsupply = 0.f;
280 }
281
282 // update status
283 batmonbus.bus_status = BATTERY_MONITOR_BUS_TEMPERATURE_REQ;
284 }
285 break;
286
287 // request temperature data
288 case BATTERY_MONITOR_BUS_TEMPERATURE_REQ:
289 // set ADC to correct temperature channel
290 batmonbus.bus_trans.buf[0] = battery_monitor_get_address(
291 battery_monitor_tempmap[batmonbus.t_idx]);
292 //set to zero so we can detect an error
293 batmonbus.bus_temp[batmonbus.t_idx] = 0;
294
295 // non-blocking transaction
296 if (i2c_transceive(&BATTERY_MONITOR_I2C_DEV, &batmonbus.bus_trans, batmonbus.addr, 1, 2)){
297 batmonbus.bus_status = BATTERY_MONITOR_BUS_TEMPERATURE_READ;
298 }
299 break;
300
301 // read temperature data
302 case BATTERY_MONITOR_BUS_TEMPERATURE_READ:
303 if (batmonbus.bus_trans.status == I2CTransDone) {
304 // read data
305 batmonbus.data = (uint16_t) (batmonbus.bus_trans.buf[0] << 8 | batmonbus.bus_trans.buf[1]);
306 // NOTE: we are not using the ALERT_FLAG at the moment,
307 // get counts
308 batmonbus.bus_tempsensors_mvolts[batmonbus.t_idx] = batmonbus.data & 0xFFF;
309 // shift right by 2 bits if 10-bit reads only
310 if (BATTERY_MONITOR_BIT_RES == 10){
311 batmonbus.bus_tempsensors_mvolts[batmonbus.t_idx] =
312 (batmonbus.bus_tempsensors_mvolts[batmonbus.t_idx]) >> 2;
313 }
314 // convert to mV
315 batmonbus.bus_tempsensors_mvolts[batmonbus.t_idx] = (uint16_t)(
316 (float)batmonbus.bus_tempsensors_mvolts[batmonbus.t_idx] * BATTERY_MONITOR_VREF_MULT);
317 // convert to temperature[C]
318 batmonbus.bus_temp[batmonbus.t_idx] =
319 ((float)batmonbus.bus_tempsensors_mvolts[batmonbus.t_idx] +
320 (float)batmon_temp_offset ) / batmon_temp_sensitivity;
321 // increment counter
322 batmonbus.t_idx++;
323 if (batmonbus.t_idx == TEMP_SENSORS_NB) {
324 batmonbus.t_idx = 0;
325 batmonbus.bus_status = BATTERY_MONITOR_BUS_CURRENT_REQ; // to loop through
326 }
327 }
328 break;
329 default:
330 // a recovery in case of a glitch
331 batmonbus.bus_status = BATTERY_MONITOR_BUS_CURRENT_REQ;
332 break;
333 }
334}

354void battery_monitor_read_balance_ports(struct BatMonBal* batmonbal) {
355 // non-blocking transaction
356 // if transaction is done, request another measurement
357 // NOTE: optionally add startup delay, but it shouldn't be necessary
358 if (batmonbal->bus_trans.status == I2CTransDone) {
359 // read data first (worst case we get some zeros)
360 batmonbal->data = (uint16_t) (batmonbal->bus_trans.buf[0] << 8 | batmonbal->bus_trans.buf[1]);
361 // NOTE: we are not using the ALERT_FLAG at the moment,
362 // get counts
363 batmonbal->bat_cell_mvolts[batmonbal->cell_index] = batmonbal->data & 0xFFF;
364 // shift right by 2 bits if 10-bit reads only
365 if (BATTERY_MONITOR_BIT_RES == 10){
366 batmonbal->bat_cell_mvolts[batmonbal->cell_index] =
367 (batmonbal->bat_cell_mvolts[batmonbal->cell_index]) >> 2;
368 }
369 // convert to mV
370 batmonbal->bat_cell_mvolts[batmonbal->cell_index] =
371 (uint16_t)((float)batmonbal->bat_cell_mvolts[batmonbal->cell_index] * BATTERY_MONITOR_VREF_MULT);
372 // convert to actual voltage
373 // we get the multiplier from battery_monitor_cellgains array, which contains multipliers
374 // for cells 1-6(in this order) battery_monitor_cellgains[0] gives us multiplier for cell_1
375 // and so on
376 batmonbal->bat_cell_mvolts[batmonbal->cell_index] =
377 (uint16_t)((float)batmonbal->bat_cell_mvolts[batmonbal->cell_index] *
378 battery_monitor_cellgains[batmonbal->cell_index]);
379
380 // increment counter
381 batmonbal->cell_index++;
382 if (batmonbal->cell_index == BATTERY_CELLS_NB) {
383 batmonbal->cell_index = 0;
384 }
385
386 // erase data
387 batmonbal->data = 0; // erase at each iteration
388
389 //set to zero so we can detect an error
390 batmonbal->bat_cell_mvolts[batmonbal->cell_index] = 0;
391
392 // set the right address into ADC
393 // cell_index is the number of the cell we want to probe
394 // battery_monitor_cellmap1 contains the mapping of channels for given cell number
395 // i.e. Cell_1 (cell_idx = 0) has channel 2
396 // this gets translated into hexadecimal number representing the channel internally
397 batmonbal->bus_trans.buf[0] =
398 battery_monitor_get_address((uint8_t)battery_monitor_cellmap[batmonbal->cell_index]);
399
400 // request more data
401 i2c_transceive(&BATTERY_MONITOR_I2C_DEV, &batmonbal->bus_trans, batmonbal->addr, 1, 2);
402 }
403}
