202void imu_cube_event(void)
203{
204#if IMU_CUBE_ORANGEPLUS_LTS
205 invensense3_456_event(&imu1);
206 invensense3_456_event(&imu2);
207 invensense3_456_event(&imu3);
208#else
209 invensense2_event(&imu1);
210
211#if IMU_CUBE_ORANGEPLUS
212 invensense3_event(&imu2);
213#else
214 mpu60x0_spi_event(&imu2);
215 if (imu2.data_available) {
216 uint32_t now_ts = get_sys_time_usec();
217
218 // set channel order
219 struct Int32Vect3 accel = {
220 (int32_t)(imu2.data_accel.value[1]),
221 (int32_t)(imu2.data_accel.value[0]),
222 -(int32_t)(imu2.data_accel.value[2])
223 };
224 struct Int32Rates rates = {
225 (int32_t)(imu2.data_rates.value[1]),
226 (int32_t)(imu2.data_rates.value[0]),
227 -(int32_t)(imu2.data_rates.value[2])
228 };
229
230 imu2.data_available = false;
231
232 // Send the scaled values over ABI
233 AbiSendMsgIMU_GYRO_RAW(IMU_CUBE2_ID, now_ts, &rates, 1, IMU_CUBE_PERIODIC_FREQ, imu2.temp);
234 AbiSendMsgIMU_ACCEL_RAW(IMU_CUBE2_ID, now_ts, &accel, 1, IMU_CUBE_PERIODIC_FREQ, imu2.temp);
235 }
236#endif
237
238 invensense2_event(&imu3);
239#endif // IMU_CUBE_ORANGEPLUS_LTS
240}
