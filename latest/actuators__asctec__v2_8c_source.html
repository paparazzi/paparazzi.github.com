47 actuators_asctec_v2.nb_err = 0;

51void actuators_asctec_v2_set(void)
52{
53#if defined ACTUATORS_START_DELAY && ! defined SITL
54 if (!actuators_delay_done) {
55 if (SysTimeTimer(actuators_delay_time) < USEC_OF_SEC(ACTUATORS_START_DELAY)) {
56#ifdef USE_I2C_ACTUATORS_REBOOT_HACK
57 //Lisa-L with Asctech v2 motors only start after reflashing when a bus error was sensed on stm32-i2c.
58 //multiple re-init solves the problem.
59 i2c1_init();
60#endif
61 return;
62 } else { actuators_delay_done = true; }
63 }
64#endif
65
66 switch (actuators_asctec_v2.i2c_trans.status) {
67 case I2CTransFailed:
68 actuators_asctec_v2.nb_err++;
69 actuators_asctec_v2.i2c_trans.status = I2CTransDone;
70 break;
71 case I2CTransSuccess:
72 case I2CTransDone:
73 actuators_asctec_v2.i2c_trans.status = I2CTransDone;
74 break;
75 default:
76 actuators_asctec_v2.nb_err++;
77 return;
78 }
79
80#ifdef KILL_MOTORS
81 actuators_asctec_v2.i2c_trans.buf[0] = 0;
82 actuators_asctec_v2.i2c_trans.buf[1] = 0;
83 actuators_asctec_v2.i2c_trans.buf[2] = 0;
84 actuators_asctec_v2.i2c_trans.buf[3] = 0;
85 actuators_asctec_v2.i2c_trans.buf[4] = 0xAA;
86#else
87 switch (actuators_asctec_v2.cmd) {
88 case TEST:
89 actuators_asctec_v2.i2c_trans.buf[0] = 251;
90 actuators_asctec_v2.i2c_trans.buf[1] = actuators_asctec_v2.cur_addr;
91 actuators_asctec_v2.i2c_trans.buf[2] = 0;
92 actuators_asctec_v2.i2c_trans.buf[3] = 231 + actuators_asctec_v2.cur_addr;
93 actuators_asctec_v2.i2c_trans.len_w = 4;
94 break;
95 case REVERSE:
96 actuators_asctec_v2.i2c_trans.buf[0] = 254;
97 actuators_asctec_v2.i2c_trans.buf[1] = actuators_asctec_v2.cur_addr;
98 actuators_asctec_v2.i2c_trans.buf[2] = 0;
99 actuators_asctec_v2.i2c_trans.buf[3] = 234 + actuators_asctec_v2.cur_addr;
100 actuators_asctec_v2.i2c_trans.len_w = 4;
101 break;
102 case SET_ADDR:
103 actuators_asctec_v2.i2c_trans.buf[0] = 250;
104 actuators_asctec_v2.i2c_trans.buf[1] = actuators_asctec_v2.cur_addr;
105 actuators_asctec_v2.i2c_trans.buf[2] = actuators_asctec_v2.new_addr;
106 actuators_asctec_v2.i2c_trans.buf[3] = 230 + actuators_asctec_v2.cur_addr +
107 actuators_asctec_v2.new_addr;
108 actuators_asctec_v2.cur_addr = actuators_asctec_v2.new_addr;
109 actuators_asctec_v2.i2c_trans.len_w = 4;
110 break;
111 case NONE:
112 actuators_asctec_v2.i2c_trans.buf[0] = actuators_asctec_v2.cmds[SERVO_FRONT_IDX];
113 actuators_asctec_v2.i2c_trans.buf[1] = actuators_asctec_v2.cmds[SERVO_BACK_IDX];
114 actuators_asctec_v2.i2c_trans.buf[2] = actuators_asctec_v2.cmds[SERVO_LEFT_IDX];
115 actuators_asctec_v2.i2c_trans.buf[3] = actuators_asctec_v2.cmds[SERVO_RIGHT_IDX];
116 actuators_asctec_v2.i2c_trans.buf[4] = 0xAA + actuators_asctec_v2.i2c_trans.buf[0] +
117 actuators_asctec_v2.i2c_trans.buf[1] +
118 actuators_asctec_v2.i2c_trans.buf[2] +
119 actuators_asctec_v2.i2c_trans.buf[3];
120 actuators_asctec_v2.i2c_trans.len_w = 5;
121 break;
122 default:
123 break;
124 }
125 actuators_asctec_v2.cmd = NONE;
126#endif
127
128 actuators_asctec_v2.i2c_trans.type = I2CTransTx; // Can be reset I2C driver
129 i2c_submit(&ACTUATORS_ASCTEC_V2_I2C_DEV, &actuators_asctec_v2.i2c_trans);
130
131}
