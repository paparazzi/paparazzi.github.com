86void sensors_hitl_parse_HITL_IMU(uint8_t *buf)
87{
88 if (DL_HITL_IMU_ac_id(buf) != AC_ID) {
89 return;
90 }
91
92 RATES_ASSIGN(imu_hitl.gyro,
93 NPS_GYRO_SIGN_P * DL_HITL_IMU_gp(buf),
94 NPS_GYRO_SIGN_Q * DL_HITL_IMU_gq(buf),
95 NPS_GYRO_SIGN_R * DL_HITL_IMU_gr(buf));

98 NPS_ACCEL_SIGN_Y * DL_HITL_IMU_ay(buf),

106 imu_hitl.gyro_available = true;
107 imu_hitl.mag_available = true;

110void sensors_hitl_parse_HITL_GPS(uint8_t *buf)
111{
112 if (DL_HITL_GPS_ac_id(buf) != AC_ID) {
113 return;
114 }
115
116 gps_hitl.week = 1794;
117 gps_hitl.tow = DL_HITL_GPS_time(buf) * 1000;
118
119 gps_hitl.ecef_vel.x = DL_HITL_GPS_ecef_vel_x(buf) * 100.;
120 gps_hitl.ecef_vel.y = DL_HITL_GPS_ecef_vel_y(buf) * 100.;
121 gps_hitl.ecef_vel.z = DL_HITL_GPS_ecef_vel_z(buf) * 100.;
122 SetBit(gps_hitl.valid_fields, GPS_VALID_VEL_ECEF_BIT);
123
124 gps_hitl.lla_pos.lat = DL_HITL_GPS_lat(buf) * 1e7;
125 gps_hitl.lla_pos.lon = DL_HITL_GPS_lon(buf) * 1e7;
126 gps_hitl.lla_pos.alt = DL_HITL_GPS_alt(buf) * 1000.;
127 SetBit(gps_hitl.valid_fields, GPS_VALID_POS_LLA_BIT);
128
129 gps_hitl.hmsl = DL_HITL_GPS_hmsl(buf) * 1000.;
130 SetBit(gps_hitl.valid_fields, GPS_VALID_HMSL_BIT);
131
132 /* calc NED speed from ECEF */
133 struct LtpDef_i ref_ltp;
134 ltp_def_from_lla_i(&ref_ltp, &gps_hitl.lla_pos);
135 struct NedCoor_i ned_vel_i;
136 ned_of_ecef_vect_i(&ned_vel_i, &ref_ltp, &gps_hitl.ecef_vel);
137 gps_hitl.ned_vel.x = ned_vel_i.x;
138 gps_hitl.ned_vel.y = ned_vel_i.y;
139 gps_hitl.ned_vel.z = ned_vel_i.z;
140 SetBit(gps_hitl.valid_fields, GPS_VALID_VEL_NED_BIT);
141 struct NedCoor_f ned_vel_f;
142 VECT3_FLOAT_OF_CM(ned_vel_f, gps_hitl.ned_vel);
143
144 /* horizontal and 3d ground speed in cm/s */
145 gps_hitl.gspeed = sqrtf(ned_vel_f.x * ned_vel_f.x + ned_vel_f.y * ned_vel_f.y) * 100;
146 gps_hitl.speed_3d = sqrtf(ned_vel_f.x * ned_vel_f.x + ned_vel_f.y * ned_vel_f.y + ned_vel_f.z * ned_vel_f.z) * 100;
147
148 /* ground course in radians * 1e7 */
149 gps_hitl.course = atan2f(ned_vel_f.y, ned_vel_f.x) * 1e7;
150 SetBit(gps_hitl.valid_fields, GPS_VALID_COURSE_BIT);
151
152 gps_hitl.pacc = 650;
153 gps_hitl.hacc = 450;
154 gps_hitl.vacc = 200;
155 gps_hitl.sacc = 100;
156 gps_hitl.pdop = 650;
157
158 if (gps_has_fix) {
159 gps_hitl.num_sv = 11;
160 gps_hitl.fix = GPS_FIX_3D;
161 } else {
162 gps_hitl.num_sv = 1;
163 gps_hitl.fix = GPS_FIX_NONE;
164 }
165
166 // publish gps data
167 uint32_t now_ts = get_sys_time_usec();
168 gps_hitl.last_msg_ticks = sys_time.nb_sec_rem;
169 gps_hitl.last_msg_time = sys_time.nb_sec;
170 if (gps_hitl.fix == GPS_FIX_3D) {
171 gps_hitl.last_3dfix_ticks = sys_time.nb_sec_rem;
172 gps_hitl.last_3dfix_time = sys_time.nb_sec;
173 }
174 AbiSendMsgGPS(GPS_SIM_ID, now_ts, &gps_hitl);
175}

177void sensors_hitl_parse_HITL_AIR_DATA(uint8_t *buf) {
178 if (DL_HITL_AIR_DATA_ac_id(buf) != AC_ID) {
179 return;
180 }
181
182 uint8_t flag = DL_HITL_AIR_DATA_update_flag(buf);
183 if (bit_is_set(flag, 0)) {
184 uint32_t now_ts = get_sys_time_usec();
185 float pressure = DL_HITL_AIR_DATA_baro(buf);
186 AbiSendMsgBARO_ABS(BARO_SIM_SENDER_ID, now_ts, pressure);
187 }
188 if (bit_is_set(flag, 1)) {
189 AbiSendMsgAIRSPEED(AIRSPEED_NPS_ID, DL_HITL_AIR_DATA_airspeed(buf));
190 }
191 if (bit_is_set(flag, 2) || bit_is_set(flag, 3)) {
192 uint8_t incidence_flag = (flag >> 2) & (0x3);
193 float aoa = DL_HITL_AIR_DATA_aoa(buf);
194 float sideslip = DL_HITL_AIR_DATA_sideslip(buf);
195 AbiSendMsgINCIDENCE(INCIDENCE_NPS_ID, incidence_flag, aoa, sideslip);
196 }
197}

200void sensors_hitl_event(void)
201{
202 uint32_t now_ts = get_sys_time_usec();
203 if (imu_hitl.gyro_available) {
204 AbiSendMsgIMU_GYRO_RAW(IMU_NPS_ID, now_ts, &imu_hitl.gyro, 1, NPS_PROPAGATE, NAN);
205 imu_hitl.gyro_available = false;
206 }
207 if (imu_hitl.accel_available) {
208 AbiSendMsgIMU_ACCEL_RAW(IMU_NPS_ID, now_ts, &imu_hitl.accel, 1, NPS_PROPAGATE, NAN);
209 imu_hitl.accel_available = false;
210 }
211 if (imu_hitl.mag_available) {
212 AbiSendMsgIMU_MAG_RAW(IMU_NPS_ID, now_ts, &imu_hitl.mag);
213 imu_hitl.mag_available = false;
214 }
215
216 // parse incoming messages
217 pprz_check_and_parse(&HITL_DEVICE.device, &sensors_hitl_tp, sensors_hitl_dl_buffer, &sensors_hitl_msg_available);
218 DlCheckAndParse(&HITL_DEVICE.device, &sensors_hitl_tp.trans_tx, sensors_hitl_dl_buffer, &sensors_hitl_msg_available, false);
219}
