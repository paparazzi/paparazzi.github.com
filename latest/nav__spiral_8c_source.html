50void nav_spiral_setup(uint8_t center_wp, uint8_t edge_wp, float radius_start, float radius_inc, float segments)
51{
52 VECT2_COPY(nav_spiral.center, waypoints[center_wp]); // center of the helix
53 nav_spiral.center.z = waypoints[center_wp].a;
54 nav_spiral.radius_start = radius_start; // start radius of the helix
55 nav_spiral.segments = segments;
56 nav_spiral.radius_min = NAV_SPIRAL_MIN_CIRCLE_RADIUS;
57 if (nav_spiral.radius_start < nav_spiral.radius_min) {
58 nav_spiral.radius_start = nav_spiral.radius_min;
59 }
60 nav_spiral.radius_increment = radius_inc; // multiplier for increasing the spiral
61
62 struct FloatVect2 edge;
63 VECT2_DIFF(edge, waypoints[edge_wp], nav_spiral.center);
64
65 nav_spiral.radius = float_vect2_norm(&edge);
66
67 // get a copy of the current position
68 struct EnuCoor_f pos_enu = *stateGetPositionEnu_f();
69
70 VECT2_DIFF(nav_spiral.trans_current, pos_enu, nav_spiral.center);
71 nav_spiral.trans_current.z = stateGetPositionUtm_f()->alt - nav_spiral.center.z;
72
73 nav_spiral.dist_from_center = FLOAT_VECT3_NORM(nav_spiral.trans_current);
74
75 // nav_spiral.alpha_limit denotes angle, where the radius will be increased
76 nav_spiral.alpha_limit = 2 * M_PI / nav_spiral.segments;
77 //current position
78 nav_spiral.fly_from.x = stateGetPositionEnu_f()->x;
79 nav_spiral.fly_from.y = stateGetPositionEnu_f()->y;
80
81 if (nav_spiral.dist_from_center > nav_spiral.radius) {
82 nav_spiral.status = SpiralOutside;
83 }
84}

86bool nav_spiral_run(void)
87{
88 struct EnuCoor_f pos_enu = *stateGetPositionEnu_f();
89
90 VECT2_DIFF(nav_spiral.trans_current, pos_enu, nav_spiral.center);
91 nav_spiral.dist_from_center = FLOAT_VECT3_NORM(nav_spiral.trans_current);
92
93 float DistanceStartEstim;
94 float CircleAlpha;
95
96 switch (nav_spiral.status) {
97 case SpiralOutside:
98 //flys until center of the helix is reached an start helix
99 nav_route_xy(nav_spiral.fly_from.x, nav_spiral.fly_from.y, nav_spiral.center.x, nav_spiral.center.y);
100 // center reached?
101 if (nav_approaching_xy(nav_spiral.center.x, nav_spiral.center.y, nav_spiral.fly_from.x, nav_spiral.fly_from.y, 0)) {
102 // nadir image
103#ifdef DIGITAL_CAM
104 dc_send_command(DC_SHOOT);
105#endif
106 nav_spiral.status = SpiralStartCircle;
107 }
108 break;
109 case SpiralStartCircle:
110 // Starts helix
111 // storage of current coordinates
112 // calculation needed, State switch to SpiralCircle
113 nav_circle_XY(nav_spiral.center.y, nav_spiral.center.y, nav_spiral.radius_start);
114 if (nav_spiral.dist_from_center >= nav_spiral.radius_start) {
115 VECT2_COPY(nav_spiral.last_circle, pos_enu);
116 nav_spiral.status = SpiralCircle;
117 // Start helix
118#ifdef DIGITAL_CAM
119 dc_Circle(360 / nav_spiral.segments);
120#endif
121 }
122 break;
123 case SpiralCircle: {
124 nav_circle_XY(nav_spiral.center.x, nav_spiral.center.y, nav_spiral.radius_start);
125 // Trigonometrische Berechnung des bereits geflogenen Winkels alpha
126 // equation:
127 // alpha = 2 * asin ( |Starting position angular - current positon| / (2* nav_spiral.radius_start)
128 // if alphamax already reached, increase radius.
129
130 //DistanceStartEstim = |Starting position angular - current positon|
131 struct FloatVect2 pos_diff;
132 VECT2_DIFF(pos_diff, nav_spiral.last_circle, pos_enu);
133 DistanceStartEstim = float_vect2_norm(&pos_diff);
134 CircleAlpha = (2.0 * asin(DistanceStartEstim / (2 * nav_spiral.radius_start)));
135 if (CircleAlpha >= nav_spiral.alpha_limit) {
136 VECT2_COPY(nav_spiral.last_circle, pos_enu);
137 nav_spiral.status = SpiralInc;
138 }
139 break;
140 }
141 case SpiralInc:
142 // increasing circle radius as long as it is smaller than max helix radius
143 if (nav_spiral.radius_start + nav_spiral.radius_increment < nav_spiral.radius) {
144 nav_spiral.radius_start = nav_spiral.radius_start + nav_spiral.radius_increment;
145#ifdef DIGITAL_CAM
146 if (dc_cam_tracing) {
147 // calculating Cam angle for camera alignment
148 nav_spiral.trans_current.z = stateGetPositionUtm_f()->alt - nav_spiral.center.z;
149 dc_cam_angle = atan(nav_spiral.radius_start / nav_spiral.trans_current.z) * 180 / M_PI;
150 }
151#endif
152 } else {
153 nav_spiral.radius_start = nav_spiral.radius;
154#ifdef DIGITAL_CAM
155 // Stopps DC
156 dc_stop();
157#endif
158 }
159 nav_spiral.status = SpiralCircle;
160 break;
161 default:
162 break;
163 }
164
165 NavVerticalAutoThrottleMode(0.); /* No pitch */
166 NavVerticalAltitudeMode(nav_spiral.center.z, 0.); /* No preclimb */
167
168 return true;
169}
