65void gpio_setup_pin_af(ioportid_t port, uint16_t pin, uint8_t af, bool is_output)
66{
67 chSysLock();
68// architecture dependent settings
69#if defined(STM32F1XX)
70// FIXME: STM32F1xx doesn't support several alternate modes, is it needed for drivers?
71 (void)port;
72 (void)pin;
73 (void)af;
74 (void)is_output;
75#else
76// STM32F4xx, STM32F3xx, STM32F7xx and STM32H7xx
77 if (af) {
78 palSetPadMode(port, pin, PAL_MODE_ALTERNATE(af));
79 } else {
80 if (is_output) {
81 palSetPadMode(port, pin, PAL_MODE_OUTPUT_PUSHPULL);
82 } else {
83 palSetPadMode(port, pin, PAL_MODE_INPUT);
84 }
85 }
86#endif
87 chSysUnlock();
88}
