84 void rover_holo_guidance_run(float *heading_sp)
85 {
86  // compute position error
87  struct FloatVect2 pos_err;
88  VECT2_DIFF(pos_err, rover_holo_guidance.sp.pos, *stateGetPositionNed_f());
89  rover_holo_guidance.speed_pid.err = float_vect2_norm(&pos_err);
90  BoundAbs(rover_holo_guidance.speed_pid.err, MAX_POS_ERR);
91 
92  // speed update when far enough
93  if (rover_holo_guidance.speed_pid.err > PROXIMITY_DIST) {
94  // compute speed error
95  rover_holo_guidance.speed_pid.d_err = - stateGetHorizontalSpeedNorm_f();
96  BoundAbs(rover_holo_guidance.speed_pid.d_err, MAX_SPEED_ERR);
97  // integral
98  rover_holo_guidance.speed_pid.sum_err = 0.f; // nothing for now
99  // run PID
100  float speed = MAX_PPRZ * compute_pid(&rover_holo_guidance.speed_pid);
101  speed = TRIM_PPRZ(speed);
102 
103  float cpsi = cosf(stateGetNedToBodyEulers_f()->psi);
104  float spsi = sinf(stateGetNedToBodyEulers_f()->psi);
105  struct FloatVect2 err_body = {
106  .x = cpsi * pos_err.x + spsi * pos_err.y,
107  .y = -spsi * pos_err.x + cpsi * pos_err.y
108  };
109  // command in rover frame = speed * normalized direction vector
110  // TODO normalize to min/max speed with pprz_t scale
111  rover_holo_guidance.cmd.motor_speed_x = speed * err_body.x / rover_holo_guidance.speed_pid.err;
112  rover_holo_guidance.cmd.motor_speed_y = speed * err_body.y / rover_holo_guidance.speed_pid.err;
113  } else {
114  rover_holo_guidance.cmd.motor_speed_x = 0.f;
115  rover_holo_guidance.cmd.motor_speed_y = 0.f;
116  }
117 
118  rover_holo_guidance.sp.heading = *heading_sp;
119  // angular error
120  rover_holo_guidance.turn_pid.err = rover_holo_guidance.sp.heading - stateGetNedToBodyEulers_f()->psi;
121  NormRadAngle(rover_holo_guidance.turn_pid.err);
122  // turn rate error
123  rover_holo_guidance.turn_pid.d_err = - stateGetBodyRates_f()->r;
124  // integral
125  rover_holo_guidance.turn_pid.sum_err = 0.f; // nothing for now
126  // run PID
127  rover_holo_guidance.cmd.motor_turn = MAX_PPRZ * compute_pid(&rover_holo_guidance.turn_pid);
128  rover_holo_guidance.cmd.motor_turn = TRIM_PPRZ(rover_holo_guidance.cmd.motor_turn);
129 }
