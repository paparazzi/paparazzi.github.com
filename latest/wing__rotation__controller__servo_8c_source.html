99 void wing_rotation_adc_to_deg(void)
100 {
101 #if !USE_NPS
102  // Read ADC
103  wing_rotation_controller.adc_wing_rotation = buf_wing_rot_pos.sum / buf_wing_rot_pos.av_nb_sample;
104 
105  wing_rotation_controller.wing_angle_deg = 0.00247111 * (float)wing_rotation_controller.adc_wing_rotation - 25.635294;
106 #else // !USE_NPS
107  // Copy setpoint as actual angle in simulation
108  wing_rotation_controller.wing_angle_deg = wing_rotation_controller.wing_angle_virtual_deg_sp;
109 #endif
110 
111  // SEND ABI Message to ctr_eff_sched and other modules that want Actuator position feedback
112  struct act_feedback_t feedback;
113  feedback.idx = SERVO_ROTATION_MECH;
114  feedback.position = 0.5 * M_PI - RadOfDeg(wing_rotation_controller.wing_angle_deg);
115  feedback.set.position = true;
116 
117  // Send ABI message
118  AbiSendMsgACT_FEEDBACK(ACT_FEEDBACK_UAVCAN_ID, &feedback, 1);
119 }

121 void wing_rotation_compute_pprz_cmd(void)
122 {
123  wing_rotation_controller.wing_angle_deg_sp = rotwing_state_skewing.wing_angle_deg_sp;
124  // Smooth accerelation and rate limited setpoint
125  float angle_error = wing_rotation_controller.wing_angle_deg_sp - wing_rotation_controller.wing_angle_virtual_deg_sp;
126  float speed_sp = wing_rotation_controller.wing_rotation_first_order_dynamics * angle_error;
127  float speed_error = speed_sp - wing_rotation_controller.wing_rotation_speed;
128  wing_rotation_controller.wing_rotation_speed += wing_rotation_controller.wing_rotation_second_order_dynamics * speed_error;
129  wing_rotation_controller.wing_angle_virtual_deg_sp += wing_rotation_controller.wing_rotation_speed;
130 
131  // Send to actuators
132  int32_t servo_pprz_cmd;
133  servo_pprz_cmd = (int32_t)(wing_rotation_controller.wing_angle_virtual_deg_sp / 90. * (float)MAX_PPRZ);
134  Bound(servo_pprz_cmd, 0, MAX_PPRZ);
135  wing_rotation_controller.servo_pprz_cmd = servo_pprz_cmd;
136 
137 #if USE_NPS
138  actuators_pprz[INDI_NUM_ACT] = servo_pprz_cmd;
