48bool sdio_connect(SDCDriver *sdc)
49{
50 if (!sdc_lld_is_card_inserted(sdc)) {
51 return FALSE;
52 }
53
54 if (cnxState == CONNECT) {
55 return TRUE;
56 }
57
58 // Try only 3 times to prevent hanging
59 for (uint8_t i = 0; i < 3; i++) {
60 sdcStart(sdc, NULL);
61 if (sdcConnect(sdc) == HAL_SUCCESS) {
62 cnxState = CONNECT;
63 return TRUE;
64 }
65 sdcStop(sdc);
66 chThdSleepMilliseconds(100);
67 }
68
69 return FALSE;
70}
71
72
73bool sdio_disconnect(SDCDriver *sdc)
74{
75 if (cnxState == STOP) {
76 return TRUE;
77 }
78 if (sdcDisconnect(sdc)) {
79 return FALSE;
80 }
81 sdcStop(sdc);
82 cnxState = STOP;
83 return TRUE;
84}
