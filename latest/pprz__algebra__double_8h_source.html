132static inline void double_quat_normalize(struct DoubleQuat *q)
133{
134 double qnorm = double_quat_norm(q);
135 if (qnorm > FLT_MIN) {
136 q->qi = q->qi / qnorm;
137 q->qx = q->qx / qnorm;
138 q->qy = q->qy / qnorm;
139 q->qz = q->qz / qnorm;
140 }
