88static struct EnuCoor_f process_new_point_lace(struct EnuCoor_f *position, float alt_sp, float uav_direction)
89{
90 struct EnuCoor_f new_point;
91 float rot_angle;
92
93 if (nav_lace.rotation == LACE_RIGHT) {
94 rot_angle = -M_PI_2;
95 nav_lace.rotation = LACE_LEFT;
96 } else {
97 rot_angle = M_PI_2;
98 nav_lace.rotation = LACE_RIGHT;
99 }
100
101 new_point.x = position->x + (cosf(rot_angle + uav_direction) * nav_lace.radius);
102 new_point.y = position->y + (sinf(rot_angle + uav_direction) * nav_lace.radius);
103 new_point.z = alt_sp;
104
105 return new_point;
106}

108static void update_target_point(struct EnuCoor_f *target, struct EnuCoor_f *border, float dt, float tau)
109{
110 // with a proper discrete transform, coeff should be exp(-dt/tau) and (1-exp(-dt/tau))
111 // but to make it simpler with the same behavior at the limits, we use tau/(dt+tau) and dt/(dt+tau)
112 // with a positive value for tau
113 if (tau > 1e-4) {
114 float alpha = dt / (dt + tau);
115 target->x = (border->x * alpha) + (target->x * (1.f - alpha));
116 target->y = (border->y * alpha) + (target->y * (1.f - alpha));
117 }
118}

124static bool nav_lace_mission(uint8_t nb, float *params, enum MissionRunFlag flag)
125{
126 if (flag == MissionInit && nb == 8) {
127 float start_x = params[0];
128 float start_y = params[1];
129 float start_z = params[2];
130 int first_turn = params[3];
131 float circle_radius = params[4];
132 float vx = params[5];
133 float vy = params[6];
134 float vz = params[7];
135 nav_lace_setup(start_x, start_y, start_z, first_turn, circle_radius, vx, vy, vz);
136 return true;
137 }
138 else if (flag == MissionUpdate && nb == 3) {
139 // update target 3D position (ENU frame, above ground alt)
140 float x = params[0];
141 float y = params[1];
142 float z = params[2];
143 VECT3_ASSIGN(nav_lace.target, x, y, z);
144 return true;
145 }
146 else if (flag == MissionUpdate && nb == 2) {
147 // update horizontal speed
148 float vx = params[0];
149 float vy = params[1];
150 nav_lace.pos_incr.x = vx * nav_dt;
151 nav_lace.pos_incr.y = vy * nav_dt;
152 return true;
153 }
154 else if (flag == MissionUpdate && nb == 1) {
155 // update vertical speed
156 float vz = params[0];
157 nav_lace.pos_incr.z = vz * nav_dt;
158 return true;
159 }
160 else if (flag == MissionRun) {
161 return nav_lace_run();
162 }
163 return false; // not a valid case
164}
165#endif

197void nav_lace_setup(float init_x, float init_y,
198 float init_z, int turn,
199 float desired_radius, float vx,
200 float vy, float vz)
201{
202 struct EnuCoor_f start = {init_x, init_y, init_z};
203 // increment based on speed
204 VECT3_ASSIGN(nav_lace.pos_incr, vx*nav_dt, vy*nav_dt, vz*nav_dt);
205
206 nav_lace.target = start;
207 nav_lace.status = LACE_ENTER;
208 nav_lace.inside_cloud = false;
209 nav_lace.radius = desired_radius;
210
211 if (turn == 1) {
212 nav_lace.rotation = LACE_RIGHT;
213 nav_lace.radius_sign = 1.0f;
214 } else {
215 nav_lace.rotation = LACE_LEFT;
216 nav_lace.radius_sign = -1.0f;
217 }
218
219 nav_lace.actual = *stateGetPositionEnu_f();
220}

222bool nav_lace_run(void)
223{
224 float pre_climb = 0.f;
225 float time = get_sys_time_float();
226
227 NavVerticalAutoThrottleMode(0.f); /* No pitch */
228
229 switch (nav_lace.status) {
230 case LACE_ENTER:
231 // init stage
232 nav_init_stage();
233 // reach target point
234 nav_route_xy(nav_lace.actual.x, nav_lace.actual.y, nav_lace.target.x, nav_lace.target.y);
235 NavVerticalAltitudeMode(nav_lace.target.z + ground_alt, pre_climb);
236
237 if (nav_lace.inside_cloud) {
238 // found border or already inside
239 nav_lace.status = LACE_INSIDE_START;
240 // update target for horizontal position
241 nav_lace.target.x = stateGetPositionEnu_f()->x;
242 nav_lace.target.y = stateGetPositionEnu_f()->y;
243 }
244 break;
245 case LACE_INSIDE_START:
246 // prepare inside circle
247 nav_lace.actual = *stateGetPositionEnu_f();
248 nav_lace.direction = change_rep(stateGetHorizontalSpeedDir_f());
249 nav_lace.circle = process_new_point_lace(&nav_lace.actual, nav_lace.target.z, nav_lace.direction);
250 // init stage
251 nav_init_stage();
252 // update border and target for recover
253 nav_lace.estim_border = nav_lace.actual;
254 update_target_point(&nav_lace.target, &nav_lace.estim_border, time - nav_lace.last_border_time, NAV_LACE_BORDER_FILTER);
255 nav_lace.last_border_time = time;
256 // fly inside
257 nav_lace.status = LACE_INSIDE;
258 break;
259 case LACE_INSIDE:
260 // increment center position
261 VECT3_ADD(nav_lace.circle, nav_lace.pos_incr);
262 nav_circle_XY(nav_lace.circle.x, nav_lace.circle.y , nav_lace.radius_sign * nav_lace.radius);
263 pre_climb = nav_lace.pos_incr.z / nav_dt;
264 NavVerticalAltitudeMode(nav_lace.circle.z + ground_alt, pre_climb);
265
266 if (!nav_lace.inside_cloud) {
267 // found border, start outside
268 nav_lace.status = LACE_OUTSIDE_START;
269 nav_lace.radius_sign = -1.0f * nav_lace.radius_sign;
270 }
271 else if (NavCircleCountNoRewind() > NAV_LACE_RECOVER_MAX_TURN) {
272 // most likely lost inside
273 nav_lace.status = LACE_RECOVER_START;
274 }
275 break;
276 case LACE_OUTSIDE_START:
277 // prepare outside circle
278 nav_lace.actual = *stateGetPositionEnu_f();
279 nav_lace.direction = change_rep(stateGetHorizontalSpeedDir_f());
280 nav_lace.circle = process_new_point_lace(&nav_lace.actual, nav_lace.circle.z, nav_lace.direction);
281 // init stage
282 nav_init_stage();
283 // upadte border and target for recover
284 nav_lace.estim_border = nav_lace.actual;
285 update_target_point(&nav_lace.target, &nav_lace.estim_border, time - nav_lace.last_border_time, NAV_LACE_BORDER_FILTER);
286 nav_lace.last_border_time = time;
287 // fly outside
288 nav_lace.status = LACE_OUTSIDE;
289 break;
290 case LACE_OUTSIDE:
291 // increment center position
292 VECT3_ADD(nav_lace.circle, nav_lace.pos_incr);
293 pre_climb = nav_lace.pos_incr.z / nav_dt;
294 nav_circle_XY(nav_lace.circle.x, nav_lace.circle.y , nav_lace.radius_sign * nav_lace.radius);
295 NavVerticalAltitudeMode(nav_lace.circle.z + ground_alt, pre_climb);
296
297 if (nav_lace.inside_cloud) {
298 // found border, start inside
299 nav_lace.status = LACE_INSIDE_START;
300 nav_lace.radius_sign = -1.0f * nav_lace.radius_sign;
301 }
302 else if (NavCircleCountNoRewind() > NAV_LACE_RECOVER_MAX_TURN) {
303 // most likely lost outside
304 nav_lace.status = LACE_RECOVER_START;
305 }
306 break;
307 case LACE_RECOVER_START:
308 // prepare recovery circle or line
309 nav_lace.recover_circle = nav_lace.target;
310 nav_lace.actual = *stateGetPositionEnu_f();
311 // initial recovery radius
312 nav_lace.recover_radius = nav_lace.radius;
313 nav_lace.max_recover_radius = 2.0f * nav_lace.recover_radius; // FIXME ?
314 // init stage
315 nav_init_stage();
316 if (nav_lace.inside_cloud) {
317 nav_lace.status = LACE_RECOVER_INSIDE;
318 }
319 else {
320 nav_lace.status = LACE_RECOVER_OUTSIDE;
321 }
322 break;
323 case LACE_RECOVER_INSIDE:
324 // increment border position
325 // fly in opposite direction to cover more space
326 nav_route_xy(nav_lace.estim_border.x, nav_lace.estim_border.y, nav_lace.actual.x, nav_lace.actual.y);
327 if (!nav_lace.inside_cloud) {
328 nav_lace.status = LACE_OUTSIDE_START;
329 nav_lace.radius_sign = -1.0f * nav_lace.radius_sign;
330 }
331 break;
332 case LACE_RECOVER_OUTSIDE:
333 // increment center position
334 VECT3_ADD(nav_lace.recover_circle, nav_lace.pos_incr);
335 nav_circle_XY(nav_lace.recover_circle.x, nav_lace.recover_circle.y , nav_lace.radius_sign * nav_lace.recover_radius);
336 // increment recover circle radius
337 if (nav_lace.recover_radius < nav_lace.max_recover_radius) {
338 nav_lace.recover_radius += 0.5;
339 }
340 // found a new border
341 if (nav_lace.inside_cloud) {
342 nav_lace.status = LACE_INSIDE_START;
343 nav_lace.radius_sign = -1.0f * nav_lace.radius_sign;
344 }
345 break;
346 default:
347 // error, leaving
348 return false;
349 }
350 // increment border and target positions
351 VECT3_ADD(nav_lace.estim_border, nav_lace.pos_incr);
352 VECT3_ADD(nav_lace.target, nav_lace.pos_incr);
353#ifdef WP_TARGET
354 nav_move_waypoint_enu(WP_TARGET, nav_lace.target.x, nav_lace.target.y, nav_lace.target.z + ground_alt);
355 RunOnceEvery(10, nav_send_waypoint(WP_TARGET));
356#endif
357
358 return true;
359}
