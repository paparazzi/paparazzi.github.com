76static inline float update_first_order_complementary(
77 struct FirstOrderComplementary *filter,
78 float value_x, float value_y)
79{
80 float x_lp_output = update_first_order_low_pass(&filter->x_lp_filter, value_x);
81 float y_lp_output = update_first_order_low_pass(&filter->y_lp_filter, value_y);
