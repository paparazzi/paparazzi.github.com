98void baro_event(void)
99{
100 if (sys_time.nb_sec > 1) {
101 ms5611_spi_event(&bb_ms5611);
102
103 if (bb_ms5611.data_available) {
104 uint32_t now_ts = get_sys_time_usec();
105
106 #if USE_BARO_MEDIAN_FILTER
107 bb_ms5611.data.temperature = update_median_filter_f(&board_baro_median_temp, bb_ms5611.data.temperature);
108 #endif
109 float temp = bb_ms5611.data.temperature / 100.0f;
110 AbiSendMsgTEMPERATURE(BARO_BOARD_SENDER_ID, temp);
111
112 #if USE_BARO_MEDIAN_FILTER
113 bb_ms5611.data.pressure = update_median_filter_f(&board_baro_median_pressure, bb_ms5611.data.pressure);
114 #endif
115 AbiSendMsgBARO_ABS(BARO_BOARD_SENDER_ID, now_ts, bb_ms5611.data.pressure);
116
117 bb_ms5611.data_available = false;
118
119#ifdef BARO_LED
120 RunOnceEvery(10, LED_TOGGLE(BARO_LED));
121#endif
122
123#if DEBUG
124 float fbaroms = bb_ms5611.data.pressure / 100.;
125 DOWNLINK_SEND_BARO_MS5611(DefaultChannel, DefaultDevice,
126 &bb_ms5611.data.d1, &bb_ms5611.data.d2,
127 &fbaroms, &temp);
128#endif
129 }
130 }
