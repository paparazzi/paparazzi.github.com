116void imu_mpu_spi_init(void)
117{
118
119#if IMU_MPU_USE_MEDIAN_FILTER
120 InitMedianFilterVect3Int(medianfilter_accel, 3);
121 InitMedianFilterRatesInt(medianfilter_rates, 3);
122#endif
123
124 mpu60x0_spi_init(&imu_mpu_spi.mpu, &IMU_MPU_SPI_DEV, IMU_MPU_SPI_SLAVE_IDX);
125 // change the default configuration
126 imu_mpu_spi.mpu.config.smplrt_div = IMU_MPU_SMPLRT_DIV;
127 imu_mpu_spi.mpu.config.dlpf_cfg = IMU_MPU_LOWPASS_FILTER;
128 imu_mpu_spi.mpu.config.dlpf_cfg_acc = IMU_MPU_ACCEL_LOWPASS_FILTER; // only for ICM sensors
129 imu_mpu_spi.mpu.config.gyro_range = IMU_MPU_GYRO_RANGE;
130 imu_mpu_spi.mpu.config.accel_range = IMU_MPU_ACCEL_RANGE;
131
132 // Set the default scaling
133 imu_set_defaults_gyro(IMU_MPU6000_ID, NULL, NULL, &MPU60X0_GYRO_SENS_F[IMU_MPU_GYRO_RANGE]);
134 imu_set_defaults_accel(IMU_MPU6000_ID, NULL, NULL, &MPU60X0_ACCEL_SENS_F[IMU_MPU_ACCEL_RANGE]);
135}

142void imu_mpu_spi_event(void)
143{
144 mpu60x0_spi_event(&imu_mpu_spi.mpu);
145 if (imu_mpu_spi.mpu.data_available) {
146 uint32_t now_ts = get_sys_time_usec();
147
148 // set channel order
149 struct Int32Vect3 accel = {
150 IMU_MPU_X_SIGN * (int32_t)(imu_mpu_spi.mpu.data_accel.value[IMU_MPU_CHAN_X]),
151 IMU_MPU_Y_SIGN * (int32_t)(imu_mpu_spi.mpu.data_accel.value[IMU_MPU_CHAN_Y]),
152 IMU_MPU_Z_SIGN * (int32_t)(imu_mpu_spi.mpu.data_accel.value[IMU_MPU_CHAN_Z])
153 };
154 struct Int32Rates rates = {
155 IMU_MPU_X_SIGN * (int32_t)(imu_mpu_spi.mpu.data_rates.value[IMU_MPU_CHAN_X]),
156 IMU_MPU_Y_SIGN * (int32_t)(imu_mpu_spi.mpu.data_rates.value[IMU_MPU_CHAN_Y]),
157 IMU_MPU_Z_SIGN * (int32_t)(imu_mpu_spi.mpu.data_rates.value[IMU_MPU_CHAN_Z])
158 };
159
160 // In case sensor exhibits faulty large spike values in raw output remove them
161#if IMU_MPU_USE_MEDIAN_FILTER
162 UpdateMedianFilterVect3Int(medianfilter_accel, accel);
163 UpdateMedianFilterRatesInt(medianfilter_rates, rates);
164#endif
165
166 imu_mpu_spi.mpu.data_available = false;
167
168 // Send the scaled values over ABI
169 AbiSendMsgIMU_GYRO_RAW(IMU_MPU6000_ID, now_ts, &rates, 1, IMU_MPU_SPI_PERIODIC_FREQ, imu_mpu_spi.mpu.temp);
170 AbiSendMsgIMU_ACCEL_RAW(IMU_MPU6000_ID, now_ts, &accel, 1, IMU_MPU_SPI_PERIODIC_FREQ, imu_mpu_spi.mpu.temp);
171 }
