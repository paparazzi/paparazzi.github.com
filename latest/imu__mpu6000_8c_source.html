115 void imu_mpu_spi_init(void)
116 {
117 
118 #if IMU_MPU_USE_MEDIAN_FILTER
119  InitMedianFilterVect3Int(medianfilter_accel, 3);
120  InitMedianFilterRatesInt(medianfilter_rates, 3);
121 #endif
122 
123  mpu60x0_spi_init(&imu_mpu_spi.mpu, &IMU_MPU_SPI_DEV, IMU_MPU_SPI_SLAVE_IDX);
124  // change the default configuration
125  imu_mpu_spi.mpu.config.smplrt_div = IMU_MPU_SMPLRT_DIV;
126  imu_mpu_spi.mpu.config.dlpf_cfg = IMU_MPU_LOWPASS_FILTER;
127  imu_mpu_spi.mpu.config.dlpf_cfg_acc = IMU_MPU_ACCEL_LOWPASS_FILTER; // only for ICM sensors
128  imu_mpu_spi.mpu.config.gyro_range = IMU_MPU_GYRO_RANGE;
129  imu_mpu_spi.mpu.config.accel_range = IMU_MPU_ACCEL_RANGE;
130 
131  // Set the default scaling
132  imu_set_defaults_gyro(IMU_MPU6000_ID, NULL, NULL, MPU60X0_GYRO_SENS_FRAC[IMU_MPU_GYRO_RANGE]);
133  imu_set_defaults_accel(IMU_MPU6000_ID, NULL, NULL, MPU60X0_ACCEL_SENS_FRAC[IMU_MPU_ACCEL_RANGE]);
134 }

141 void imu_mpu_spi_event(void)
142 {
143  mpu60x0_spi_event(&imu_mpu_spi.mpu);
144  if (imu_mpu_spi.mpu.data_available) {
145  uint32_t now_ts = get_sys_time_usec();
146 
147  // set channel order
148  struct Int32Vect3 accel = {
149  IMU_MPU_X_SIGN * (int32_t)(imu_mpu_spi.mpu.data_accel.value[IMU_MPU_CHAN_X]),
150  IMU_MPU_Y_SIGN * (int32_t)(imu_mpu_spi.mpu.data_accel.value[IMU_MPU_CHAN_Y]),
151  IMU_MPU_Z_SIGN * (int32_t)(imu_mpu_spi.mpu.data_accel.value[IMU_MPU_CHAN_Z])
152  };
153  struct Int32Rates rates = {
154  IMU_MPU_X_SIGN * (int32_t)(imu_mpu_spi.mpu.data_rates.value[IMU_MPU_CHAN_X]),
155  IMU_MPU_Y_SIGN * (int32_t)(imu_mpu_spi.mpu.data_rates.value[IMU_MPU_CHAN_Y]),
156  IMU_MPU_Z_SIGN * (int32_t)(imu_mpu_spi.mpu.data_rates.value[IMU_MPU_CHAN_Z])
157  };
158 
159  // In case sensor exhibits faulty large spike values in raw output remove them
160 #if IMU_MPU_USE_MEDIAN_FILTER
161  UpdateMedianFilterVect3Int(medianfilter_accel, accel);
162  UpdateMedianFilterRatesInt(medianfilter_rates, rates);
163 #endif
164 
165  imu_mpu_spi.mpu.data_available = false;
166 
167  // Send the scaled values over ABI
168  AbiSendMsgIMU_GYRO_RAW(IMU_MPU6000_ID, now_ts, &rates, 1, imu_mpu_spi.mpu.temp);
169  AbiSendMsgIMU_ACCEL_RAW(IMU_MPU6000_ID, now_ts, &accel, 1, imu_mpu_spi.mpu.temp);
170  }
