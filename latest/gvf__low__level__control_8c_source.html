35 void gvf_low_level_getState(void)
36 {
37  #if defined(FIXEDWING_FIRMWARE)
38  float ground_speed = stateGetHorizontalSpeedNorm_f();
39  gvf_state.course = stateGetHorizontalSpeedDir_f();
40  gvf_state.px_dot = ground_speed * sinf(gvf_state.course);
41  gvf_state.py_dot = ground_speed * cosf(gvf_state.course);
42 
43  #elif defined(ROVER_FIRMWARE) || defined(ROTORCRAFT_FIRMWARE)
44  // We assume that the course and psi
