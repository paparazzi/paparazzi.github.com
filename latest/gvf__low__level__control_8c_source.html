35void gvf_low_level_getState(void)
36{
37 #if defined(FIXEDWING_FIRMWARE)
38 float ground_speed = stateGetHorizontalSpeedNorm_f();
39 gvf_state.course = stateGetHorizontalSpeedDir_f();
40 gvf_state.px_dot = ground_speed * sinf(gvf_state.course);
41 gvf_state.py_dot = ground_speed * cosf(gvf_state.course);
42
43 #elif defined(ROVER_FIRMWARE) || defined(ROTORCRAFT_FIRMWARE)
44 // We assume that the course and psi
45 // of the rover (steering wheel) are the same
46 gvf_state.course = stateGetNedToBodyEulers_f()->psi;
47 gvf_state.px_dot = stateGetSpeedEnu_f()->x;
48 gvf_state.py_dot = stateGetSpeedEnu_f()->y;
49 #endif
50}

52void gvf_low_level_control_2D(float omega)
53{
54
55#if defined(FIXEDWING_FIRMWARE)
56 if (autopilot_get_mode() == AP_MODE_AUTO2) {
57
58 // Coordinated turn
59 struct FloatEulers *att = stateGetNedToBodyEulers_f();
60 float ground_speed = stateGetHorizontalSpeedNorm_f();
61
62 lateral_mode = LATERAL_MODE_ROLL;
63
64 h_ctl_roll_setpoint =
65 -atanf(omega * ground_speed / GVF_GRAVITY / cosf(att->theta));
66 BoundAbs(h_ctl_roll_setpoint, h_ctl_roll_max_setpoint);
67 }
68#endif
69
70 (void)(omega); // Avoid unused parameter warning
71}
