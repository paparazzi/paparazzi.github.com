81 void imu_ppzuav_init(void)
82 {
83  /* Set accel configuration */
84  adxl345_i2c_init(&imu_ppzuav.acc_adxl, &(IMU_PPZUAV_I2C_DEV), ADXL345_ADDR);
85  /* set the data rate */
86  imu_ppzuav.acc_adxl.config.rate = IMU_PPZUAV_ACCEL_RATE;
87 
88  /* Gyro configuration and initalization */
89  itg3200_init(&imu_ppzuav.gyro_itg, &(IMU_PPZUAV_I2C_DEV), ITG3200_ADDR);
90  /* change the default config */
91  imu_ppzuav.gyro_itg.config.smplrt_div = IMU_PPZUAV_GYRO_SMPLRT_DIV;
92  imu_ppzuav.gyro_itg.config.dlpf_cfg = IMU_PPZUAV_GYRO_LOWPASS;
93 
94  /* initialize mag and set default options */
95  hmc58xx_init(&imu_ppzuav.mag_hmc, &(IMU_PPZUAV_I2C_DEV), HMC58XX_ADDR);
96  /* set the type to the old HMC5843 */
97  imu_ppzuav.mag_hmc.type = HMC_TYPE_5843;
98 }

101 void imu_ppzuav_periodic(void)
102 {
103  adxl345_i2c_periodic(&imu_ppzuav.acc_adxl);
104 
105  // Start reading the latest gyroscope data
106  itg3200_periodic(&imu_ppzuav.gyro_itg);
107 
108  // Read HMC58XX at 50Hz (main loop for rotorcraft: 512Hz)
109  RunOnceEvery(10, hmc58xx_periodic(&imu_ppzuav.mag_hmc));
110 }

112 void imu_ppzuav_event(void)
113 {
114  uint32_t now_ts = get_sys_time_usec();
115 
116  adxl345_i2c_event(&imu_ppzuav.acc_adxl);
117  if (imu_ppzuav.acc_adxl.data_available) {
118  imu.accel_unscaled.x = -imu_ppzuav.acc_adxl.data.vect.x;
119  imu.accel_unscaled.y = imu_ppzuav.acc_adxl.data.vect.y;
120  imu.accel_unscaled.z = -imu_ppzuav.acc_adxl.data.vect.z;
121  imu_ppzuav.acc_adxl.data_available = false;
122  imu_scale_accel(&imu);
123  AbiSendMsgIMU_ACCEL_INT32(IMU_PPZUAV_ID, now_ts, &imu.accel);
124  }
125 
126  /* If the itg3200 I2C transaction has succeeded: convert the data */
127  itg3200_event(&imu_ppzuav.gyro_itg);
128  if (imu_ppzuav.gyro_itg.data_available) {
129  imu.gyro_unscaled.p = -imu_ppzuav.gyro_itg.data.rates.p;
130  imu.gyro_unscaled.q = imu_ppzuav.gyro_itg.data.rates.q;
131  imu.gyro_unscaled.r = -imu_ppzuav.gyro_itg.data.rates.r;
132  imu_ppzuav.gyro_itg.data_available = false;
133  imu_scale_gyro(&imu);
134  AbiSendMsgIMU_GYRO_INT32(IMU_PPZUAV_ID, now_ts, &imu.gyro);
135  }
136 
137  /* HMC58XX event task */
138  hmc58xx_event(&imu_ppzuav.mag_hmc);
139  if (imu_ppzuav.mag_hmc.data_available) {
140  imu.mag_unscaled.x = -imu_ppzuav.mag_hmc.data.vect.y;
141  imu.mag_unscaled.y = -imu_ppzuav.mag_hmc.data.vect.x;
142  imu.mag_unscaled.z = -imu_ppzuav.mag_hmc.data.vect.z;
143  imu_ppzuav.mag_hmc.data_available = false;
144  imu_scale_mag(&imu);
145  AbiSendMsgIMU_MAG_INT32(IMU_PPZUAV_ID, now_ts, &imu.mag);
146  }
147 }
