43 tid_t sys_time_register_timer(float duration, sys_time_cb cb)
44 {
45  uint32_t start_time = sys_time.nb_tick;
46  for (tid_t i = 0; i < SYS_TIME_NB_TIMER; i++) {
47  if (!sys_time.timer[i].in_use) {
48  sys_time.timer[i].cb = cb;
