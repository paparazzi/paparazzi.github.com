228 void guidance_h_mode_changed(uint8_t new_mode)
229 {
230  if (new_mode == guidance_h.mode) {
231  return;
232  }
233 
234 #if HYBRID_NAVIGATION
235  guidance_hybrid_norm_ref_airspeed = 0;
236 #endif
237 
238  switch (new_mode) {
239  case GUIDANCE_H_MODE_RC_DIRECT:
240  stabilization_none_enter();
241  break;
242 
243 #if USE_STABILIZATION_RATE
244  case GUIDANCE_H_MODE_RATE:
245  stabilization_rate_enter();
246  break;
247 #endif
248 
249  case GUIDANCE_H_MODE_CARE_FREE:
250  stabilization_attitude_reset_care_free_heading();
251  /* Falls through. */
252  case GUIDANCE_H_MODE_FORWARD:
253  case GUIDANCE_H_MODE_ATTITUDE:
254 #if NO_ATTITUDE_RESET_ON_MODE_CHANGE
255  /* reset attitude stabilization if previous mode was not using it */
256  if (guidance_h.mode == GUIDANCE_H_MODE_KILL ||
257  guidance_h.mode == GUIDANCE_H_MODE_RATE ||
258  guidance_h.mode == GUIDANCE_H_MODE_RC_DIRECT)
259 #endif
260  stabilization_attitude_enter();
261  break;
262 
263  case GUIDANCE_H_MODE_GUIDED:
264  case GUIDANCE_H_MODE_HOVER:
265 #if GUIDANCE_INDI
266  guidance_indi_enter();
267 #endif
268  guidance_h_hover_enter();
269 #if NO_ATTITUDE_RESET_ON_MODE_CHANGE
270  /* reset attitude stabilization if previous mode was not using it */
271  if (guidance_h.mode == GUIDANCE_H_MODE_KILL ||
272  guidance_h.mode == GUIDANCE_H_MODE_RATE ||
273  guidance_h.mode == GUIDANCE_H_MODE_RC_DIRECT)
274 #endif
275  stabilization_attitude_enter();
276  break;
277 
278 #if GUIDANCE_H_MODE_MODULE_SETTING == GUIDANCE_H_MODE_MODULE
279  case GUIDANCE_H_MODE_MODULE:
280  guidance_h_module_enter();
281  break;
282 #endif
283 
284  case GUIDANCE_H_MODE_NAV:
285  guidance_h_nav_enter();
286 #if NO_ATTITUDE_RESET_ON_MODE_CHANGE
287  /* reset attitude stabilization if previous mode was not using it */
288  if (guidance_h.mode == GUIDANCE_H_MODE_KILL ||
289  guidance_h.mode == GUIDANCE_H_MODE_RATE ||
290  guidance_h.mode == GUIDANCE_H_MODE_RC_DIRECT)
291 #endif
292  stabilization_attitude_enter();
293  break;
294 
295  case GUIDANCE_H_MODE_FLIP:
296  guidance_flip_enter();
297  break;
298 
299  default:
300  break;
301  }
302 
303  guidance_h.mode = new_mode;
304 
305 }

308 void guidance_h_read_rc(bool in_flight)
309 {
310 
311  switch (guidance_h.mode) {
312 
313  case GUIDANCE_H_MODE_RC_DIRECT:
314  stabilization_none_read_rc();
315  break;
316 
317 #if USE_STABILIZATION_RATE
318  case GUIDANCE_H_MODE_RATE:
319 #if SWITCH_STICKS_FOR_RATE_CONTROL
320  stabilization_rate_read_rc_switched_sticks();
321 #else
322  stabilization_rate_read_rc();
323 #endif
324  break;
325 #endif
326 
327  case GUIDANCE_H_MODE_CARE_FREE:
328  stabilization_attitude_read_rc(in_flight, TRUE, FALSE);
329  break;
330  case GUIDANCE_H_MODE_FORWARD:
331  stabilization_attitude_read_rc(in_flight, FALSE, TRUE);
332  break;
333  case GUIDANCE_H_MODE_ATTITUDE:
334  stabilization_attitude_read_rc(in_flight, FALSE, FALSE);
335  break;
336  case GUIDANCE_H_MODE_HOVER:
337  stabilization_attitude_read_rc_setpoint_eulers_f(&guidance_h.rc_sp, in_flight, FALSE, FALSE);
338 #if GUIDANCE_H_USE_SPEED_REF
339  read_rc_setpoint_speed_i(&guidance_h.sp.speed, in_flight);
340  /* enable x,y velocity setpoints */
341  SetBit(guidance_h.sp.mask, 5);
342 #endif
343  break;
344 
345 #if GUIDANCE_H_MODE_MODULE_SETTING == GUIDANCE_H_MODE_MODULE
346  case GUIDANCE_H_MODE_MODULE:
347  guidance_h_module_read_rc();
348  break;
349 #endif
350 
351  case GUIDANCE_H_MODE_NAV:
352  if (radio_control.status == RC_OK) {
353  stabilization_attitude_read_rc_setpoint_eulers_f(&guidance_h.rc_sp, in_flight, FALSE, FALSE);
354  } else {
355  FLOAT_EULERS_ZERO(guidance_h.rc_sp);
356  }
357  break;
358  case GUIDANCE_H_MODE_FLIP:
359  stabilization_attitude_read_rc(in_flight, FALSE, FALSE);
360  break;
361  default:
362  break;
363  }

367 void guidance_h_run(bool in_flight)
368 {
369  switch (guidance_h.mode) {
370 
371  case GUIDANCE_H_MODE_RC_DIRECT:
372  stabilization_none_run(in_flight);
373  break;
374 
375 #if USE_STABILIZATION_RATE
376  case GUIDANCE_H_MODE_RATE:
377  stabilization_rate_run(in_flight);
378  break;
379 #endif
380 
381  case GUIDANCE_H_MODE_FORWARD:
382  if (transition_percentage < (100 << INT32_PERCENTAGE_FRAC)) {
383  transition_run(true);
384  }
385  /* Falls through. */
386  case GUIDANCE_H_MODE_CARE_FREE:
387  case GUIDANCE_H_MODE_ATTITUDE:
388  if ((!(guidance_h.mode == GUIDANCE_H_MODE_FORWARD)) && transition_percentage > 0) {
389  transition_run(false);
390  }
391  stabilization_attitude_run(in_flight);
392 #if (STABILIZATION_FILTER_CMD_ROLL_PITCH || STABILIZATION_FILTER_CMD_YAW)
393  if (in_flight) {
394  stabilization_filter_commands();
395  }
396 #endif
397 
398  break;
399 
400  case GUIDANCE_H_MODE_HOVER:
401  /* set psi command from RC */
402  guidance_h.sp.heading = guidance_h.rc_sp.psi;
403  /* fall trough to GUIDED to update ref, run traj and set final attitude setpoint */
404 
405  /* Falls through. */
406  case GUIDANCE_H_MODE_GUIDED:
407  guidance_h_guided_run(in_flight);
408  break;
409 
410  case GUIDANCE_H_MODE_NAV:
411  guidance_h_from_nav(in_flight);
412  break;
413 
414 #if GUIDANCE_H_MODE_MODULE_SETTING == GUIDANCE_H_MODE_MODULE
415  case GUIDANCE_H_MODE_MODULE:
416  guidance_h_module_run(in_flight);
417  break;
418 #endif
419 
420  case GUIDANCE_H_MODE_FLIP:
421  guidance_flip_run();
422  break;
423 
424  default:
425  break;
426  }
427 }

430 static void guidance_h_update_reference(void)
431 {
432  /* compute reference even if usage temporarily disabled via guidance_h_use_ref */
433 #if GUIDANCE_H_USE_REF
434  if (bit_is_set(guidance_h.sp.mask, 5)) {
435  struct FloatVect2 sp_speed;
436  sp_speed.x = SPEED_FLOAT_OF_BFP(guidance_h.sp.speed.x);
437  sp_speed.y = SPEED_FLOAT_OF_BFP(guidance_h.sp.speed.y);
438  gh_update_ref_from_speed_sp(sp_speed);
439  } else {
440  gh_update_ref_from_pos_sp(guidance_h.sp.pos);
441  }
442 #endif
443 
444  /* either use the reference or simply copy the pos setpoint */
445  if (guidance_h.use_ref) {
446  /* convert our reference to generic representation */
447  INT32_VECT2_RSHIFT(guidance_h.ref.pos, gh_ref.pos, (GH_POS_REF_FRAC - INT32_POS_FRAC));
448  guidance_h.ref.speed.x = SPEED_BFP_OF_REAL(gh_ref.speed.x);
449  guidance_h.ref.speed.y = SPEED_BFP_OF_REAL(gh_ref.speed.y);
450  guidance_h.ref.accel.x = ACCEL_BFP_OF_REAL(gh_ref.accel.x);
451  guidance_h.ref.accel.y = ACCEL_BFP_OF_REAL(gh_ref.accel.y);
452  } else {
453  VECT2_COPY(guidance_h.ref.pos, guidance_h.sp.pos);
454  INT_VECT2_ZERO(guidance_h.ref.speed);
455  INT_VECT2_ZERO(guidance_h.ref.accel);
456  }
457 
458 #if GUIDANCE_H_USE_SPEED_REF
459  if (guidance_h.mode == GUIDANCE_H_MODE_HOVER) {
460  VECT2_COPY(guidance_h.sp.pos, guidance_h.ref.pos); // for display only
461  }
462 #endif
463 
464  /* update heading setpoint from rate */
465  if (bit_is_set(guidance_h.sp.mask, 7)) {
466  guidance_h.sp.heading += guidance_h.sp.heading_rate / PERIODIC_FREQUENCY;
467  FLOAT_ANGLE_NORMALIZE(guidance_h.sp.heading);
468  }
469 }
470 
471 #define MAX_POS_ERR POS_BFP_OF_REAL(16.)
472 #define MAX_SPEED_ERR SPEED_BFP_OF_REAL(16.)
473 
474 #ifndef GUIDANCE_H_THRUST_CMD_FILTER
475 #define GUIDANCE_H_THRUST_CMD_FILTER 10
476 #endif
477 
478 /* with a pgain of 100 and a scale of 2,
479  * you get an angle of 5.6 degrees for 1m pos error */
480 #define GH_GAIN_SCALE 2
481 
482 #if !GUIDANCE_INDI
483 static void guidance_h_traj_run(bool in_flight)
484 {
485  /* maximum bank angle: default 20 deg, max 40 deg*/
486  static const int32_t traj_max_bank = Min(BFP_OF_REAL(GUIDANCE_H_MAX_BANK, INT32_ANGLE_FRAC),
487  BFP_OF_REAL(RadOfDeg(40), INT32_ANGLE_FRAC));
488  static const int32_t total_max_bank = BFP_OF_REAL(RadOfDeg(45), INT32_ANGLE_FRAC);
489 
490  /* compute position error */
491  VECT2_DIFF(guidance_h_pos_err, guidance_h.ref.pos, *stateGetPositionNed_i());
492  /* saturate it */
493  VECT2_STRIM(guidance_h_pos_err, -MAX_POS_ERR, MAX_POS_ERR);
494 
495  /* compute speed error */
496  VECT2_DIFF(guidance_h_speed_err, guidance_h.ref.speed, *stateGetSpeedNed_i());
497  /* saturate it */
498  VECT2_STRIM(guidance_h_speed_err, -MAX_SPEED_ERR, MAX_SPEED_ERR);
499 
500  /* run PID */
501  int32_t pd_x =
502  ((guidance_h.gains.p * guidance_h_pos_err.x) >> (INT32_POS_FRAC - GH_GAIN_SCALE)) +
503  ((guidance_h.gains.d * (guidance_h_speed_err.x >> 2)) >> (INT32_SPEED_FRAC - GH_GAIN_SCALE - 2));
504  int32_t pd_y =
505  ((guidance_h.gains.p * guidance_h_pos_err.y) >> (INT32_POS_FRAC - GH_GAIN_SCALE)) +
506  ((guidance_h.gains.d * (guidance_h_speed_err.y >> 2)) >> (INT32_SPEED_FRAC - GH_GAIN_SCALE - 2));
507  guidance_h_cmd_earth.x = pd_x +
508  ((guidance_h.gains.v * guidance_h.ref.speed.x) >> (INT32_SPEED_FRAC - GH_GAIN_SCALE)) + /* speed feedforward gain */
509  ((guidance_h.gains.a * guidance_h.ref.accel.x) >> (INT32_ACCEL_FRAC -
510  GH_GAIN_SCALE)); /* acceleration feedforward gain */
511  guidance_h_cmd_earth.y = pd_y +
512  ((guidance_h.gains.v * guidance_h.ref.speed.y) >> (INT32_SPEED_FRAC - GH_GAIN_SCALE)) + /* speed feedforward gain */
513  ((guidance_h.gains.a * guidance_h.ref.accel.y) >> (INT32_ACCEL_FRAC -
514  GH_GAIN_SCALE)); /* acceleration feedforward gain */
515 
516  /* trim max bank angle from PD */
517  VECT2_STRIM(guidance_h_cmd_earth, -traj_max_bank, traj_max_bank);
518 
519  /* Update pos & speed error integral, zero it if not in_flight.
520  * Integrate twice as fast when not only POS but also SPEED are wrong,
521  * but do not integrate POS errors when the SPEED is already catching up.
522  */
523  if (in_flight) {
524  /* ANGLE_FRAC (12) * GAIN (8) * LOOP_FREQ (9) -> INTEGRATOR HIGH RES ANGLE_FRAX (28) */
525  guidance_h_trim_att_integrator.x += (guidance_h.gains.i * pd_x);
526  guidance_h_trim_att_integrator.y += (guidance_h.gains.i * pd_y);
527  /* saturate it */
528  VECT2_STRIM(guidance_h_trim_att_integrator, -(traj_max_bank << (INT32_ANGLE_FRAC + GH_GAIN_SCALE * 2)),
529  (traj_max_bank << (INT32_ANGLE_FRAC + GH_GAIN_SCALE * 2)));
530  /* add it to the command */
531  guidance_h_cmd_earth.x += (guidance_h_trim_att_integrator.x >> (INT32_ANGLE_FRAC + GH_GAIN_SCALE * 2));
532  guidance_h_cmd_earth.y += (guidance_h_trim_att_integrator.y >> (INT32_ANGLE_FRAC + GH_GAIN_SCALE * 2));
533  } else {
534  INT_VECT2_ZERO(guidance_h_trim_att_integrator);
535  }
536 
537  /* compute a better approximation of force commands by taking thrust into account */
538  if (guidance_h.approx_force_by_thrust && in_flight) {
539  static int32_t thrust_cmd_filt;
540  int32_t vertical_thrust = (stabilization_cmd[COMMAND_THRUST] * guidance_v_thrust_coeff) >> INT32_TRIG_FRAC;
541  thrust_cmd_filt = (thrust_cmd_filt * GUIDANCE_H_THRUST_CMD_FILTER + vertical_thrust) /
542  (GUIDANCE_H_THRUST_CMD_FILTER + 1);
543  guidance_h_cmd_earth.x = ANGLE_BFP_OF_REAL(atan2f((guidance_h_cmd_earth.x * MAX_PPRZ / INT32_ANGLE_PI_2),
544  thrust_cmd_filt));
545  guidance_h_cmd_earth.y = ANGLE_BFP_OF_REAL(atan2f((guidance_h_cmd_earth.y * MAX_PPRZ / INT32_ANGLE_PI_2),
546  thrust_cmd_filt));

662 static void read_rc_setpoint_speed_i(struct Int32Vect2 *speed_sp, bool in_flight)
663 {
664  if (in_flight) {
665  // negative pitch is forward
666  int64_t rc_x = -radio_control.values[RADIO_PITCH];
667  int64_t rc_y = radio_control.values[RADIO_ROLL];
668  DeadBand(rc_x, MAX_PPRZ / 20);
669  DeadBand(rc_y, MAX_PPRZ / 20);
670 
671  // convert input from MAX_PPRZ range to SPEED_BFP
672  int32_t max_speed = SPEED_BFP_OF_REAL(GUIDANCE_H_REF_MAX_SPEED);
674  //int32_t rc_norm = sqrtf(rc_x * rc_x + rc_y * rc_y);
675  //int32_t max_pprz = rc_norm * MAX_PPRZ / Max(abs(rc_x), abs(rc_y);
676  rc_x = rc_x * max_speed / MAX_PPRZ;
677  rc_y = rc_y * max_speed / MAX_PPRZ;
678 
679  /* Rotate from body to NED frame by negative psi angle */
680  int32_t psi = -stateGetNedToBodyEulers_i()->psi;
681  int32_t s_psi, c_psi;
682  PPRZ_ITRIG_SIN(s_psi, psi);
683  PPRZ_ITRIG_COS(c_psi, psi);
684  speed_sp->x = (int32_t)(((int64_t)c_psi * rc_x + (int64_t)s_psi * rc_y) >> INT32_TRIG_FRAC);
685  speed_sp->y = (int32_t)((-(int64_t)s_psi * rc_x + (int64_t)c_psi * rc_y) >> INT32_TRIG_FRAC);
686  } else {
687  speed_sp->x = 0;
688  speed_sp->y = 0;
689  }
690 }
