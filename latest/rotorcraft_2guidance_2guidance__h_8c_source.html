167static void read_rc_setpoint_heading(struct HorizontalGuidanceRCInput *rc_sp, bool in_flight, struct RadioControl *rc)
168{
169 if (in_flight) {
170 /* calculate dt for yaw integration */
171 float dt = get_sys_time_float() - rc_sp->last_ts;
172 /* make sure nothing drastically weird happens, bound dt to 0.5sec */
173 Bound(dt, 0, 0.5);
174
175 /* do not advance yaw setpoint if within a small deadband around stick center or if throttle is zero */
176 if (YAW_DEADBAND_EXCEEDED(rc) && !THROTTLE_STICK_DOWN_FROM_RC(rc)) {
177 float heading_rate = (float) rc->values[RADIO_YAW] * GUIDANCE_H_SP_MAX_R / MAX_PPRZ;
178 rc_sp->heading += heading_rate * dt;
179 FLOAT_ANGLE_NORMALIZE(rc_sp->heading);
180 }
181 } else { /* if not flying, use current yaw as setpoint */
182 rc_sp->heading = stateGetNedToBodyEulers_f()->psi;
183 }
184 /* update timestamp for dt calculation */
185 rc_sp->last_ts = get_sys_time_float();
186}

189static void read_rc_setpoint_speed_i(struct Int32Vect2 *speed_sp, bool in_flight, struct RadioControl *rc)
190{
191 if (in_flight) {
192 // negative pitch is forward
193 int64_t rc_x = -rc->values[RADIO_PITCH];
194 int64_t rc_y = rc->values[RADIO_ROLL];
195 DeadBand(rc_x, MAX_PPRZ / 20);
196 DeadBand(rc_y, MAX_PPRZ / 20);
197
198 // convert input from MAX_PPRZ range to SPEED_BFP
199 int32_t max_speed = SPEED_BFP_OF_REAL(GUIDANCE_H_REF_MAX_SPEED);
201 //int32_t rc_norm = sqrtf(rc_x * rc_x + rc_y * rc_y);
202 //int32_t max_pprz = rc_norm * MAX_PPRZ / Max(abs(rc_x), abs(rc_y);
203 rc_x = rc_x * max_speed / MAX_PPRZ;
204 rc_y = rc_y * max_speed / MAX_PPRZ;
205
206 /* Rotate from body to NED frame by negative psi angle */
207 int32_t psi = -stateGetNedToBodyEulers_i()->psi;
208 int32_t s_psi, c_psi;
209 PPRZ_ITRIG_SIN(s_psi, psi);
210 PPRZ_ITRIG_COS(c_psi, psi);
211 speed_sp->x = (int32_t)(((int64_t)c_psi * rc_x + (int64_t)s_psi * rc_y) >> INT32_TRIG_FRAC);
212 speed_sp->y = (int32_t)((-(int64_t)s_psi * rc_x + (int64_t)c_psi * rc_y) >> INT32_TRIG_FRAC);
213 } else {
214 speed_sp->x = 0;
215 speed_sp->y = 0;
216 }
217}

219static void rc_cb(uint8_t sender_id UNUSED, struct RadioControl *rc)
220{
221 switch (guidance_h.mode) {
222
223 case GUIDANCE_H_MODE_HOVER:
224 read_rc_setpoint_heading(&guidance_h.rc_sp, autopilot_in_flight(), rc);
225 read_rc_setpoint_speed_i(&guidance_h.rc_sp.vect, autopilot_in_flight(), rc);
226#if GUIDANCE_H_USE_SPEED_REF
227 /* enable x,y velocity setpoints */
228 guidance_h.sp.speed = guidance_h.rc_sp.vect;
229 guidance_h.sp.h_mask = GUIDANCE_H_SP_SPEED;
230#endif
231 break;
232 case GUIDANCE_H_MODE_NAV:
233 INT_VECT2_ZERO(guidance_h.rc_sp.vect);
234 if (radio_control.status == RC_OK) {
235 read_rc_setpoint_heading(&guidance_h.rc_sp, autopilot_in_flight(), rc);
236 }
237 break;
238 default:
239 break;
240 }

244struct StabilizationSetpoint guidance_h_run(bool in_flight)
245{
246 struct StabilizationSetpoint sp;
247 STAB_SP_SET_EULERS_ZERO(sp);
248
249 switch (guidance_h.mode) {
250
251 case GUIDANCE_H_MODE_HOVER:
252 /* set psi command from RC */
253 guidance_h.sp.heading = guidance_h.rc_sp.heading;
254 /* fall trough to GUIDED to update ref, run traj and set final attitude setpoint */
255
256 /* Falls through. */
257 case GUIDANCE_H_MODE_GUIDED:
258 sp = guidance_h_guided_run(in_flight);
259 break;
260
261 case GUIDANCE_H_MODE_NAV:
262 sp = guidance_h_from_nav(in_flight);
263 break;
264
265 default:
266 break;
267 }
268
269 return sp;
270}

273static void guidance_h_update_reference(void)
274{
275 /* compute reference even if usage temporarily disabled via guidance_h_use_ref */
276#if GUIDANCE_H_USE_REF
277 if (guidance_h.sp.h_mask == GUIDANCE_H_SP_ACCEL) {
278 struct FloatVect2 sp_accel_local;
279 sp_accel_local.x = ACCEL_FLOAT_OF_BFP(guidance_h.sp.accel.x);
280 sp_accel_local.y = ACCEL_FLOAT_OF_BFP(guidance_h.sp.accel.y);
281 gh_update_ref_from_accel_sp(sp_accel_local);
282 }
283 else if (guidance_h.sp.h_mask == GUIDANCE_H_SP_SPEED) {
284 struct FloatVect2 sp_speed;
285 sp_speed.x = SPEED_FLOAT_OF_BFP(guidance_h.sp.speed.x);
286 sp_speed.y = SPEED_FLOAT_OF_BFP(guidance_h.sp.speed.y);
287 gh_update_ref_from_speed_sp(sp_speed);
288 } else {
289 gh_update_ref_from_pos_sp(guidance_h.sp.pos);
290 }
291#endif
292
293 /* either use the reference or simply copy the pos setpoint */
294 if (guidance_h.use_ref) {
295 /* convert our reference to generic representation */
296 INT32_VECT2_RSHIFT(guidance_h.ref.pos, gh_ref.pos, (GH_POS_REF_FRAC - INT32_POS_FRAC));
297 guidance_h.ref.speed.x = SPEED_BFP_OF_REAL(gh_ref.speed.x);
298 guidance_h.ref.speed.y = SPEED_BFP_OF_REAL(gh_ref.speed.y);
299 guidance_h.ref.accel.x = ACCEL_BFP_OF_REAL(gh_ref.accel.x);
300 guidance_h.ref.accel.y = ACCEL_BFP_OF_REAL(gh_ref.accel.y);
301 } else {
302 switch (nav.setpoint_mode) {
303 case NAV_SETPOINT_MODE_SPEED:
304 guidance_h.ref.pos.x = stateGetPositionNed_i()->x;
305 guidance_h.ref.pos.y = stateGetPositionNed_i()->y;
306 guidance_h.ref.speed.x = guidance_h.sp.speed.x;
307 guidance_h.ref.speed.y = guidance_h.sp.speed.y;
308 guidance_h.ref.accel.x = 0;
309 guidance_h.ref.accel.y = 0;
310 break;
311
312 case NAV_SETPOINT_MODE_ACCEL:
313 guidance_h.ref.pos.x = stateGetPositionNed_i()->x;
314 guidance_h.ref.pos.y = stateGetPositionNed_i()->y;
315 guidance_h.ref.speed.x = stateGetSpeedNed_i()->x;
316 guidance_h.ref.speed.y = stateGetSpeedNed_i()->y;
317 guidance_h.ref.accel.x = guidance_h.sp.accel.x;
318 guidance_h.ref.accel.y = guidance_h.sp.accel.y;
319 break;
320
321 case NAV_SETPOINT_MODE_POS:
322 default: // Fallback is guidance by pos
323 VECT2_COPY(guidance_h.ref.pos, guidance_h.sp.pos);
324 INT_VECT2_ZERO(guidance_h.ref.speed);
325 INT_VECT2_ZERO(guidance_h.ref.accel);
326 break;
327 }
328 }
329
330#if GUIDANCE_H_USE_SPEED_REF
331 if (guidance_h.mode == GUIDANCE_H_MODE_HOVER) {
332 VECT2_COPY(guidance_h.sp.pos, guidance_h.ref.pos); // for display only
333 }
334#endif
335
336 /* update heading setpoint from rate */
337 if (guidance_h.sp.yaw_mask == GUIDANCE_H_SP_YAW_RATE) {
338 guidance_h.sp.heading += guidance_h.sp.heading_rate / PERIODIC_FREQUENCY;
339 FLOAT_ANGLE_NORMALIZE(guidance_h.sp.heading);
340 }
341}

343void guidance_h_hover_enter(void)
344{
345 /* reset speed setting */
346 guidance_h.sp.speed.x = 0;
347 guidance_h.sp.speed.y = 0;
348
349 /* set horizontal setpoint to current position */
350 guidance_h_set_pos(
351 stateGetPositionNed_f()->x,
352 stateGetPositionNed_f()->y);
353 /* reset guidance reference */
354 reset_guidance_reference_from_current_position();
355
356 /* set guidance to current heading and position */
357 guidance_h.rc_sp.heading = stateGetNedToBodyEulers_f()->psi;
358 guidance_h_set_heading(stateGetNedToBodyEulers_f()->psi);
359
360 /* call specific implementation */
361 guidance_h_run_enter();
362}

375struct StabilizationSetpoint guidance_h_from_nav(bool in_flight)
376{
377 if (!in_flight) {
378 guidance_h_nav_enter();
379 }
380
381 if (nav.fp_max_speed > 0.f) {
382 guidance_h_SetMaxSpeed(nav.fp_max_speed);
383 }
384 if (nav.horizontal_mode == NAV_HORIZONTAL_MODE_NONE) {
385 struct StabilizationSetpoint sp;
386 STAB_SP_SET_EULERS_ZERO(sp);
387 return sp; // don't call guidance, still return attitude zero
388 } else if (nav.horizontal_mode == NAV_HORIZONTAL_MODE_ATTITUDE) {
389 if (nav.setpoint_mode == NAV_SETPOINT_MODE_QUAT) {
390 return stab_sp_from_quat_f(&nav.quat);
391 }
392 else {
393 // it should be nav.setpoint_mode == NAV_SETPOINT_MODE_ATTITUDE
394 // TODO error handling ?
395 struct FloatEulers sp_cmd_f = {
396 .phi = nav.roll,
397 .theta = nav.pitch,
398 .psi = nav.heading
399 };
400 return stab_sp_from_eulers_f(&sp_cmd_f);
401 }
402 } else if (nav.horizontal_mode == NAV_HORIZONTAL_MODE_GUIDED) {
403 return guidance_h_guided_run(in_flight);
404 } else {
405 // update carrot for GCS display and convert ENU float -> NED int
406 // even if sp is changed later
407 guidance_h.sp.pos.x = POS_BFP_OF_REAL(nav.carrot.y);
408 guidance_h.sp.pos.y = POS_BFP_OF_REAL(nav.carrot.x);
409
410 switch (nav.setpoint_mode) {
411 case NAV_SETPOINT_MODE_POS:
412 guidance_h_set_pos(nav.carrot.y, nav.carrot.x); // nav pos is in ENU frame, convert to NED
413 guidance_h_update_reference();
414 guidance_h_set_heading(nav.heading);
415 guidance_h_cmd = guidance_h_run_pos(in_flight, &guidance_h);
416 break;
417
418 case NAV_SETPOINT_MODE_SPEED:
419 guidance_h_set_vel(nav.speed.y, nav.speed.x); // nav speed is in ENU frame, convert to NED
420 guidance_h_update_reference();
421 guidance_h_set_heading(nav.heading);
422 guidance_h_cmd = guidance_h_run_speed(in_flight, &guidance_h);
423 break;
424
425 case NAV_SETPOINT_MODE_ACCEL:
426 guidance_h_set_acc(nav.accel.y, nav.accel.x); // nav acc is in ENU frame, convert to NED
427 guidance_h_update_reference();
428 guidance_h_set_heading(nav.heading);
429 guidance_h_cmd = guidance_h_run_accel(in_flight, &guidance_h);
430 break;
431
432 default:
433 // nothing to do for other cases at the moment
434 break;
435 }
436 /* return final attitude setpoint */
437 return guidance_h_cmd;
438 }
439}

441struct StabilizationSetpoint guidance_h_guided_run(bool in_flight)
442{
443 /* guidance_h.sp.pos and guidance_h.sp.heading need to be set from external source */
444 if (!in_flight) {
445 guidance_h_hover_enter();
446 }
447
448 guidance_h_update_reference();
449
450 guidance_h_cmd = guidance_h_run_pos(in_flight, &guidance_h);
451 /* return final attitude setpoint */
452 return guidance_h_cmd;
