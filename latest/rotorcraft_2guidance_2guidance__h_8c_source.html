145 void guidance_h_mode_changed(uint8_t new_mode)
146 {
147  if (new_mode == guidance_h.mode) {
148  return;
149  }
150 
151  switch (new_mode) {
152  case GUIDANCE_H_MODE_RC_DIRECT:
153  stabilization_none_enter();
154  break;
155 
156 #if USE_STABILIZATION_RATE
157  case GUIDANCE_H_MODE_RATE:
158  stabilization_rate_enter();
159  break;
160 #endif
161 
162  case GUIDANCE_H_MODE_CARE_FREE:
163  stabilization_attitude_reset_care_free_heading();
164  /* Falls through. */
165  case GUIDANCE_H_MODE_FORWARD:
166  case GUIDANCE_H_MODE_ATTITUDE:
167 #if NO_ATTITUDE_RESET_ON_MODE_CHANGE
168  /* reset attitude stabilization if previous mode was not using it */
169  if (guidance_h.mode == GUIDANCE_H_MODE_KILL ||
170  guidance_h.mode == GUIDANCE_H_MODE_RATE ||
171  guidance_h.mode == GUIDANCE_H_MODE_RC_DIRECT)
172 #endif
173  stabilization_attitude_enter();
174  break;
175 
176  case GUIDANCE_H_MODE_GUIDED:
177  case GUIDANCE_H_MODE_HOVER:
178  guidance_h_hover_enter();
179 #if NO_ATTITUDE_RESET_ON_MODE_CHANGE
180  /* reset attitude stabilization if previous mode was not using it */
181  if (guidance_h.mode == GUIDANCE_H_MODE_KILL ||
182  guidance_h.mode == GUIDANCE_H_MODE_RATE ||
183  guidance_h.mode == GUIDANCE_H_MODE_RC_DIRECT)
184 #endif
185  stabilization_attitude_enter();
186  break;
187 
188 #if GUIDANCE_H_MODE_MODULE_SETTING == GUIDANCE_H_MODE_MODULE
189  case GUIDANCE_H_MODE_MODULE:
190  guidance_h_module_enter();
191  break;
192 #endif
193 
194  case GUIDANCE_H_MODE_NAV:
195  guidance_h_nav_enter();
196 #if NO_ATTITUDE_RESET_ON_MODE_CHANGE
197  /* reset attitude stabilization if previous mode was not using it */
198  if (guidance_h.mode == GUIDANCE_H_MODE_KILL ||
199  guidance_h.mode == GUIDANCE_H_MODE_RATE ||
200  guidance_h.mode == GUIDANCE_H_MODE_RC_DIRECT)
201 #endif
202  stabilization_attitude_enter();
203  break;
204 
205  case GUIDANCE_H_MODE_FLIP:
206  guidance_flip_enter();
207  break;
208 
209  default:
210  break;
211  }
212 
213  guidance_h.mode = new_mode;
214

218 void guidance_h_read_rc(bool in_flight)
219 {
220 
221  switch (guidance_h.mode) {
222 
223  case GUIDANCE_H_MODE_RC_DIRECT:
224  stabilization_none_read_rc();
225  break;
226 
227 #if USE_STABILIZATION_RATE
228  case GUIDANCE_H_MODE_RATE:
229 #if SWITCH_STICKS_FOR_RATE_CONTROL
230  stabilization_rate_read_rc_switched_sticks();
231 #else
232  stabilization_rate_read_rc();
233 #endif
234  break;
235 #endif
236 
237  case GUIDANCE_H_MODE_CARE_FREE:
238  stabilization_attitude_read_rc(in_flight, TRUE, FALSE);
239  break;
240  case GUIDANCE_H_MODE_FORWARD:
241  stabilization_attitude_read_rc(in_flight, FALSE, TRUE);
242  break;
243  case GUIDANCE_H_MODE_ATTITUDE:
244  stabilization_attitude_read_rc(in_flight, FALSE, FALSE);
245  break;
246  case GUIDANCE_H_MODE_HOVER:
247  stabilization_attitude_read_rc_setpoint_eulers_f(&guidance_h.rc_sp, in_flight, FALSE, FALSE);
248 #if GUIDANCE_H_USE_SPEED_REF
249  read_rc_setpoint_speed_i(&guidance_h.sp.speed, in_flight);
250  /* enable x,y velocity setpoints */
251  SetBit(guidance_h.sp.mask, 5);
252 #endif
253  break;
254 
255 #if GUIDANCE_H_MODE_MODULE_SETTING == GUIDANCE_H_MODE_MODULE
256  case GUIDANCE_H_MODE_MODULE:
257  guidance_h_module_read_rc();
258  break;
259 #endif
260 
261  case GUIDANCE_H_MODE_NAV:
262  if (radio_control.status == RC_OK) {
263  stabilization_attitude_read_rc_setpoint_eulers_f(&guidance_h.rc_sp, in_flight, FALSE, FALSE);
264  } else {
265  FLOAT_EULERS_ZERO(guidance_h.rc_sp);
266  }
267  break;
268  case GUIDANCE_H_MODE_FLIP:
269  stabilization_attitude_read_rc(in_flight, FALSE, FALSE);
270  break;
271  default:
272  break;
273  }

277 void guidance_h_run(bool in_flight)
278 {
279  switch (guidance_h.mode) {
280 
281  case GUIDANCE_H_MODE_RC_DIRECT:
282  stabilization_none_run(in_flight);
283  break;
284 
285 #if USE_STABILIZATION_RATE
286  case GUIDANCE_H_MODE_RATE:
287  stabilization_rate_run(in_flight);
288  break;
289 #endif
290 
291  case GUIDANCE_H_MODE_FORWARD:
292  if (transition_percentage < (100 << INT32_PERCENTAGE_FRAC)) {
293  transition_run(true);
294  }
295  /* Falls through. */
296  case GUIDANCE_H_MODE_CARE_FREE:
297  case GUIDANCE_H_MODE_ATTITUDE:
298  if ((!(guidance_h.mode == GUIDANCE_H_MODE_FORWARD)) && transition_percentage > 0) {
299  transition_run(false);
300  }
301  stabilization_attitude_run(in_flight);
302 #if (STABILIZATION_FILTER_CMD_ROLL_PITCH || STABILIZATION_FILTER_CMD_YAW)
303  if (in_flight) {
304  stabilization_filter_commands();
305  }
306 #endif
307 
308  break;
309 
310  case GUIDANCE_H_MODE_HOVER:
311  /* set psi command from RC */
312  guidance_h.sp.heading = guidance_h.rc_sp.psi;
313  /* fall trough to GUIDED to update ref, run traj and set final attitude setpoint */
314 
315  /* Falls through. */
316  case GUIDANCE_H_MODE_GUIDED:
317  guidance_h_guided_run(in_flight);
318  break;
319 
320  case GUIDANCE_H_MODE_NAV:
321  guidance_h_from_nav(in_flight);
322  break;
323 
324 #if GUIDANCE_H_MODE_MODULE_SETTING == GUIDANCE_H_MODE_MODULE
325  case GUIDANCE_H_MODE_MODULE:
326  guidance_h_module_run(in_flight);
327  break;
328 #endif
329 
330  case GUIDANCE_H_MODE_FLIP:
331  guidance_flip_run();
332  break;
333 
334  default:
335  break;
336  }
337 }

340 static void guidance_h_update_reference(void)
341 {
342  /* compute reference even if usage temporarily disabled via guidance_h_use_ref */
343 #if GUIDANCE_H_USE_REF
344  if (bit_is_set(guidance_h.sp.mask, 5)) {
345  struct FloatVect2 sp_speed;
346  sp_speed.x = SPEED_FLOAT_OF_BFP(guidance_h.sp.speed.x);
347  sp_speed.y = SPEED_FLOAT_OF_BFP(guidance_h.sp.speed.y);
348  gh_update_ref_from_speed_sp(sp_speed);
349  } else {
350  gh_update_ref_from_pos_sp(guidance_h.sp.pos);
351  }
352 #endif
353 
354  /* either use the reference or simply copy the pos setpoint */
355  if (guidance_h.use_ref) {
356  /* convert our reference to generic representation */
357  INT32_VECT2_RSHIFT(guidance_h.ref.pos, gh_ref.pos, (GH_POS_REF_FRAC - INT32_POS_FRAC));
358  guidance_h.ref.speed.x = SPEED_BFP_OF_REAL(gh_ref.speed.x);
359  guidance_h.ref.speed.y = SPEED_BFP_OF_REAL(gh_ref.speed.y);
360  guidance_h.ref.accel.x = ACCEL_BFP_OF_REAL(gh_ref.accel.x);
361  guidance_h.ref.accel.y = ACCEL_BFP_OF_REAL(gh_ref.accel.y);
362  } else {
363  if (nav.setpoint_mode == NAV_SETPOINT_MODE_POS) {
364  VECT2_COPY(guidance_h.ref.pos, guidance_h.sp.pos);
365  INT_VECT2_ZERO(guidance_h.ref.speed);
366  INT_VECT2_ZERO(guidance_h.ref.accel);
367  } else { //(nav.setpoint_mode == NAV_SETPOINT_MODE_SPEED)
368  guidance_h.ref.pos.x = stateGetPositionNed_i()->x;
369  guidance_h.ref.pos.y = stateGetPositionNed_i()->y;
370  guidance_h.ref.speed.x = guidance_h.sp.speed.x;
371  guidance_h.ref.speed.y = guidance_h.sp.speed.y;
372  guidance_h.ref.accel.x = 0;
373  guidance_h.ref.accel.y = 0;
374  } // TODO: make accel ref set
375  }
376 
377 #if GUIDANCE_H_USE_SPEED_REF
378  if (guidance_h.mode == GUIDANCE_H_MODE_HOVER) {
379  VECT2_COPY(guidance_h.sp.pos, guidance_h.ref.pos); // for display only
380  }
381 #endif
382 
383  /* update heading setpoint from rate */
384  if (bit_is_set(guidance_h.sp.mask, 7)) {
385  guidance_h.sp.heading += guidance_h.sp.heading_rate / PERIODIC_FREQUENCY;
386  FLOAT_ANGLE_NORMALIZE(guidance_h.sp.heading);
387  }
388 }

390 void guidance_h_hover_enter(void)
391 {
392  /* reset speed setting */
393  guidance_h.sp.speed.x = 0;
394  guidance_h.sp.speed.y = 0;
395 
396  /* set horizontal setpoint to current position */
397  guidance_h_set_pos(
398  stateGetPositionNed_f()->x,
399  stateGetPositionNed_f()->y);
400  /* reset guidance reference */
401  reset_guidance_reference_from_current_position();
402 
403  /* set guidance to current heading and position */
404  guidance_h.rc_sp.psi = stateGetNedToBodyEulers_f()->psi;
405  guidance_h_set_heading(stateGetNedToBodyEulers_f()->psi);
406 
407  /* call specific implementation */
408  guidance_h_run_enter();
409 }

422 void guidance_h_from_nav(bool in_flight)
423 {
424  if (!in_flight) {
425  guidance_h_nav_enter();
426  }
427 
428  if (nav.horizontal_mode == NAV_HORIZONTAL_MODE_MANUAL) {
429  stabilization_cmd[COMMAND_ROLL] = nav.cmd_roll;
430  stabilization_cmd[COMMAND_PITCH] = nav.cmd_pitch;
431  stabilization_cmd[COMMAND_YAW] = nav.cmd_yaw;
432  } else if (nav.horizontal_mode == NAV_HORIZONTAL_MODE_ATTITUDE) {
433  if (nav.setpoint_mode == NAV_SETPOINT_MODE_QUAT) {
434  // directly apply quat setpoint
435  struct Int32Quat quat_i;
436  QUAT_BFP_OF_REAL(quat_i, nav.quat);
437  stabilization_attitude_set_quat_setpoint_i(&quat_i);
438  stabilization_attitude_run(in_flight);
439  }
440  else {
441  // it should be nav.setpoint_mode == NAV_SETPOINT_MODE_ATTITUDE
442  // TODO error handling ?
443  struct Int32Eulers sp_cmd_i;
444  sp_cmd_i.phi = ANGLE_BFP_OF_REAL(nav.roll);
445  sp_cmd_i.theta = ANGLE_BFP_OF_REAL(nav.pitch);
446  sp_cmd_i.psi = ANGLE_BFP_OF_REAL(nav.heading);
447  stabilization_attitude_set_rpy_setpoint_i(&sp_cmd_i);
448  stabilization_attitude_run(in_flight);
449  }
450  } else if (nav.horizontal_mode == NAV_HORIZONTAL_MODE_GUIDED) {
451  guidance_h_guided_run(in_flight);
452  } else {
453  // update carrot for display, even if sp is changed in speed mode
454  guidance_h_set_pos(nav.carrot.y, nav.carrot.x);
455  switch (nav.setpoint_mode) {
456  case NAV_SETPOINT_MODE_POS:
457  // set guidance in NED
458  guidance_h_update_reference();
459  guidance_h_set_heading(nav.heading);
460  guidance_h_cmd = guidance_h_run_pos(in_flight, &guidance_h);
461  break;
462 
463  case NAV_SETPOINT_MODE_SPEED:
464  guidance_h_set_vel(nav.speed.y, nav.speed.x); // nav speed is in ENU frame, convert to NED
465  guidance_h_update_reference();
466  guidance_h_set_heading(nav.heading);
467  guidance_h_cmd = guidance_h_run_speed(in_flight, &guidance_h);
468  break;
469 
470  case NAV_SETPOINT_MODE_ACCEL:
471  // TODO set_accel ref
472  guidance_h_set_heading(nav.heading);
473  guidance_h_cmd = guidance_h_run_accel(in_flight, &guidance_h);
474  break;
475 
476  default:
477  // nothing to do for other cases at the moment
478  break;
479  }
480  /* set final attitude setpoint */
481  stabilization_attitude_set_stab_sp(&guidance_h_cmd);
482  stabilization_attitude_run(in_flight);
483 
484  }

487 static inline void transition_run(bool to_forward)
488 {
489  if (to_forward) {
490  //Add 0.00625%
491  transition_percentage += 1 << (INT32_PERCENTAGE_FRAC - 4);
492  } else {
493  //Subtract 0.00625%
494  transition_percentage -= 1 << (INT32_PERCENTAGE_FRAC - 4);
495  }
496 
497 #ifdef TRANSITION_MAX_OFFSET
498  const int32_t max_offset = ANGLE_BFP_OF_REAL(TRANSITION_MAX_OFFSET);
499  transition_theta_offset = INT_MULT_RSHIFT((transition_percentage << (INT32_ANGLE_FRAC - INT32_PERCENTAGE_FRAC)) / 100,
500  max_offset, INT32_ANGLE_FRAC);
501 #endif
502 }

505 static void read_rc_setpoint_speed_i(struct Int32Vect2 *speed_sp, bool in_flight)
506 {
507  if (in_flight) {
508  // negative pitch is forward
509  int64_t rc_x = -radio_control.values[RADIO_PITCH];
510  int64_t rc_y = radio_control.values[RADIO_ROLL];
511  DeadBand(rc_x, MAX_PPRZ / 20);
512  DeadBand(rc_y, MAX_PPRZ / 20);
513 
514  // convert input from MAX_PPRZ range to SPEED_BFP
515  int32_t max_speed = SPEED_BFP_OF_REAL(GUIDANCE_H_REF_MAX_SPEED);
517  //int32_t rc_norm = sqrtf(rc_x * rc_x + rc_y * rc_y);
518  //int32_t max_pprz = rc_norm * MAX_PPRZ / Max(abs(rc_x), abs(rc_y);
519  rc_x = rc_x * max_speed / MAX_PPRZ;
520  rc_y = rc_y * max_speed / MAX_PPRZ;
521 
522  /* Rotate from body to NED frame by negative psi angle */
523  int32_t psi = -stateGetNedToBodyEulers_i()->psi;
524  int32_t s_psi, c_psi;
525  PPRZ_ITRIG_SIN(s_psi, psi);
526  PPRZ_ITRIG_COS(c_psi, psi);
527  speed_sp->x = (int32_t)(((int64_t)c_psi * rc_x + (int64_t)s_psi * rc_y) >> INT32_TRIG_FRAC);
528  speed_sp->y = (int32_t)((-(int64_t)s_psi * rc_x + (int64_t)c_psi * rc_y) >> INT32_TRIG_FRAC);
529  } else {
530  speed_sp->x = 0;
531  speed_sp->y = 0;
532  }
533 }
