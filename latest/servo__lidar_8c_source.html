56#ifdef COMMAND_SERVO
57void servoLidar_periodic(void) {
58 if (get_sys_time_msec() > servoLidar.last_update + servoLidar.speed) {
59 servoLidar.last_update = get_sys_time_msec();
60
61 if(servoLidar.enabled) {
62 // Update servo position
63 servoLidar.position += (servoLidar.direction == 0) ? 100 : -100;
64 if (servoLidar.position >= MAX_PPRZ*0.8 || servoLidar.position <= -MAX_PPRZ*0.8) {
65 servoLidar.direction ^= 1;
66 }
67
68 // Set servo command
69 commands[COMMAND_SERVO] = servoLidar.position;
70 servoLidar.angle = PWM2ANGLE(servoLidar.position);
71
72 // Send ABI message
73 AbiSendMsgOBSTACLE_DETECTION(AGL_LIDAR_TFMINI_ID, tfmini.distance, servoLidar.angle, 0);
74 }
75 else {
76 commands[COMMAND_SERVO] = 0; // Center servo when disabled
77 }
78 }
79}
80#else
81void servoLidar_periodic(void) {
82 // No servo functionality; do nothing
83 #error "You need to define COMMAND_SERVO to use servoLidar_periodic()"
84}
