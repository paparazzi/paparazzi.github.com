59 static void nav_hybrid_goto(struct EnuCoor_f *wp)
60 {
61  nav_rotorcraft_base.goto_wp.to = *wp;
62  nav_rotorcraft_base.goto_wp.dist2_to_wp = get_dist2_to_point(wp);
63  VECT2_COPY(nav.target, *wp);
64 
65  // Calculate position error
66  struct FloatVect2 pos_error;
67  struct EnuCoor_f *pos = stateGetPositionEnu_f();
68  VECT2_DIFF(pos_error, nav.target, *pos);
69 
70  struct FloatVect2 speed_sp;
71  VECT2_SMUL(speed_sp, pos_error, gih_params.pos_gain);
72 
73  if (force_forward) {
74  float_vect2_scale_in_2d(&speed_sp, nav_max_speed);
75  } else {
76  // Calculate distance to waypoint
77  float dist_to_wp = float_vect2_norm(&pos_error);
78  // Calculate max speed to decelerate from
79  float max_speed_decel2 = fabsf(2.f * dist_to_wp * MAX_DECELERATION); // dist_to_wp can only be positive, but just in case
80  float max_speed_decel = sqrtf(max_speed_decel2);
81  // Bound the setpoint velocity vector
82  float max_h_speed = Min(nav_max_speed, max_speed_decel);
83  float_vect2_bound_in_2d(&speed_sp, max_h_speed);
84  }
85 
86  VECT2_COPY(nav.speed, speed_sp);
87  nav.horizontal_mode = NAV_HORIZONTAL_MODE_WAYPOINT;
88  nav.setpoint_mode = NAV_SETPOINT_MODE_SPEED;
89 }

91 static void nav_hybrid_route(struct EnuCoor_f *wp_start, struct EnuCoor_f *wp_end)
92 {
93  struct FloatVect2 wp_diff, pos_diff;
94  VECT2_DIFF(wp_diff, *wp_end, *wp_start);
95  VECT2_DIFF(pos_diff, *wp_end, *stateGetPositionEnu_f());
96 
97  // Calculate magnitude of the desired speed vector based on distance to waypoint
98  float dist_to_target = float_vect2_norm(&pos_diff);
99  float desired_speed;
100  if (force_forward) {
101  desired_speed = nav_max_speed;
102  } else {
103  desired_speed = dist_to_target * gih_params.pos_gain;
104  Bound(desired_speed, 0.0f, nav_max_speed);
105  }
106 
107  // Calculate length of line segment
108  float length_line = Max(float_vect2_norm(&wp_diff), 0.01f);
109  // Normal vector to the line, with length of the line
110  struct FloatVect2 normalv;
111  VECT2_ASSIGN(normalv, -wp_diff.y, wp_diff.x);
112  // Length of normal vector is the same as of the line segment
113  float length_normalv = length_line; // >= 0.01
114  // Distance along the normal vector
115  float dist_to_line = (pos_diff.x * normalv.x + pos_diff.y * normalv.y) / length_normalv;
116  // Normal vector scaled to be the distance to the line
117  struct FloatVect2 v_to_line, v_along_line;
118  v_to_line.x = dist_to_line * normalv.x / length_normalv * guidance_indi_line_gain;
119  v_to_line.y = dist_to_line * normalv.y / length_normalv * guidance_indi_line_gain;
120  // The distance that needs to be traveled along the line
121  v_along_line.x = wp_diff.x / length_line * GUIDANCE_INDI_NAV_LINE_DIST;
122  v_along_line.y = wp_diff.y / length_line * GUIDANCE_INDI_NAV_LINE_DIST;
123  // Calculate the desired direction to converge to the line
124  struct FloatVect2 direction;
125  VECT2_SMUL(direction, v_along_line, (1.f / (1.f + fabsf(dist_to_line) * 0.05f)));
126  VECT2_ADD(direction, v_to_line);
127  float length_direction = Max(float_vect2_norm(&direction), 0.01f);
128  // Scale to have the desired speed
129  VECT2_SMUL(nav.speed, direction, desired_speed / length_direction);
130  // final target position, should be on the line, for display
131  VECT2_SUM(nav.target, *stateGetPositionEnu_f(), direction);
132 
133  nav_rotorcraft_base.goto_wp.from = *wp_start;
134  nav_rotorcraft_base.goto_wp.to = *wp_end;
135  nav_rotorcraft_base.goto_wp.dist2_to_wp = get_dist2_to_point(wp_end);
136  nav.horizontal_mode = NAV_HORIZONTAL_MODE_ROUTE;
137  nav.setpoint_mode = NAV_SETPOINT_MODE_SPEED;
138 }

140 static bool nav_hybrid_approaching(struct EnuCoor_f *wp, struct EnuCoor_f *from, float approaching_time)
141 {
142  float dist_to_point;
143  struct FloatVect2 diff;
144  struct EnuCoor_f *pos = stateGetPositionEnu_f();
145 
146  /* if an approaching_time is given, estimate diff after approching_time secs */
147  if (approaching_time > 0.f) {
148  struct FloatVect2 estimated_pos;
149  struct FloatVect2 estimated_progress;
150  struct EnuCoor_f *speed = stateGetSpeedEnu_f();
151  VECT2_SMUL(estimated_progress, *speed, approaching_time);
152  VECT2_SUM(estimated_pos, *pos, estimated_progress);
153  VECT2_DIFF(diff, *wp, estimated_pos);
154  }
155  /* else use current position */
156  else {
157  VECT2_DIFF(diff, *wp, *pos);
158  }
159  /* compute distance of estimated/current pos to target wp
160  */
161  dist_to_point = float_vect2_norm(&diff);
162 
163  /* return TRUE if we have arrived */
164  if (dist_to_point < ARRIVED_AT_WAYPOINT) {
165  return true;
166  }
167 
168  /* if coming from a valid waypoint */
169  if (from != NULL) {
170  /* return TRUE if normal line at the end of the segment is crossed */
171  struct FloatVect2 from_diff;
172  VECT2_DIFF(from_diff, *wp, *from);
173  return (diff.x * from_diff.x + diff.y * from_diff.y < 0.f);
174  }
175 
176  return false;
177 }

179 static void nav_hybrid_circle(struct EnuCoor_f *wp_center, float radius)
180 {
181  struct FloatVect2 pos_diff;
182  float desired_speed;
183 
184  VECT2_DIFF(pos_diff, *stateGetPositionEnu_f(), *wp_center);
185  // direction of rotation
186  float sign_radius = radius > 0.f ? 1.f : -1.f;
187  // absolute radius
188  float abs_radius = fabsf(radius);
189 
190  if (abs_radius > 0.1f) {
191  // store last qdr
192  float last_qdr = nav_rotorcraft_base.circle.qdr;
193  // compute qdr
194  nav_rotorcraft_base.circle.qdr = atan2f(pos_diff.y, pos_diff.x);
195  // increment circle radians
196  float trigo_diff = nav_rotorcraft_base.circle.qdr - last_qdr;
197  NormRadAngle(trigo_diff);
198  nav_rotorcraft_base.circle.radians += trigo_diff;
199  // progress angle
200  float progress_angle = GUIDANCE_INDI_NAV_CIRCLE_DIST / abs_radius;
201  Bound(progress_angle, M_PI / 16.f, M_PI / 4.f);
202  float alpha = nav_rotorcraft_base.circle.qdr - sign_radius * progress_angle;
203  // final target position, should be on the circle, for display
204  nav.target.x = wp_center->x + cosf(alpha) * abs_radius;
205  nav.target.y = wp_center->y + sinf(alpha) * abs_radius;
206  }
207  else {
208  // radius is too small, direct to center
209  VECT2_COPY(nav.target, *wp_center);
210  }
211  // compute desired speed
212  if (force_forward) {
213  desired_speed = nav_max_speed;
214  } else {
215  float radius_diff = fabsf(float_vect2_norm(&pos_diff) - abs_radius);
216  if (radius_diff > GUIDANCE_INDI_NAV_CIRCLE_DIST) {
217  // far from circle, speed proportional to diff
218  desired_speed = radius_diff * gih_params.pos_gain;
219  } else {
220  // close to circle, speed function of radius for a feasible turn
221  // MAX_BANK / 2 gives some margins for the turns
222  desired_speed = sqrtf(PPRZ_ISA_GRAVITY * abs_radius * tanf(GUIDANCE_H_MAX_BANK / 2.f));
223  }
224  Bound(desired_speed, 0.0f, nav_max_speed);
225  }
226  // compute speed vector from target position
227  struct FloatVect2 speed_unit;
228  VECT2_DIFF(speed_unit, nav.target, *stateGetPositionEnu_f());
229  float_vect2_normalize(&speed_unit);
230  VECT2_SMUL(nav.speed, speed_unit, desired_speed);
231 
232  nav_rotorcraft_base.circle.center = *wp_center;
233  nav_rotorcraft_base.circle.radius = radius;
234  nav.horizontal_mode = NAV_HORIZONTAL_MODE_CIRCLE;
235  nav.setpoint_mode = NAV_SETPOINT_MODE_SPEED;
236 }
