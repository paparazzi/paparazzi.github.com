65void baro_scp_event(void)
66{
67
68 if (scp_trans.status == I2CTransSuccess) {
69
70 if (baro_scp_status == BARO_SCP_RD_TEMP) {
71
72 /* read two byte temperature */
73 baro_scp_temperature = scp_trans.buf[0] << 8;
74 baro_scp_temperature |= scp_trans.buf[1];
75 if (baro_scp_temperature & 0x2000) {

84 }
85
86 else if (baro_scp_status == BARO_SCP_RD_PRESS_0) {
87
88 /* read one byte pressure */
