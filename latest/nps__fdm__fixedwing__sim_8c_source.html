156void nps_fdm_run_step(bool launch, double *commands, int commands_nb __attribute__((unused)))
157{
158
159 // commands
160 sim_state.delta_roll = commands[COMMAND_ROLL] * MAX_PPRZ;
161 sim_state.delta_pitch = -commands[COMMAND_PITCH] * MAX_PPRZ; // invert back to correct sign
162 sim_state.delta_thrust = commands[COMMAND_THROTTLE];
163
164 static bool already_launched = false;
165 if (launch && !already_launched) {
166 sim_state.airspeed = NOMINAL_AIRSPEED;
167 already_launched = true;
168 }
169 if (sim_state.airspeed == 0. && sim_state.delta_thrust > 0.) {
170 sim_state.airspeed = NOMINAL_AIRSPEED;
171 }
172
173 if (sim_state.pos.z >= -3. && sim_state.airspeed > 0.) {
174 double v2 = sim_state.airspeed * sim_state.airspeed;
175 double vn2 = NOMINAL_AIRSPEED * NOMINAL_AIRSPEED;
176
177 // roll dynamic
178 double c_l = 4e-5 * sim_state.delta_roll;
179 double phi_dot_dot = ROLL_RESPONSE_FACTOR * c_l * v2 / vn2 - sim_state.rates.phi;
180 sim_state.rates.phi = sim_state.rates.phi + phi_dot_dot * fdm.curr_dt;
181 BoundAbs(sim_state.rates.phi, ROLL_RATE_MAX_SETPOINT);
182 sim_state.attitude.phi = sim_state.attitude.phi + sim_state.rates.phi * fdm.curr_dt;
183 NormRadAngle(sim_state.attitude.phi);
184 BoundAbs(sim_state.attitude.phi, ROLL_MAX_SETPOINT);
185
186 // yaw dynamic
187 sim_state.rates.psi = PPRZ_ISA_GRAVITY / sim_state.airspeed * tan(YAW_RESPONSE_FACTOR * sim_state.attitude.phi);
188 sim_state.attitude.psi = sim_state.attitude.psi + sim_state.rates.psi * fdm.curr_dt;
189 NormRadAngle(sim_state.attitude.psi);
190
191 // Aerodynamic pitching moment coeff, proportional to elevator
192 // No Thrust moment, so null (0) for steady flight
193 double c_m = 5e-7 * sim_state.delta_pitch;
194 double theta_dot_dot = PITCH_RESPONSE_FACTOR * c_m * v2 - sim_state.rates.theta;
195 sim_state.rates.theta = sim_state.rates.theta + theta_dot_dot * fdm.curr_dt;
196 sim_state.attitude.theta = sim_state.attitude.theta + sim_state.rates.theta * fdm.curr_dt;
197
198 // Flight path angle
199 double gamma = atan2(sim_state.speed.z, sim_state.airspeed);
200
201 // Cz proportional to angle of attack
202 double alpha = sim_state.attitude.theta - gamma;
203 double c_z = 0.2 * alpha + PPRZ_ISA_GRAVITY / vn2;
204
205 // Lift
206 double lift = c_z * v2;
207 double z_dot_dot = lift / WEIGHT * cos(sim_state.attitude.theta) * cos(sim_state.attitude.phi) - PPRZ_ISA_GRAVITY;
208 sim_state.speed.z = sim_state.speed.z + z_dot_dot * fdm.curr_dt;
209 sim_state.pos.z = sim_state.pos.z + sim_state.speed.z * fdm.curr_dt;
210
211 // Constant Cx, drag to get expected cruise and maximum throttle
212 double cruise_th = AUTO_THROTTLE_NOMINAL_CRUISE_THROTTLE;
213 double drag = cruise_th + (v2 - vn2) * (1. - cruise_th) / (MAXIMUM_AIRSPEED*MAXIMUM_AIRSPEED - vn2);
214 double airspeed_dot = MAXIMUM_POWER / sim_state.airspeed * (sim_state.delta_thrust - drag) / WEIGHT - PPRZ_ISA_GRAVITY * sin(gamma);
215 sim_state.airspeed = sim_state.airspeed + airspeed_dot * fdm.curr_dt;
216 sim_state.airspeed = Max(sim_state.airspeed, 10.); // avoid stall
217
218 // Wind effect (FIXME apply to forces ?)
219 sim_state.speed.x = sim_state.airspeed * sin(sim_state.attitude.psi) + sim_state.wind.x;
220 sim_state.speed.y = sim_state.airspeed * cos(sim_state.attitude.psi) + sim_state.wind.y;
221 sim_state.pos.x = sim_state.pos.x + sim_state.speed.x * fdm.curr_dt;
222 sim_state.pos.y = sim_state.pos.y + sim_state.speed.y * fdm.curr_dt;
223 sim_state.pos.z = sim_state.pos.z + sim_state.wind.z * fdm.curr_dt;
224 }

226 /****************************************************************************/
227 // Coordenates transformations |
228 // ----------------------------|
229
230 /* in ECEF */
231 ecef_of_enu_point_d(&fdm.ecef_pos, &ltpdef, &sim_state.pos);
232 lla_of_ecef_d(&fdm.lla_pos, &fdm.ecef_pos);
233 ecef_of_enu_vect_d(&fdm.ecef_ecef_vel, &ltpdef, &sim_state.speed);
234 //ecef_of_enu_vect_d(&fdm.ecef_ecef_accel, &ltpdef, &rover_acc);
235
236 /* in NED */
237 ned_of_ecef_point_d(&fdm.ltpprz_pos, &ltpdef, &fdm.ecef_pos);
238 ned_of_ecef_vect_d(&fdm.ltpprz_ecef_vel, &ltpdef, &fdm.ecef_ecef_vel);
239 ned_of_ecef_vect_d(&fdm.ltpprz_ecef_accel, &ltpdef, &fdm.ecef_ecef_accel);
240 fdm.hmsl = -fdm.ltpprz_pos.z + NAV_ALT0 / 1000.;
241 fdm.agl = -fdm.ltpprz_pos.z; // flat Earth
242
243 /* Eulers */
244 fdm.ltp_to_body_eulers = sim_state.attitude;
245 // Exporting Eulers to AHRS (in quaternions)
246 double_quat_of_eulers(&fdm.ltp_to_body_quat, &fdm.ltp_to_body_eulers);
247
248 // Angular vel & acc
249 fdm.body_ecef_rotvel.p = sim_state.rates.phi;
250 fdm.body_ecef_rotvel.q = sim_state.rates.theta;
251 fdm.body_ecef_rotvel.r = sim_state.rates.psi;
252}

259static void init_ltp(void)
260{
261
262 struct LlaCoor_d llh_nav0; /* Height above the ellipsoid */
263 llh_nav0.lat = RadOfDeg((double)NAV_LAT0 / 1e7);
264 llh_nav0.lon = RadOfDeg((double)NAV_LON0 / 1e7);
265 llh_nav0.alt = (double)(NAV_ALT0) / 1000.0;
266
267 struct EcefCoor_d ecef_nav0;
268
269 ecef_of_lla_d(&ecef_nav0, &llh_nav0);
270
271 ltp_def_from_ecef_d(&ltpdef, &ecef_nav0);
272 fdm.ecef_pos = ecef_nav0;
273
274 // accel and mag data should not be used
275 // AHRS and INS are bypassed
276 fdm.ltp_g.x = 0.;
277 fdm.ltp_g.y = 0.;
278 fdm.ltp_g.z = 0.;
279
280 fdm.ltp_h.x = 1.;
281 fdm.ltp_h.y = 0.;
282 fdm.ltp_h.z = 0.;
283
284}
