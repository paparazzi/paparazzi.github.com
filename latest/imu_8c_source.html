375static void cmd_imu(shell_stream_t *sh, int argc, const char *const argv[])
376{
377 (void) argv;
378 int i = 0;
379 if (argc > 0) {
380 chprintf(sh, "Usage: imu\r\n");
381 return;
382 }
383 chprintf(sh, "GYRO IMU data\r\n");
384 for (i = 0; i < IMU_MAX_SENSORS; i++) {
385 if (imu.gyros[i].abi_id != 0) {
386 chprintf(sh, " Gyro id: %u, time %lu\r\n", imu.gyros[i].abi_id, imu.gyros[i].last_stamp);
387 show_calibrated(sh, &imu.gyros[i].calibrated);
388 show_rates(sh, "neutral", &imu.gyros[i].neutral);
389 show_ratesf(sh, "scale", &imu.gyros[i].scale_f);
390 show_matrix(sh, "body_to_sensor", &imu.gyros[i].body_to_sensor);
391 show_rates(sh, "unscaled", &imu.gyros[i].unscaled);
392 show_rates(sh, "scaled", &imu.gyros[i].scaled);
393 struct FloatRates gf;
394 RATES_FLOAT_OF_BFP(gf, imu.gyros[i].scaled);
395 chprintf(sh, " -> gyro (rad/s): %.3f, %.3f, %.3f\r\n", gf.p, gf.q, gf.r);
396 }
397 }
398 chprintf(sh, "ACCEL IMU data\r\n");
399 for (i = 0; i < IMU_MAX_SENSORS; i++) {
400 if (imu.accels[i].abi_id != 0) {
401 chprintf(sh, " Accel id: %u, time %lu\r\n", imu.accels[i].abi_id, imu.accels[i].last_stamp);
402 show_calibrated(sh, &imu.accels[i].calibrated);
403 show_vect3(sh, "neutral", &imu.accels[i].neutral);
404 show_vect3f(sh, "scale", &imu.accels[i].scale_f);
405 show_matrix(sh, "body_to_sensor", &imu.accels[i].body_to_sensor);
406 show_vect3(sh, "unscaled", &imu.accels[i].unscaled);
407 show_vect3(sh, "scaled", &imu.accels[i].scaled);
408 struct FloatVect3 af;
409 ACCELS_FLOAT_OF_BFP(af, imu.accels[i].scaled);
410 chprintf(sh, " -> accel (m/s2): %.3f, %.3f, %.3f\r\n", af.x, af.y, af.z);
411 }
412 }
413 chprintf(sh, "MAG IMU data\r\n");
414 for (i = 0; i < IMU_MAX_SENSORS; i++) {
415 if (imu.mags[i].abi_id != 0) {
416 chprintf(sh, " Mag id: %u\r\n", imu.mags[i].abi_id);
417 show_calibrated(sh, &imu.mags[i].calibrated);
418 show_vect3(sh, "neutral", &imu.mags[i].neutral);
419 show_vect3f(sh, "scale", &imu.mags[i].scale_f);
420 show_matrix(sh, "body_to_sensor", &imu.mags[i].body_to_sensor);
421 show_vect3(sh, "unscaled", &imu.mags[i].unscaled);
422 show_vect3(sh, "scaled", &imu.mags[i].scaled);
423 struct FloatVect3 mf;
424 MAGS_FLOAT_OF_BFP(mf, imu.mags[i].scaled);
425 chprintf(sh, " -> mag (unit): %.3f, %.3f, %.3f\r\n", mf.x, mf.y, mf.z);
426 }
427 }

473 }

541

546 imu.mags[i].calibrated.scale = false;

548 imu.mags[i].calibrated.rotation = false;

551 imu.mags[i] = mag_calib[i];
552 }
553
554 // Set the default safe values if not calibrated
555 if(!imu.mags[i].calibrated.neutral) {

558

695static void imu_gyro_raw_cb(uint8_t sender_id, uint32_t stamp, struct Int32Rates *data, uint8_t samples, float rate, float temp)
696{
697 // Find the correct gyro
698 struct imu_gyro_t *gyro = imu_get_gyro(sender_id, true);
699 if(gyro == NULL || samples < 1)
700 return;
701
702 // Filter the gyro
703 struct Int32Rates data_filtered[samples];
704 if(gyro->calibrated.filter) {
705 for(uint8_t i = 0; i < samples; i++) {
706 data_filtered[i].p = update_butterworth_2_low_pass(&gyro->filter[0], data[i].p);
707 data_filtered[i].q = update_butterworth_2_low_pass(&gyro->filter[1], data[i].q);
708 data_filtered[i].r = update_butterworth_2_low_pass(&gyro->filter[2], data[i].r);
709 }
710 data = data_filtered;
711 }
712
713 // Copy last sample as unscaled
714 RATES_COPY(gyro->unscaled, data[samples-1]);
715
716 // Scale the gyro
717 struct Int32Rates scaled, scaled_rot;
718 scaled.p = (gyro->unscaled.p - gyro->neutral.p) * gyro->scale_f.p;
719 scaled.q = (gyro->unscaled.q - gyro->neutral.q) * gyro->scale_f.q;
720 scaled.r = (gyro->unscaled.r - gyro->neutral.r) * gyro->scale_f.r;
721
722 // Rotate the sensor
723 int32_rmat_transp_ratemult(&scaled_rot, &gyro->body_to_sensor, &scaled);
724
725#if IMU_INTEGRATION
726 // Only integrate if we have gotten a previous measurement and didn't overflow the timer
727 if(!isnan(rate) && gyro->last_stamp > 0 && stamp > gyro->last_stamp) {
728 struct FloatRates integrated;
729
730 // Trapezoidal integration (TODO: coning correction)
731 integrated.p = RATE_FLOAT_OF_BFP(gyro->scaled.p + scaled_rot.p) * 0.5f;
732 integrated.q = RATE_FLOAT_OF_BFP(gyro->scaled.q + scaled_rot.q) * 0.5f;
733 integrated.r = RATE_FLOAT_OF_BFP(gyro->scaled.r + scaled_rot.r) * 0.5f;
734
735 // Only perform multiple integrations in sensor frame if needed
736 if(samples > 1) {
737 struct FloatRates integrated_sensor;
738 struct FloatRMat body_to_sensor;
739 // Rotate back to sensor frame
740 RMAT_FLOAT_OF_BFP(body_to_sensor, gyro->body_to_sensor);
741 float_rmat_ratemult(&integrated_sensor, &body_to_sensor, &integrated);
742
743 // Add all the other samples
744 for(uint8_t i = 0; i < samples-1; i++) {
745 struct FloatRates f_sample;
746 f_sample.p = RATE_FLOAT_OF_BFP((data[i].p - gyro->neutral.p) * gyro->scale_f.p);
747 f_sample.q = RATE_FLOAT_OF_BFP((data[i].q - gyro->neutral.q) * gyro->scale_f.q);
748 f_sample.r = RATE_FLOAT_OF_BFP((data[i].r - gyro->neutral.r) * gyro->scale_f.r);
749
750
751#if IMU_LOG_HIGHSPEED
752 pprz_msg_send_IMU_GYRO(&pprzlog_tp.trans_tx, &(IMU_LOG_HIGHSPEED_DEVICE).device, AC_ID, &sender_id, &f_sample.p, &f_sample.q, &f_sample.r);
753#endif
754
755 integrated_sensor.p += f_sample.p;
756 integrated_sensor.q += f_sample.q;
757 integrated_sensor.r += f_sample.r;
758 }
759
760 // Rotate to body frame
761 float_rmat_transp_ratemult(&integrated, &body_to_sensor, &integrated_sensor);
762 }
763
764 // Divide by the time of the collected samples
765 integrated.p = integrated.p * (1.f / rate);
766 integrated.q = integrated.q * (1.f / rate);
767 integrated.r = integrated.r * (1.f / rate);
768
769 // Send the integrated values
770 uint16_t dt = (1e6 / rate) * samples;
771 AbiSendMsgIMU_GYRO_INT(sender_id, stamp, &integrated, dt);
772 }
773#else
774 (void)rate; // Surpress compile warning not used
775#endif
776
777#if IMU_LOG_HIGHSPEED
778 struct FloatRates f_sample;
779 RATES_FLOAT_OF_BFP(f_sample, scaled);
780 pprz_msg_send_IMU_GYRO(&pprzlog_tp.trans_tx, &(IMU_LOG_HIGHSPEED_DEVICE).device, AC_ID, &sender_id, &f_sample.p, &f_sample.q, &f_sample.r);
781#endif
782
783 // Copy and send
784 gyro->temperature = temp;
785 RATES_COPY(gyro->scaled, scaled_rot);
786 AbiSendMsgIMU_GYRO(sender_id, stamp, &gyro->scaled);
787 gyro->last_stamp = stamp;
788}

887{
888 // Find the correct mag
889 struct imu_mag_t *mag = imu_get_mag(sender_id, true);
890 if(mag == NULL)
891 return;
892
893 // Calculate current compensation
894 struct Int32Vect3 mag_correction;
895 mag_correction.x = (int32_t)(mag->current_scale.x * (float) electrical.current);
896 mag_correction.y = (int32_t)(mag->current_scale.y * (float) electrical.current);
897 mag_correction.z = (int32_t)(mag->current_scale.z * (float) electrical.current);
898
899 // Copy last sample as unscaled
900 VECT3_COPY(mag->unscaled, *data);
901
902 // Scale the mag
903 struct Int32Vect3 scaled;
904 scaled.x = (mag->unscaled.x - mag_correction.x - mag->neutral.x) * mag->scale_f.x;
905 scaled.y = (mag->unscaled.y - mag_correction.y - mag->neutral.y) * mag->scale_f.y;
906 scaled.z = (mag->unscaled.z - mag_correction.z - mag->neutral.z) * mag->scale_f.z;
907
908
909 // Rotate the sensor
910 int32_rmat_transp_vmult(&mag->scaled, &mag->body_to_sensor, &scaled);
911 AbiSendMsgIMU_MAG(sender_id, stamp, &mag->scaled);
912}
