118void baro_amsys_read_periodic(void)
119{
120 // Initiate next read
121 if (baro_amsys_i2c_trans.status == I2CTransDone) {
122#ifndef MEASURE_AMSYS_TEMPERATURE
123 i2c_receive(&BARO_AMSYS_I2C_DEV, &baro_amsys_i2c_trans, BARO_AMSYS_ADDR, 2);
124#else
125 i2c_receive(&BARO_AMSYS_I2C_DEV, &baro_amsys_i2c_trans, BARO_AMSYS_ADDR, 4);
126#endif
127 }
128
129#ifdef BARO_AMSYS_SYNC_SEND
130 DOWNLINK_SEND_AMSYS_BARO(DefaultChannel, DefaultDevice, &pBaroRaw, &baro_amsys_p, &baro_amsys_offset, &ref_alt_init,
131 &baro_amsys_abs_altitude, &baro_amsys_altitude, &baro_amsys_temp);
132#else
133 RunOnceEvery(10, DOWNLINK_SEND_AMSYS_BARO(DefaultChannel, DefaultDevice, &pBaroRaw, &baro_amsys_p, &baro_amsys_offset,
134 &ref_alt_init, &baro_amsys_abs_altitude, &baro_amsys_altitude, &baro_amsys_temp));
135#endif
136}

138void baro_amsys_read_event(void)
139{
140 pBaroRaw = 0;
141 // Get raw altimeter from buffer
142 pBaroRaw = (baro_amsys_i2c_trans.buf[0] << 8) | baro_amsys_i2c_trans.buf[1];
143#ifdef MEASURE_AMSYS_TEMPERATURE
144 tBaroRaw = (baro_amsys_i2c_trans.buf[2] << 8) | baro_amsys_i2c_trans.buf[3];
145 const float temp_off_scale = (float)(TEMPERATURE_AMSYS_MAX - TEMPERATURE_AMSYS_MIN) /
146 (TEMPERATURE_AMSYS_OFFSET_MAX - TEMPERATURE_AMSYS_OFFSET_MIN);
147 // Tmin=-25, Tmax=85
148 baro_amsys_temp = temp_off_scale * (tBaroRaw - TEMPERATURE_AMSYS_OFFSET_MIN) +
149 TEMPERATURE_AMSYS_MIN;
150#endif
151 // Check if this is valid altimeter
152 if (pBaroRaw == 0) {
153 baro_amsys_valid = false;
154 } else {
155 baro_amsys_valid = true;
156 }
157
158 baro_amsys_adc = pBaroRaw;
159
160 // Continue only if a new altimeter value was received
161 //if (baro_amsys_valid && GpsFixValid()) {
162 if (baro_amsys_valid) {
163 uint32_t now_ts = get_sys_time_usec();
164
165 //Cut RAW Min and Max
166 if (pBaroRaw < BARO_AMSYS_OFFSET_MIN) {
167 pBaroRaw = BARO_AMSYS_OFFSET_MIN;
168 }
169 if (pBaroRaw > BARO_AMSYS_OFFSET_MAX) {
170 pBaroRaw = BARO_AMSYS_OFFSET_MAX;
171 }
172
173 //Convert to pressure
174 baro_amsys_p = (float)(pBaroRaw - BARO_AMSYS_OFFSET_MIN) * BARO_AMSYS_MAX_PRESSURE / (float)(
175 BARO_AMSYS_OFFSET_MAX - BARO_AMSYS_OFFSET_MIN);
176 // Send pressure over ABI
177 AbiSendMsgBARO_ABS(BARO_AMSYS_SENDER_ID, now_ts, baro_amsys_p);
178 // compute altitude localy
179 if (!baro_amsys_offset_init) {
180 --baro_amsys_cnt;
181 // Check if averaging completed
182 if (baro_amsys_cnt == 0) {
183 // Calculate average
184 baro_amsys_offset = (float)(baro_amsys_offset_tmp / BARO_AMSYS_OFFSET_NBSAMPLES_AVRG);
185 ref_alt_init = GROUND_ALT;
186 baro_amsys_offset_init = true;
187
188 // hight over Sea level at init point
189 //baro_amsys_offset_altitude = 288.15 / 0.0065 * (1 - pow((baro_amsys_p)/1013.25 , 1/5.255));
190 }
191 // Check if averaging needs to continue
192 else if (baro_amsys_cnt <= BARO_AMSYS_OFFSET_NBSAMPLES_AVRG) {
193 baro_amsys_offset_tmp += baro_amsys_p;
194 }
195
196 baro_amsys_altitude = 0.0;
197
198 } else {
199 // Lowpassfiltering and convert pressure to altitude
200 baro_amsys_altitude = baro_filter * baro_old + (1 - baro_filter) * (baro_amsys_offset - baro_amsys_p) * baro_scale /
201 (1.2041 * 9.81);
202 baro_old = baro_amsys_altitude;
203 //New value available
204 }
205 baro_amsys_abs_altitude = baro_amsys_altitude + ref_alt_init;
206 } /*else {
207 baro_amsys_abs_altitude = 0.0;
208 }*/
209
210
211 // Transaction has been read
212 baro_amsys_i2c_trans.status = I2CTransDone;
213}
